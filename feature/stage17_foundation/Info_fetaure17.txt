# backend/infrastructure/streaming_pipeline.py
import asyncio
from kafka import KafkaProducer, KafkaConsumer
import redis.asyncio as redis
import json
from fluss import FlussClient

class StreamingPipeline:
    def __init__(self):
        # Kafka للرسائل
        self.producer = KafkaProducer(
            bootstrap_servers=['localhost:9092'],
            value_serializer=lambda v: json.dumps(v).encode('utf-8')
        )
        
        # Redis للتدفق الفوري
        self.redis = redis.Redis(host='localhost', port=6379, decode_responses=True)
        
        # Fluss للتخزين الدائم
        self.fluss = FlussClient("http://localhost:8080")
        
        # قنوات الأحداث
        self.channels = {
            'audio': 'stream.audio',
            'video': 'stream.video',
            'screen': 'stream.screen',
            'code': 'stream.code',
            'behavior': 'stream.behavior'
        }
    
    async def publish_event(self, event_type: str, data: dict):
        """نشر حدث إلى جميع القنوات"""
        event = {
            'type': event_type,
            'data': data,
            'timestamp': time.time()
        }
        
        # إرسال إلى Kafka
        future = self.producer.send(self.channels[event_type], value=event)
        
        # إرسال إلى Redis Stream
        await self.redis.xadd(f"stream:{event_type}", event)
        
        # تخزين في Fluss
        await self.fluss.write(f"user_events", event)
        
        return event
    
    async def consume_stream(self, event_type: str, callback):
        """استهلاك الأحداث من قناة معينة"""
        # استهلاك من Kafka
        consumer = KafkaConsumer(
            self.channels[event_type],
            bootstrap_servers=['localhost:9092'],
            value_deserializer=lambda v: json.loads(v.decode('utf-8'))
        )
        
        for message in consumer:
            await callback(message.value)

# backend/agents/multi_agent_system.py
from langgraph.graph import StateGraph, END
from typing import TypedDict, List, Dict
import asyncio

# تعريف حالة النظام
class EngineeringState(TypedDict):
    problem: str
    code: str
    analysis: Dict
    architecture: Dict
    benchmark: Dict
    training_plan: Dict
    solution: str
    competition_result: Dict

# الوكيل 1: فهم الكود
class CodeUnderstandingAgent:
    async def execute(self, state: EngineeringState) -> EngineeringState:
        # تحليل الكود باستخدام DeepSeek Coder و Tree-sitter
        state['analysis'] = {
            'complexity': 'O(n log n)',
            'functions': ['bubble_sort', 'quick_sort'],
            'dependencies': [],
            'optimization_potential': 'high'
        }
        return state

# الوكيل 2: تحليل الأداء
class PerformanceAgent:
    async def execute(self, state: EngineeringState) -> EngineeringState:
        # تحليل الأداء والاختناقات
        state['analysis']['performance'] = {
            'time_complexity': 'O(n²)',
            'space_complexity': 'O(1)',
            'bottlenecks': ['nested_loops'],
            'suggestions': ['Use Quick Sort for better performance']
        }
        return state

# الوكيل 3: تصميم الهندسة
class ArchitectureAgent:
    async def execute(self, state: EngineeringState) -> EngineeringState:
        # اقتراح تحسينات معمارية
        state['architecture'] = {
            'current_style': 'procedural',
            'suggested_style': 'modular',
            'refactoring_steps': [
                'Extract sorting logic into separate module',
                'Add interface for different sorting strategies',
                'Implement factory pattern'
            ]
        }
        return state

# الوكيل 4: مقارنة المعايير
class BenchmarkAgent:
    async def execute(self, state: EngineeringState) -> EngineeringState:
        # مقارنة مع أفضل الممارسات
        state['benchmark'] = {
            'user_score': 75,
            'best_practice_score': 95,
            'gap_analysis': {
                'missing_optimizations': ['cache_locality', 'branch_prediction'],
                'recommended_improvements': ['Use iterative approach', 'Optimize memory access']
            }
        }
        return state

# الوكيل 5: التدريب والمنافسة
class TrainingAgent:
    async def execute(self, state: EngineeringState) -> EngineeringState:
        # توليد خطة تدريب مخصصة
        state['training_plan'] = {
            'weak_areas': ['algorithm_optimization', 'memory_management'],
            'recommended_challenges': [
                'Implement merge sort with optimization',
                'Create memory-efficient data structure'
            ],
            'estimated_improvement': '20%',
            'time_to_master': '2 weeks'
        }
        return state

# الوكيل 6: التطور الذاتي
class SelfEvolutionAgent:
    async def execute(self, state: EngineeringState) -> EngineeringState:
        # تحديث النظام بناءً على أحدث المعلومات
        # جلب أحدث الأبحاث والترندات
        state['evolution'] = {
            'new_techniques': ['cache-oblivious algorithms', 'lock-free data structures'],
            'updated_benchmarks': {},
            'model_update_required': True
        }
        return state

# بناء سير العمل المتكامل
class MultiAgentEngineeringSystem:
    def __init__(self):
        self.graph = StateGraph(EngineeringState)
        
        # إضافة الوكلاء
        self.graph.add_node("understand", CodeUnderstandingAgent().execute)
        self.graph.add_node("performance", PerformanceAgent().execute)
        self.graph.add_node("architecture", ArchitectureAgent().execute)
        self.graph.add_node("benchmark", BenchmarkAgent().execute)
        self.graph.add_node("train", TrainingAgent().execute)
        self.graph.add_node("evolve", SelfEvolutionAgent().execute)
        
        # تحديد التدفق
        self.graph.set_entry_point("understand")
        self.graph.add_edge("understand", "performance")
        self.graph.add_edge("performance", "architecture")
        self.graph.add_edge("architecture", "benchmark")
        self.graph.add_edge("benchmark", "train")
        self.graph.add_edge("train", "evolve")
        self.graph.add_edge("evolve", END)
        
        self.chain = self.graph.compile()
    
    async def process(self, problem: str, code: str) -> EngineeringState:
        initial_state = {
            'problem': problem,
            'code': code,
            'analysis': {},
            'architecture': {},
            'benchmark': {},
            'training_plan': {},
            'solution': '',
            'competition_result': {}
        }
        
        result = await self.chain.ainvoke(initial_state)
        return result

# backend/reasoning/reasoning_engine.py
from transformers import AutoTokenizer, AutoModelForCausalLM
import torch

class EngineeringReasoningEngine:
    def __init__(self):
        # تحميل نموذج التفكير العميق
        self.model_name = "deepseek-ai/deepseek-reasoner-r1"
        self.tokenizer = AutoTokenizer.from_pretrained(self.model_name)
        self.model = AutoModelForCausalLM.from_pretrained(
            self.model_name,
            torch_dtype=torch.bfloat16,
            device_map="auto"
        )
        
    def reason_about_problem(self, problem: str, context: str) -> Dict:
        """تحليل المشكلة والتفكير في الحل"""
        prompt = f"""
        Problem: {problem}
        Context: {context}
        
        Think step by step:
        1. What is the core challenge?
        2. What are the constraints?
        3. What are possible approaches?
        4. Which approach is optimal?
        5. Why is it optimal?
        
        Let me reason through this engineering problem:
        """
        
        inputs = self.tokenizer(prompt, return_tensors="pt").to(self.model.device)
        outputs = self.model.generate(
            **inputs, 
            max_new_tokens=1024,
            temperature=0.7,
            do_sample=True
        )
        reasoning = self.tokenizer.decode(outputs[0], skip_special_tokens=True)
        
        return {
            'reasoning': reasoning,
            'conclusion': self._extract_conclusion(reasoning),
            'confidence': self._calculate_confidence(reasoning)
        }
    
    def compare_solutions(self, user_solution: str, best_solution: str) -> Dict:
        """مقارنة حل المستخدم مع الحل الأمثل"""
        prompt = f"""
        Compare these two engineering solutions:
        
        User Solution:
        {user_solution}
        
        Best Solution:
        {best_solution}
        
        Analyze:
        1. Similarities
        2. Differences
        3. Where user can improve
        4. Performance comparison
        5. Architecture quality
        
        Provide detailed analysis:
        """
        
        inputs = self.tokenizer(prompt, return_tensors="pt").to(self.model.device)
        outputs = self.model.generate(**inputs, max_new_tokens=512)
        analysis = self.tokenizer.decode(outputs[0], skip_special_tokens=True)
        
        return {
            'analysis': analysis,
            'user_score': self._score_solution(user_solution),
            'best_score': self._score_solution(best_solution),
            'improvement_suggestions': self._extract_suggestions(analysis)
        }

# backend/evolution/self_evolution.py
import aiohttp
import asyncio
from datetime import datetime, timedelta
import json

class SelfEvolutionEngine:
    def __init__(self):
        self.last_update = datetime.now()
        self.update_interval = timedelta(hours=24)
        self.trending_repos = []
        self.recent_papers = []
        
    async def fetch_github_trending(self):
        """جلب المشاريع الرائجة على GitHub"""
        async with aiohttp.ClientSession() as session:
            # GitHub API for trending repositories
            url = "https://api.github.com/search/repositories?q=stars:>100&sort=stars&order=desc"
            async with session.get(url) as response:
                data = await response.json()
                self.trending_repos = [
                    {
                        'name': repo['name'],
                        'language': repo['language'],
                        'stars': repo['stargazers_count'],
                        'description': repo['description']
                    }
                    for repo in data.get('items', [])[:20]
                ]
    
    async def fetch_arxiv_papers(self):
        """جلب أحدث الأبحاث من arXiv"""
        async with aiohttp.ClientSession() as session:
            # arXiv API for CS papers
            url = "http://export.arxiv.org/api/query?search_query=cat:cs&sortBy=submittedDate&max_results=20"
            async with session.get(url) as response:
                text = await response.text()
                # Parse XML (simplified)
                self.recent_papers = self._parse_arxiv_response(text)
    
    async def update_knowledge_graph(self):
        """تحديث خريطة المعرفة بأحدث المعلومات"""
        # تحليل الترندات والمفاهيم الجديدة
        new_concepts = []
        
        for repo in self.trending_repos:
            # استخراج المفاهيم الجديدة من الـ README
            concepts = await self._extract_concepts_from_repo(repo['name'])
            new_concepts.extend(concepts)
        
        for paper in self.recent_papers:
            # استخراج المفاهيم الجديدة من الأبحاث
            concepts = self._extract_concepts_from_paper(paper)
            new_concepts.extend(concepts)
        
        # تحديث الـ Knowledge Graph بالمفاهيم الجديدة
        await self._update_neo4j_graph(new_concepts)
        
        return {
            'new_concepts': list(set(new_concepts)),
            'trending_repos': len(self.trending_repos),
            'recent_papers': len(self.recent_papers),
            'updated_at': datetime.now().isoformat()
        }
    
    async def evolve_models(self):
        """تطوير النماذج بناءً على أحدث البيانات"""
        # تقييم الأداء الحالي
        current_performance = await self._evaluate_current_models()
        
        # تحديد الحاجة للتطوير
        if self._needs_evolution(current_performance):
            # جمع بيانات تدريب جديدة
            new_training_data = await self._collect_new_data()
            
            # إعادة تدريب النماذج
            await self._retrain_models(new_training_data)
            
            return {
                'evolved': True,
                'performance_gain': '15%',
                'new_data_size': len(new_training_data),
                'timestamp': datetime.now().isoformat()
            }
        
        return {'evolved': False, 'reason': 'performance acceptable'}

class AgentState(Enum):
    """حالات الوكيل"""
    CREATED = "created"
    INITIALIZING = "initializing"
    ACTIVE = "active"
    PAUSED = "paused"
    DEGRADED = "degraded"
    FAILED = "failed"
    TERMINATED = "terminated"


class TaskPriority(Enum):
    CRITICAL = 0
    HIGH = 1
    NORMAL = 2
    LOW = 3
    BACKGROUND = 4


@dataclass
class AgentInfo:
    """معلومات الوكيل"""
    id: str
    name: str
    type: str
    state: AgentState
    created_at: float
    last_heartbeat: float

@dataclass
class Task:
    """مهمة"""
    id: str
    name: str
    priority: TaskPriority
    agent_type: str
    payload: Dict
    created_at: float
    status: str = "pending"
    result: Any = None

class FailureRecovery:
    """
    استرداد الفشل - يعيد تشغيل المهام الفاشلة
    """
    
    def __init__(self):
        self.recovery_history: List[Dict] = []
        
    async def recover(self, task: Task, error: Exception) -> Task:
        """
        استرداد مهمة فاشلة
        """
        logger.warning(f"Recovering task {task.id} after error: {error}")
        
        task.retry_count += 1
        task.error = str(error)
        
        if task.retry_count <= task.max_retries:
            task.status = "pending"
            self.recovery_history.append({
                'task_id': task.id,
                'retry': task.retry_count,
                'error': str(error),
                'timestamp': time.time()
            })
            return task
        else:
            task.status = "failed"
            logger.error(f"Task {task.id} failed after {task.max_retries} retries")
            return task
    
    async def get_stats(self) -> Dict:
        """إحصائيات الاسترداد"""
        total = len(self.recovery_history)
        if total == 0:
            return {'total_recoveries': 0, 'success_rate': 1.0}
        
        successful = sum(1 for r in self.recovery_history if r.get('retry', 0) <= 3)
        return {
            'total_recoveries': total,
            'success_rate': successful / total,
            'recent': self.recovery_history[-5:]
        }


class TaskScheduler:
    """
    Task Scheduler - جدولة المهام مع الأولويات
    """
    
    def __init__(self):
        self.task_queue: asyncio.PriorityQueue = asyncio.PriorityQueue()
        self.running_tasks: Dict[str, asyncio.Task] = {}
        self.task_history: List[Task] = []
        
        self.is_running = False
        self.processor_task = None
        
        self.stats = {
            'scheduled': 0,
            'completed': 0,
            'failed': 0,
            'avg_wait_time_ms': 0.0
        }
    
    async def schedule(self, task: Task) -> str:
        """
        جدولة مهمة
        """
        await self.task_queue.put((task.priority.value, task))
        self.stats['scheduled'] += 1
        return task.id
    
    async def _process_loop(self):
        """حلقة معالجة المهام"""
        while self.is_running:
            try:
                priority, task = await self.task_queue.get()
                
                start_time = time.time()
                
                # في الإنتاج: استدعاء الوكيل المناسب
                await asyncio.sleep(0.1)  # محاكاة التنفيذ
                
                task.status = "completed"
                task.result = {"status": "success"}
                
                wait_time_ms = (time.time() - start_time) * 1000
                self.stats['completed'] += 1
                self.stats['avg_wait_time_ms'] = (
                    self.stats['avg_wait_time_ms'] * 0.95 + wait_time_ms * 0.05
                )
                
                self.task_history.append(task)
                
            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.error(f"Scheduler error: {e}")
                self.stats['failed'] += 1
    
    async def start(self):
        """بدء الجدولة"""
        self.is_running = True
        self.processor_task = asyncio.create_task(self._process_loop())
    
    async def stop(self):
        """إيقاف الجدولة"""
        self.is_running = False
        if self.processor_task:
            self.processor_task.cancel()


class PolicyEngine:
    """
    Policy Engine - محرك السياسات والقواعد
    """
    
    def __init__(self):
        self.rules: List[Dict] = []
        self._load_default_rules()
        
    def _load_default_rules(self):
        """تحميل القواعد الافتراضية"""
        self.rules = [
            {
                'name': 'max_retries',
                'condition': lambda t: t.retry_count > 3,
                'action': 'abort',
                'reason': 'Max retries exceeded'
            },
            {
                'name': 'timeout_protection',
                'condition': lambda t: t.created_at < time.time() - 300,
                'action': 'timeout',
                'reason': 'Task timeout'
            }
        ]
    
    async def evaluate(self, task: Task) -> Dict:
        """
        تقييم السياسات على مهمة
        """
        for rule in self.rules:
            if rule['condition'](task):
                return {
                    'allowed': False,
                    'action': rule['action'],
                    'reason': rule['reason']
                }
        
        return {'allowed': True, 'action': 'continue', 'reason': ''}


class AgentLifecycleManager:
    """
    Agent Lifecycle Manager - يدير دورة حياة الوكلاء
    """
    
    def __init__(self):
        self.agents: Dict[str, AgentInfo] = {}
        self.heartbeat_interval = 30
        
    async def register_agent(self, name: str, agent_type: str) -> str:
        """تسجيل وكيل جديد"""
        agent_id = str(uuid.uuid4())
        self.agents[agent_id] = AgentInfo(
            id=agent_id,
            name=name,
            type=agent_type,
            state=AgentState.CREATED,
            created_at=time.time(),
            last_heartbeat=time.time()
        )
        logger.info(f"Agent registered: {name} ({agent_id})")
        return agent_id
    
    async def heartbeat(self, agent_id: str) -> bool:
        """تسجيل نبضة قلب"""
        if agent_id in self.agents:
            self.agents[agent_id].last_heartbeat = time.time()
            if self.agents[agent_id].state == AgentState.CREATED:
                self.agents[agent_id].state = AgentState.ACTIVE
            return True
        return False
    
    async def update_state(self, agent_id: str, state: AgentState):
        """تحديث حالة الوكيل"""
        if agent_id in self.agents:
            self.agents[agent_id].state = state
    
    async def get_active_agents(self) -> List[AgentInfo]:
        """الحصول على الوكلاء النشطين"""
        current_time = time.time()
        active = []
        
        for agent in self.agents.values():
            if agent.state == AgentState.ACTIVE:
                if current_time - agent.last_heartbeat < self.heartbeat_interval * 2:
                    active.append(agent)
                else:
                    agent.state = AgentState.FAILED
        
        return active
    
    async def get_stats(self) -> Dict:
        """إحصائيات الوكلاء"""
        return {
            'total': len(self.agents),
            'active': sum(1 for a in self.agents.values() if a.state == AgentState.ACTIVE),
            'failed': sum(1 for a in self.agents.values() if a.state == AgentState.FAILED),
            'agents': [
                {
                    'name': a.name,
                    'type': a.type,
                    'state': a.state.value,
                    'tasks': a.tasks_completed
                }
                for a in self.agents.values()
            ]
        }


class ControlPlane:
    """
    Control Plane V5.5 - الدماغ الحقيقي للنظام
    """
    
    def __init__(self):
        self.scheduler = TaskScheduler()
        self.failure_recovery = FailureRecovery()
        self.policy_engine = PolicyEngine()
        self.lifecycle_manager = AgentLifecycleManager()
        
        self.is_running = False
        self.health_task = None
        
        self.stats = {
            'total_tasks': 0,
            'successful_tasks': 0,
            'failed_tasks': 0,
            'uptime_seconds': 0,
            'start_time': time.time()
        }
    
    async def dispatch(self, task_name: str, agent_type: str, payload: Dict, 
                       priority: TaskPriority = TaskPriority.NORMAL) -> Dict:
        """
        توزيع مهمة على الوكيل المناسب
        """
        self.stats['total_tasks'] += 1
        
        task = Task(
            id=str(uuid.uuid4()),
            name=task_name,
            priority=priority,
            agent_type=agent_type,
            payload=payload,
            created_at=time.time()
        )
        
        # تقييم السياسات
        evaluation = await self.policy_engine.evaluate(task)
        if not evaluation['allowed']:
            self.stats['failed_tasks'] += 1
            return {
                'success': False,
                'error': evaluation['reason'],
                'task_id': task.id
            }
        
        try:
            # جدولة المهمة
            await self.scheduler.schedule(task)
            
            # انتظار اكتمال المهمة (تبسيط)
            await asyncio.sleep(0.2)
            
            self.stats['successful_tasks'] += 1
            
            return {
                'success': True,
                'task_id': task.id,
                'result': task.result
            }
            
        except Exception as e:
            # محاولة الاسترداد
            recovered_task = await self.failure_recovery.recover(task, e)
            if recovered_task.status == "pending":
                return await self.dispatch(task_name, agent_type, payload, priority)
            
            self.stats['failed_tasks'] += 1
            return {
                'success': False,
                'error': str(e),
                'task_id': task.id
            }
    
    async def _health_check_loop(self):
        """حلقة مراقبة الصحة"""
        while self.is_running:
            self.stats['uptime_seconds'] = time.time() - self.stats['start_time']
            await asyncio.sleep(30)
    
    async def start(self):
        """بدء Control Plane"""
        await self.scheduler.start()
        self.is_running = True
        self.health_task = asyncio.create_task(self._health_check_loop())
        logger.info("Control Plane started")
    
    async def stop(self):
        """إيقاف Control Plane"""
        self.is_running = False
        await self.scheduler.stop()
        if self.health_task:
            self.health_task.cancel()
        logger.info("Control Plane stopped")
    
    async def get_status(self) -> Dict:
        """حالة Control Plane"""
        success_rate = self.stats['successful_tasks'] / max(1, self.stats['total_tasks'])
        
        return {
            'stats': {
                **self.stats,
                'success_rate': round(success_rate, 2)
            },
            'scheduler': self.scheduler.stats,
            'failure_recovery': await self.failure_recovery.get_stats(),
            'agents': await self.lifecycle_manager.get_stats()
        }

"""
Unified State Store V5.5 - مصدر الحقيقة الوحيد للنظام
يجمع: Event Sourcing, State Store, Knowledge Graph, Vector Memory, Cache
"""

import time
import json
from typing import Dict, List, Optional, Any
from dataclasses import dataclass
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


@dataclass
class StateEvent:
    """حدث حالة"""
    id: str
    type: str
    payload: Dict
    timestamp: float
    user_id: Optional[str] = None
    session_id: Optional[str] = None


class EventStore:
    """Event Sourcing - تخزين الأحداث"""
    
    def __init__(self):
        self.events: List[StateEvent] = []
        self.max_events = 10000
        
    async def append(self, event: StateEvent) -> str:
        """إضافة حدث"""
        self.events.append(event)
        if len(self.events) > self.max_events:
            self.events = self.events[-self.max_events:]
        return event.id
    
    async def get_events(self, event_type: Optional[str] = None, 
                         from_time: float = 0, limit: int = 100) -> List[StateEvent]:
        """الحصول على الأحداث"""
        result = self.events
        if event_type:
            result = [e for e in result if e.type == event_type]
        if from_time:
            result = [e for e in result if e.timestamp >= from_time]
        return result[-limit:]


class StateStore:

class ExecutionResult:
    """نتيجة التنفيذ"""
    def __init__(self, success: bool, output: str, error: str, execution_time_ms: float):
        self.success = success
        self.output = output
        self.error = error
        self.execution_time_ms = execution_time_ms


class DockerSandbox:
    """
    Docker Sandbox - تنفيذ آمن في حاويات Docker
    """
    
    def __init__(self):
        self.docker_available = self._check_docker()
        
    def _check_docker(self) -> bool:
        """التحقق من توفر Docker"""
        try:
            import docker
            client = docker.from_env()
            client.ping()
            return True
        except:
            logger.warning("Docker not available, using subprocess fallback")
            return False
    
    async def execute(self, code: str, timeout_sec: int = 30,
                      memory_limit_mb: int = 512, cpu_limit: float = 0.5) -> ExecutionResult:
        """
        تنفيذ كود في بيئة معزولة
        """
        if self.docker_available:
            return await self._execute_docker(code, timeout_sec, memory_limit_mb, cpu_limit)
        else:
            return await self._execute_subprocess(code, timeout_sec)
    
    async def _execute_docker(self, code: str, timeout_sec: int,
                               memory_limit_mb: int, cpu_limit: float) -> ExecutionResult:
        """تنفيذ في Docker"""
        try:
            import docker
            
            client = docker.from_env()
            
            with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
                f.write(code)
                temp_file = f.name
            
            container = client.containers.run(
                image="python:3.11-slim",
                command=f"python /code/{os.path.basename(temp_file)}",
                volumes={os.path.dirname(temp_file): {'bind': '/code', 'mode': 'ro'}},
                working_dir='/code',
                mem_limit=f"{memory_limit_mb}m",
                nano_cpus=int(cpu_limit * 1e9),
                network_disabled=True,
                read_only=True,
                detach=True,
                remove=False
            )
            
            try:
                result = container.wait(timeout=timeout_sec)
                logs = container.logs().decode()
                
                return ExecutionResult(
                    success=result['StatusCode'] == 0,
                    output=logs,
                    error="" if result['StatusCode'] == 0 else logs,
                    execution_time_ms=0
                )
            finally:
                container.remove(force=True)
                os.unlink(temp_file)
                
        except Exception as e:
            return ExecutionResult(False, "", str(e), 0)
    
    async def _execute_subprocess(self, code: str, timeout_sec: int) -> ExecutionResult:
        """تنفيذ في subprocess (fallback)"""
        import subprocess
        
        with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
            f.write(code)
            temp_file = f.name
        
        try:
            result = subprocess.run(
                ["python3", temp_file],
                capture_output=True,
                text=True,
                timeout=timeout_sec
            )
            
            return ExecutionResult(
                success=result.returncode == 0,
                output=result.stdout,
                error=result.stderr,
                execution_time_ms=0
            )
        except subprocess.TimeoutExpired:
            return ExecutionResult(False, "", f"Timeout after {timeout_sec}s", 0)
        except Exception as e:
            return ExecutionResult(False, "", str(e), 0)
        finally:
            if os.path.exists(temp_file):
                os.unlink(temp_file)

class Tracer:
    """
    Tracer - تتبع الطلبات عبر النظام
    """
    
    def __init__(self, service_name: str):
        self.service_name = service_name
        self.spans: List[Span] = []
        self.active_spans: Dict[str, Span] = {}
        
    def start_span(self, operation: str, parent_span_id: Optional[str] = None) -> str:
        """بدء امتداد جديد"""
        trace_id = current_trace_id.get()
        if not trace_id:
            trace_id = self._generate_id()
            current_trace_id.set(trace_id)
        
        span_id = self._generate_id()
        
        span = Span(
            trace_id=trace_id,
            span_id=span_id,
            parent_span_id=parent_span_id or current_span_id.get(),
            operation=operation,
            start_time=time.time()
        )
        
        self.active_spans[span_id] = span
        current_span_id.set(span_id)
        
        return span_id
    
    def finish_span(self, span_id: str, tags: Dict = None):
        """إنهاء امتداد"""
        if span_id in self.active_spans:
            span = self.active_spans[span_id]
            span.end_time = time.time()
            if tags:
                span.tags.update(tags)
            self.spans.append(span)
            del self.active_spans[span_id]
            
            if span.parent_span_id:
                current_span_id.set(span.parent_span_id)
            else:
                current_span_id.set('')
    
    def _generate_id(self) -> str:
        """توليد معرف فريد"""
        import random
        import string
        return ''.join(random.choices(string.hexdigits, k=16))
    
    def get_traces(self) -> List[Dict]:
        """الحصول على جميع التتبعات"""
        return [
            {
                'trace_id': s.trace_id,
                'span_id': s.span_id,
                'operation': s.operation,
                'duration_ms': (s.end_time - s.start_time) * 1000 if s.end_time else 0,
                'tags': s.tags
            }
            for s in self.spans
        ]


class MetricsCollector:

    Metrics Collector - جمع المقاييس
    """

        """تسجيل توزيع"""
        if name not in self.histograms:
            self.histograms[name] = []
        self.histograms[name].append(value)
        if len(self.histograms[name]) > 1000:
            self.histograms[name] = self.histograms[name][-1000:]
    
    def get_metrics(self) -> Dict:

        result = {**self.counters, **self.gauges}
        
        for name, values in self.histograms.items():
            if values:
                result[f"{name}_avg"] = sum(values) / len(values)
                result[f"{name}_max"] = max(values)
                result[f"{name}_p95"] = sorted(values)[int(len(values) * 0.95)]
        
        return result

    """

            'input_tokens': input_tokens,
            'output_tokens': output_tokens,
            'cost': cost,
            'timestamp': time.time()
        })
    
    def record_compute(self, duration_seconds: float):
        """تسجيل استخدام الحوسبة"""
        cost = duration_seconds * self.compute_cost_per_second
        self.costs.append({
            'type': 'compute',
            'duration_seconds': duration_seconds,
            'cost': cost,
            'timestamp': time.time()
        })
    
    def get_total_cost(self, since: Optional[float] = None) -> float:
        """الحصول على إجمالي التكلفة"""
        costs = self.costs
        if since:
            costs = [c for c in costs if c['timestamp'] >= since]
        return sum(c['cost'] for c in costs)
    
    def get_cost_breakdown(self) -> Dict:
        """توزيع التكاليف"""

        compute_cost = sum(c['cost'] for c in self.costs if c['type'] == 'compute')

            'llm': round(llm_cost, 4),
            'compute': round(compute_cost, 4),
            'total': round(llm_cost + compute_cost, 4)

class MultiAgentSystem:
    """
    Multi-Agent System V5.5 - النظام المتكامل مع Control Plane
    """
    
    def __init__(self):
        self.control_plane = ControlPlane()
        self.state_store = UnifiedStateStore()
        self.tool_gateway = ToolExecutionGateway()
        self.observability = ObservabilityStack("multi-agent-system")
        
        # الوكلاء
        self.agents = {
            'code': CodeUnderstandingAgent(),
            'performance': PerformanceAgent(),
            'architecture': ArchitectureAgent(),
            'benchmark': BenchmarkAgent(),
            'training': TrainingAgent(),
            'evolution': SelfEvolutionAgent()
        }
        
        self.agent_results: List[AgentResult] = []
        
    async def initialize(self):
        """تهيئة النظام"""
        await self.control_plane.start()
        
        for agent in self.agents.values():
            await agent.initialize(self.control_plane, self.state_store)
        
        logger.info("Multi-Agent System initialized")
    
    async def process(self, problem: str, code: str, user_id: str = None) -> Dict:
        """
        معالجة مشكلة باستخدام جميع الوكلاء
        """
        span_id = self.observability.start_span("process_problem")
        start_time = time.time()
        
        try:
            results = {}
            
            # تنفيذ الوكلاء بالتسلسل
            agent_chain = ['code', 'performance', 'architecture', 'benchmark', 'training', 'evolution']
            
            for agent_name in agent_chain:
                agent = self.agents.get(agent_name)
                if agent:
                    # توزيع المهمة عبر Control Plane
                    dispatch_result = await self.control_plane.dispatch(
                        task_name=f"{agent_name}_analysis",
                        agent_type=agent_name,
                        payload={'problem': problem, 'code': code, 'user_id': user_id},
                        priority=TaskPriority.NORMAL
                    )
                    
                    if dispatch_result['success']:
                        # محاكاة نتيجة الوكيل
                        agent_result = await agent.execute(problem, {'user_id': user_id})
                        results[agent_name] = agent_result.output
                        self.agent_results.append(agent_result)
                        
                        # تسجيل في المراقبة
                        self.observability.record_metric(f"agent_{agent_name}_success", 1, "counter")
            
            # تسجيل الحدث في State Store
            await self.state_store.commit(
                event_type="problem_processed",
                payload={
                    'problem': problem[:200],
                    'results': {k: str(v)[:100] for k, v in results.items()},
                    'user_id': user_id
                },
                state_key=f"process_{int(time.time())}",
                user_id=user_id
            )
            
            execution_time_ms = (time.time() - start_time) * 1000
            self.observability.record_metric("process_duration_ms", execution_time_ms, "histogram")
            
            return {
                'success': True,
                'results': results,
                'execution_time_ms': execution_time_ms,
                'agents_used': agent_chain
            }
            
        except Exception as e:
            self.observability.logger.error(f"Processing failed: {e}")
            return {
                'success': False,
                'error': str(e),
                'execution_time_ms': (time.time() - start_time) * 1000
            }
        finally:
            self.observability.finish_span(span_id)
    
    async def get_status(self) -> Dict:
        """حالة النظام"""
        return {
            'control_plane': await self.control_plane.get_status(),
            'state_store': await self.state_store.get_stats(),
            'tool_gateway': await self.tool_gateway.get_stats(),
            'observability': self.observability.get_status(),
            'agent_results': len(self.agent_results)
        }
    
    async def shutdown(self):
        """إيقاف النظام"""
        await self.control_plane.stop()
        logger.info("Multi-Agent System shutdown")
