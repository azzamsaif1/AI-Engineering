Unified Cognitive State Vector (CSV) - Heart of the Digital Twin
Combines all representations into a single vector
"""

import torch

import numpy as np
from typing import Dict, List, Optional
from dataclasses import dataclass

@dataclass
class CognitiveState:
    """Unified cognitive state of the user"""
    skill_graph_vector: np.ndarray      # 256-dim
    fingerprint_vector: np.ndarray      # 128-dim
    memory_embedding: np.ndarray        # 768-dim
    rl_state: np.ndarray                # 64-dim
    recent_actions: List[int]           # last 5 actions
    timestamp: float
    
    def to_tensor(self) -> torch.Tensor:
        """Convert to tensor for neural networks"""
        components = [
            torch.tensor(self.skill_graph_vector),
            torch.tensor(self.fingerprint_vector),
            torch.tensor(self.memory_embedding),
            torch.tensor(self.rl_state),
            torch.tensor(self.recent_actions + [0] * (5 - len(self.recent_actions)))
        ]
        return torch.cat(components)

class CognitiveStateBuilder:
    """
    Builds unified cognitive state from disparate system components
    """
    
    def __init__(self, state_dim: int = 1280):

"""
Dynamic Weighted Skill Graph with automatic updates
"""

from neo4j import GraphDatabase
import numpy as np
from typing import Dict, List, Optional
from datetime import datetime

class DynamicSkillGraph:
    """
    Skill graph with dynamic weights that update based on user performance
    skill_score = f(reward, time_decay, consistency, evidence_count)
    """
    
    def __init__(self, uri="bolt://localhost:7687", user="neo4j", password="password"):
        self.driver = GraphDatabase.driver(uri, auth=(user, password))
        self._ensure_constraints()
    
    def _ensure_constraints(self):
        with self.driver.session() as session:
            session.run("CREATE CONSTRAINT IF NOT EXISTS FOR (u:User) REQUIRE u.id IS UNIQUE")
            session.run("CREATE CONSTRAINT IF NOT EXISTS FOR (s:Skill) REQUIRE s.name IS UNIQUE")
    
    def add_skill(self, user_id: str, skill_name: str, category: str = "general"):
        with self.driver.session() as session:
            session.run(
                """
                MERGE (u:User {id: $user_id})
                MERGE (s:Skill {name: $skill_name})
                SET s.category = $category
                MERGE (u)-[r:HAS_SKILL]->(s)
                SET r.level = 0.1,
                    r.confidence = 0.5,
                    r.decay_rate = 0.05,
                    r.last_updated = datetime(),
                    r.evidence_count = 0
                """,
                user_id=user_id, skill_name=skill_name, category=category
            )
    
    def update_skill_score(self, user_id: str, skill_name: str, reward: float):
        """
        Dynamic skill score update based on RL reward
        skill_new = skill_old + learning_rate * (reward - skill_old)
        """
        with self.driver.session() as session:
            result = session.run(
                """
                MATCH (u:User {id: $user_id})-[r:HAS_SKILL]->(s:Skill {name: $skill_name})
                RETURN r.level as old_level, r.evidence_count as evidence, r.decay_rate as decay
                """,
                user_id=user_id, skill_name=skill_name
            )
            
            record = result.single()
            if not record:
                self.add_skill(user_id, skill_name)
                old_level = 0.1
                evidence = 0
                decay = 0.05
            else:
                old_level = record['old_level']
                evidence = record['evidence']
                decay = record['decay']
            
            # Update formula
            learning_rate = 0.1 * (1 / (1 + evidence))
            new_level = old_level + learning_rate * (reward - old_level)
            new_level = max(0.0, min(1.0, new_level))
            
            # Apply decay
            new_level = new_level * (1 - decay)
            
            session.run(
                """
                MATCH (u:User {id: $user_id})-[r:HAS_SKILL]->(s:Skill {name: $skill_name})
                SET r.level = $new_level,
                    r.confidence = $new_level,
                    r.evidence_count = r.evidence_count + 1,
                    r.last_updated = datetime(),
                    r.decay_rate = $decay_rate
                """,
                user_id=user_id, skill_name=skill_name,
                new_level=new_level, decay_rate=decay
            )
            
            return new_level
    
    def get_skill_vector(self, user_id: str) -> np.ndarray:
        """Get skill graph as vector for CSV"""
        with self.driver.session() as session:
            result = session.run(
                """
                MATCH (u:User {id: $user_id})-[r:HAS_SKILL]->(s:Skill)
                RETURN s.name as skill, r.level as level
                ORDER BY s.name
                """,
                user_id=user_id
            )
            
            skills = list(result)
            if not skills:
                return np.zeros(256)
            
            # Create vector (simplified)
            vector = np.array([s['level'] for s in skills[:256]])
            if len(vector) < 256:
                vector = np.pad(vector, (0, 256 - len(vector)))
            
            return vector
    
    def get_weak_skills(self, user_id: str, threshold: float = 0.4) -> List[Dict]:
        """Get skills below threshold for targeted improvement"""
        with self.driver.session() as session:
            result = session.run(
                """
                MATCH (u:User {id: $user_id})-[r:HAS_SKILL]->(s:Skill)
                WHERE r.level < $threshold
                RETURN s.name as skill, r.level as level, r.evidence_count as evidence
                ORDER BY r.level ASC
                LIMIT 10
                """,
                user_id=user_id, threshold=threshold
            )
            return [{'skill': r['skill'], 'level': r['level'], 'evidence': r['evidence']} for r in result]

"""
User Policy Model (UPM) - The actual digital twin
Not just LLM - learns user's decision patterns via contrastive learning
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
from sentence_transformers import SentenceTransformer
from typing import Dict, List, Optional
import numpy as np

class UserPolicyModel(nn.Module):
    """
    Core of the Digital Twin - learns user's decision patterns
    input = (CSV + problem_embedding)
    output = solution_embedding (not text)
    """
    
    def __init__(self, state_dim: int = 1280, embedding_dim: int = 768, hidden_dim: int = 512):
        super().__init__()
        
        # Problem encoder
        self.problem_encoder = SentenceTransformer("BAAI/bge-m3")
        
        # State encoder
        self.state_encoder = nn.Sequential(
            nn.Linear(state_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU()
        )
        
        # Policy network
        self.policy = nn.Sequential(
            nn.Linear(hidden_dim + 768, hidden_dim),  # state + problem
            nn.ReLU(),
            nn.Linear(hidden_dim, embedding_dim),
            nn.Tanh()
        )
        
        # LoRA adapters (will be fine-tuned)
        self.lora_A = nn.Parameter(torch.randn(hidden_dim, 32))
        self.lora_B = nn.Parameter(torch.randn(32, hidden_dim))
        self.use_lora = True
        
    def forward(self, csv_vector: torch.Tensor, problem_text: str) -> torch.Tensor:
        """
        Forward pass: predict solution embedding
        """
        # Encode state
        state_emb = self.state_encoder(csv_vector)
        
        # Encode problem
        problem_emb = self.problem_encoder.encode(problem_text, normalize_embeddings=True)
        problem_emb = torch.tensor(problem_emb).float()
        
        # Combine
        combined = torch.cat([state_emb, problem_emb], dim=-1)
        
        # Policy
        solution_emb = self.policy(combined)
        
        return solution_emb
    
    def get_solution_text(self, solution_emb: torch.Tensor, context: str) -> str:
        """Decode embedding to text (using LLM decoder)"""
        # This would call an LLM with the embedding as context
        # Simplified for now
        return "Decoded solution from embedding"
    
    def compute_contrastive_loss(self, user_emb: torch.Tensor, twin_emb: torch.Tensor) -> torch.Tensor:
        """
        Contrastive loss between user and twin predictions
        This is how the twin learns
        """
        # Cosine similarity
        sim = F.cosine_similarity(user_emb, twin_emb, dim=-1)
        
        # Contrastive loss: want similarity high
        loss = 1 - sim.mean()
        
        return loss
    
    def update_with_lora(self, gradients):
        """Update LoRA adapters instead of full model"""
        if self.use_lora:
            with torch.no_grad():
                self.lora_A += gradients['lora_A']
                self.lora_B += gradients['lora_B']

# backend/stage3_digital_twin/user_policy_model/lora_finetune.py
"""
LoRA Fine-tuning for User Policy Model
Updates only low-rank adapters, not full model
"""

import torch
import torch.optim as optim
from typing import Dict, List
import numpy as np

class LoRAFineTuner:
    """
    Fine-tune only LoRA adapters for each user
    Preserves base model while personalizing
    """
    
    def __init__(self, user_policy_model, lr: float = 1e-4):
        self.model = user_policy_model
        self.optimizer = optim.Adam([
            {'params': [self.model.lora_A], 'lr': lr},
            {'params': [self.model.lora_B], 'lr': lr}
        ])
        
        self.training_history = []
        
    def train_step(self, user_solution_emb: torch.Tensor, twin_prediction_emb: torch.Tensor) -> float:
        """
        Single training step using contrastive loss
        """
        self.optimizer.zero_grad()
        
        # Compute loss
        loss = self.model.compute_contrastive_loss(user_solution_emb, twin_prediction_emb)
        
        # Backward
        loss.backward()
        
        # Update LoRA adapters only
        self.optimizer.step()
        
        self.training_history.append(loss.item())
        
        return loss.item()
    
    def get_adapter_state(self) -> Dict:
        """Get current LoRA adapter weights for storage"""
        return {
            'lora_A': self.model.lora_A.detach().cpu().numpy(),
            'lora_B': self.model.lora_B.detach().cpu().numpy()
        }
    
    def load_adapter_state(self, adapter_state: Dict):
        """Load saved LoRA adapter weights"""
        self.model.lora_A.data = torch.tensor(adapter_state['lora_A'])
        self.model.lora_B.data = torch.tensor(adapter_state['lora_B'])

"""
Episodic Memory - stores events and experiences
"""

from typing import Dict, List, Any, Optional
from datetime import datetime
import json
import numpy as np

class EpisodicMemory:
    """
    Stores events with temporal context
    Used for recall of past experiences
    """
    
    def __init__(self, max_size: int = 1000):
        self.episodes: List[Dict] = []
        self.max_size = max_size
        
    def store(self, event: Dict):
        """Store an episode"""
        event['timestamp'] = datetime.now().isoformat()
        self.episodes.append(event)
        
        # Prune if too large
        if len(self.episodes) > self.max_size:
            self.episodes = self.episodes[-self.max_size:]
    
    def recall_recent(self, limit: int = 10) -> List[Dict]:
        """Recall most recent episodes"""
        return self.episodes[-limit:]
    
    def recall_by_type(self, event_type: str, limit: int = 10) -> List[Dict]:
        """Recall episodes by type"""
        filtered = [e for e in self.episodes if e.get('type') == event_type]
        return filtered[-limit:]
    
    def recall_similar(self, query: Dict, limit: int = 5) -> List[Dict]:
        """Recall episodes similar to query (simplified)"""
        # In production, use embedding similarity
        return self.episodes[-limit:]
    
    def get_stats(self) -> Dict:
        return {
            'total_episodes': len(self.episodes),
            'max_size': self.max_size,
            'oldest': self.episodes[0]['timestamp'] if self.episodes else None,
            'newest': self.episodes[-1]['timestamp'] if self.episodes else None
        }

import uuid

class SemanticMemory:
    """
    Vector-based semantic memory using Qdrant
    Stores knowledge embeddings for semantic search
    """
    
    def __init__(self, collection_name: str = "semantic_memory"):
        self.encoder = SentenceTransformer("BAAI/bge-m3")
        self.client = QdrantClient(host="localhost", port=6333)
        self.collection_name = collection_name
        self._ensure_collection()
    
    def _ensure_collection(self):
        collections = self.client.get_collections().collections
        if not any(c.name == self.collection_name for c in collections):
            self.client.create_collection(
                collection_name=self.collection_name,
                vectors_config=models.VectorParams(
                    size=768,
                    distance=models.Distance.COSINE
                )
            )
    
    def store(self, text: str, metadata: Dict) -> str:
        """Store knowledge in semantic memory"""
        embedding = self.encoder.encode(text, normalize_embeddings=True)
        point_id = str(uuid.uuid4())
        
        self.client.upsert(
            collection_name=self.collection_name,
            points=[
                models.PointStruct(
                    id=point_id,
                    vector=embedding.tolist(),
                    payload={
                        'text': text,
                        'metadata': metadata,
                        'timestamp': __import__('time').time()
                    }
                )
            ]
        )
        return point_id

            limit=limit
        )

                if resp.status != 200:

                data = await resp.json()

        filtered = []

                filtered.append(repo)

        topics = repo.get('topics', [])
        name = repo.get('name', '').lower()
        description = repo.get('description', '').lower()

            if topic in name or topic in description or any(topic in t for t in topics):

        return False

        """Basic security check"""
        # Check for malicious patterns

        return True

            'score': outcome.get('score', outcome.get('confidence', 0.5))

        if len(self.patterns[action_type]) > self.max_patterns:
            self.patterns[action_type] = self.patterns[action_type][-self.max_patterns:]

        patterns = self.patterns[action_type]

"""
Disagreement Detector - Measures divergence between model outputs
"""

import numpy as np
from typing import Dict, List, Any
from scipy.spatial.distance import cosine
from sklearn.feature_extraction.text import TfidfVectorizer


class DisagreementDetector:
    """
    Detects and quantifies disagreement between model outputs
    """
    
    def __init__(self):
        self.vectorizer = TfidfVectorizer(max_features=100)
        self.disagreement_threshold = 0.35
    
    def compute_disagreement_matrix(self, outputs: Dict[str, str]) -> np.ndarray:
        """
        Compute pairwise disagreement between all models
        """
        model_names = list(outputs.keys())
        texts = list(outputs.values())
        
        if len(texts) < 2:
            return np.array([[0.0]])
        
        # TF-IDF vectorization
        try:
            vectors = self.vectorizer.fit_transform(texts).toarray()
        except ValueError:
            # If no variation in texts
            return np.zeros((len(model_names), len(model_names)))
        
        # Compute pairwise cosine distances (disagreement)
        disagreement_matrix = np.zeros((len(model_names), len(model_names)))
        for i in range(len(model_names)):
            for j in range(len(model_names)):
                if i == j:
                    disagreement_matrix[i, j] = 0.0
                else:
                    disagreement_matrix[i, j] = cosine(vectors[i], vectors[j])
        
        return disagreement_matrix
    
    def compute_aggregate_disagreement(self, outputs: Dict[str, str]) -> Dict[str, Any]:
        """
        Compute aggregate disagreement metrics
        """
        matrix = self.compute_disagreement_matrix(outputs)
        
        if matrix.shape[0] < 2:
            return {
                'mean_disagreement': 0.0,
                'max_disagreement': 0.0,
                'min_disagreement': 0.0,
                'std_disagreement': 0.0,
                'is_high_disagreement': False
            }
        
        # Get upper triangle excluding diagonal
        upper_triangle = matrix[np.triu_indices_from(matrix, k=1)]
        
        return {
            'mean_disagreement': float(np.mean(upper_triangle)),
            'max_disagreement': float(np.max(upper_triangle)),
            'min_disagreement': float(np.min(upper_triangle)),
            'std_disagreement': float(np.std(upper_triangle)),
            'is_high_disagreement': float(np.mean(upper_triangle)) > self.disagreement_threshold,
            'disagreement_matrix': matrix.tolist()
        }
    
    def find_disagreement_sources(self, outputs: Dict[str, str]) -> List[Dict]:
        """
        Identify specific points of disagreement between models
        """
        texts = list(outputs.values())
        model_names = list(outputs.keys())
        
        if len(texts) < 2:
            return []
        
        # Simple word-based disagreement detection
        all_words = set()
        word_sets = []
        
        for text in texts:
            words = set(text.lower().split())
            word_sets.append(words)
            all_words.update(words)
        
        # Find words that appear in some but not all
        disagreement_sources = []
        for word in all_words:
            appearances = [word in word_set for word_set in word_sets]
            if not all(appearances) and not any(appearances):
                # Word appears in some but not all
                appearing_models = [model_names[i] for i, appears in enumerate(appearances) if appears]
                missing_models = [model_names[i] for i, appears in enumerate(appearances) if not appears]
                
                disagreement_sources.append({
                    'word': word,
                    'appearing_in': appearing_models,
                    'missing_in': missing_models,
                    'disagreement_type': 'term_presence'
                })
        
        # Limit to top disagreements
        return disagreement_sources[:20]
    
    def get_confidence_weight(self, outputs: Dict[str, str], model_name: str) -> float:
        """
        Calculate confidence weight for a model based on disagreement patterns
        """
        matrix = self.compute_disagreement_matrix(outputs)
        model_indices = list(outputs.keys())
        
        if model_name not in model_indices:
            return 0.5
        
        idx = model_indices.index(model_name)
        
        if matrix.shape[0] < 2:
            return 1.0
        
        # Average disagreement with other models
        disagreements = [matrix[idx, j] for j in range(matrix.shape[1]) if j != idx]
        
        if not disagreements:
            return 1.0
        
        avg_disagreement = np.mean(disagreements)
        
        # Lower disagreement = higher confidence
        confidence = 1.0 - avg_disagreement
        
        return max(0.1, min(1.0, confidence))

            'name': 'Binary Search',
            'description': 'Implement binary search',
            'type': 'algorithm',
            'topics': ['binary_search', 'arrays']
        }
        
        user_solution = """
        def binary_search(arr, target):
            left, right = 0, len(arr) - 1
            while left <= right:
                mid = (left + right) // 2
                if arr[mid] == target:
                    return mid
                elif arr[mid] < target:
                    left = mid + 1
                else:
                    right = mid - 1
            return -1
        """
        
        result = await twin.learn_from_user(problem, user_solution, score=95)
        
        assert result['status'] == 'learned'
        assert 'loss' in result
        assert 'updated_skills' in result
        assert isinstance(result['loss'], float)
    
    @pytest.mark.asyncio
    async def test_competition(self):
        """Test competition mode"""
        twin = DigitalTwinPipeline(user_id="test_user_004")
        
        problem = {
            'id': 'prob_003',
            'name': 'Reverse Linked List',
            'description': 'Reverse a singly linked list',
            'type': 'algorithm',
            'topics': ['linked_lists', 'recursion']
        }
        
        user_solution = """
        def reverse_list(head):
            prev = None
            curr = head
            while curr:
                next_temp = curr.next
                curr.next = prev
                prev = curr
                curr = next_temp
            return prev
        """
