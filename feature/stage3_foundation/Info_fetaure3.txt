# backend/digital_twin/engineering_fingerprint.py
import spacy
from collections import Counter
import re

class EngineeringFingerprint:
    def __init__(self, user_id):
        self.user_id = user_id
        self.nlp = spacy.load("en_core_web_sm")
        self.patterns = {
            'naming': [],
            'structure': [],
            'debugging_style': [],
            'architecture_decisions': []
        }
    
    def analyze_code_style(self, code):
        """تحليل أسلوب كتابة الكود"""
        doc = self.nlp(code)
        
        # تحليل التسمية (camelCase vs snake_case)
        variable_names = re.findall(r'\b[a-z_][a-z0-9_]*\b', code)
        naming_style = 'snake_case' if any('_' in n for n in variable_names) else 'camelCase'
        
        # تحليل التعليقات
        comments = re.findall(r'#.*$|//.*$', code, re.MULTILINE)
        comment_style = 'descriptive' if any(len(c) > 30 for c in comments) else 'minimal'
        
        # تحليل الهيكل
        functions = re.findall(r'def\s+(\w+)', code)
        classes = re.findall(r'class\s+(\w+)', code)
        
        return {
            'naming_style': naming_style,
            'comment_style': comment_style,
            'functions_count': len(functions),
            'classes_count': len(classes),
            'avg_line_length': sum(len(l) for l in code.split('\n')) / max(len(code.split('\n')), 1)
        }
    
    def extract_decision_patterns(self, commit_history):
        """تحليل أنماط القرارات من تاريخ المشاريع"""
        patterns = {
            'refactoring_frequency': 0,
            'testing_approach': 'unknown',
            'error_handling_style': 'unknown'
        }
        
        for commit in commit_history:
            if 'refactor' in commit['message'].lower():
                patterns['refactoring_frequency'] += 1
            if 'test' in commit['message'].lower():
                patterns['testing_approach'] = 'test_inclusive'
            if 'error' in commit['message'].lower() or 'bug' in commit['message'].lower():
                patterns['error_handling_style'] = 'reactive'
        
        return patterns

# backend/digital_twin/skill_graph.py
from neo4j import GraphDatabase

class SkillGraph:
    def __init__(self, uri="bolt://localhost:7687", user="neo4j", password="password"):
        self.driver = GraphDatabase.driver(uri, auth=(user, password))
    
    def add_skill(self, user_id, skill_name, level, category):
        """إضافة مهارة للمستخدم"""
        with self.driver.session() as session:
            session.run(
                """
                MERGE (u:User {id: $user_id})
                MERGE (s:Skill {name: $skill_name, category: $category})
                MERGE (u)-[r:HAS_SKILL]->(s)
                SET r.level = $level, r.updated_at = datetime()
                """,
                user_id=user_id, skill_name=skill_name, level=level, category=category
            )
    
    def add_prerequisite(self, skill1, skill2):
        """إضافة علاقة prerequisite بين مهارتين"""
        with self.driver.session() as session:
            session.run(
                """
                MATCH (s1:Skill {name: $skill1})
                MATCH (s2:Skill {name: $skill2})
                MERGE (s1)-[:PREREQUISITE_FOR]->(s2)
                """,
                skill1=skill1, skill2=skill2
            )
    
    def get_skill_graph(self, user_id):
        """الحصول على خريطة المهارات الكاملة للمستخدم"""
        with self.driver.session() as session:
            result = session.run(
                """
                MATCH (u:User {id: $user_id})-[r:HAS_SKILL]->(s:Skill)
                RETURN s.name as skill, s.category as category, r.level as level
                ORDER BY r.level DESC
                """,
                user_id=user_id
            )
            return [dict(record) for record in result]
    
    def get_weak_areas(self, user_id, threshold=50):
        """الحصول على نقاط الضعف"""
        with self.driver.session() as session:
            result = session.run(
                """
                MATCH (u:User {id: $user_id})-[r:HAS_SKILL]->(s:Skill)
                WHERE r.level < $threshold
                RETURN s.name as skill, r.level as level
                ORDER BY r.level ASC
                """,
                user_id=user_id, threshold=threshold
            )
            return [dict(record) for record in result]
    
    def get_recommended_learning_path(self, user_id, target_skill):
        """توليد مسار تعلم مخصص للوصول إلى مهارة معينة"""
        with self.driver.session() as session:
            result = session.run(
                """
                MATCH path = shortestPath((u:User {id: $user_id})-[:HAS_SKILL*]->(s:Skill {name: $target}))
                RETURN [node in nodes(path) | node.name] as path
                """,
                user_id=user_id, target=target_skill
            )
            record = result.single()
            return record['path'] if record else []

# backend/digital_twin/cognitive_profile.py
from sentence_transformers import SentenceTransformer
import numpy as np
from collections import defaultdict

class CognitiveProfile:
    def __init__(self, user_id):
        self.user_id = user_id
        self.encoder = SentenceTransformer("BAAI/bge-m3")
        self.profile = {
            'thinking_style_embedding': None,
            'decision_vectors': [],
            'problem_solving_patterns': defaultdict(list),
            'abstraction_level': 0.5,
            'optimization_focus': 0.5
        }
    
    def extract_thinking_style(self, problem_solutions):
        """استخراج أسلوب التفكير من حلول المشكلات"""
        descriptions = []
        for solution in problem_solutions:
            # تحليل طريقة حل المشكلة
            desc = self._analyze_solution_approach(solution)
            descriptions.append(desc)
        
        # تحويل أسلوب التفكير إلى متجه
        style_text = " ".join(descriptions)
        self.profile['thinking_style_embedding'] = self.encoder.encode([style_text])[0]
        
        return self.profile['thinking_style_embedding']
    
    def _analyze_solution_approach(self, solution):
        """تحليل نهج حل المشكلة"""
        approach = []
        
        # تحليل استخدام الخوارزميات
        if 'sort' in solution.lower():
            approach.append('uses sorting algorithms')
        if 'hash' in solution.lower():
            approach.append('uses hash-based approach')
        if 'recurs' in solution.lower():
            approach.append('uses recursion')
        
        # تحليل الـ optimization
        if 'cache' in solution.lower():
            approach.append('optimizes with caching')
        if 'vector' in solution.lower():
            approach.append('uses vectorization')
        
        return " ".join(approach) if approach else "standard approach"
    
    def update_from_solution(self, problem, solution, score, time_taken):
        """تحديث الملف المعرفي بناءً على حل جديد"""
        # تحليل أسلوب الحل
        approach = self._analyze_solution_approach(solution)
        self.profile['problem_solving_patterns'][problem['category']].append({
            'approach': approach,
            'score': score,
            'time': time_taken
        })
        
        # تحديث مستوى التجريد (abstraction level)
        if 'class' in solution or 'interface' in solution:
            self.profile['abstraction_level'] = min(1.0, self.profile['abstraction_level'] + 0.05)
        
        # تحديث optimization focus
        if 'cache' in solution or 'optim' in solution:
            self.profile['optimization_focus'] = min(1.0, self.profile['optimization_focus'] + 0.05)
    
    def get_cognitive_fingerprint(self):
        """الحصول على البصمة المعرفية الكاملة"""
        return {
            'thinking_style': self.profile['thinking_style_embedding'].tolist() if self.profile['thinking_style_embedding'] is not None else None,
            'abstraction_level': self.profile['abstraction_level'],
            'optimization_focus': self.profile['optimization_focus'],
            'problem_solving_patterns': dict(self.profile['problem_solving_patterns'])
        }

# backend/digital_twin/digital_twin.py
from transformers import AutoTokenizer, AutoModelForCausalLM
import torch

class DigitalTwin:
    def __init__(self, user_id, cognitive_profile, skill_graph):
        self.user_id = user_id
        self.cognitive_profile = cognitive_profile
        self.skill_graph = skill_graph
        
        # تحميل نموذج مخصص (يمكن تدريبه لاحقاً)
        self.model_name = "deepseek-ai/deepseek-coder-6.7b-instruct"
        self.tokenizer = AutoTokenizer.from_pretrained(self.model_name, trust_remote_code=True)
        self.model = AutoModelForCausalLM.from_pretrained(
            self.model_name,
            torch_dtype=torch.bfloat16,
            device_map="auto",
            trust_remote_code=True
        )
        
        # ذاكرة التوأم (أخطاء وقرارات سابقة)
        self.memory = []
    
    def predict_solution(self, problem_description):
        """توقع الحل الذي سيكتبه المستخدم"""
        # إضافة أسلوب المستخدم من الملف المعرفي
        style_context = f"""
        User's thinking style: {self.cognitive_profile.get_cognitive_fingerprint()}
        
        Based on the user's previous solutions, they prefer:
        - Abstraction level: {self.cognitive_profile.profile['abstraction_level']}
        - Optimization focus: {self.cognitive_profile.profile['optimization_focus']}
        
        Problem: {problem_description}
        
        Predict the user's solution approach:
        """
        
        inputs = self.tokenizer(style_context, return_tensors="pt").to(self.model.device)
        outputs = self.model.generate(**inputs, max_new_tokens=256)
        prediction = self.tokenizer.decode(outputs[0], skip_special_tokens=True)
        
        return prediction
    
    def compare_with_user(self, user_solution, twin_solution):
        """مقارنة حل المستخدم مع حل التوأم"""
        diff = self._calculate_diff(user_solution, twin_solution)
        
        return {
            'similarity': diff['similarity'],
            'differences': diff['differences'],
            'twin_was_better': diff['twin_better']
        }
    
    def learn_from_user(self, problem, user_solution, score):
        """التعلم من حل المستخدم"""
        self.memory.append({
            'problem': problem,
            'solution': user_solution,
            'score': score,
            'timestamp': datetime.now()
        })
        
        # تحديث الملف المعرفي
        self.cognitive_profile.update_from_solution(problem, user_solution, score, 0)
    
    def generate_challenge_for_user(self, weak_area=None):
        """توليد تحدٍ مخصص للمستخدم بناءً على نقاط ضعفه"""
        if not weak_area:
            # الحصول على أضعف مهارة من Skill Graph
            weak_areas = self.skill_graph.get_weak_areas(self.user_id)
            weak_area = weak_areas[0]['skill'] if weak_areas else 'general'
        
        challenge = f"""
        Challenge: Improve your skill in {weak_area}
        
        Based on your digital twin's analysis, you need to focus on:
        - Understanding {weak_area} fundamentals
        - Practicing with real-world examples
        - Optimizing your approach
        
        Write a solution that demonstrates mastery of {weak_area}.
        """
        
        return challenge

# backend/digital_twin/self_evolution.py
import requests
from datetime import datetime, timedelta
import json

class SelfEvolutionEngine:
    def __init__(self, digital_twin, skill_graph):
        self.digital_twin = digital_twin
        self.skill_graph = skill_graph
        self.trending_topics = []
        self.last_update = datetime.now()
    
    def fetch_trending_technologies(self):
        """جلب أحدث التقنيات من GitHub و arXiv"""
        trends = []
        
        # GitHub Trending
        try:
            response = requests.get('https://github-trending-api.com/repositories')
            if response.status_code == 200:
                repos = response.json()
                for repo in repos[:10]:
                    trends.append({
                        'name': repo['name'],
                        'language': repo['language'],
                        'stars': repo['stars'],
                        'description': repo['description']
                    })
        except:
            pass
        
        # arXiv papers (CS only)
        try:
            response = requests.get(
                'http://export.arxiv.org/api/query?search_query=cat:cs&sortBy=submittedDate&max_results=10'
            )
            # Parse XML response
            # في التطبيق الفعلي، تحتاج إلى parsing XML
        except:
            pass
        
        self.trending_topics = trends
        return trends
    
    def analyze_skill_gaps(self, user_id):
        """تحليل الفجوات بين مهارات المستخدم والتقنيات الجديدة"""
        user_skills = self.skill_graph.get_skill_graph(user_id)
        user_skill_names = [s['skill'].lower() for s in user_skills]
        
        gaps = []
        for trend in self.trending_topics:
            trend_lower = trend['name'].lower()
            if not any(skill in trend_lower for skill in user_skill_names):
                gaps.append({
                    'technology': trend['name'],
                    'description': trend.get('description', ''),
                    'importance': 'high' if trend.get('stars', 0) > 1000 else 'medium'
                })
        
        return gaps
    
    def evolve_twin(self, user_id):
        """تطوير التوأم الرقمي بناءً على أحدث المعلومات"""
        # جلب أحدث التقنيات
        trends = self.fetch_trending_technologies()
        
        # تحليل الفجوات
        gaps = self.analyze_skill_gaps(user_id)
        
        # تحديث المهارات الموصى بها
        recommended_skills = [gap['technology'] for gap in gaps if gap['importance'] == 'high']
        
        # إعادة بناء جزء من Knowledge Graph
        for skill in recommended_skills[:5]:
            self.skill_graph.add_skill(user_id, skill, 0, 'recommended')
        
        return {
            'trending': trends[:5],
            'skill_gaps': gaps[:5],
            'recommended_skills': recommended_skills[:5],
            'evolution_date': datetime.now().isoformat()
        }

# backend/digital_twin/coevolution.py
import random
import numpy as np

class CoevolutionEngine:
    def __init__(self, user_id, digital_twin):
        self.user_id = user_id
        self.digital_twin = digital_twin
        self.competition_history = []
        self.user_elo = 1200  # Elo rating للمستخدم
        self.twin_elo = 1200  # Elo rating للتوأم
        self.competition_count = 0
    
    def start_competition_round(self, problem):
        """بدء جولة منافسة جديدة"""
        # التوأم يحل المشكلة
        twin_solution = self.digital_twin.predict_solution(problem['description'])
        
        # تقييم الحلين (سيتم بعد حل المستخدم)
        return {
            'round_id': self.competition_count + 1,
            'problem': problem,
            'twin_solution': twin_solution,
            'user_solution': None,
            'winner': None,
            'elo_change': None
        }
    
    def evaluate_round(self, round_data, user_solution, user_score, twin_score):
        """تقييم الجولة وتحديد الفائز"""
        if user_score > twin_score:
            winner = 'user'
            elo_change = self._calculate_elo_change(self.user_elo, self.twin_elo, True)
        elif twin_score > user_score:
            winner = 'twin'
            elo_change = self._calculate_elo_change(self.user_elo, self.twin_elo, False)
        else:
            winner = 'tie'
            elo_change = {'user': 0, 'twin': 0}
        
        # تحديث التصنيف
        self.user_elo += elo_change['user']
        self.twin_elo += elo_change['twin']
        
        # تسجيل التاريخ
        self.competition_history.append({
            'round_id': self.competition_count + 1,
            'problem': round_data['problem'],
            'winner': winner,
            'user_elo': self.user_elo,
            'twin_elo': self.twin_elo,
            'timestamp': datetime.now().isoformat()
        })
        
        self.competition_count += 1
        
        # التوأم يتعلم من الحل الفائز
        if winner == 'user':
            self.digital_twin.learn_from_user(round_data['problem'], user_solution, user_score)
        
        return {
            'winner': winner,
            'user_elo': self.user_elo,
            'twin_elo': self.twin_elo,
            'elo_change': elo_change
        }
    
    def _calculate_elo_change(self, user_elo, twin_elo, user_won):
        """حساب تغير Elo rating"""
        K = 32  # عامل التغيير
        expected_user = 1 / (1 + 10 ** ((twin_elo - user_elo) / 400))
        
        if user_won:
            user_change = K * (1 - expected_user)
            twin_change = -user_change
        else:
            user_change = K * (0 - expected_user)
            twin_change = -user_change
        
        return {'user': round(user_change), 'twin': round(twin_change)}
    
    def get_competition_stats(self):
        """الحصول على إحصائيات المنافسة"""
        user_wins = sum(1 for h in self.competition_history if h['winner'] == 'user')
        twin_wins = sum(1 for h in self.competition_history if h['winner'] == 'twin')
        ties = sum(1 for h in self.competition_history if h['winner'] == 'tie')
        
        return {
            'total_rounds': self.competition_count,
            'user_wins': user_wins,
            'twin_wins': twin_wins,
            'ties': ties,
            'user_win_rate': user_wins / self.competition_count if self.competition_count > 0 else 0,
            'user_elo': self.user_elo,
            'twin_elo': self.twin_elo,
            'user_is_better': self.user_elo > self.twin_elo
        }
    
    def generate_next_challenge(self):
        """توليد التحدي التالي بناءً على مستوى المنافسة"""
        if self.user_elo > self.twin_elo + 100:
            # المستخدم أفضل بكثير → تحدى التوأم
            difficulty = 'expert'
            message = "You're ahead! Can your twin catch up?"
        elif self.twin_elo > self.user_elo + 100:
            # التوأم أفضل بكثير → تحدى المستخدم
            difficulty = 'challenging'
            message = "Your twin is winning! Time to prove yourself!"
        else:
            # متقاربين
            difficulty = 'balanced'
            message = "Close competition! Next round decides the leader!"
        
        return {
            'difficulty': difficulty,
            'message': message,
            'suggested_topic': self._suggest_topic_based_on_weakness()
        }
    
    def _suggest_topic_based_on_weakness(self):
        """اقتراح موضوع بناءً على نقاط ضعف المستخدم"""
        # تحليل أضعف المهارات من Skill Graph
        weak_areas = self.digital_twin.skill_graph.get_weak_areas(self.user_id)
        if weak_areas:
            return weak_areas[0]['skill']
        return 'general programming'
