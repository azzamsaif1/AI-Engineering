# backend/nlp/term_extractor.py
import spacy
import re

class TechnicalTermExtractor:
    def __init__(self):
        self.nlp_en = spacy.load("en_core_web_sm")
        self.nlp_de = spacy.load("de_core_news_sm")
        
        # المصطلحات التقنية المعروفة مسبقاً ######needs to solve
        self.technical_patterns = {
            'Hardware': ['CPU', 'GPU', 'RAM', 'NIC', 'DMA', 'PHY', 'SSD'],
            'Protocols': ['TCP', 'UDP', 'HTTP', 'HTTPS', 'FTP', 'SSH'],
            'Concepts': ['Algorithm', 'Database', 'Server', 'Client', 'Socket']
        }
    
    def extract_terms(self, text, language='en'):
        nlp = self.nlp_en if language == 'en' else self.nlp_de
        doc = nlp(text)
        
        terms = []
        relations = []
        
        # استخراج الكيانات
        for ent in doc.ents:
            if self.is_technical_entity(ent.label_):
                terms.append({
                    'text': ent.text,
                    'label': ent.label_,
                    'start': ent.start_char,
                    'end': ent.end_char
                })
        
        # استخراج العلاقات (dependency parsing)
        for token in doc:
            if token.dep_ in ['nsubj', 'dobj', 'attr']:
                relations.append({
                    'head': token.head.text,
                    'dependent': token.text,
                    'relation': token.dep_
                })
        
        return {'terms': terms, 'relations': relations}
    
    def is_technical_entity(self, label):
        technical_labels = ['PRODUCT', 'ORG', 'WORK_OF_ART', 'LAW']
        return label in technical_labels

# backend/embeddings/sentence_encoder.py
from sentence_transformers import SentenceTransformer
import numpy as np

class TechnicalEncoder:
    def __init__(self, model_name='BAAI/bge-m3'):
        print(f"Loading model: {model_name}")
        self.model = SentenceTransformer(model_name)
        self.dimension = self.model.get_sentence_embedding_dimension()
    
    def encode(self, texts):
        """تحويل النصوص إلى متجهات"""
        if isinstance(texts, str):
            texts = [texts]
        embeddings = self.model.encode(texts, normalize_embeddings=True)
        return embeddings
    
    def similarity(self, text1, text2):
        """حساب التشابه الدلالي بين نصين"""
        emb1 = self.encode([text1])[0]
        emb2 = self.encode([text2])[0]
        return np.dot(emb1, emb2)
    
    def find_similar(self, query, candidates, top_k=5):
        """البحث عن أكثر النصوص تشابهاً"""
        query_emb = self.encode([query])[0]
        candidate_embs = self.encode(candidates)
        
        similarities = np.dot(candidate_embs, query_emb)
        top_indices = np.argsort(similarities)[-top_k:][::-1]
        
        return [(candidates[i], similarities[i]) for i in top_indices]

# backend/knowledge_graph/graph_builder.py
from neo4j import GraphDatabase

class KnowledgeGraphBuilder:
    def __init__(self, uri="bolt://localhost:7687", user="neo4j", password="password"):
        self.driver = GraphDatabase.driver(uri, auth=(user, password))
    
    def add_concept(self, name, category, level='beginner'):
        with self.driver.session() as session:
            session.run(
                "MERGE (c:Concept {name: $name}) "
                "SET c.category = $category, c.level = $level",
                name=name, category=category, level=level
            )
    
    def add_relation(self, concept1, concept2, relation_type):
        """إضافة علاقة بين مفهومين"""
        with self.driver.session() as session:
            session.run(
                f"MATCH (a:Concept {{name: $c1}}), (b:Concept {{name: $c2}}) "
                f"MERGE (a)-[:{relation_type}]->(b)",
                c1=concept1, c2=concept2
            )
    
    def get_prerequisites(self, concept):
        """الحصول على المتطلبات الأساسية لمفهوم"""
        with self.driver.session() as session:
            result = session.run(
                "MATCH (p:Concept)-[:REQUIRES]->(c:Concept {name: $concept}) "
                "RETURN p.name as prerequisite",
                concept=concept
            )
            return [record['prerequisite'] for record in result]
    
    def get_learning_path(self, target_concept):
        """الحصول على مسار التعلم الأمثل"""
        with self.driver.session() as session:
            result = session.run(
                "MATCH path = shortestPath((start)-[:REQUIRES*]->(end {name: $target})) "
                "RETURN [node in nodes(path) | node.name] as learning_path",
                target=target_concept
            )
            record = result.single()
            return record['learning_path'] if record else []

# backend/topic_modeling/topic_detector.py
from bertopic import BERTopic
from sentence_transformers import SentenceTransformer

class SmartTopicDetector:
    def __init__(self):
        self.embedding_model = SentenceTransformer("BAAI/bge-m3")
        self.topic_model = BERTopic(
            embedding_model=self.embedding_model,
            language="multilingual",
            calculate_probabilities=True,
            nr_topics='auto'
        )
        self.topics = None
    
    def detect_topics(self, texts):
        """اكتشاف المواضيع في النصوص"""
        topics, probs = self.topic_model.fit_transform(texts)
        topic_info = self.topic_model.get_topic_info()
        
        results = []
        for topic_id in set(topics):
            if topic_id != -1:
                words = self.topic_model.get_topic(topic_id)
                if words:
                    results.append({
                        'topic_id': topic_id,
                        'name': " ".join([w for w, _ in words[:3]]),
                        'keywords': [w for w, _ in words[:10]],
                        'size': sum(1 for t in topics if t == topic_id)
                    })
        
        return results
    
    def get_topic_for_text(self, text):
        """تحديد موضوع نص واحد"""
        topic, prob = self.topic_model.transform([text])
        topic_id = topic[0]
        
        if topic_id != -1:
            words = self.topic_model.get_topic(topic_id)
            return {
                'topic_id': topic_id,
                'name': " ".join([w for w, _ in words[:3]]),
                'confidence': float(prob[0][topic_id]) if len(prob[0]) > topic_id else 0.5
            }
        return {'topic_id': -1, 'name': 'General', 'confidence': 0.3}

# backend/nlp/smart_understanding_layer.py
"""
Smart Self-Adaptive Understanding Layer
No hardcoded patterns - discovers everything automatically
"""

import spacy
from langdetect import detect
from transformers import pipeline
from sentence_transformers import SentenceTransformer
from sklearn.cluster import KMeans
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np
from collections import defaultdict
import re


class SmartContextDetector:
    """يكتشف اللغة والمجال تلقائياً بدون تدخل المستخدم"""
    
    def __init__(self):
        self.domain_classifier = pipeline(
            "zero-shot-classification",
            model="facebook/bart-large-mnli",
            device=-1
        )
        
        self.domains = [
            "software engineering and programming",
            "computer networking and protocols",
            "cybersecurity and encryption",
            "databases and data management",
            "algorithms and data structures",
            "operating systems and kernels",
            "cloud computing and devops",
            "artificial intelligence and machine learning",
            "web development and frontend",
            "mobile development",
            "mathematics and statistics",
            "general technical content"
        ]
    
    def detect_all(self, text: str) -> dict:
        """يكتشف اللغة والمجال ونوع المحتوى"""
        # كشف اللغة
        try:
            language = detect(text)
        except:
            language = "en"  # fallback
        
        # كشف المجال
        domain_result = self.domain_classifier(text[:1000], self.domains)
        primary_domain = domain_result['labels'][0]
        domain_confidence = domain_result['scores'][0]
        
        # كشف مستوى النص (تقديري)
        level = self._estimate_level(text)
        
        return {
            'language': language,
            'domain': primary_domain,
            'domain_confidence': round(domain_confidence * 100, 2),
            'level': level,
            'text_length': len(text)
        }
    
    def _estimate_level(self, text: str) -> str:
        """تقدير مستوى النص (beginner/intermediate/advanced)"""
        words = text.split()
        avg_word_len = sum(len(w) for w in words) / max(len(words), 1)
        
        # كلمات دالة على المستوى المتقدم
        advanced_indicators = ['kernel', 'protocol', 'encryption', 'distributed', 'concurrency', 'asynchronous']
        advanced_count = sum(1 for w in advanced_indicators if w in text.lower())
        
        if avg_word_len > 7 or advanced_count > 3:
            return 'advanced'
        elif avg_word_len > 5 or advanced_count > 1:
            return 'intermediate'
        return 'beginner'

            r'\b[A-Z][a-z]+(?:[A-Z][a-z]+)*\b',  # CamelCase
            r'\b[A-Z]{2,}\b',                     # Acronyms (TCP, API)
            r'\b[a-z]+[0-9]+\b',                  # مع أرقام (Ethernet1)
            r'\b[a-z]+(?:[_-][a-z]+)+\b',         # snake-case or kebab-case
        ]
    
    def extract(self, text: str, language: str = 'en') -> dict:
        """يستخرج الكيانات باستخدام طرق متعددة"""
        nlp = self.nlp if language == 'en' else self.nlp_de
        doc = nlp(text)
        
        entities = []
        
        # 1. استخراج الكيانات التقليدية من spaCy
        for ent in doc.ents:
            entities.append({
                'text': ent.text,
                'type': ent.label_,
                'method': 'spacy',
                'start': ent.start_char,
                'end': ent.end_char
            })
        
        # 2. استخراج العبارات الاسمية (noun chunks)
        for chunk in doc.noun_chunks:
            if len(chunk.text.split()) >= 2 and len(chunk.text) > 3:
                entities.append({
                    'text': chunk.text,
                    'type': 'concept_phrase',
                    'method': 'chunking',
                    'start': chunk.start_char,
                    'end': chunk.end_char
                })
        
        # 3. استخراج باستخدام الـ Regex
        for pattern in self.tech_patterns:
            matches = re.findall(pattern, text)
            for match in matches:
                if len(match) > 2 and not any(e['text'] == match for e in entities):
                    entities.append({
                        'text': match,
                        'type': 'tech_term',
                        'method': 'regex',
                        'start': text.find(match),
                        'end': text.find(match) + len(match)
                    })
        
        # إزالة التكرارات
        unique_entities = []
        seen = set()
        for e in entities:
            if e['text'].lower() not in seen and len(e['text']) > 2:
                seen.add(e['text'].lower())
                unique_entities.append(e)
        
        return {'entities': unique_entities, 'count': len(unique_entities)}


class SemanticConceptExtractor:
    """يستخرج المفاهيم الدلالية باستخدام embeddings"""
    
    def __init__(self):
        self.model = SentenceTransformer("BAAI/bge-m3")
        
    def extract(self, text: str) -> dict:
        """يحول النص إلى متجهات دلالية"""
        # تقسيم إلى جمل
        sentences = [s.strip() for s in text.split('.') if len(s.strip()) > 10]
        
        if not sentences:
            sentences = [text[:500]]
        
        # توليد embeddings
        embeddings = self.model.encode(sentences, normalize_embeddings=True)
        
        return {
            'sentences': sentences,
            'embeddings': embeddings.tolist(),
            'dimension': embeddings.shape[1] if len(embeddings.shape) > 1 else 768,
            'num_sentences': len(sentences)
        }
    
    def get_sentence_vectors(self, text: str) -> np.ndarray:
        """يحصل على متجهات الجمل"""
        sentences = [s.strip() for s in text.split('.') if len(s.strip()) > 10]
        if not sentences:
            sentences = [text[:500]]
        return self.model.encode(sentences, normalize_embeddings=True)


class AdaptiveConceptClusterer:
    """يجمع المفاهيم المتشابهة لاكتشاف موضوعات جديدة"""
    
    def __init__(self):
        self.encoder = SentenceTransformer("BAAI/bge-m3")
        
    def cluster_concepts(self, concepts: list, embeddings: np.ndarray = None) -> dict:
        """يجمع المفاهيم المتشابهة باستخدام KMeans"""
        if not concepts or len(concepts) < 2:
            return {'clusters': [], 'concept_clusters': {}}
        
        # توليد embeddings إذا لم تكن موجودة
        if embeddings is None:
            embeddings = self.encoder.encode(concepts, normalize_embeddings=True)
        
        # عدد المجموعات المناسب
        n_clusters = min(max(2, len(concepts) // 3), 8)
        
        # KMeans clustering
        kmeans = KMeans(n_clusters=n_clusters, random_state=42, n_init=10)
        labels = kmeans.fit_predict(embeddings)
        
        # تنظيم النتائج
        clusters = {}
        for i, (concept, label) in enumerate(zip(concepts, labels)):
            if label not in clusters:
                clusters[label] = []
            clusters[label].append(concept)
        
        # حساب المركز الدلالي لكل مجموعة
        cluster_centers = kmeans.cluster_centers_
        
        # إضافة معلومات المجموعات
        cluster_list = []
        for label, members in clusters.items():
            cluster_list.append({
                'cluster_id': int(label),
                'size': len(members),
                'members': members[:10],  # أول 10 أعضاء
                'center': cluster_centers[label].tolist() if label < len(cluster_centers) else [],
                'suggested_name': self._suggest_cluster_name(members)
            })
        
        return {
            'clusters': cluster_list,
            'concept_clusters': {c: int(l) for c, l in zip(concepts, labels)},
            'num_clusters': n_clusters,
            'total_concepts': len(concepts)
        }
    
    def _suggest_cluster_name(self, members: list) -> str:
        """يقترح اسماً للمجموعة بناءً على أعضائها"""
        if not members:
            return "Unknown"
        # نأخذ أطول عضو كاقتراح أو نستخدم LLM لاحقاً
        return max(members, key=len)[:30]


class RelationDetector:
    """يكتشف العلاقات بين المفاهيم تلقائياً"""
    
    def __init__(self):
        self.encoder = SentenceTransformer("BAAI/bge-m3")
        
    def detect_relations(self, concepts: list, text: str) -> list:
        """يكتشف العلاقات بين المفاهيم بناءً على السياق"""
        relations = []
        
        # البحث عن أنماط العلاقات في النص
        relation_patterns = [
            (r'(\w+)\s+uses\s+(\w+)', 'USES'),
            (r'(\w+)\s+depends on\s+(\w+)', 'DEPENDS_ON'),
            (r'(\w+)\s+is a\s+(\w+)', 'IS_A'),
            (r'(\w+)\s+contains\s+(\w+)', 'CONTAINS'),
            (r'(\w+)\s+has\s+(\w+)', 'HAS'),
            (r'(\w+)\s+calls\s+(\w+)', 'CALLS'),
            (r'(\w+)\s+communicates with\s+(\w+)', 'COMMUNICATES_WITH'),
        ]
        
        for pattern, rel_type in relation_patterns:
            matches = re.findall(pattern, text, re.IGNORECASE)
            for match in matches:
                if len(match) == 2:
                    # التحقق من وجود المفاهيم في قائمة المفاهيم المستخرجة
                    concept1 = match[0]
                    concept2 = match[1]
                    relations.append({
                        'source': concept1,
                        'target': concept2,
                        'type': rel_type,
                        'detected_by': 'pattern_matching'
                    })
        
        # إزالة التكرارات
        unique_relations = []
        seen = set()
        for r in relations:
            key = f"{r['source']}_{r['target']}_{r['type']}"
            if key not in seen:
                seen.add(key)
                unique_relations.append(r)
        
        return unique_relations


class SelfAdaptiveUnderstandingLayer:
    """
    الطبقة الذكية ذاتية التكيف
    تكتشف كل شيء تلقائياً بدون إعدادات مسبقة
    """
    
    def __init__(self):
        self.context_detector = SmartContextDetector()
        self.entity_extractor = DynamicEntityExtractor()
        self.semantic_extractor = SemanticConceptExtractor()
        self.concept_clusterer = AdaptiveConceptClusterer()
        self.relation_detector = RelationDetector()
        
    def analyze(self, text: str) -> dict:
        """
        تحليل كامل للنص بدون أي افتراضات مسبقة
        """
        if not text or len(text) < 20:
            return {'error': 'Text too short for analysis'}
        
        # 1. كشف السياق (لغة، مجال، مستوى)
        context = self.context_detector.detect_all(text)
        
        # 2. استخراج الكيانات الديناميكي
        entities = self.entity_extractor.extract(text, context['language'])
        
        # 3. استخراج المفاهيم الدلالية
        semantic = self.semantic_extractor.extract(text)
        
        # 4. تجميع المفاهيم المتشابهة
        concept_texts = [e['text'] for e in entities['entities']]
        if len(concept_texts) > 1:
            clusters = self.concept_clusterer.cluster_concepts(concept_texts)
        else:
            clusters = {'clusters': [], 'concept_clusters': {}, 'num_clusters': 0}
        
        # 5. اكتشاف العلاقات
        relations = self.relation_detector.detect_relations(concept_texts, text)
        
        return {
            'context': context,
            'entities': entities,
            'semantic': semantic,
            'clusters': clusters,
            'relations': relations,
            'summary': {
                'total_entities': entities['count'],
                'num_clusters': clusters.get('num_clusters', 0),
                'num_relations': len(relations),
                'has_semantic_data': len(semantic.get('embeddings', [])) > 0
            }
        }
