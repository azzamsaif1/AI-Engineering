import aiohttp
import asyncio
from typing import Dict, List
from qdrant_client import QdrantClient
from neo4j import GraphDatabase
from sentence_transformers import SentenceTransformer

class AcademicIntelligence:
    """
    النسخة التعليمية - ذكاء أكاديمي مستقل للتعليم والمعرفة
    """
    def __init__(self):
        # مصادر المعرفة
        self.sources = {
            'mit_ocw': MITOCWSource(),
            'stanford': StanfordOnlineSource(),
            'coursera': CourseraSource(),
            'edx': EdXSource(),
            'youtube': YouTubeEduSource(),
            'books': BooksSource()
        }
        
        # تخزين المعرفة
        self.knowledge_graph = Neo4jConnection()
        self.vector_store = QdrantClient(host="localhost", port=6333)
        self.encoder = SentenceTransformer("BAAI/bge-m3")
        
        # خريطة المعرفة العالمية
        self.global_knowledge_graph = {}
        
    async def build_knowledge_graph(self) -> Dict:
        """بناء خريطة المعرفة العالمية من جميع المصادر"""
        
        # جلب المحتوى من جميع المصادر
        tasks = [source.fetch() for source in self.sources.values()]
        results = await asyncio.gather(*tasks)
        
        # معالجة كل مصدر
        for result in results:
            await self._process_source(result)
        
        # بناء العلاقات بين المفاهيم
        await self._build_relations()
        
        # تحسين خريطة المعرفة
        await self._optimize_knowledge_graph()
        
        return self.global_knowledge_graph
    
    async def get_learning_path(self, topic: str, user_level: str) -> Dict:
        """الحصول على مسار تعلم مخصص"""
        
        # البحث عن المفاهيم الأساسية للموضوع
        concepts = await self._find_concepts(topic)
        
        # ترتيب المفاهيم حسب المتطلبات الأساسية
        ordered_concepts = self._topological_sort(concepts)
        
        # اختيار أفضل المصادر لكل مفهوم
        sources = await self._select_best_sources(ordered_concepts, user_level)
        
        # إنشاء خريطة التعلم
        learning_path = {
            'topic': topic,
            'user_level': user_level,
            'estimated_duration': self._estimate_duration(ordered_concepts),
            'prerequisites': await self._get_prerequisites(topic),
            'concepts': [],
            'projects': await self._get_projects(topic)
        }
        
        for concept in ordered_concepts:
            learning_path['concepts'].append({
                'name': concept['name'],
                'difficulty': concept['difficulty'],
                'sources': sources.get(concept['name'], []),
                'estimated_time': concept['estimated_time'],
                'prerequisites': concept.get('prerequisites', [])
            })
        
        return learning_path

            'top_skills': sorted(self.skill_demand.items(), key=lambda x: x[1], reverse=True)[:20],

# backend/specialized/security_intelligence.py
import aiohttp
import asyncio
from typing import Dict, List
from joern import JoernAPI
from semgrep import Semgrep
import subprocess

class SecurityIntelligence:
    """
    النسخة الأمنية - ذكاء دفاعي أخلاقي للتحليل الأمني
    """
    def __init__(self):
        # مصادر الثغرات
        self.sources = {
            'cve': CVESource(),
            'nvd': NVDSource(),
            'owasp': OWASPSource(),
            'exploit_db': ExploitDBSource()
        }
        
        # أدوات التحليل الأمني
        self.joern = JoernAPI()
        self.semgrep = Semgrep()
        
        # قاعدة المعرفة الأمنية
        self.security_knowledge_graph = {}
        self.vulnerability_patterns = []
        self.mitigations = []
        
    async def analyze_code_security(self, code: str, language: str = "python") -> Dict:
        """تحليل أمني للكود"""
        
        # تحليل باستخدام Semgrep
        semgrep_results = await self.semgrep.scan(code, language)
        
        # تحليل تدفق البيانات باستخدام Joern
        joern_results = await self.joern.analyze(code)
        
        # البحث عن ثغرات مشابهة في قاعدة البيانات
        similar_vulnerabilities = await self._find_similar_vulnerabilities(code)
        
        # اقتراح إصلاحات
        mitigations = await self._suggest_mitigations(semgrep_results, joern_results)
        
        return {
            'semgrep_results': semgrep_results,
            'joern_results': joern_results,
            'similar_vulnerabilities': similar_vulnerabilities,
            'mitigations': mitigations,
            'risk_score': self._calculate_risk_score(semgrep_results, joern_results),
            'recommendations': await self._generate_recommendations(semgrep_results, joern_results)
        }
    
    async def predict_vulnerabilities(self, architecture: Dict) -> List:
        """توقع الثغرات المستقبلية في architecture معينة"""
        
        # تحليل نقاط الضعف المحتملة
        potential_weaknesses = self._analyze_potential_weaknesses(architecture)
        
        # مقارنة مع أنماط الثغرات المعروفة
        matched_patterns = await self._match_vulnerability_patterns(potential_weaknesses)
        
        # ترتيب حسب الخطورة
        ranked = sorted(matched_patterns, key=lambda x: x['severity'], reverse=True)
        
        return ranked
    
    async def get_security_knowledge(self, topic: str) -> Dict:
        """الحصول على المعرفة الأمنية حول موضوع معين"""
        
        # البحث في قاعدة المعرفة الأمنية
        knowledge = await self._search_security_knowledge(topic)
        
        # تنظيم المعرفة
        return {
            'topic': topic,
            'vulnerabilities': knowledge.get('vulnerabilities', []),
            'mitigations': knowledge.get('mitigations', []),
            'best_practices': knowledge.get('best_practices', []),
            'tools': knowledge.get('tools', []),
            'references': knowledge.get('references', [])
        }

# backend/specialized/unified_core.py
from langgraph import StateGraph
from typing import Dict

class UnifiedCognitiveCore:
    """
    العقل المركزي الموحد - ينسق بين النسخ المتخصصة الثلاث
    """
    def __init__(self):
        self.academic = AcademicIntelligence()
        self.career = CareerIntelligence()
        self.security = SecurityIntelligence()
        
        # تنسيق الوكلاء
        self.orchestrator = StateGraph(UnifiedState)
        self._setup_orchestration()
        
    async def process_request(self, request: Dict) -> Dict:
        """معالجة الطلب باستخدام النسخة المناسبة"""
        
        request_type = request.get('type', 'general')
        
        if request_type == 'academic':
            result = await self.academic.process(request)
        elif request_type == 'career':
            result = await self.career.process(request)
        elif request_type == 'security':
            result = await self.security.process(request)
        else:
            # دمج نتائج جميع النسخ
            result = await self._integrate_all(request)
        
        return result
    
    async def _integrate_all(self, request: Dict) -> Dict:
        """دمج نتائج جميع النسخ الثلاث"""
        
        # المعرفة الأكاديمية
        academic_result = await self.academic.process(request)
        
        # التوجيه المهني
        career_result = await self.career.process(request)
        
        # التحليل الأمني
        security_result = await self.security.process(request)
        
        return {
            'academic': academic_result,
            'career': career_result,
            'security': security_result,
            'integrated_recommendation': await self._generate_integrated_recommendation(
                academic_result, career_result, security_result
            )
        }

class EventType(Enum):

class EventBus:
    """
    Event Bus V12 - ناقل الأحداث المركزي
    """
    
    def __init__(self):
        self.subscribers: Dict[EventType, List[Callable]] = {}
        self.event_history: List[Event] = []
        self.max_history = 10000
        self.stats = {
            'events_published': 0,
            'events_processed': 0,
            'subscribers_count': 0
        }
    
    def subscribe(self, event_type: EventType):
        """ديكورator للتسجيل في الأحداث"""
        def decorator(handler: Callable):
            if event_type not in self.subscribers:
                self.subscribers[event_type] = []
            self.subscribers[event_type].append(handler)
            self.stats['subscribers_count'] += 1
            return handler
        return decorator
    
    async def emit(self, event_type: EventType, payload: Dict, 
                   source: str = "system") -> Event:
        """نشر حدث"""
        event = Event(
            id=f"{event_type.value}_{int(time.time()*1000)}_{self.stats['events_published']}",
            type=event_type,
            payload=payload,
            source=source,
            timestamp=time.time()
        )
        
        self.event_history.append(event)
        if len(self.event_history) > self.max_history:
            self.event_history = self.event_history[-self.max_history:]
        
        self.stats['events_published'] += 1
        
        # استدعاء المعالجات
        if event_type in self.subscribers:
            for handler in self.subscribers[event_type]:
                try:
                    if asyncio.iscoroutinefunction(handler):
                        await handler(event)
                    else:
                        handler(event)
                    self.stats['events_processed'] += 1
                except Exception as e:
                    logger.error(f"Handler error for {event_type.value}: {e}")
        
        return event


class UnifiedMemory:
    """
    Unified Memory V12 - دالة ذاكرة موحدة
    تدمج: Vector DB (Qdrant), Graph DB (Neo4j), Cache (Redis)
    """
    
    def __init__(self):
        # في الإنتاج: اتصالات حقيقية
        self.vector_store: Dict[str, List[float]] = {}  # محاكاة Qdrant
        self.graph_store: Dict[str, Dict] = {}  # محاكاة Neo4j
        self.cache: Dict[str, Any] = {}
        
        self.stats = {
            'vector_stores': 0,
            'graph_stores': 0,
            'cache_hits': 0,
            'cache_misses': 0
        }
    
    async def store_vector(self, key: str, vector: List[float], metadata: Dict = None) -> str:
        """تخزين متجه"""
        self.vector_store[key] = vector
        self.stats['vector_stores'] += 1
        return key
    
    async def store_graph(self, node_id: str, properties: Dict, relations: List = None) -> str:
        """تخزين في الرسم البياني"""
        self.graph_store[node_id] = {
            'properties': properties,
            'relations': relations or []
        }
        self.stats['graph_stores'] += 1
        return node_id
    
    async def store_cache(self, key: str, value: Any, ttl_seconds: int = 3600) -> str:
        """تخزين في الكاش"""
        self.cache[key] = {
            'value': value,
            'expires_at': time.time() + ttl_seconds
        }
        return key
    
    async def retrieve_vector(self, query_vector: List[float], top_k: int = 10) -> List[Dict]:
        """استرجاع من المتجهات"""
        # في الإنتاج: بحث تشابه حقيقي
        results = []
        for key, vec in list(self.vector_store.items())[:top_k]:
            results.append({'id': key, 'score': 0.8, 'metadata': {}})
        return results
    
    async def retrieve_graph(self, query: str) -> List[Dict]:
        """استرجاع من الرسم البياني"""
        results = []
        for node_id, data in list(self.graph_store.items())[:10]:
            results.append({'id': node_id, 'properties': data['properties']})
        return results
    
    async def retrieve_cache(self, key: str) -> Optional[Any]:
        """استرجاع من الكاش"""
        if key in self.cache:
            cache_item = self.cache[key]
            if cache_item['expires_at'] > time.time():
                self.stats['cache_hits'] += 1
                return cache_item['value']
            else:
                del self.cache[key]
        self.stats['cache_misses'] += 1
        return None
    
    async def get_stats(self) -> Dict:
        """إحصائيات الذاكرة"""
        return {
            **self.stats,
            'vector_count': len(self.vector_store),
            'graph_count': len(self.graph_store),
            'cache_count': len(self.cache)
        }


class TaskScheduler:
    """
    Task Scheduler V12 - جدولة المهام بين الوكلاء
    """
    
    def __init__(self):
        self.task_queue: asyncio.Queue = asyncio.Queue()
        self.running_tasks: Dict[str, asyncio.Task] = {}
        self.is_running = False
        self.processor_task = None
        
        self.stats = {
            'tasks_scheduled': 0,
            'tasks_completed': 0,
            'tasks_failed': 0
        }
    
    async def schedule(self, agent_name: str, task_func, priority: int = 2) -> str:
        """جدولة مهمة"""
        task_id = f"{agent_name}_{int(time.time()*1000)}_{self.stats['tasks_scheduled']}"
        await self.task_queue.put({
            'id': task_id,
            'agent': agent_name,
            'func': task_func,
            'priority': priority,
            'timestamp': time.time()
        })
        self.stats['tasks_scheduled'] += 1
        return task_id
    
    async def _process_loop(self):
        """حلقة معالجة المهام"""
        while self.is_running:
            try:
                task = await self.task_queue.get()
                
                try:
                    result = await task['func']()
                    self.stats['tasks_completed'] += 1
                except Exception as e:
                    logger.error(f"Task {task['id']} failed: {e}")
                    self.stats['tasks_failed'] += 1
                
            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.error(f"Scheduler error: {e}")
    
    async def start(self):
        """بدء الجدولة"""
        self.is_running = True
        self.processor_task = asyncio.create_task(self._process_loop())
    
    async def stop(self):
        """إيقاف الجدولة"""
        self.is_running = False
        if self.processor_task:
            self.processor_task.cancel()


class LearningLoop:
    """
    Learning Loop V12 - حلقة التعلم الذاتي
    Evaluate → Score → Update → Reinforce
    """
    
    def __init__(self):
        self.performance_history: Dict[str, List[float]] = {}
        self.learning_rate = 0.1
        self.stats = {
            'evaluations': 0,
            'updates_performed': 0,
            'avg_score': 0.0
        }
    
    async def evaluate(self, agent_name: str, output: Any, expected: Any) -> float:
        """تقييم خرج الوكيل"""
        # تبسيطاً: حساب التشابه
        score = 0.7  # قيمة افتراضية
        
        if agent_name not in self.performance_history:
            self.performance_history[agent_name] = []
        
        self.performance_history[agent_name].append(score)
        self.stats['evaluations'] += 1
        self.stats['avg_score'] = (
            self.stats['avg_score'] * 0.95 + score * 0.05
        )
        
        return score
    
    async def update(self, agent_name: str, score: float):
        """تحديث الوكيل بناءً على التقييم"""
        if score < 0.5:
            # يحتاج تحسين
            self.stats['updates_performed'] += 1
            logger.info(f"Agent {agent_name} needs improvement (score={score:.2f})")
        else:
            # تعزيز
            logger.debug(f"Agent {agent_name} performing well (score={score:.2f})")
    
    async def get_agent_performance(self, agent_name: str) -> Dict:
        """الحصول على أداء وكيل معين"""
        history = self.performance_history.get(agent_name, [])
        if not history:
            return {'avg_score': 0.0, 'samples': 0}
        
        return {
            'avg_score': sum(history) / len(history),
            'samples': len(history),
            'trend': history[-10:] if len(history) > 10 else history
        }


class AgentOS:
    """
    Agent Operating System V12 - نظام تشغيل الوكلاء
    هذا هو القلب الحقيقي للنظام
    """
    
    def __init__(self):
        self.event_bus = EventBus()
        self.memory = UnifiedMemory()
        self.scheduler = TaskScheduler()
        self.learning_loop = LearningLoop()
        
        self.agents: Dict[str, AgentInfo] = {}
        self.agent_instances: Dict[str, Any] = {}
        
        self.is_running = False
        self.health_task = None
        
        self.stats = {
            'start_time': 0,
            'total_events': 0,
            'active_agents': 0,
            'uptime_seconds': 0
        }
    
    async def register_agent(self, name: str, agent_instance, 
                              description: str, capabilities: List[str]):
        """تسجيل وكيل في نظام التشغيل"""
        self.agents[name] = AgentInfo(
            name=name,
            description=description,
            capabilities=capabilities
        )
        self.agent_instances[name] = agent_instance
        
        # ربط الوكيل بنظام التشغيل
        if hasattr(agent_instance, 'initialize'):
            await agent_instance.initialize(self)
        
        await self.event_bus.emit(EventType.AGENT_REGISTERED, {
            'name': name,
            'capabilities': capabilities
        }, source="agent_os")
        
        logger.info(f"Agent registered: {name}")
    
    async def start(self):
        """بدء نظام التشغيل"""
        self.is_running = True
        self.stats['start_time'] = time.time()
        
        # بدء الجدولة
        await self.scheduler.start()
        
        # بدء مراقبة الصحة
        self.health_task = asyncio.create_task(self._health_loop())
        
        # بدء جميع الوكلاء
        for name, agent in self.agent_instances.items():
            if hasattr(agent, 'start'):
                await agent.start()
            self.agents[name].is_active = True
            self.stats['active_agents'] += 1
            await self.event_bus.emit(EventType.AGENT_STARTED, {
                'name': name
            }, source="agent_os")
        
        logger.info(f"AgentOS started with {len(self.agents)} agents")
    
    async def _health_loop(self):
        """حلقة مراقبة الصحة"""
        while self.is_running:
            self.stats['uptime_seconds'] = time.time() - self.stats['start_time']
            self.stats['total_events'] = self.event_bus.stats['events_published']
            
            # تحديث نبضات القلب للوكلاء
            for name, info in self.agents.items():
                info.last_heartbeat = time.time()
            
            await self.event_bus.emit(EventType.SYSTEM_HEALTH, {
                'stats': self.stats,
                'agents': {k: {'active': v.is_active} for k, v in self.agents.items()}
            }, source="agent_os")
            
            await asyncio.sleep(30)  # كل 30 ثانية
    
    async def stop(self):
        """إيقاف نظام التشغيل"""
        logger.info("Stopping AgentOS...")
        self.is_running = False
        
        # إيقاف جميع الوكلاء
        for name, agent in self.agent_instances.items():
            if hasattr(agent, 'stop'):
                await agent.stop()
            self.agents[name].is_active = False
        
        # إيقاف الجدولة
        await self.scheduler.stop()
        
        if self.health_task:
            self.health_task.cancel()
        
        logger.info("AgentOS stopped")
    
    async def get_status(self) -> Dict:
        """الحالة الكاملة للنظام"""
        memory_stats = await self.memory.get_stats()
        
        return {
            'is_running': self.is_running,
            'stats': self.stats,
            'event_bus': self.event_bus.stats,
            'memory': memory_stats,
            'scheduler': self.scheduler.stats,
            'learning_loop': self.learning_loop.stats,
            'agents': {
                name: {
                    'description': info.description,
                    'capabilities': info.capabilities,
                    'is_active': info.is_active,
                    'performance': await self.learning_loop.get_agent_performance(name)
                }
                for name, info in self.agents.items()
            }
        }

class SecurityAnalyzer:
    """محلل أمني للكود"""
    
    def __init__(self):
        self.patterns = {
            'sql_injection': r'(?i)(execute|executemany|raw_connection).*?\+.*?(request|input|get)',
            'xss': r'(?i)(innerHTML|outerHTML|document\.write).*?(request|input|get)',
            'command_injection': r'(?i)(os\.system|subprocess\.call|eval|exec).*?(request|input|get)',
            'hardcoded_secrets': r'(?i)(password|secret|key|token)\s*=\s*[\'\"][^\'\"]+[\'\"]'
        }
    
    async def scan_code(self, code: str) -> List[SecurityFinding]:
        """مسح الكود بحثاً عن ثغرات"""
        findings = []
        lines = code.split('\n')
        
        for i, line in enumerate(lines):
            for rule_id, pattern in self.patterns.items():
                if re.search(pattern, line):
                    findings.append(SecurityFinding(
                        line=i + 1,
                        column=0,
                        rule_id=rule_id,
                        message=f"Potential {rule_id} detected",
                        severity=self._get_severity(rule_id),
                        fix_suggestion=self._get_fix_suggestion(rule_id)
                    ))
        
        return findings
    
    def _get_severity(self, rule_id: str) -> str:
        """الحصول على شدة الثغرة"""
        severities = {
            'sql_injection': 'CRITICAL',
            'command_injection': 'CRITICAL',
            'xss': 'HIGH',
            'hardcoded_secrets': 'HIGH'
        }
        return severities.get(rule_id, 'MEDIUM')
    
    def _get_fix_suggestion(self, rule_id: str) -> str:
        """الحصول على اقتراح إصلاح"""
        suggestions = {
            'sql_injection': 'Use parameterized queries instead of string concatenation',
            'command_injection': 'Avoid shell commands. Use safe alternatives or sanitize input',
            'xss': 'Use textContent instead of innerHTML, or sanitize output',
            'hardcoded_secrets': 'Use environment variables or secret management services'
        }
        return suggestions.get(rule_id, 'Review and fix the issue')

class ReActPlanner:
    """
    ReAct Planner (Reasoning + Acting) - مخطط ReAct
    """
    
    def __init__(self, max_iterations: int = 10):
        self.max_iterations = max_iterations
        self.thought_history: List[Thought] = []
        
    async def plan(self, goal: str, context: Dict, available_tools: List[str]) -> List[Dict]:
        """
        تخطيط باستخدام ReAct
        """
        plan = []
        
        for i in range(self.max_iterations):
            # 1. Thought - التفكير
            thought = await self._think(goal, context, self.thought_history)
            
            # 2. Action - اختيار إجراء
            action = await self._select_action(thought, available_tools)
            
            # 3. Action Input - مدخلات الإجراء
            action_input = await self._get_action_input(action, context)
            
            # 4. Observation - مراقبة النتيجة
            observation = await self._simulate_action(action, action_input)
            
            # تسجيل
            self.thought_history.append(Thought(
                step=i,
                thought=thought,
                action=action,
                action_input=action_input,
                observation=observation,
                timestamp=time.time()
            ))
            
            # إضافة إلى الخطة
            plan.append({
                'step': i,
                'action': action,
                'input': action_input,
                'expected_outcome': observation
            })
            
            # التحقق من اكتمال الهدف
            if self._is_goal_achieved(observation, goal):
                break
        
        return plan
    
    async def _think(self, goal: str, context: Dict, history: List[Thought]) -> str:
        """توليد تفكير"""
        prompt = f"""
        Goal: {goal}
        Context: {json.dumps(context)[:500]}
        
        Previous steps:
        {chr(10).join([f"Step {t.step}: {t.thought} -> {t.action}" for t in history[-3:]])}
        
        What should I do next?
        """
        # في الإنتاج: استدعاء LLM
        return f"Need to analyze the problem and break it down into sub-tasks"
    
    async def _select_action(self, thought: str, available_tools: List[str]) -> str:
        """اختيار إجراء"""
        # تبسيطاً: إرجاع أول أداة متاحة
        return available_tools[0] if available_tools else "search"
    
    async def _get_action_input(self, action: str, context: Dict) -> Dict:
        """الحصول على مدخلات الإجراء"""
        return {"query": context.get('topic', 'general')}
    
    async def _simulate_action(self, action: str, action_input: Dict) -> str:
        """محاكاة تنفيذ الإجراء"""
        return f"Executed {action} with input {action_input}"
    
    def _is_goal_achieved(self, observation: str, goal: str) -> bool:
        """التحقق من تحقيق الهدف"""
        return "completed" in observation.lower() or "success" in observation.lower()


class TreeOfThoughtsPlanner:
    """
    Tree of Thoughts Planner - مخطط أشجار التفكير المتقدمة
    """
    
    def __init__(self, branching_factor: int = 3, depth: int = 3):
        self.branching_factor = branching_factor
        self.depth = depth
        
    async def plan(self, goal: str, context: Dict, available_tools: List[str]) -> List[Dict]:
        """
        تخطيط باستخدام أشجار التفكير
        """
        # بناء شجرة التفكير
        root = await self._create_node(goal, context, 0)
        
        # توسيع الشجرة
        best_path = await self._expand_tree(root, available_tools, self.depth)
        
        # تحويل المسار إلى خطة
        plan = []
        for node in best_path:
            plan.append({
                'step': node['depth'],
                'thought': node['thought'],
                'action': node['action'],
                'input': node['action_input']
            })
        
        return plan
    
    async def _create_node(self, thought: str, context: Dict, depth: int) -> Dict:
        """إنشاء عقدة في الشجرة"""
        return {
            'thought': thought,
            'context': context,
            'depth': depth,
            'children': [],
            'score': 0.0
        }
    
    async def _expand_tree(self, node: Dict, tools: List[str], max_depth: int) -> List[Dict]:
        """توسيع الشجرة"""
        if node['depth'] >= max_depth:
            return [node]
        
        # توليد أفكار متعددة
        for i in range(self.branching_factor):
            child_thought = f"Alternative approach {i+1}: {node['thought']}"
            child = await self._create_node(child_thought, node['context'], node['depth'] + 1)
            child['action'] = tools[i % len(tools)] if tools else "search"
            node['children'].append(child)
        
        # تقييم وتصنيف الأطفال
        for child in node['children']:
            child['score'] = await self._evaluate_node(child)
        
        # ترتيب حسب الدرجة
        node['children'].sort(key=lambda x: x['score'], reverse=True)
        
        # اختيار أفضل مسار
        best_child = node['children'][0] if node['children'] else None
        if best_child:
            rest_path = await self._expand_tree(best_child, tools, max_depth)
            return [node] + rest_path
        
        return [node]
    
    async def _evaluate_node(self, node: Dict) -> float:
        """تقييم عقدة"""
        # تبسيطاً: درجة عشوائية
        import random
        return random.uniform(0.5, 1.0)


class GoalStack:
    """
    Goal Stack - إدارة الأهداف الهرمية
    """
    
    def __init__(self):
        self.goals: Dict[str, Goal] = {}
        self.active_goal_id: Optional[str] = None
        self.goal_history: List[Goal] = []
        
    async def push_goal(self, description: str, priority: int = 5, parent_id: Optional[str] = None) -> str:
        """إضافة هدف جديد"""
        goal_id = f"goal_{int(time.time()*1000)}_{len(self.goals)}"
        
        goal = Goal(
            id=goal_id,
            description=description,
            priority=priority,
            parent_id=parent_id,
            status=GoalStatus.PENDING
        )
        
        self.goals[goal_id] = goal
        
        # إذا كان هناك هدف أب، أضف كهدف فرعي
        if parent_id and parent_id in self.goals:
            self.goals[parent_id].sub_goals.append(goal_id)
        
        # تحديث الهدف النشط (الأولوية القصوى)
        await self._update_active_goal()
        
        return goal_id
    
    async def complete_goal(self, goal_id: str, result: Any = None):
        """إكمال هدف"""
        if goal_id in self.goals:
            self.goals[goal_id].status = GoalStatus.COMPLETED
            self.goals[goal_id].completed_at = time.time()
            self.goals[goal_id].result = result
            self.goal_history.append(self.goals[goal_id])
            await self._update_active_goal()
    
    async def fail_goal(self, goal_id: str, error: str):
        """فشل هدف"""
        if goal_id in self.goals:
            self.goals[goal_id].status = GoalStatus.FAILED
            self.goals[goal_id].result = {'error': error}
            await self._update_active_goal()
    
    async def _update_active_goal(self):
        """تحديث الهدف النشط (أعلى أولوية pending)"""
        pending_goals = [g for g in self.goals.values() if g.status == GoalStatus.PENDING]
        
        if pending_goals:
            # ترتيب حسب الأولوية (الأقل رقم = أعلى أولوية)
            pending_goals.sort(key=lambda x: x.priority)
            self.active_goal_id = pending_goals[0].id
        else:
            self.active_goal_id = None
    
    async def get_active_goal(self) -> Optional[Goal]:
        """الحصول على الهدف النشط"""
        if self.active_goal_id:
            return self.goals.get(self.active_goal_id)
        return None
    
    async def get_status(self) -> Dict:
        """الحالة الكاملة للأهداف"""
        return {
            'total_goals': len(self.goals),
            'completed': sum(1 for g in self.goals.values() if g.status == GoalStatus.COMPLETED),
            'failed': sum(1 for g in self.goals.values() if g.status == GoalStatus.FAILED),
            'pending': sum(1 for g in self.goals.values() if g.status == GoalStatus.PENDING),
            'active_goal': self.active_goal_id,
            'goals': [
                {
                    'id': g.id,
                    'description': g.description[:100],
                    'status': g.status.value,
                    'priority': g.priority
                }
                for g in list(self.goals.values())[-10:]
            ]
        }


class AgentPlanningEngine:
    """

    """
    
    def __init__(self, agent_name: str, planner_type: str = "react"):
        self.agent_name = agent_name
        self.planner_type = planner_type

        self.goal_stack = GoalStack()
        self.react_planner = ReActPlanner() if planner_type == "react" else None
        self.tot_planner = TreeOfThoughtsPlanner() if planner_type == "tot" else None
        
        self.current_plan: List[Dict] = []
        self.execution_history: List[Dict] = []
        
        # حلقة التنفيذ الرئيسية
        self.is_running = False
        self.execution_task = None
        
        self.stats = {
            'plans_created': 0,
            'plans_executed': 0,
            'plans_failed': 0,
            'avg_plan_duration_ms': 0.0
        }
    
    async def set_goal(self, description: str, priority: int = 5) -> str:
        """تعيين هدف جديد للوكيل"""
        goal_id = await self.goal_stack.push_goal(description, priority)
        
        # بدء التخطيط إذا لم يكن هناك خطة نشطة
        if not self.current_plan:
            asyncio.create_task(self._plan_and_execute())
        
        return goal_id
    
    async def _plan_and_execute(self):
        """التخطيط والتنفيذ المستمر"""
        while True:
            active_goal = await self.goal_stack.get_active_goal()
            if not active_goal:
                await asyncio.sleep(1)
                continue
            
            # 1. التخطيط
            plan = await self._create_plan(active_goal)
            self.current_plan = plan
            self.stats['plans_created'] += 1
            
            # 2. التنفيذ
            start_time = time.time()
            success = await self._execute_plan(plan)
            duration_ms = (time.time() - start_time) * 1000
            
            # 3. تحديث الإحصائيات
            if success:
                await self.goal_stack.complete_goal(active_goal.id)
                self.stats['plans_executed'] += 1
            else:
                await self.goal_stack.fail_goal(active_goal.id, "Plan execution failed")
                self.stats['plans_failed'] += 1
            
            self.stats['avg_plan_duration_ms'] = (
                self.stats['avg_plan_duration_ms'] * 0.9 + duration_ms * 0.1
            )
            
            await asyncio.sleep(0.1)
    
    async def _create_plan(self, goal: Goal) -> List[Dict]:
        """إنشاء خطة لتحقيق الهدف"""
        context = {'goal': goal.description, 'agent': self.agent_name}
        available_tools = await self._get_available_tools()
        
        if self.planner_type == "react" and self.react_planner:
            return await self.react_planner.plan(goal.description, context, available_tools)
        elif self.planner_type == "tot" and self.tot_planner:
            return await self.tot_planner.plan(goal.description, context, available_tools)
        else:
            # خطة بسيطة
            return [{'step': 0, 'action': 'analyze', 'input': {'goal': goal.description}}]
    
    async def _execute_plan(self, plan: List[Dict]) -> bool:
        """تنفيذ الخطة"""
        for step in plan:
            try:
                # تنفيذ الخطوة
                result = await self._execute_step(step)
                self.execution_history.append({
                    'timestamp': time.time(),
                    'step': step,
                    'result': result
                })
                
                if not result.get('success', True):
                    return False
                    
            except Exception as e:
                logger.error(f"Plan execution failed at step {step.get('step')}: {e}")
                return False
        
        return True
    
    async def _execute_step(self, step: Dict) -> Dict:
        """تنفيذ خطوة واحدة"""
        action = step.get('action', 'unknown')
        action_input = step.get('input', {})
        
        # في الإنتاج: استدعاء أداة حقيقية
        return {'success': True, 'action': action, 'result': f"Executed {action}"}
    
    async def _get_available_tools(self) -> List[str]:
        """الحصول على الأدوات المتاحة"""
        return ['search', 'analyze', 'generate', 'validate', 'retrieve']
    
    async def get_status(self) -> Dict:
        """حالة المحرك"""
        return {
            'agent': self.agent_name,
            'planner_type': self.planner_type,
            'stats': self.stats,
            'goal_stack': await self.goal_stack.get_status(),
            'current_plan_length': len(self.current_plan)
        }
    
    async def start(self):
        """بدء المحرك"""
        self.is_running = True
        asyncio.create_task(self._plan_and_execute())
    
    async def stop(self):
        """إيقاف المحرك"""
        self.is_running = False

class KafkaEventBus:
    """
    Kafka Event Bus - للإنتاج الفعلي
    """
    
    def __init__(self, bootstrap_servers: List[str] = None):
        self.bootstrap_servers = bootstrap_servers or ['localhost:9092']
        self.producer = None
        self.consumers: Dict[str, Any] = {}
        self.handlers: Dict[EventType, List[Callable]] = {}
        
        self.stats = {
            'events_produced': 0,
            'events_consumed': 0,
            'errors': 0
        }
    
    async def initialize(self):
        """تهيئة اتصال Kafka"""
        try:
            from aiokafka import AIOKafkaProducer, AIOKafkaConsumer
            import msgpack
            
            self.producer = AIOKafkaProducer(
                bootstrap_servers=self.bootstrap_servers,
                value_serializer=lambda v: msgpack.packb(v)
            )
            await self.producer.start()
            logger.info("KafkaEventBus initialized")
        except ImportError:
            logger.warning("aiokafka not installed, using memory fallback")
            self.producer = None
    
    async def emit(self, event_type: EventType, payload: Dict, 
                   source: str = "system", partition_key: str = None) -> Event:
        """نشر حدث إلى Kafka"""
        event = Event(
            id=f"{event_type.value}_{int(time.time()*1000)}_{self.stats['events_produced']}",
            type=event_type,
            payload=payload,
            source=source,
            timestamp=time.time(),
            partition_key=partition_key or event_type.value
        )
        
        if self.producer:
            try:
                await self.producer.send(
                    topic=event_type.value,
                    value=event.__dict__,
                    key=event.partition_key.encode()
                )
                self.stats['events_produced'] += 1
            except Exception as e:
                logger.error(f"Kafka emit error: {e}")
                self.stats['errors'] += 1
        
        return event
    
    def subscribe(self, event_type: EventType):
        """ديكورator للتسجيل في الأحداث"""
        def decorator(handler: Callable):
            if event_type not in self.handlers:
                self.handlers[event_type] = []
            self.handlers[event_type].append(handler)
            return handler
        return decorator
    
    async def start_consumer(self, event_types: List[EventType], group_id: str):
        """بدء مستهلك لأحداث معينة"""
        try:
            from aiokafka import AIOKafkaConsumer
            import msgpack
            
            topics = [et.value for et in event_types]
            consumer = AIOKafkaConsumer(
                *topics,
                bootstrap_servers=self.bootstrap_servers,
                group_id=group_id,
                value_deserializer=lambda v: msgpack.unpackb(v)
            )
            await consumer.start()
            
            # بدء حلقة الاستهلاك
            asyncio.create_task(self._consume_loop(consumer, event_types))
            
            self.consumers[group_id] = consumer
            logger.info(f"Started consumer for {topics}")
            
        except ImportError:
            logger.warning("aiokafka not installed, skipping consumer")
    
    async def _consume_loop(self, consumer, event_types: List[EventType]):
        """حلقة استهلاك الأحداث"""
        async for msg in consumer:
            try:
                event_data = msg.value
                event = Event(**event_data)
                
                if event.type in self.handlers:
                    for handler in self.handlers[event.type]:
                        if asyncio.iscoroutinefunction(handler):
                            await handler(event)
                        else:
                            handler(event)
                
                self.stats['events_consumed'] += 1
                
            except Exception as e:
                logger.error(f"Consume error: {e}")
                self.stats['errors'] += 1
    
    async def shutdown(self):
        """إيقاف ناقل الأحداث"""
        if self.producer:
            await self.producer.stop()
        
        for consumer in self.consumers.values():
            await consumer.stop()
        
        logger.info("KafkaEventBus shutdown")


class NATSEventBus:
    """
    NATS Event Bus - بديل أخف من Kafka
    """
    
    def __init__(self, servers: List[str] = None):
        self.servers = servers or ['nats://localhost:4222']
        self.nc = None
        self.js = None
        self.handlers: Dict[str, List[Callable]] = {}
        
        self.stats = {
            'events_produced': 0,
            'events_consumed': 0
        }
    
    async def initialize(self):
        """تهيئة اتصال NATS"""
        try:
            import nats
            from nats.js import JetStreamContext
            
            self.nc = await nats.connect(servers=self.servers)
            self.js = self.nc.jetstream()
            logger.info("NATSEventBus initialized")
        except ImportError:
            logger.warning("nats-py not installed, using memory fallback")
    
    async def emit(self, event_type: EventType, payload: Dict,
                   source: str = "system", partition_key: str = None) -> Event:
        """نشر حدث إلى NATS"""
        event = Event(
            id=f"{event_type.value}_{int(time.time()*1000)}_{self.stats['events_produced']}",
            type=event_type,
            payload=payload,
            source=source,
            timestamp=time.time(),
            partition_key=partition_key
        )
        
        if self.js:
            await self.js.publish(
                subject=event_type.value,
                payload=json.dumps(event.__dict__).encode()
            )
            self.stats['events_produced'] += 1
        
        return event
    
    def subscribe(self, subject: str):
        """ديكورator للتسجيل في موضوع"""
        def decorator(handler: Callable):
            if subject not in self.handlers:
                self.handlers[subject] = []
            self.handlers[subject].append(handler)
            return handler
        return decorator
    
    async def start_subscriber(self, subjects: List[str]):
        """بدء مشترك لمواضيع معينة"""
        if not self.nc:
            return
        
        for subject in subjects:
            await self.nc.subscribe(subject, cb=self._handle_message)
        
        logger.info(f"Started subscribers for {subjects}")
    
    async def _handle_message(self, msg):
        """معالجة الرسالة الواردة"""
        try:
            data = json.loads(msg.data.decode())
            event = Event(**data)
            
            if event.type.value in self.handlers:
                for handler in self.handlers[event.type.value]:
                    if asyncio.iscoroutinefunction(handler):
                        await handler(event)
                    else:
                        handler(event)
            
            self.stats['events_consumed'] += 1
            
        except Exception as e:
            logger.error(f"Message handling error: {e}")
    
    async def shutdown(self):
        """إيقاف ناقل الأحداث"""
        if self.nc:
            await self.nc.close()
        logger.info("NATSEventBus shutdown")


class MemoryEventBus:
    """
    Memory Event Bus - للتطوير والاختبار
    """
    
    def __init__(self):
        self.handlers: Dict[EventType, List[Callable]] = {}
        self.event_history: List[Event] = []
        self.max_history = 10000
        
        self.stats = {
            'events_produced': 0,
            'events_consumed': 0
        }
    
    async def emit(self, event_type: EventType, payload: Dict,
                   source: str = "system", partition_key: str = None) -> Event:
        """نشر حدث"""
        event = Event(
            id=f"{event_type.value}_{int(time.time()*1000)}_{self.stats['events_produced']}",
            type=event_type,
            payload=payload,
            source=source,
            timestamp=time.time(),
            partition_key=partition_key
        )
        
        self.event_history.append(event)
        if len(self.event_history) > self.max_history:
            self.event_history = self.event_history[-self.max_history:]
        
        self.stats['events_produced'] += 1
        
        # استدعاء المعالجات
        if event_type in self.handlers:
            for handler in self.handlers[event_type]:
                try:
                    if asyncio.iscoroutinefunction(handler):
                        await handler(event)
                    else:
                        handler(event)
                    self.stats['events_consumed'] += 1
                except Exception as e:
                    logger.error(f"Handler error: {e}")
        
        return event
    
    def subscribe(self, event_type: EventType):
        """ديكورator للتسجيل في الأحداث"""
        def decorator(handler: Callable):
            if event_type not in self.handlers:
                self.handlers[event_type] = []
            self.handlers[event_type].append(handler)
            return handler
        return decorator
    
    async def replay(self, event_type: Optional[EventType] = None, from_time: float = 0):
        """إعادة تشغيل الأحداث السابقة"""
        events = self.event_history
        if event_type:
            events = [e for e in events if e.type == event_type]
        if from_time:
            events = [e for e in events if e.timestamp >= from_time]
        
        for event in events:
            if event.type in self.handlers:
                for handler in self.handlers[event.type]:
                    if asyncio.iscoroutinefunction(handler):
                        await handler(event)
                    else:

                    MATCH (a {{id: $from_id}})
                    MATCH (b {{id: $to_id}})
                    MERGE (a)-[r:{relation_type}]->(b)
                    SET r += $props
                    """,
                    from_id=from_id, to_id=to_id, props=properties or {}

class SandboxExecutor:
    """
    Sandbox Executor - تنفيذ آمن في بيئة معزولة
    """
    
    def __init__(self, sandbox_type: str = "docker"):
        self.sandbox_type = sandbox_type
        self.active_sandboxes: Dict[str, Any] = {}
        
    async def execute(self, code: str, language: str = "python", timeout_sec: int = 30) -> Dict:
        """
        تنفيذ كود في بيئة معزولة
        """
        if self.sandbox_type == "docker":
            return await self._execute_docker(code, language, timeout_sec)
        else:
            return await self._execute_subprocess(code, language, timeout_sec)
    
    async def _execute_docker(self, code: str, language: str, timeout_sec: int) -> Dict:
        """تنفيذ في Docker container"""
        try:
            import docker
            
            client = docker.from_env()
            
            # إنشاء container مؤقت
            container = client.containers.run(
                image=f"python:3.11-slim",
                command=f"python -c {code[:1000]}",
                detach=True,
                remove=True,
                mem_limit="512m",
                network_disabled=True,
                read_only=True
            )
            
            # انتظار النتيجة مع timeout
            result = container.wait(timeout=timeout_sec)
            logs = container.logs().decode()
            
            return {
                'success': result['StatusCode'] == 0,
                'output': logs,
                'exit_code': result['StatusCode']
            }
            
        except ImportError:
            logger.warning("docker not installed, falling back to subprocess")
            return await self._execute_subprocess(code, language, timeout_sec)
        except Exception as e:
            return {'success': False, 'error': str(e)}
    
    async def _execute_subprocess(self, code: str, language: str, timeout_sec: int) -> Dict:
        """تنفيذ في subprocess (أقل أماناً)"""
        import subprocess
        
        try:
            result = await asyncio.wait_for(
                asyncio.to_thread(
                    subprocess.run,
                    ["python", "-c", code],
                    capture_output=True,
                    timeout=timeout_sec
                ),
                timeout=timeout_sec + 5
            )
            
            return {
                'success': result.returncode == 0,
                'output': result.stdout.decode(),
                'error': result.stderr.decode(),
                'exit_code': result.returncode
            }
            
        except asyncio.TimeoutError:
            return {'success': False, 'error': f'Timeout after {timeout_sec}s'}
        except Exception as e:
            return {'success': False, 'error': str(e)}

# متغيرات السياق للتتبع
current_trace_id: ContextVar[str] = ContextVar('trace_id', default='')
current_span_id: ContextVar[str] = ContextVar('span_id', default='')


@dataclass
class Span:
    """امتداد للتتبع"""
    trace_id: str
    span_id: str
    parent_span_id: Optional[str]
    operation_name: str
    start_time: float
    end_time: Optional[float] = None
    tags: Dict = field(default_factory=dict)
    logs: List[Dict] = field(default_factory=list)


class Tracer:
    """
    Tracer - تتبع الطلبات عبر النظام
    """
    
    def __init__(self, service_name: str):
        self.service_name = service_name
        self.spans: List[Span] = []
        self.active_spans: Dict[str, Span] = {}
        
    def start_span(self, operation_name: str, parent_span_id: Optional[str] = None) -> str:
        """بدء امتداد جديد"""
        trace_id = current_trace_id.get()
        if not trace_id:
            trace_id = self._generate_id()
            current_trace_id.set(trace_id)
        
        span_id = self._generate_id()
        
        span = Span(
            trace_id=trace_id,
            span_id=span_id,
            parent_span_id=parent_span_id or current_span_id.get(),
            operation_name=operation_name,
            start_time=time.time()
        )
        
        self.active_spans[span_id] = span
        current_span_id.set(span_id)
        
        return span_id
    
    def finish_span(self, span_id: str, tags: Dict = None):
        """إنهاء امتداد"""
        if span_id in self.active_spans:
            span = self.active_spans[span_id]
            span.end_time = time.time()
            if tags:
                span.tags.update(tags)
            self.spans.append(span)
            del self.active_spans[span_id]
            
            # استعادة الـ span الأب
            if span.parent_span_id:
                current_span_id.set(span.parent_span_id)
            else:
                current_span_id.set('')
    
    def add_log(self, span_id: str, message: str, fields: Dict = None):
        """إضافة سجل إلى امتداد"""
        if span_id in self.active_spans:
            self.active_spans[span_id].logs.append({
                'timestamp': time.time(),
                'message': message,
                'fields': fields or {}
            })
    
    def _generate_id(self) -> str:
        """توليد معرف فريد"""
        import random
        import string
        return ''.join(random.choices(string.hexdigits, k=16))
    
    def get_traces(self) -> List[Dict]:
        """الحصول على جميع التتبعات"""
        return [
            {
                'trace_id': span.trace_id,
                'span_id': span.span_id,
                'parent_span_id': span.parent_span_id,
                'operation': span.operation_name,
                'duration_ms': (span.end_time - span.start_time) * 1000 if span.end_time else 0,
                'tags': span.tags
            }
            for span in self.spans
        ]


class MetricsCollector:
    """
    Metrics Collector - جمع المقاييس لـ Prometheus
    """
    
    def __init__(self):
        self.metrics: Dict[str, List[float]] = {}
        self.counters: Dict[str, int] = {}
        
    def counter(self, name: str, value: int = 1, tags: Dict = None):
        """زيادة عداد"""
        key = name
        if tags:
            key = f"{name}_{json.dumps(tags, sort_keys=True)}"
        self.counters[key] = self.counters.get(key, 0) + value
    
    def gauge(self, name: str, value: float, tags: Dict = None):
        """تعيين مقياس"""
        key = name
        if tags:
            key = f"{name}_{json.dumps(tags, sort_keys=True)}"
        if key not in self.metrics:
            self.metrics[key] = []
        self.metrics[key].append(value)
        # الاحتفاظ بآخر 1000 قيمة
        if len(self.metrics[key]) > 1000:
            self.metrics[key] = self.metrics[key][-1000:]
    
    def histogram(self, name: str, value: float, buckets: List[float] = None, tags: Dict = None):
        """تسجيل توزيع القيم"""
        self.gauge(name, value, tags)
    
    def get_metrics(self) -> Dict:
        """الحصول على جميع المقاييس"""
        result = {}
        
        for key, values in self.metrics.items():
            if values:
                result[f"{key}_avg"] = sum(values) / len(values)
                result[f"{key}_max"] = max(values)
                result[f"{key}_min"] = min(values)
                result[f"{key}_p95"] = sorted(values)[int(len(values) * 0.95)] if len(values) > 0 else 0
        
        for key, value in self.counters.items():
            result[key] = value
        
        return result
