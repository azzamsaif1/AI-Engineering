"""
TechLingua Mentor - Main Flask Application
Integrated with SADE v3.0 (Semantic Adaptive Discovery Engine)
"""

from flask import Flask, render_template, request, jsonify, session
from flask_sqlalchemy import SQLAlchemy
from flask_login import LoginManager, UserMixin, login_user, login_required, logout_user, current_user
from werkzeug.security import generate_password_hash, check_password_hash
from datetime import datetime
import os
import json
import re
from flask_socketio import SocketIO, emit

# Import SADE v3.0
from stage1_sade_v3.pipeline import SADEPipelineV3

app = Flask(__name__)
app.config['SECRET_KEY'] = os.environ.get('SECRET_KEY', 'techlingua-secret-key-2024')
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///database.db'
app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False

db = SQLAlchemy(app)
socketio = SocketIO(app, cors_allowed_origins="*")
login_manager = LoginManager()
login_manager.init_app(app)
login_manager.login_view = 'login'

# Initialize SADE v3.0 Pipeline
sade_pipeline = SADEPipelineV3()

# ==================== MODELS ====================

class User(UserMixin, db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(80), unique=True, nullable=False)
    password = db.Column(db.String(200), nullable=False)
    email = db.Column(db.String(120), unique=True, nullable=True)
    created_at = db.Column(db.DateTime, default=datetime.now)
    level = db.Column(db.String(20), default='A1')
    points = db.Column(db.Integer, default=0)

class Session(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'))
    session_id = db.Column(db.String(100), unique=True)
    start_time = db.Column(db.DateTime, default=datetime.now)
    end_time = db.Column(db.DateTime)
    duration = db.Column(db.Integer, default=0)
    status = db.Column(db.String(20), default='active')
    title = db.Column(db.String(200), default='')
    analysis_result = db.Column(db.Text, default='{}')
    created_at = db.Column(db.DateTime, default=datetime.now)

class Transcript(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    session_id = db.Column(db.Integer, db.ForeignKey('session.id'))
    text = db.Column(db.Text, nullable=False)
    translation = db.Column(db.Text)
    is_final = db.Column(db.Boolean, default=False)
    timestamp = db.Column(db.Integer, default=0)
    created_at = db.Column(db.DateTime, default=datetime.now)

class Concept(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'))
    session_id = db.Column(db.Integer, db.ForeignKey('session.id'))
    text = db.Column(db.String(200), nullable=False)
    concept_type = db.Column(db.String(50))
    confidence = db.Column(db.Float, default=0.5)
    created_at = db.Column(db.DateTime, default=datetime.now)

class Relation(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'))
    session_id = db.Column(db.Integer, db.ForeignKey('session.id'))
    source = db.Column(db.String(200), nullable=False)
    target = db.Column(db.String(200), nullable=False)
    relation_type = db.Column(db.String(50))
    confidence = db.Column(db.Float, default=0.5)
    created_at = db.Column(db.DateTime, default=datetime.now)

# ==================== CREATE TABLES ====================
with app.app_context():
    db.create_all()

# ==================== AUTHENTICATION ====================

@login_manager.user_loader
def load_user(user_id):
    return User.query.get(int(user_id))

@app.route('/')
def index():
    return render_template('index.html', current_user=current_user if current_user.is_authenticated else None)

@app.route('/login', methods=['POST'])
def login():
    data = request.get_json()
    username = data.get('username')
    password = data.get('password')
    user = User.query.filter_by(username=username).first()
    if user and check_password_hash(user.password, password):
        login_user(user)
        return jsonify({'success': True, 'username': user.username})
    return jsonify({'success': False, 'error': 'Invalid credentials'})

@app.route('/register', methods=['POST'])
def register():
    data = request.get_json()
    username = data.get('username')
    password = data.get('password')
    email = data.get('email', '')
    
    if User.query.filter_by(username=username).first():
        return jsonify({'success': False, 'error': 'Username exists'})
    
    hashed_password = generate_password_hash(password)
    new_user = User(username=username, password=hashed_password, email=email)
    db.session.add(new_user)
    db.session.commit()
    login_user(new_user)
    return jsonify({'success': True, 'username': username})

@app.route('/logout')
def logout():
    logout_user()
    return jsonify({'success': True})

# ==================== SESSION API ====================

@app.route('/api/start_session', methods=['POST'])
@login_required
def start_session():
    import uuid
    session_id = str(uuid.uuid4())
    new_session = Session(user_id=current_user.id, session_id=session_id)
    db.session.add(new_session)
    db.session.commit()
    return jsonify({'session_id': new_session.id, 'session_uuid': session_id, 'success': True})

@app.route('/api/stop_session/<int:session_id>', methods=['POST'])
@login_required
def stop_session(session_id):
    session_obj = Session.query.get(session_id)
    if session_obj and session_obj.user_id == current_user.id:
        session_obj.end_time = datetime.now()
        session_obj.duration = int((session_obj.end_time - session_obj.start_time).total_seconds())
        session_obj.status = 'completed'
        db.session.commit()
        return jsonify({'success': True, 'duration': session_obj.duration})
    return jsonify({'error': 'Session not found'}), 404

# ==================== TRANSCRIPT API ====================

@app.route('/api/save_transcript', methods=['POST'])
@login_required
def save_transcript():
    data = request.json
    session_id = data.get('session_id')
    text = data.get('text')
    is_final = data.get('is_final', False)
    translation = data.get('translation', '')
    
    if not session_id or not text:
        return jsonify({'error': 'Missing data'}), 400
    
    session_obj = Session.query.get(session_id)
    if not session_obj or session_obj.user_id != current_user.id:
        return jsonify({'error': 'Unauthorized'}), 401
    
    transcript = Transcript(
        session_id=session_id,
        text=text,
        translation=translation,
        is_final=is_final
    )
    db.session.add(transcript)
    db.session.commit()
    
    # If enough text accumulated, run SADE analysis
    all_text = ' '.join([t.text for t in Transcript.query.filter_by(session_id=session_id).all()])
    if len(all_text) > 200:
        analysis = sade_pipeline.analyze(all_text)
        session_obj.analysis_result = json.dumps(analysis)
        session_obj.title = analysis.get('concepts', [{}])[0].get('text', '')[:100] if analysis.get('concepts') else ''
        db.session.commit()
        
        # Emit analysis results via SocketIO
        socketio.emit('analysis_update', {
            'session_id': session_id,
            'concepts': analysis.get('concepts', [])[:10],
            'relations': analysis.get('relations', [])[:5],
            'topics': analysis.get('cluster_count', 0),
            'confidence': analysis.get('confidence', {})
        }, room=f"session_{session_id}")
    
    return jsonify({'success': True})

# ==================== SADE ANALYSIS API ====================

@app.route('/api/analyze_text', methods=['POST'])
@login_required
def analyze_text():
    """Run SADE v3.0 analysis on text"""
    data = request.json
    text = data.get('text', '')
    
    if not text or len(text) < 20:
        return jsonify({'error': 'Text too short for analysis'}), 400
    
    analysis = sade_pipeline.analyze(text)
    
    return jsonify(analysis)

@app.route('/api/session_analysis/<int:session_id>', methods=['GET'])
@login_required
def get_session_analysis(session_id):
    """Get stored analysis for a session"""
    session_obj = Session.query.get(session_id)
    if not session_obj or session_obj.user_id != current_user.id:
        return jsonify({'error': 'Unauthorized'}), 401
    
    if session_obj.analysis_result:
        return jsonify(json.loads(session_obj.analysis_result))
    
    # Run analysis if not available
    transcripts = Transcript.query.filter_by(session_id=session_id).all()
    all_text = ' '.join([t.text for t in transcripts])
    if len(all_text) > 200:
        analysis = sade_pipeline.analyze(all_text)
        session_obj.analysis_result = json.dumps(analysis)
        db.session.commit()
        return jsonify(analysis)
    
    return jsonify({'error': 'Not enough text for analysis'}), 400

@app.route('/api/extract_concepts', methods=['POST'])
@login_required
def extract_concepts():
    """Extract concepts from text using SADE"""
    data = request.json
    text = data.get('text', '')
    
    if not text:
        return jsonify({'error': 'No text provided'}), 400
    
    analysis = sade_pipeline.analyze(text)
    
    return jsonify({
        'concepts': analysis.get('concepts', []),
        'relations': analysis.get('relations', []),
        'confidence': analysis.get('confidence', {})
    })

# ==================== TRANSLATION API ====================

@app.route('/api/translate', methods=['POST'])
@login_required
def translate():
    data = request.json
    text = data.get('text', '')
    target_lang = data.get('target_lang', 'ar')
    
    if not text:
        return jsonify({'error': 'No text provided'}), 400
    
    try:
        from deep_translator import GoogleTranslator
        translated = GoogleTranslator(source='auto', target=target_lang).translate(text[:1000])
        return jsonify({'translated_text': translated})
    except Exception as e:
        return jsonify({'translated_text': text, 'error': str(e)})
