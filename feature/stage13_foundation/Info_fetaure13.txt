# backend/evaluation/engineering_capability_index.py
import numpy as np
from typing import Dict, List
from datetime import datetime, timedelta

class EngineeringCapabilityIndex:
    """
    مؤشر القدرات الهندسية التكيفي - نظام عادل وشفاف لقياس التطور الهندسي
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        
        # أبعاد القياس
        self.dimensions = {
            'learning_speed': LearningSpeedMetric(),
            'engineering_depth': EngineeringDepthMetric(),
            'execution_capability': ExecutionCapabilityMetric(),
            'continuous_evolution': ContinuousEvolutionMetric(),
            'human_twin_sync': HumanTwinSyncMetric()
        }
        
        # أوزان الأبعاد (قابلة للتعديل)
        self.weights = {
            'learning_speed': 0.25,
            'engineering_depth': 0.25,
            'execution_capability': 0.20,
            'continuous_evolution': 0.15,
            'human_twin_sync': 0.15
        }
        
        # تاريخ القياسات
        self.history = []
        
    async def evaluate(self, user_data: Dict) -> Dict:
        """
        تقييم المستخدم بشكل عادل وشفاف
        """
        scores = {}
        
        for name, metric in self.dimensions.items():
            scores[name] = await metric.calculate(user_data)
        
        # حساب المؤشر الكلي
        total_score = sum(scores[name] * self.weights[name] for name in scores)
        
        # إنشاء التقرير
        report = {
            'user_id': self.user_id,
            'timestamp': datetime.now().isoformat(),
            'total_score': total_score,
            'dimensions': scores,
            'strengths': self._identify_strengths(scores),
            'weaknesses': self._identify_weaknesses(scores),
            'recommendations': self._generate_recommendations(scores),
            'progress_trend': self._calculate_progress_trend()
        }
        
        self.history.append(report)
        
        return report
    
    def get_engineering_profile(self) -> Dict:
        """الحصول على الملف الهندسي للمستخدم (للشركات)"""
        
        if not self.history:
            return {'error': 'No data available'}
        
        latest = self.history[-1]
        
        return {
            'user_id': self.user_id,
            'systems_thinking': latest['dimensions']['engineering_depth'] * 100,
            'distributed_systems': self._calculate_specialty_score('distributed_systems'),
            'research_capability': latest['dimensions']['learning_speed'] * 100,
            'execution_consistency': latest['dimensions']['execution_capability'] * 100,
            'adaptive_learning_speed': latest['dimensions']['learning_speed'] * 100,
            'human_twin_sync': latest['dimensions']['human_twin_sync'] * 100,
            'overall_grade': self._get_grade(latest['total_score'])
        }

class LearningSpeedMetric:
    """قياس سرعة التعلم"""
    
    async def calculate(self, user_data: Dict) -> float:
        """حساب سرعة التعلم من 0 إلى 1"""
        # تحليل وقت تعلم المفاهيم الجديدة
        concept_learning_times = user_data.get('concept_learning_times', [])
        
        if not concept_learning_times:
            return 0.5
        
        # سرعة الانتقال بين المستويات
        level_transition_speed = user_data.get('level_transition_speed', 30)  # أيام
        
        # التكيف مع التقنيات الجديدة
        adaptation_speed = user_data.get('adaptation_speed', 0.7)
        
        # حساب النتيجة
        score = 0.4 * (1 / (1 + np.mean(concept_learning_times) / 7))  # كلما قل الوقت زادت النتيجة
        score += 0.3 * (1 / (1 + level_transition_speed / 30))
        score += 0.3 * adaptation_speed
        
        return min(1.0, score)

class HumanTwinSyncMetric:
    """قياس التكامل بين الإنسان والنسخة الرقمية"""
    
    async def calculate(self, user_data: Dict) -> float:
        """حساب مدى تكامل المستخدم مع نسخته الرقمية"""
        
        human_performance = user_data.get('human_performance', [])
        twin_performance = user_data.get('twin_performance', [])
        
        if not human_performance or not twin_performance:
            return 0.5
        
        # تحسين النسخة بسبب المستخدم
        twin_improvement = self._calculate_twin_improvement(twin_performance)
        
        # تحسين المستخدم بسبب النسخة
        human_improvement = self._calculate_human_improvement(human_performance)
        
        # التطور المشترك
        co_evolution = (twin_improvement + human_improvement) / 2
        
        # مدى قرب أسلوب النسخة من المستخدم
        style_similarity = user_data.get('style_similarity', 0.7)
        
        return 0.4 * co_evolution + 0.3 * style_similarity + 0.3 * human_improvement

# backend/execution/autonomous_project_engine.py
from langgraph import StateGraph
from typing import Dict, List
import asyncio

class AutonomousProjectEngine:
    """
    المحرك التنفيذي المستقل - قادر على بناء مشاريع كاملة بشكل شبه مستقل
    """
    
    def __init__(self):
        # قدرات المحرك
        self.capabilities = {
            'planning': ProjectPlanningAgent(),
            'architecture': ArchitectureDesignAgent(),
            'backend': BackendDevelopmentAgent(),
            'frontend': FrontendDevelopmentAgent(),
            'api': APIDevelopmentAgent(),
            'database': DatabaseDesignAgent(),
            'devops': DevOpsAgent(),
            'testing': TestingAgent(),
            'deployment': DeploymentAgent(),
            'optimization': OptimizationAgent()
        }
        
        # تكامل الأدوات الضخمة
        self.tool_integration = MassiveToolIntegration()
        
        # معمل البرمجيات المستمر
        self.laboratory = ContinuousSoftwareLaboratory()
        
        # تنسيق الوكلاء
        self.orchestrator = StateGraph(ProjectState)
        self._setup_orchestration()
        
    async def build_project(self, requirements: str) -> Dict:
        """
        بناء مشروع كامل من المتطلبات
        """
        project_state = {
            'requirements': requirements,
            'plan': {},
            'architecture': {},
            'backend': {},
            'frontend': {},
            'api': {},
            'database': {},
            'devops': {},
            'testing': {},
            'deployment': {},
            'optimization': {},
            'final_product': None
        }
        
        # تنفيذ جميع المراحل
        result = await self.orchestrator.ainvoke(project_state)
        
        return result

class MassiveToolIntegration:
    """
    تكامل الأدوات الضخمة - تتعلم من GitHub, npm, PyPI, Docker, OSS
    """
    
    async def find_best_tools(self, project_type: str) -> List:
        """البحث عن أفضل الأدوات للمشروع"""
        
        tools = []
        
        # البحث في GitHub
        github_tools = await self._search_github(project_type)
        tools.extend(github_tools)
        
        # البحث في npm (للجافاسكريبت)
        npm_tools = await self._search_npm(project_type)
        tools.extend(npm_tools)
        
        # البحث في PyPI (للبايثون)
        pypi_tools = await self._search_pypi(project_type)
        tools.extend(pypi_tools)
        
        # البحث في Docker Hub
        docker_tools = await self._search_docker(project_type)
        tools.extend(docker_tools)
        
        # ترتيب حسب الشعبية والجودة
        ranked = sorted(tools, key=lambda x: x['score'], reverse=True)
        
        return ranked[:10]

class ContinuousSoftwareLaboratory:
    """
    معمل البرمجيات المستمر - بيئة تجريب مستمرة تبني، تختبر، وتحسن
    """
    
    async def experiment(self, code: str) -> Dict:
        """إجراء تجارب على الكود وتحسينه"""
        
        # بناء
        build_result = await self._build(code)
        
        # اختبار
        test_result = await self._test(code)
        
        # قياس الأداء
        benchmark_result = await self._benchmark(code)
        
        # تحسين
        improved_code = await self._optimize(code, test_result, benchmark_result)
        
        # مقارنة
        comparison = await self._compare(code, improved_code)
        
        return {
            'original_code': code,
            'improved_code': improved_code,
            'build_result': build_result,
            'test_result': test_result,
            'benchmark_result': benchmark_result,
            'improvement': comparison
        }

class EventType(Enum):

    METRIC_UPDATED = "evaluation.metric_updated"

    EXECUTION_STARTED = "execution.started"

@dataclass
class Event:
    """حدث قابل للتخزين"""
    id: str

    payload: Dict
    source: str

    user_id: Optional[str] = None

class EventStore:
    """
    Event Store - تخزين الأحداث للتحليل والاسترجاع
    """
    
    def __init__(self, storage_path: str = "./data/events"):
        self.storage_path = storage_path
        self.events: List[Event] = []
        self.max_events_in_memory = 10000
        
    async def store(self, event: Event) -> str:
        """تخزين حدث"""
        self.events.append(event)
        
        # تنظيف الذاكرة
        if len(self.events) > self.max_events_in_memory:
            self.events = self.events[-self.max_events_in_memory:]
        
        # في الإنتاج: تخزين في ملف أو قاعدة بيانات
        return event.id
    
    async def get_events(self, event_type: Optional[EventType] = None,
                          user_id: Optional[str] = None,
                          from_time: Optional[float] = None,
                          to_time: Optional[float] = None,
                          limit: int = 100) -> List[Event]:
        """استرجاع الأحداث حسب المعايير"""
        result = self.events
        
        if event_type:
            result = [e for e in result if e.type == event_type]
        
        if user_id:
            result = [e for e in result if e.user_id == user_id]
        
        if from_time:
            result = [e for e in result if e.timestamp >= from_time]
        
        if to_time:
            result = [e for e in result if e.timestamp <= to_time]
        
        return result[-limit:]
    
    async def get_metrics(self, user_id: str, days: int = 30) -> Dict:
        """الحصول على مقاييس المستخدم من الأحداث"""
        from_time = time.time() - (days * 86400)
        
        events = await self.get_events(
            user_id=user_id,
            from_time=from_time,
            limit=10000
        )
        
        metrics = {
            'total_events': len(events),
            'evaluations': len([e for e in events if e.type == EventType.EVALUATION_COMPLETED]),
            'executions': len([e for e in events if e.type == EventType.EXECUTION_COMPLETED]),
            'success_rate': 0.0,
            'avg_score': 0.0
        }
        
        # حساب معدل النجاح
        executions = [e for e in events if e.type == EventType.EXECUTION_COMPLETED]
        failed = [e for e in events if e.type == EventType.EXECUTION_FAILED]
        
        if executions or failed:
            metrics['success_rate'] = len(executions) / (len(executions) + len(failed))
        
        # حساب متوسط الدرجات
        scores = []
        for e in events:
            if e.type == EventType.SCORE_CHANGED:
                scores.append(e.payload.get('score', 0))
        
        if scores:
            metrics['avg_score'] = sum(scores) / len(scores)
        
        return metrics


class EventBus:
    """
    Event Bus - ناقل الأحداث المركزي
    """
    
    def __init__(self, use_kafka: bool = False):
        self.use_kafka = use_kafka
        self.subscribers: Dict[EventType, List[Callable]] = {}
        self.event_store = EventStore()
        self._event_queue: asyncio.Queue = asyncio.Queue()
        self._is_running = False
        self._processor_task = None
        
        self.stats = {
            'events_emitted': 0,
            'events_processed': 0,
            'subscribers_count': 0
        }
    
    def subscribe(self, event_type: EventType):
        """ديكورator للتسجيل في الأحداث"""
        def decorator(handler: Callable):
            if event_type not in self.subscribers:
                self.subscribers[event_type] = []
            self.subscribers[event_type].append(handler)
            self.stats['subscribers_count'] += 1
            return handler
        return decorator
    
    async def emit(self, event_type: EventType, payload: Dict,
                   source: str = "system", user_id: str = None,
                   session_id: str = None) -> Event:
        """نشر حدث"""
        event = Event(
            id=f"{event_type.value}_{int(time.time()*1000)}_{self.stats['events_emitted']}",
            type=event_type,
            payload=payload,
            source=source,
            timestamp=time.time(),
            user_id=user_id,
            session_id=session_id
        )
        
        # تخزين الحدث
        await self.event_store.store(event)
        
        # إضافة إلى قائمة الانتظار
        await self._event_queue.put(event)
        
        self.stats['events_emitted'] += 1
        
        return event
    
    async def _process_loop(self):
        """حلقة معالجة الأحداث"""
        while self._is_running:
            try:
                event = await self._event_queue.get()
                
                if event.type in self.subscribers:
                    for handler in self.subscribers[event.type]:
                        try:
                            if asyncio.iscoroutinefunction(handler):
                                await handler(event)
                            else:
                                handler(event)
                            self.stats['events_processed'] += 1
                        except Exception as e:
                            logger.error(f"Handler error for {event.type.value}: {e}")
                
            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.error(f"Event processing error: {e}")
    
    async def start(self):
        """بدء ناقل الأحداث"""
        self._is_running = True
        self._processor_task = asyncio.create_task(self._process_loop())
        logger.info("EventBus started")
    
    async def stop(self):
        """إيقاف ناقل الأحداث"""
        self._is_running = False
        if self._processor_task:
            self._processor_task.cancel()
        logger.info("EventBus stopped")
    
    async def get_events(self, **kwargs) -> List[Event]:
        """الحصول على الأحداث المخزنة"""
        return await self.event_store.get_events(**kwargs)
    
    async def get_user_metrics(self, user_id: str, days: int = 30) -> Dict:
        """الحصول على مقاييس المستخدم"""
        return await self.event_store.get_metrics(user_id, days)

@dataclass
class MemoryEntry:
    """عنصر ذاكرة للوكيل"""
    id: str
    agent_id: str
    content: str
    embedding: List[float]
    metadata: Dict
    timestamp: float
    importance: float
    access_count: int = 0


class VectorMemory:
    """
    Vector Memory - تخزين متجهات للبحث الدلالي
    """
    
    def __init__(self, dimension: int = 768):
        self.dimension = dimension
        self.entries: Dict[str, MemoryEntry] = {}
        
    async def store(self, entry: MemoryEntry) -> str:
        """تخزين متجه"""
        self.entries[entry.id] = entry
        return entry.id
    
    async def search(self, query_embedding: List[float], top_k: int = 10) -> List[MemoryEntry]:
        """بحث عن أقرب الجيران"""
        if not self.entries:
            return []
        
        similarities = []
        for entry in self.entries.values():
            sim = self._cosine_similarity(query_embedding, entry.embedding)
            similarities.append((entry, sim))
        
        similarities.sort(key=lambda x: x[1], reverse=True)
        return [entry for entry, _ in similarities[:top_k]]
    
    def _cosine_similarity(self, a: List[float], b: List[float]) -> float:
        """حساب التشابه جيب التمام"""
        dot = sum(x * y for x, y in zip(a, b))
        norm_a = sum(x * x for x in a) ** 0.5
        norm_b = sum(x * x for x in b) ** 0.5
        
        if norm_a == 0 or norm_b == 0:
            return 0.0
        
        return dot / (norm_a * norm_b)


class GraphMemory:
    """
    Graph Memory - تخزين العلاقات بين الذكريات
    """
    
    def __init__(self):
        self.nodes: Dict[str, Dict] = {}
        self.edges: List[Tuple[str, str, str, float]] = []  # from, to, relation, weight
        
    async def add_node(self, node_id: str, properties: Dict) -> str:
        """إضافة عقدة"""
        self.nodes[node_id] = properties
        return node_id
    
    async def add_edge(self, from_id: str, to_id: str, relation: str, weight: float = 1.0):
        """إضافة علاقة"""
        self.edges.append((from_id, to_id, relation, weight))
    
    async def get_related(self, node_id: str, relation: Optional[str] = None) -> List[Tuple[str, Dict, float]]:
        """الحصول على العقد المرتبطة"""
        results = []
        for from_id, to_id, rel, weight in self.edges:
            if from_id == node_id and (relation is None or rel == relation):
                if to_id in self.nodes:
                    results.append((to_id, self.nodes[to_id], weight))
        
        return sorted(results, key=lambda x: x[2], reverse=True)


class CacheMemory:
    """
    Cache Memory - تخزين مؤقت سريع
    """
    
    def __init__(self, ttl_seconds: int = 3600):
        self.cache: Dict[str, Dict] = {}
        self.ttl_seconds = ttl_seconds
        
    async def get(self, key: str) -> Optional[Any]:
        """الحصول من الكاش"""
        if key in self.cache:
            entry = self.cache[key]
            if time.time() - entry['timestamp'] < self.ttl_seconds:
                return entry['value']
            else:
                del self.cache[key]
        return None
    
    async def set(self, key: str, value: Any) -> str:
        """تخزين في الكاش"""
        self.cache[key] = {
            'value': value,
            'timestamp': time.time()
        }
        return key
    
    async def delete(self, key: str) -> bool:
        """حذف من الكاش"""
        if key in self.cache:
            del self.cache[key]
            return True
        return False

        }

        """تحديث الإحصائيات العالمية"""
        if not all_users_metrics:

        performances = [m.performance_over_time for m in all_users_metrics]

        successes = [m.success_rate for m in all_users_metrics]
        adaptations = [m.adaptation_rate for m in all_users_metrics]
        failures = [m.failure_rate for m in all_users_metrics]
        
        self.global_stats = {
            'performance_mean': np.mean(performances),
            'performance_std': max(0.1, np.std(performances)),
            'complexity_mean': np.mean(complexities),
            'complexity_std': max(0.1, np.std(complexities)),

            'success_std': max(0.1, np.std(successes)),
            'adaptation_mean': np.mean(adaptations),
            'adaptation_std': max(0.1, np.std(adaptations)),
            'failure_mean': np.mean(failures),
            'failure_std': max(0.1, np.std(failures))
        }
    
    def normalize(self, metrics: CapabilityMetrics) -> CapabilityMetrics:
        """تطبيع المقاييس"""
        def z_score(value, mean, std):
            return (value - mean) / std
        
        normalized = CapabilityMetrics(
            performance_over_time=max(0.0, min(1.0, 0.5 + 0.2 * z_score(
                metrics.performance_over_time,
                self.global_stats['performance_mean'],
                self.global_stats['performance_std']
            ))),
            complexity_handled=max(0.0, min(1.0, 0.5 + 0.2 * z_score(
                metrics.complexity_handled,
                self.global_stats['complexity_mean'],
                self.global_stats['complexity_std']
            ))),
            success_rate=max(0.0, min(1.0, 0.5 + 0.2 * z_score(
                metrics.success_rate,
                self.global_stats['success_mean'],
                self.global_stats['success_std']
            ))),
            adaptation_rate=max(0.0, min(1.0, 0.5 + 0.2 * z_score(
                metrics.adaptation_rate,
                self.global_stats['adaptation_mean'],
                self.global_stats['adaptation_std']
            ))),
            failure_rate=max(0.01, min(0.99, 0.2 + 0.2 * z_score(
                metrics.failure_rate,
                self.global_stats['failure_mean'],
                self.global_stats['failure_std']
            )))
        )
        
        return normalized

class ExecutionResult:

    success: bool
    output: str

class DockerSandbox:
    """
    Docker Sandbox - تنفيذ آمن في حاويات Docker
    """
    
    def __init__(self):
        self.docker_available = False
        self._check_docker()
    
    def _check_docker(self):
        """التحقق من توفر Docker"""
        try:
            import docker
            self.docker_client = docker.from_env()
            self.docker_client.ping()
            self.docker_available = True
            logger.info("Docker is available")
        except ImportError:
            logger.warning("Docker not available, using subprocess fallback")
            self.docker_available = False
        except Exception as e:
            logger.warning(f"Docker error: {e}")
            self.docker_available = False
    
    async def execute(self, code: str, language: str = "python",
                      timeout_sec: int = 30,
                      memory_limit_mb: int = 512,
                      cpu_limit: float = 1.0) -> ExecutionResult:
        """
        تنفيذ كود في بيئة معزولة
        """
        start_time = time.time()
        
        if self.docker_available:
            return await self._execute_docker(code, language, timeout_sec, memory_limit_mb, cpu_limit)
        else:
            return await self._execute_subprocess(code, language, timeout_sec)
    
    async def _execute_docker(self, code: str, language: str,
                               timeout_sec: int, memory_limit_mb: int,
                               cpu_limit: float) -> ExecutionResult:
        """تنفيذ في Docker container"""
        try:
            import docker
            
            # إنشاء ملف مؤقت للكود
            with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
                f.write(code)
                temp_file = f.name
            
            # تشغيل الحاوية
            container = self.docker_client.containers.run(
                image="python:3.11-slim",
                command=f"python {os.path.basename(temp_file)}",
                volumes={os.path.dirname(temp_file): {'bind': '/code', 'mode': 'ro'}},
                working_dir='/code',
                mem_limit=f"{memory_limit_mb}m",
                nano_cpus=int(cpu_limit * 1e9),
                network_disabled=True,
                read_only=True,
                detach=True,
                remove=False
            )
            
            # انتظار النتيجة مع timeout
            try:
                result = container.wait(timeout=timeout_sec)
                logs = container.logs().decode()
                
                # جمع إحصائيات الموارد
                stats = container.stats(stream=False)
                
                execution_time_ms = (time.time() - start_time) * 1000
                
                return ExecutionResult(
                    success=result['StatusCode'] == 0,
                    output=logs,
                    error="" if result['StatusCode'] == 0 else logs,
                    execution_time_ms=execution_time_ms,
                    exit_code=result['StatusCode'],
                    resource_usage={
                        'cpu_usage': stats.get('cpu_stats', {}).get('cpu_usage', {}),
                        'memory_usage': stats.get('memory_stats', {}).get('usage', 0)
                    }
                )
                
            except docker.errors.APIError as e:
                return ExecutionResult(
                    success=False,
                    output="",
                    error=str(e),
                    execution_time_ms=(time.time() - start_time) * 1000,
                    exit_code=-1,
                    resource_usage={}
                )
            finally:
                # تنظيف
                container.remove(force=True)
                os.unlink(temp_file)
                
        except Exception as e:
            return ExecutionResult(
                success=False,
                output="",
                error=str(e),
                execution_time_ms=(time.time() - start_time) * 1000,
                exit_code=-1,
                resource_usage={}
            )
    
    async def _execute_subprocess(self, code: str, language: str,
                                   timeout_sec: int) -> ExecutionResult:
        """تنفيذ في subprocess (fallback أقل أماناً)"""
        import subprocess
        
        start_time = time.time()
        
        with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
            f.write(code)
            temp_file = f.name
        
        try:
            result = await asyncio.wait_for(
                asyncio.to_thread(
                    subprocess.run,
                    ["python", temp_file],
                    capture_output=True,
                    timeout=timeout_sec
                ),
                timeout=timeout_sec + 5
            )
            
            execution_time_ms = (time.time() - start_time) * 1000
            
            return ExecutionResult(
                success=result.returncode == 0,
                output=result.stdout.decode(),
                error=result.stderr.decode(),
                execution_time_ms=execution_time_ms,
                exit_code=result.returncode,
                resource_usage={}
            )
            
        except asyncio.TimeoutError:
            return ExecutionResult(
                success=False,
                output="",
                error=f"Timeout after {timeout_sec}s",
                execution_time_ms=(time.time() - start_time) * 1000,
                exit_code=-1,
                resource_usage={}
            )
        except Exception as e:
            return ExecutionResult(

                output="",
                error=str(e),
                execution_time_ms=(time.time() - start_time) * 1000,
                exit_code=-1,
                resource_usage={}
            )
        finally:
            if os.path.exists(temp_file):
                os.unlink(temp_file)

class UserBehaviorEncoder:

    User Behavior Encoder - يحول سلوك المستخدم إلى تضمين
    """

        self.dimension = dimension
        
    async def encode(self, behavior_data: Dict) -> List[float]:
        """توليد تضمين لسلوك المستخدم"""
        # في الإنتاج: نموذج تعلم عميق حقيقي

        preferences = behavior_data.get('preferences', {})

        while len(features) < self.dimension:
            features.append(0.0)
        
        # تسوية

class TwinDistanceCalculator:
    """
    Twin Distance Calculator - يحسب المسافة بين الإنسان والتوأم الرقمي
    """
    
    def __init__(self):
        self.history: deque = deque(maxlen=100)
        
    def cosine_similarity(self, embedding1: List[float], embedding2: List[float]) -> float:
        """حساب التشابه جيب التمام"""
        dot = sum(x * y for x, y in zip(embedding1, embedding2))
        norm1 = sum(x * x for x in embedding1) ** 0.5
        norm2 = sum(x * x for x in embedding2) ** 0.5
        
        if norm1 == 0 or norm2 == 0:
            return 0.0
        
        return dot / (norm1 * norm2)
    
    def calculate(self, user_embedding: List[float], twin_embedding: List[float]) -> float:
        """حساب المسافة (التشابه) بين الإنسان والتوأم"""
        similarity = self.cosine_similarity(user_embedding, twin_embedding)
        self.history.append(similarity)
        return similarity

        user_embedding = await self.encoder.encode(user_data.get('behavior', {}))
        twin_embedding = await self.encoder.encode(twin_data.get('behavior', {}))
        
        self.user_embeddings.append(user_embedding)
        self.twin_embeddings.append(twin_embedding)
        
        # حساب التشابه الحالي
        similarity = self.distance_calculator.calculate(user_embedding, twin_embedding)
        
        # حساب تحسن الإنسان بفضل التوأم
        human_improvement = self._calculate_human_improvement(user_data, twin_data)
        
        # حساب تحسن التوأم بفضل الإنسان
        twin_improvement = self._calculate_twin_improvement(twin_data, user_data)
        
        # اتجاه الاختلاف
        divergence_trend = self._calculate_divergence_trend()
        
        # درجة التطور المشترك
        co_evolution = (human_improvement + twin_improvement) / 2
        
        # حساب درجة التكامل الكلية
        overall_sync = (
            similarity * 0.3 +
            co_evolution * 0.3 +
            human_improvement * 0.2 +
            twin_improvement * 0.2
        )
        
        # تحديث الإحصائيات
        self.stats['sync_updates'] += 1
        self.stats['avg_similarity'] = self.stats['avg_similarity'] * 0.95 + similarity * 0.05
        self.stats['co_evolution_strength'] = self.stats['co_evolution_strength'] * 0.95 + co_evolution * 0.05
        
        # تسجيل الأداء
        if 'performance' in user_data:
            self.performance_history['user'].append(user_data['performance'])
        if 'performance' in twin_data:
            self.performance_history['twin'].append(twin_data['performance'])
        
        return SyncMetrics(
            similarity_score=similarity,
            co_evolution_score=co_evolution,
            human_improvement=human_improvement,
            twin_improvement=twin_improvement,
            divergence_trend=divergence_trend,
            overall_sync=overall_sync
        )
    
    def _calculate_human_improvement(self, user_data: Dict, twin_data: Dict) -> float:
        """حساب تحسن الإنسان بفضل التوأم"""
        # مقارنة أداء الإنسان قبل وبعد التفاعل مع التوأم
        user_performance = user_data.get('performance', 0.5)
        baseline = user_data.get('baseline_performance', 0.3)
        
        if user_performance > baseline:
            return min(1.0, (user_performance - baseline) / 0.5)
        return 0.0
    
    def _calculate_twin_improvement(self, twin_data: Dict, user_data: Dict) -> float:
        """حساب تحسن التوأم بفضل الإنسان"""
        twin_performance = twin_data.get('performance', 0.5)
        baseline = twin_data.get('baseline_performance', 0.3)
        
        if twin_performance > baseline:
            return min(1.0, (twin_performance - baseline) / 0.5)
        return 0.0
    
    def _calculate_divergence_trend(self) -> float:
        """حساب اتجاه الاختلاف بين الإنسان والتوأم"""
        if len(self.user_embeddings) < 5 or len(self.twin_embeddings) < 5:
            return 0.0
        
        recent_user = self.user_embeddings[-1]
        recent_twin = self.twin_embeddings[-1]
        older_user = self.user_embeddings[-5]
        older_twin = self.twin_embeddings[-5]
        
        recent_sim = self.distance_calculator.cosine_similarity(recent_user, recent_twin)
        older_sim = self.distance_calculator.cosine_similarity(older_user, older_twin)
        
        # إذا كان التشابه يتناقص، الاختلاف في ازدياد
        if older_sim > recent_sim:
            return (older_sim - recent_sim) / older_sim
        return 0.0
    
    async def get_report(self) -> Dict:
        """تقرير التكامل"""
        if self.stats['sync_updates'] == 0:
            return {'error': 'No sync data available'}
        
        return {
            'user_id': self.user_id,
            'overall_sync': self.stats['co_evolution_strength'],
            'similarity_score': self.stats['avg_similarity'],
            'co_evolution_strength': self.stats['co_evolution_strength'],
            'human_performance_trend': self._get_performance_trend('user'),
            'twin_performance_trend': self._get_performance_trend('twin'),
            'recommendation': self._generate_recommendation()
        }
    
    def _get_performance_trend(self, key: str) -> str:
        """الحصول على اتجاه أداء"""
        history = self.performance_history.get(key, [])
        if len(history) < 5:
            return 'stable'
        
        recent = np.mean(history[-5:])
        older = np.mean(history[-10:-5]) if len(history) >= 10 else recent
        
        if recent > older + 0.05:
            return 'improving'
        elif recent < older - 0.05:
            return 'declining'
        return 'stable'
    
    def _generate_recommendation(self) -> str:
        """توليد توصية للتحسين"""
        if self.stats['co_evolution_strength'] > 0.8:
            return "Excellent human-twin synchronization. Challenge both with harder problems."
        elif self.stats['co_evolution_strength'] > 0.6:
            return "Good synchronization. Focus on areas where the twin diverges from your style."
        elif self.stats['co_evolution_strength'] > 0.4:
            return "Moderate synchronization. Increase interaction with your digital twin."
        else:
            return "Low synchronization. Your digital twin needs more data about your behavior."

class AutonomousProjectEngine:
    """
    Autonomous Project Engine - المحرك التنفيذي المستقل
    """
    
    def __init__(self, event_bus: EventBus, memory: AgentMemoryLayer, sandbox: ExecutionSandbox):
        self.event_bus = event_bus
        self.memory = memory
        self.sandbox = sandbox
        self.is_running = False
        
    async def execute_project(self, requirements: str) -> Dict:
        """تنفيذ مشروع كامل"""
        project_id = f"project_{int(time.time())}"
        
        await self.event_bus.emit(EventType.EXECUTION_STARTED, {
            'project_id': project_id,
            'requirements': requirements
        }, source="project_engine")
        
        try:
            # محاكاة مراحل المشروع
            stages = ['planning', 'architecture', 'backend', 'frontend', 'testing', 'deployment']
            results = {}
            
            for stage in stages:
                await self.event_bus.emit(EventType.STEP_COMPLETED, {
                    'project_id': project_id,
                    'stage': stage,
                    'status': 'in_progress'
                }, source="project_engine")
                
                await asyncio.sleep(0.5)  # محاكاة العمل
                
                results[stage] = {'status': 'completed', 'duration_ms': 500}
            
            await self.event_bus.emit(EventType.EXECUTION_COMPLETED, {
                'project_id': project_id,
                'results': results
            }, source="project_engine")
            
            return {
                'project_id': project_id,
                'success': True,
                'results': results
            }
            
        except Exception as e:
            await self.event_bus.emit(EventType.EXECUTION_FAILED, {
                'project_id': project_id,
                'error': str(e)
            }, source="project_engine")
            
            return {
                'project_id': project_id,
                'success': False,
                'error': str(e)
            }

        await self.event_bus.start()
        
        # تسجيل معالجات الأحداث
        @self.event_bus.subscribe(EventType.EXECUTION_COMPLETED)
        async def on_execution_completed(event):

        @self.event_bus.subscribe(EventType.EXECUTION_FAILED)
        async def on_execution_failed(event):
            logger.error(f"Execution failed: {event.payload.get('error')}")
        
        @self.event_bus.subscribe(EventType.SCORE_CHANGED)
