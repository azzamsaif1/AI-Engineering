# backend/realtime/streaming_asr.py
import asyncio
import numpy as np
from faster_whisper import WhisperModel
from collections import deque
import re

class StreamingASR:
    def __init__(self, model_size="small", device="cuda"):
        self.model = WhisperModel(model_size, device=device, compute_type="float16")
        self.audio_buffer = deque(maxlen=44100 * 30)  # 30 seconds buffer
        self.sample_rate = 16000
        self.previous_transcript = ""
        self.confirmed_text = ""
        
    async def process_audio_chunk(self, audio_chunk: np.ndarray):
        """معالجة مقطع صوتي جديد"""
        # إضافة إلى المخزن المؤقت
        self.audio_buffer.extend(audio_chunk)
        
        # تحويل إلى numpy array
        audio_array = np.array(list(self.audio_buffer), dtype=np.float32)
        
        # تشغيل Whisper
        segments, info = await asyncio.to_thread(
            self.model.transcribe, 
            audio_array, 
            beam_size=5,
            language="de"
        )
        
        # تجميع النص
        full_transcript = " ".join([seg.text for seg in segments])
        
        # Local Agreement Algorithm - الحصول على النص المConfirmed
        common_prefix = self._longest_common_prefix(self.previous_transcript, full_transcript)
        
        if common_prefix != self.confirmed_text:
            # نص جديد مؤكد
            new_text = common_prefix[len(self.confirmed_text):]
            if new_text:
                yield {"type": "final", "text": new_text}
            self.confirmed_text = common_prefix
        
        # عرض النص الجزئي (غير المؤكد)
        if len(full_transcript) > len(self.confirmed_text):
            partial = full_transcript[len(self.confirmed_text):]
            if partial:
                yield {"type": "partial", "text": partial}
        
        self.previous_transcript = full_transcript
    
    def _longest_common_prefix(self, str1: str, str2: str) -> str:
        """العثور على أطول بادئة مشتركة"""
        min_len = min(len(str1), len(str2))
        for i in range(min_len):
            if str1[i] != str2[i]:
                return str1[:i]
        return str1[:min_len]

# backend/realtime/fluss_client.py
import asyncio
from fluss import FlussClient
from fluss.schema import Schema, Field

class FlussStreamingStorage:
    def __init__(self, host="localhost", port=8080):
        self.client = FlussClient(f"http://{host}:{port}")
        self.schema = Schema([
            Field("timestamp", "long"),
            Field("user_id", "string"),
            Field("session_id", "string"),
            Field("type", "string"),  # audio, text, code, screen
            Field("data", "json"),
            Field("embedding", "vector")
        ])
        
    async def write_event(self, user_id: str, session_id: str, event_type: str, data: dict):
        """كتابة حدث إلى دفق البيانات"""
        event = {
            "timestamp": int(time.time() * 1000),
            "user_id": user_id,
            "session_id": session_id,
            "type": event_type,
            "data": data
        }
        
        # تخزين في Fluss
        await self.client.write("user_events", event)
        
        # إرجاع الإزاحة للتتبع
        return event
    
    async def get_user_stream(self, user_id: str, session_id: str, from_time: int = None):
        """الحصول على دفق المستخدم"""
        query = f"""
        SELECT * FROM user_events 
        WHERE user_id = '{user_id}' 
        AND session_id = '{session_id}'
        """
        if from_time:
            query += f" AND timestamp > {from_time}"
        query += " ORDER BY timestamp ASC"
        
        return await self.client.query(query)
    
    async def get_cognitive_profile(self, user_id: str):
        """الحصول على البصمة المعرفية للمستخدم"""
        query = f"""
        SELECT data FROM user_events 
        WHERE user_id = '{user_id}' 
        AND type IN ('code_pattern', 'decision', 'error')
        ORDER BY timestamp DESC
        LIMIT 1000
        """
        return await self.client.query(query)

import cv2
import mediapipe as mp
import numpy as np
from PIL import Image
import torch
from transformers import AutoModel, AutoProcessor

class ScreenBehaviorAnalyzer:
    def __init__(self):
        # نموذج فهم الصور
        self.processor = AutoProcessor.from_pretrained("Qwen/Qwen-VL-Chat")
        self.model = AutoModel.from_pretrained("Qwen/Qwen-VL-Chat", device_map="auto")
        
        # تتبع العين والوجه
        self.face_mesh = mp.solutions.face_mesh.FaceMesh()
        self.mp_drawing = mp.solutions.drawing_utils

        self.last_mouse_time = time.time()
        
    def analyze_screen(self, screenshot: np.ndarray, mouse_x: int, mouse_y: int):
        """تحليل محتوى الشاشة وحركة المستخدم"""
        # تتبع حركة المؤشر
        current_time = time.time()
        self.mouse_positions.append({
            'x': mouse_x, 
            'y': mouse_y, 
            'time': current_time,
            'speed': self._calculate_speed(mouse_x, mouse_y)
        })
        
        # تحليل منطقة التركيز
        focus_area = self._get_focus_area(mouse_x, mouse_y)
        focus_content = self._extract_area_content(screenshot, focus_area)
        
        # تحليل ما يقرأه المستخدم
        reading_pattern = self._analyze_reading_pattern()
        
        return {
            'focus_area': focus_area,
            'focus_content': focus_content,
            'reading_pattern': reading_pattern,
            'mouse_speed': self._calculate_avg_speed(),
            'attention_score': self._calculate_attention_score()
        }
    
    def _calculate_attention_score(self) -> float:
        """حساب درجة التركيز بناءً على حركة المؤشر"""
        if len(self.mouse_positions) < 10:
            return 0.5
        
        # حساب التغير في السرعة
        speeds = [p['speed'] for p in self.mouse_positions[-50:]]
        speed_variance = np.var(speeds) if len(speeds) > 1 else 0
        
        # التركيز العالي = حركة مستقرة
        attention = 1.0 - min(1.0, speed_variance / 1000)
        
        # إضافة عامل الوقت
        time_since_last = time.time() - self.last_mouse_time
        if time_since_last > 30:
            attention *= 0.5
        
        return max(0.0, min(1.0, attention))

# backend/cognitive/integrated_model.py
import torch
import torch.nn as nn
from transformers import AutoModel

class IntegratedCognitiveModel(nn.Module):
    def __init__(self, input_dim=768, hidden_dim=512, output_dim=256):
        super().__init__()
        
        # مشفرات للمدخلات المختلفة
        self.audio_encoder = nn.LSTM(input_dim, hidden_dim, batch_first=True)
        self.vision_encoder = nn.LSTM(input_dim, hidden_dim, batch_first=True)
        self.code_encoder = nn.LSTM(input_dim, hidden_dim, batch_first=True)
        self.behavior_encoder = nn.LSTM(input_dim, hidden_dim, batch_first=True)
        
        # آلية الانتباه المتعدد الوسائط
        self.cross_attention = nn.MultiheadAttention(hidden_dim, num_heads=8)
        
        # المصنف النهائي
        self.classifier = nn.Sequential(
            nn.Linear(hidden_dim * 4, hidden_dim),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(hidden_dim, output_dim)
        )
        
    def forward(self, audio_features, vision_features, code_features, behavior_features):
        # تشفير كل وسيط على حدة
        audio_out, _ = self.audio_encoder(audio_features)
        vision_out, _ = self.vision_encoder(vision_features)
        code_out, _ = self.code_encoder(code_features)
        behavior_out, _ = self.behavior_encoder(behavior_features)
        
        # دمج المتجهات
        combined = torch.cat([
            audio_out[:, -1, :],  # آخر مخرجات LSTM
            vision_out[:, -1, :],
            code_out[:, -1, :],
            behavior_out[:, -1, :]
        ], dim=1)
        
        # تصنيف
        cognitive_profile = self.classifier(combined)
        
        return cognitive_profile

class RealtimeCognitiveProfile:
    def __init__(self, user_id: str):
        self.user_id = user_id
        self.model = IntegratedCognitiveModel()
        self.model.eval()
        
        # مخازن مؤقتة
        self.audio_buffer = []
        self.vision_buffer = []
        self.code_buffer = []
        self.behavior_buffer = []
        
        # البصمة المعرفية الحالية
        self.current_profile = None
        self.last_update = time.time()
        
    def update(self, audio_data, vision_data, code_data, behavior_data):
        """تحديث البصمة المعرفية ببيانات جديدة"""
        # إضافة إلى المخازن
        self.audio_buffer.append(audio_data)
        self.vision_buffer.append(vision_data)
        self.code_buffer.append(code_data)
        self.behavior_buffer.append(behavior_data)
        
        # تحديث كل 5 ثوانٍ
        if time.time() - self.last_update > 5:
            self._compute_profile()
            self.last_update = time.time()
    
    def _compute_profile(self):
        """حساب البصمة المعرفية الحالية"""
        with torch.no_grad():
            # تحويل المخازن إلى tensors
            audio_tensor = torch.tensor(self.audio_buffer[-50:]).unsqueeze(0)
            vision_tensor = torch.tensor(self.vision_buffer[-50:]).unsqueeze(0)
            code_tensor = torch.tensor(self.code_buffer[-50:]).unsqueeze(0)
            behavior_tensor = torch.tensor(self.behavior_buffer[-50:]).unsqueeze(0)
            
            # حساب البصمة
            self.current_profile = self.model(
                audio_tensor, vision_tensor, code_tensor, behavior_tensor
            )
    
    def get_cognitive_state(self) -> dict:
        """الحصول على الحالة المعرفية الحالية للمستخدم"""
        return {
            'profile': self.current_profile.tolist() if self.current_profile is not None else None,
            'confidence': self._calculate_confidence(),
            'attention_level': self._estimate_attention(),
            'cognitive_load': self._estimate_cognitive_load(),
            'recommendation': self._generate_recommendation()
        }

import torch

class CognitiveDigitalTwin:
    def __init__(self, user_id: str, cognitive_profile):
        self.user_id = user_id
        self.cognitive_profile = cognitive_profile
        
        # تحميل نموذج أساسي
        self.model_name = "deepseek-ai/deepseek-coder-6.7b-instruct"
        self.tokenizer = AutoTokenizer.from_pretrained(self.model_name, trust_remote_code=True)

        )

        self.memory = []
        self.prediction_accuracy = 0.7
        
    def predict_next_action(self, current_state: dict) -> dict:
        """توقع الإجراء التالي للمستخدم"""
        # إضافة البصمة المعرفية للسياق
        prompt = f"""
        User cognitive profile: {self.cognitive_profile}
        Current state: {current_state}
        Based on the user's previous behavior, predict the next action:
        """
        
        inputs = self.tokenizer(prompt, return_tensors="pt").to(self.model.device)
        outputs = self.model.generate(**inputs, max_new_tokens=128)
        prediction = self.tokenizer.decode(outputs[0], skip_special_tokens=True)
        
        return {
            'predicted_action': prediction,
            'confidence': self.prediction_accuracy,
            'timestamp': time.time()
        }
    
    def learn_from_actual(self, predicted: dict, actual: dict):
        """التعلم من الفرق بين التوقع والفعل الفعلي"""
        if predicted.get('predicted_action') == actual.get('action'):
            self.prediction_accuracy = min(0.95, self.prediction_accuracy + 0.01)
        else:
            self.prediction_accuracy = max(0.5, self.prediction_accuracy - 0.02)
        
        # تخزين في الذاكرة
        self.memory.append({
            'predicted': predicted,
            'actual': actual,
            'accuracy': self.prediction_accuracy,
            'timestamp': time.time()
        })
        
        # تحديث البصمة المعرفية
        self._update_cognitive_profile(predicted, actual)
    
    def generate_challenge(self, weak_area: str = None) -> dict:
        """توليد تحدي مخصص بناءً على نقاط ضعف المستخدم"""
        if not weak_area:
            # تحليل نقاط الضعف من البصمة المعرفية
            weak_area = self._identify_weak_area()
        
        prompt = f"""
        Generate a programming challenge for a user with these characteristics:
        - Weak area: {weak_area}
        - Cognitive profile: {self.cognitive_profile}
        
        Challenge should be:
        1. Appropriate difficulty level
        2. Focus on {weak_area}
        3. Time-limited (30 minutes)
        4. Include test cases
        
        Return JSON format.
        """
        
        inputs = self.tokenizer(prompt, return_tensors="pt").to(self.model.device)
        outputs = self.model.generate(**inputs, max_new_tokens=512)
        challenge = self.tokenizer.decode(outputs[0], skip_special_tokens=True)
        
        return json.loads(challenge)
