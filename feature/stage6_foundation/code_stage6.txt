class TaskState(Enum):

    PLANNING = "planning"

@dataclass

    """هدف فرعي في خطة التنفيذ"""

    dependencies: List[str]

    execution_trace: List[Dict] = field(default_factory=list)

class HTNPlanner:
    """
    Hierarchical Task Network Planner - يقسم الأهداف الكبيرة إلى مهام فرعية
    """
    
    def __init__(self):
        self.methods = self._register_methods()
        
    def _register_methods(self) -> Dict:
        """تسجيل طرق decomposition للأهداف المختلفة"""
        return {
            "design_system": self._decompose_design_system,
            "implement_feature": self._decompose_implement_feature,
            "optimize_performance": self._decompose_optimize_performance,
            "security_audit": self._decompose_security_audit,
            "write_tests": self._decompose_write_tests
        }
    
    async def plan(self, goal: str, requirements: Dict) -> List[Subgoal]:
        """توليد خطة من الأهداف الفرعية"""
        goal_type = self._classify_goal(goal)
        
        if goal_type in self.methods:
            subgoals = await self.methods[goal_type](goal, requirements)
        else:
            subgoals = await self._generic_decompose(goal, requirements)
        
        # ترتيب حسب التبعيات
        subgoals = self._order_by_dependencies(subgoals)
        
        return subgoals

        """تحليل عام لأي هدف"""

        return [
            Subgoal(id=f"{base_id}_0", description="Understand the goal",
                    agent_type="research", dependencies=[], tools_needed=["analyzer"],
                    expected_output="understanding"),
            Subgoal(id=f"{base_id}_1", description="Plan approach",
                    agent_type="architecture", dependencies=["sg_0"], tools_needed=["planner"],
                    expected_output="plan"),
            Subgoal(id=f"{base_id}_2", description="Execute plan",
                    agent_type="backend", dependencies=["sg_1"], tools_needed=["executor"],
                    expected_output="result"),
            Subgoal(id=f"{base_id}_3", description="Verify result",
                    agent_type="security", dependencies=["sg_2"], tools_needed=["verifier"],
                    expected_output="verification")
        ]
    
    def _classify_goal(self, goal: str) -> str:
        """تصنيف الهدف"""
        goal_lower = goal.lower()
        if "design" in goal_lower or "architecture" in goal_lower:
            return "design_system"
        elif "implement" in goal_lower or "build" in goal_lower or "create" in goal_lower:
            return "implement_feature"
        elif "optimize" in goal_lower or "performance" in goal_lower:
            return "optimize_performance"
        elif "security" in goal_lower or "audit" in goal_lower:
            return "security_audit"
        elif "test" in goal_lower:
            return "write_tests"
        return "generic"
    
    def _order_by_dependencies(self, subgoals: List[Subgoal]) -> List[Subgoal]:
        """ترتيب حسب التبعيات"""
        ordered = []
        remaining = subgoals.copy()
        
        while remaining:
            available = [sg for sg in remaining if all(dep in [s.id for s in ordered] for dep in sg.dependencies)]
            if not available:
                available = [remaining[0]]
            ordered.extend(available)
            remaining = [sg for sg in remaining if sg not in available]
        
        return ordered


class MCTSPlanner:
    """
    Monte Carlo Tree Search - يستكشف أفضل مسار تنفيذ
    """
    
    def __init__(self, exploration_constant: float = 1.414):
        self.exploration_constant = exploration_constant
        self.simulation_results = []
        
    async def optimize_path(self, subgoals: List[Subgoal]) -> List[Subgoal]:
        """
        تحسين مسار التنفيذ باستخدام MCTS
        """
        if len(subgoals) < 2:
            return subgoals
        
        # محاكاة مسارات متعددة
        paths = []
        for _ in range(min(10, len(subgoals))):
            path = await self._simulate_path(subgoals)
            paths.append(path)
        
        # اختيار أفضل مسار
        best_path = max(paths, key=lambda x: x.get('score', 0))
        
        return best_path.get('subgoals', subgoals)
    
    async def _simulate_path(self, subgoals: List[Subgoal]) -> Dict:
        """محاكاة مسار واحد"""
        import random
        
        # محاكاة عشوائية
        shuffled = random.sample(subgoals, len(subgoals))
        
        # حساب درجة المسار (تعقيد أقل = أفضل)
        complexity = sum(len(sg.tools_needed) for sg in shuffled)
        score = 1.0 / (1.0 + complexity / 10)
        
        return {
            'subgoals': shuffled,
            'score': score
        }

class EventBus:
    """
    Event Bus - ناقل الأحداث للنظام الموزع
    """
    
    def __init__(self):
        self.topics: Dict[str, List] = {}
        self.handlers: Dict[str, List] = {}
        
    async def publish(self, topic: str, event: Dict):
        """نشر حدث إلى topic"""
        if topic not in self.topics:
            self.topics[topic] = []
        
        event['published_at'] = time.time()
        self.topics[topic].append(event)
        
        # استدعاء المعالجات
        if topic in self.handlers:
            for handler in self.handlers[topic]:
                asyncio.create_task(handler(event))
        
        logger.info(f"Event published: {topic} - {event.get('type', 'unknown')}")
    
    async def subscribe(self, topic: str, handler):
        """الاشتراك في topic"""
        if topic not in self.handlers:
            self.handlers[topic] = []
        self.handlers[topic].append(handler)
    
    async def get_events(self, topic: str, last_n: int = 100) -> List[Dict]:

class AgentRegistry:
    """
    سجل الوكلاء - يدير جميع الوكلاء المتاحين
    """
    
    def __init__(self):
        self.agents: Dict[str, Dict] = {}
        
    def register(self, agent_name: str, agent_type: str, capabilities: List[str], endpoint: str):
        """تسجيل وكيل جديد"""
        self.agents[agent_name] = {
            'type': agent_type,
            'capabilities': capabilities,
            'endpoint': endpoint,
            'status': 'active',
            'registered_at': time.time(),
            'tasks_completed': 0,
            'avg_latency_ms': 0.0

        """الحصول على وكيل مناسب لمهمة"""
        for name, info in self.agents.items():

                return name
        return None
    
    def update_status(self, agent_name: str, status: str):
        """تحديث حالة وكيل"""
        if agent_name in self.agents:
            self.agents[agent_name]['status'] = status

        if agent_name in self.agents:
            agent = self.agents[agent_name]
            agent['tasks_completed'] += 1
            agent['avg_latency_ms'] = agent['avg_latency_ms'] * 0.9 + latency_ms * 0.1

        """

            goal=goal,
            requirements=requirements,
            context=context or {}
        )

        self.stats['total_tasks'] += 1
        
        # بدء التنفيذ
        asyncio.create_task(self._execute_task(task_id))
        
        # نشر حدث
        await self.event_bus.publish('task.events', {

            'goal': goal,
            'timestamp': time.time()
        })
        
        return task_id
    
    async def _execute_task(self, task_id: str):
        """
        تنفيذ مهمة - العملية الأساسية للنظام
        """
        task = self.active_tasks.get(task_id)
        if not task:
            return
        
        start_time = time.time()
        
        try:
            # ========== PHASE 1: PLANNING ==========
            task.state = TaskState.PLANNING
            task.execution_trace.append({
                'phase': 'planning',
                'timestamp': time.time(),
                'goal': task.goal
            })
            
            # HTN Planning
            subgoals = await self.planner.plan(task.goal, task.requirements)
            
            # MCTS Optimization
            optimized_subgoals = await self.mcts.optimize_path(subgoals)
            task.subgoals = optimized_subgoals
            
            self.stats['avg_subgoals_per_task'] = (
                self.stats['avg_subgoals_per_task'] * 0.9 + len(task.subgoals) * 0.1
            )
            
            # ========== PHASE 2: DISPATCH ==========
            task.state = TaskState.DISPATCHED
            
            # نشر خطة التنفيذ
            await self.event_bus.publish('task.events', {
                'type': 'plan_ready',
                'task_id': task_id,
                'subgoals': [
                    {'id': sg.id, 'description': sg.description, 'agent_type': sg.agent_type}
                    for sg in task.subgoals
                ]
            })
            
            # ========== PHASE 3: EXECUTION ==========
            task.state = TaskState.EXECUTING
            
            # تنفيذ الأهداف الفرعية بالتوازي حيث أمكن
            results = await self._execute_subgoals_parallel(task)
            
            # ========== PHASE 4: VERIFICATION ==========
            task.state = TaskState.VERIFYING
            
            verification = await self._verify_results(task, results)
            
            if not verification.get('passed', False):
                # فشل التحقق - محاولة استرداد
                if self.config['retry_attempts'] > 0:
                    await self._retry_failed_subgoals(task, verification)
            
            # ========== PHASE 5: COMPLETE ==========
            task.state = TaskState.COMPLETED
            task.completed_at = time.time()
            task.result = {
                'success': True,
                'outputs': results,
                'verification': verification,
                'execution_time_ms': (time.time() - start_time) * 1000
            }
            
            self.stats['completed_tasks'] += 1
            self.stats['avg_execution_time_ms'] = (
                self.stats['avg_execution_time_ms'] * 0.95 + 
                task.result['execution_time_ms'] * 0.05
            )
            
            # ========== PHASE 6: COMPETITION (اختياري) ==========
            if self.config['enable_competition']:
                competition_result = await self._run_agent_competition(task)
                task.result['competition'] = competition_result
            
            # نشر حدث الإكمال
            await self.event_bus.publish('task.events', {
                'type': 'task_completed',
                'task_id': task_id,
                'duration_ms': task.result['execution_time_ms'],
                'timestamp': time.time()
            })
            
        except Exception as e:
            task.state = TaskState.FAILED
            task.completed_at = time.time()
            task.error = str(e)
            self.stats['failed_tasks'] += 1
            
            await self.event_bus.publish('task.events', {
                'type': 'task_failed',
                'task_id': task_id,
                'error': str(e),
                'timestamp': time.time()
            })
            
            logger.error(f"Task {task_id} failed: {e}")
        
        finally:
            # نقل إلى التاريخ
            self.task_history.append(task)
            if task_id in self.active_tasks:
                del self.active_tasks[task_id]
    
    async def _execute_subgoals_parallel(self, task: EngineeringTask) -> Dict:
        """
        تنفيذ الأهداف الفرعية بالتوازي
        """
        results = {}
        completed = set()
        
        while len(completed) < len(task.subgoals):
            # ابحث عن الأهداف الجاهزة للتنفيذ
            ready = []
            for sg in task.subgoals:
                if sg.id in completed:
                    continue
                if sg.state != TaskState.PENDING:
                    continue
                # تحقق من التبعيات
                deps_met = all(dep in completed for dep in sg.dependencies)
                if deps_met:
                    ready.append(sg)
            
            if not ready:
                # حلقة تبعية - اكسرها
                remaining = [sg for sg in task.subgoals if sg.id not in completed]
                if remaining:
                    ready = [remaining[0]]
            
            # تنفيذ الدفعة الحالية بالتوازي
            tasks = [self._execute_subgoal(sg, task) for sg in ready]
            batch_results = await asyncio.gather(*tasks, return_exceptions=True)
            
            for sg, result in zip(ready, batch_results):
                if isinstance(result, Exception):
                    results[sg.id] = {'error': str(result), 'success': False}
                    sg.state = TaskState.FAILED
                    sg.error = str(result)
                else:
                    results[sg.id] = result
                    sg.state = TaskState.COMPLETED
                    sg.result = result
                completed.add(sg.id)
        
        return results

    async def call_tool(self, tool_name: str, parameters: Dict, context: Dict) -> Dict:
        """
        استدعاء أداة عبر MCP
        """
        # التحقق من وجود الأداة
        if tool_name not in self.tools:
            return {
                'success': False,
                'error': f"Tool '{tool_name}' not found",
                'available_tools': list(self.tools.keys())
            }
        
        tool = self.tools[tool_name]
        
        # التحقق من معدل الاستدعاء
        if not self._check_rate_limit(tool_name, tool.rate_limit_per_minute):
            return {
                'success': False,
                'error': f"Rate limit exceeded for tool '{tool_name}'. Max {tool.rate_limit_per_minute} calls per minute"
            }
        
        call_id = str(uuid.uuid4())
        start_time = time.time()
        
        tool_call = ToolCall(
            id=call_id,
            tool_name=tool_name,
            parameters=parameters,
            context=context
        )
        
        self.tool_calls[call_id] = tool_call
        
        try:
            # تنفيذ الأداة مع timeout
            result = await asyncio.wait_for(
                tool.handler(parameters, context),
                timeout=tool.timeout_seconds
            )
            
            execution_time_ms = (time.time() - start_time) * 1000
            
            tool_call.status = "completed"
            tool_call.result = result
            tool_call.execution_time_ms = execution_time_ms
            
            # تسجيل في التاريخ
            self.call_history.append({
                'tool': tool_name,
                'success': True,
                'duration_ms': execution_time_ms,
                'timestamp': time.time()
            })
            
            # تحديث سجل المعدل
            self.rate_limits[tool_name].append(time.time())
            self._clean_rate_limits(tool_name)
            
            return {
                'success': True,
                'call_id': call_id,
                'result': result,
                'tool': tool_name,
                'execution_time_ms': execution_time_ms
            }
            
        except asyncio.TimeoutError:
            tool_call.status = "timeout"
            tool_call.error = f"Timeout after {tool.timeout_seconds}s"
            
            return {
                'success': False,
                'call_id': call_id,
                'error': tool_call.error,
                'tool': tool_name
            }
            
        except Exception as e:
            tool_call.status = "failed"
            tool_call.error = str(e)
            
            return {
                'success': False,
                'call_id': call_id,
                'error': str(e),
                'tool': tool_name
            }

        return len(self.rate_limits[tool_name]) < limit_per_minute

"""
Agent Sandbox V6 - عزل تنفيذ الكود مع:
- Firecracker MicroVMs
- Resource limits
- Network isolation
"""

import asyncio
import time
import uuid
from typing import Dict, Optional, Any
from dataclasses import dataclass, field
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


@dataclass
class Sandbox:
    """بيئة تنفيذ معزولة"""
    id: str
    tool_name: str
    created_at: float = field(default_factory=time.time)
    expires_at: float = field(default_factory=lambda: time.time() + 300)
    status: str = "running"
    resource_usage: Dict = field(default_factory=lambda: {
        'cpu_used_ms': 0,
        'memory_used_mb': 0,
        'network_bytes': 0
    })


class SandboxManagerV6:
    """
    Sandbox Manager - يدير بيئات التنفيذ المعزولة
    """
    
    def __init__(self):
        self.sandboxes: Dict[str, Sandbox] = {}
        
        # الحدود القصوى للموارد
        self.default_limits = {
            'cpu_max_ms': 10000,  # 10 seconds max CPU
            'memory_max_mb': 512,  # 512 MB max memory
            'network_allowed': False,
            'filesystem_allowed': False,
            'max_duration_sec': 60
        }
        
        # إحصائيات
        self.stats = {
            'sandboxes_created': 0,
            'sandboxes_destroyed': 0,
            'active_sandboxes': 0,
            'violations': 0
        }
        
        # خلفية التنظيف
        self.cleanup_task = None
        self.running = False
    
    async def start(self):
        """بدء خدمة Sandbox"""
        self.running = True
        self.cleanup_task = asyncio.create_task(self._cleanup_loop())
        logger.info("SandboxManager started")
    
    async def create_sandbox(self, tool_name: str, 
                              timeout_sec: int = 30,
                              memory_limit_mb: int = 512,
                              cpu_limit: float = 1.0) -> str:
        """
        إنشاء بيئة Sandbox جديدة
        """
        sandbox_id = str(uuid.uuid4())
        
        # في الإنتاج: استدعاء Firecracker أو Docker
        # تبسيطاً: إنشاء كائن محاكاة
        
        sandbox = Sandbox(
            id=sandbox_id,
            tool_name=tool_name,
            expires_at=time.time() + timeout_sec
        )
        
        self.sandboxes[sandbox_id] = sandbox
        self.stats['sandboxes_created'] += 1
        self.stats['active_sandboxes'] += 1
        
        logger.info(f"Sandbox created: {sandbox_id} for tool {tool_name}")
        
        return sandbox_id
    
    async def execute_in_sandbox(self, sandbox_id: str, code: str, params: Dict) -> Dict:
        """
        تنفيذ كود في بيئة معزولة
        """
        if sandbox_id not in self.sandboxes:
            return {
                'success': False,
                'error': f"Sandbox {sandbox_id} not found"
            }
        
        sandbox = self.sandboxes[sandbox_id]
        
        # التحقق من انتهاء الصلاحية
        if time.time() > sandbox.expires_at:
            await self.destroy_sandbox(sandbox_id)
            return {
                'success': False,
                'error': "Sandbox expired"
            }
        
        start_time = time.time()
        
        try:
            # في الإنتاج: تنفيذ فعلي في Sandbox
            # تبسيطاً: محاكاة آمنة
            
            # فحص الأمان الأساسي
            dangerous_patterns = ['exec(', 'eval(', '__import__', 'open(', 'file(']
            for pattern in dangerous_patterns:
                if pattern in code:
                    self.stats['violations'] += 1
                    return {
                        'success': False,
                        'error': f"Security violation: {pattern} not allowed in sandbox"
                    }
            
            # محاكاة التنفيذ
            await asyncio.sleep(min(0.5, len(code) / 1000))
            
            result = {
                'output': f"Executed in sandbox {sandbox_id}",
                'exit_code': 0,
                'execution_time_ms': (time.time() - start_time) * 1000
            }
            
            # تحديث استخدام الموارد
            sandbox.resource_usage['cpu_used_ms'] += result['execution_time_ms']
            
            return result
            
        except Exception as e:
            return {
                'success': False,
                'error': str(e)
            }
    
    async def destroy_sandbox(self, sandbox_id: str):
        """
        تدمير بيئة Sandbox
        """
        if sandbox_id in self.sandboxes:
            # في الإنتاج: إيقاف Firecracker VM
            
            del self.sandboxes[sandbox_id]
            self.stats['sandboxes_destroyed'] += 1
            self.stats['active_sandboxes'] -= 1
            
            logger.info(f"Sandbox destroyed: {sandbox_id}")
    
    async def _cleanup_loop(self):
        """حلقة تنظيف الخلفية"""
        while self.running:
            current_time = time.time()
            expired = []
            
            for sandbox_id, sandbox in self.sandboxes.items():
                if current_time > sandbox.expires_at:
                    expired.append(sandbox_id)
                elif sandbox.resource_usage['cpu_used_ms'] > self.default_limits['cpu_max_ms']:
                    expired.append(sandbox_id)
                    self.stats['violations'] += 1
            
            for sandbox_id in expired:
                await self.destroy_sandbox(sandbox_id)
            
            await asyncio.sleep(10)  # كل 10 ثوانٍ
    
    async def get_sandbox_status(self, sandbox_id: str) -> Optional[Dict]:
        """الحصول على حالة Sandbox"""
        if sandbox_id not in self.sandboxes:
            return None
        
        sandbox = self.sandboxes[sandbox_id]
        return {
            'id': sandbox.id,
            'tool_name': sandbox.tool_name,
            'status': sandbox.status,
            'created_at': sandbox.created_at,
            'expires_at': sandbox.expires_at,
            'resource_usage': sandbox.resource_usage
        }
    
    async def get_stats(self) -> Dict:
        """إحصائيات Sandbox Manager"""
        return {
            'sandboxes_created': self.stats['sandboxes_created'],
            'sandboxes_destroyed': self.stats['sandboxes_destroyed'],
            'active_sandboxes': self.stats['active_sandboxes'],
            'violations': self.stats['violations']
        }
    
    async def shutdown(self):
        """إيقاف Sandbox Manager"""
        self.running = False
        if self.cleanup_task:
            self.cleanup_task.cancel()
        
        # تدمير جميع الـ Sandboxes
        for sandbox_id in list(self.sandboxes.keys()):
            await self.destroy_sandbox(sandbox_id)
        
        logger.info("SandboxManager shutdown")
