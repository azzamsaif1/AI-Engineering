"""
Cognitive Orchestrator V8 - المنسق المعرفي المركزي
مسؤول عن: scheduling, prioritization, GPU allocation, cognitive budget, routing
"""

import asyncio
import time
import psutil
import GPUtil
from typing import Dict, List, Optional, Any, Tuple
from enum import Enum
from dataclasses import dataclass, field
from collections import defaultdict
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


class Modality(Enum):
    AUDIO = "audio"
    VISION = "vision"
    CODE = "code"
    BEHAVIOR = "behavior"
    PREDICTIVE = "predictive"


class Priority(Enum):
    CRITICAL = 0
    HIGH = 1
    NORMAL = 2
    LOW = 3
    BACKGROUND = 4


@dataclass
class CognitiveBudget:
    """ميزانية المعالجة المعرفية"""
    total_gpu_memory_mb: float = 24000
    available_gpu_memory_mb: float = 24000
    total_cpu_percent: float = 100.0
    available_cpu_percent: float = 100.0
    allocated_resources: Dict[str, float] = field(default_factory=dict)
    
    def can_allocate(self, required_mb: float, required_cpu: float) -> bool:
        return (self.available_gpu_memory_mb >= required_mb and 
                self.available_cpu_percent >= required_cpu)
    
    def allocate(self, modality: str, required_mb: float, required_cpu: float) -> bool:
        if self.can_allocate(required_mb, required_cpu):
            self.available_gpu_memory_mb -= required_mb
            self.available_cpu_percent -= required_cpu
            self.allocated_resources[modality] = required_mb
            return True
        return False
    
    def release(self, modality: str):
        if modality in self.allocated_resources:
            self.available_gpu_memory_mb += self.allocated_resources[modality]
            del self.allocated_resources[modality]


class DynamicCognitiveRouter:
    """
    التوجيه المعرفي الديناميكي - يقرر أي النماذج يتم تفعيلها
    """
    
    def __init__(self):
        self.routing_rules = self._load_routing_rules()
        self.active_stack: List[str] = []
        
    def _load_routing_rules(self) -> Dict:
        """قواعد التوجيه الديناميكية"""
        return {
            'coding': {
                'trigger': ['code', 'implementation', 'function', 'class', 'algorithm'],
                'activate': ['code', 'behavior', 'predictive'],
                'deactivate': ['audio', 'vision']
            },
            'debugging': {
                'trigger': ['error', 'bug', 'debug', 'exception', 'traceback', 'backspace'],
                'activate': ['code', 'behavior', 'predictive', 'vision'],
                'deactivate': ['audio']
            },
            'meeting': {
                'trigger': ['meeting', 'discussion', 'presentation', 'slides'],
                'activate': ['audio', 'memory', 'summarization'],
                'deactivate': ['code', 'vision']
            },
            'research': {
                'trigger': ['paper', 'arxiv', 'research', 'benchmark', 'state-of-the-art'],
                'activate': ['code', 'predictive', 'memory'],
                'deactivate': ['audio', 'behavior']
            },
            'design': {
                'trigger': ['architecture', 'design', 'diagram', 'flowchart', 'UML'],
                'activate': ['vision', 'code', 'predictive'],
                'deactivate': ['audio', 'behavior']
            }
        }
    
    async def route(self, context: Dict) -> List[str]:
        """
        تحديد النماذج التي يجب تفعيلها بناءً على السياق
        """
        # تحليل السياق
        context_text = str(context).lower()
        
        best_match = None
        best_score = 0.0
        
        for mode, rule in self.routing_rules.items():
            score = sum(1 for trigger in rule['trigger'] if trigger in context_text)
            if score > best_score:
                best_score = score
                best_match = mode
        
        if best_match and best_score > 0:
            self.active_stack = self.routing_rules[best_match]['activate']
            logger.info(f"Dynamic routing: {best_match} mode activated -> {self.active_stack}")
        else:
            # الوضع الافتراضي
            self.active_stack = ['code', 'behavior']
        
        return self.active_stack


class CognitiveOrchestratorV8:
    """
    Cognitive Orchestrator V8 - المنسق المعرفي المركزي
    """
    
    def __init__(self):
        self.budget = CognitiveBudget()
        self.router = DynamicCognitiveRouter()
        
        # جداول المهام
        self.task_queue: asyncio.Queue = asyncio.Queue()
        self.running_tasks: Dict[str, asyncio.Task] = {}
        self.task_results: Dict[str, Any] = {}
        
        # أولويات المهام
        self.priorities: Dict[str, Priority] = {}
        
        # إحصائيات
        self.stats = {
            'tasks_scheduled': 0,
            'tasks_completed': 0,
            'tasks_failed': 0,
            'avg_wait_time_ms': 0.0,
            'gpu_utilization_history': []
        }
        
        # حلقة المعالجة
        self.processing_task = None
        self.running = False
        
        # تكاليف النماذج التقريبية (MB VRAM, % CPU)
        self.model_costs = {
            Modality.AUDIO: (2048, 15),      # WhisperX
            Modality.VISION: (4096, 20),     # ScreenAI
            Modality.CODE: (3072, 25),       # Tree-sitter + Joern
            Modality.BEHAVIOR: (1024, 10),   # Mamba
            Modality.PREDICTIVE: (2048, 15)  # Predictive models
        }
    
    async def schedule_task(self, modality: Modality, task_func, 
                            priority: Priority = Priority.NORMAL,
                            context: Dict = None) -> str:
        """
        جدولة مهمة معرفية
        """
        task_id = f"{modality.value}_{int(time.time() * 1000)}_{self.stats['tasks_scheduled']}"
        
        # التحقق من الميزانية
        cost_mb, cost_cpu = self.model_costs.get(modality, (1024, 10))
        if not self.budget.can_allocate(cost_mb, cost_cpu):
            logger.warning(f"Insufficient resources for {modality.value}")
            return None
        
        # إضافة إلى قائمة الانتظار
        await self.task_queue.put({
            'id': task_id,
            'modality': modality,
            'task_func': task_func,
            'priority': priority,
            'context': context or {},
            'enqueued_at': time.time()
        })
        
        self.priorities[task_id] = priority
        self.stats['tasks_scheduled'] += 1
        
        return task_id
    
    async def _processing_loop(self):
        """حلقة المعالجة الرئيسية - تستهلك المهام من قائمة الانتظار"""
        while self.running:
            try:
                # الحصول على المهمة التالية (مع أولوية)
                task_item = await self._get_next_task()
                if not task_item:
                    await asyncio.sleep(0.01)
                    continue
                
                task_id = task_item['id']
                modality = task_item['modality']
                cost_mb, cost_cpu = self.model_costs.get(modality, (1024, 10))
                
                # تخصيص الموارد
                if self.budget.allocate(modality.value, cost_mb, cost_cpu):
                    wait_time = (time.time() - task_item['enqueued_at']) * 1000
                    self.stats['avg_wait_time_ms'] = (
                        self.stats['avg_wait_time_ms'] * 0.95 + wait_time * 0.05
                    )
                    
                    # تنفيذ المهمة
                    try:
                        result = await task_item['task_func'](task_item['context'])
                        self.task_results[task_id] = result
                        self.stats['tasks_completed'] += 1
                    except Exception as e:
                        self.task_results[task_id] = {'error': str(e)}
                        self.stats['tasks_failed'] += 1
                        logger.error(f"Task {task_id} failed: {e}")
                    finally:
                        # تحرير الموارد
                        self.budget.release(modality.value)
                else:
                    # إعادة المهمة إلى قائمة الانتظار
                    await self.task_queue.put(task_item)
                    await asyncio.sleep(0.05)
                    
            except Exception as e:
                logger.error(f"Processing loop error: {e}")
                await asyncio.sleep(0.1)
    
    async def _get_next_task(self) -> Optional[Dict]:
        """الحصول على المهمة التالية مع مراعاة الأولوية"""
        # تجميع المهام حسب الأولوية
        priority_buckets = defaultdict(list)
        temp_queue = []
        
        # استخراج جميع المهام من قائمة الانتظار
        while not self.task_queue.empty():
            try:
                task = self.task_queue.get_nowait()
                priority_buckets[self.priorities.get(task['id'], Priority.NORMAL)].append(task)
                temp_queue.append(task)
            except asyncio.QueueEmpty:
                break
        
        # إعادة المهام إلى قائمة الانتظار (إذا لم نأخذ مهمة)
        if not priority_buckets:
            return None
        
        # اختيار المهمة ذات الأولوية القصوى
        for priority in [Priority.CRITICAL, Priority.HIGH, Priority.NORMAL, Priority.LOW, Priority.BACKGROUND]:
            if priority in priority_buckets and priority_buckets[priority]:
                selected = priority_buckets[priority][0]
                # إعادة باقي المهام
                remaining = [t for t in temp_queue if t['id'] != selected['id']]
                for task in remaining:
                    await self.task_queue.put(task)
                return selected
        
        return None
    
    async def get_task_result(self, task_id: str, timeout_ms: int = 5000) -> Optional[Any]:
        """انتظار نتيجة مهمة"""
        start_time = time.time()
        while task_id not in self.task_results:
            if (time.time() - start_time) * 1000 > timeout_ms:
                return None
            await asyncio.sleep(0.01)
        return self.task_results.pop(task_id)
    
    async def start(self):
        """بدء المنسق"""
        self.running = True
        self.processing_task = asyncio.create_task(self._processing_loop())
        logger.info("Cognitive Orchestrator started")
    
    async def get_system_health(self) -> Dict:
        """الحصول على صحة النظام"""
        try:
            gpus = GPUtil.getGPUs()
            gpu_util = gpus[0].load * 100 if gpus else 0
            gpu_memory = gpus[0].memoryUsed if gpus else 0
        except:
            gpu_util = 0
            gpu_memory = 0
        
        self.stats['gpu_utilization_history'].append({
            'timestamp': time.time(),
            'utilization': gpu_util
        })
        if len(self.stats['gpu_utilization_history']) > 100:
            self.stats['gpu_utilization_history'] = self.stats['gpu_utilization_history'][-100:]
        
        return {

@dataclass

class CrossModalAttention(nn.Module):
    """
    الانتباه متعدد الحواس - يربط بين الإدراكات المختلفة
    """
    
    def __init__(self, d_model: int = 512, n_heads: int = 8):
        super().__init__()
        self.d_model = d_model
        self.n_heads = n_heads
        
        self.audio_proj = nn.Linear(256, d_model)
        self.vision_proj = nn.Linear(256, d_model)
        self.code_proj = nn.Linear(256, d_model)
        self.behavior_proj = nn.Linear(256, d_model)
        
        self.cross_attention = nn.MultiheadAttention(d_model, n_heads, batch_first=True)
        
    def forward(self, audio_feat, vision_feat, code_feat, behavior_feat):
        # إسقاط جميع الميزات إلى نفس البعد
        audio_emb = self.audio_proj(audio_feat).unsqueeze(1)
        vision_emb = self.vision_proj(vision_feat).unsqueeze(1)
        code_emb = self.code_proj(code_feat).unsqueeze(1)
        behavior_emb = self.behavior_proj(behavior_feat).unsqueeze(1)
        
        # دمج جميع الوسائط
        combined = torch.cat([audio_emb, vision_emb, code_emb, behavior_emb], dim=1)
        
        # الانتباه المتقاطع
        attended, attention_weights = self.cross_attention(combined, combined, combined)
        
        return attended, attention_weights

class MemoryConsolidationStrategy:
    """
    استراتيجية دمج الذاكرة - تمنع الـ catastrophic forgetting
    """
    
    def __init__(self):
        self.consolidation_interval = 3600  # كل ساعة
        self.last_consolidation = time.time()
        self.pending_consolidations: List[MemoryEntry] = []
        
    async def should_consolidate(self) -> bool:
        """هل حان وقت الدمج؟"""
        return (time.time() - self.last_consolidation) > self.consolidation_interval
    
    async def consolidate(self, shadow_memory: List[MemoryEntry], 
                           main_memory: List[MemoryEntry]) -> List[MemoryEntry]:
        """
        دمج الذاكرة الظلية مع الذاكرة الرئيسية
        - يتم الدمج فقط بعد التحقق من الاستقرار
        - يمنع الـ catastrophic forgetting
        """
        if not shadow_memory:
            return main_memory
        
        # 1. التحقق من استقرار التغييرات
        stable_entries = []
        for entry in shadow_memory:
            if await self._is_stable(entry):
                stable_entries.append(entry)
        
        # 2. دمج الإدخالات المستقرة فقط
        for entry in stable_entries:
            existing = self._find_entry(main_memory, entry.id)
            if existing:
                # دمج مع وجود سابق
                existing.importance = max(existing.importance, entry.importance)
                existing.access_count += entry.access_count
                existing.consolidation_count += 1
            else:
                # إضافة جديدة
                main_memory.append(entry)
        
        # 3. تنظيف الذاكرة الظلية
        shadow_memory.clear()
        self.last_consolidation = time.time()
        
        logger.info(f"Memory consolidation: merged {len(stable_entries)} entries")
        
        return main_memory
    
    async def _is_stable(self, entry: MemoryEntry) -> bool:
        """التحقق من استقرار إدخال الذاكرة"""
        # الإدخال مستقر إذا:
        # 1. تم الوصول إليه عدة مرات
        # 2. عمره كافٍ (ليس جديداً جداً)
        # 3. ليس له تناقضات مع ذكريات أخرى
        
        if entry.access_count < 3:
            return False
        
        if time.time() - entry.timestamp < 300:  # أقل من 5 دقائق
            return False
        
        return True
    
    def _find_entry(self, memory: List[MemoryEntry], entry_id: str) -> Optional[MemoryEntry]:
        """البحث عن إدخال في الذاكرة"""
        for entry in memory:
            if entry.id == entry_id:
                return entry
        return None

            ],
            'credential_leakage': [
                r'password\s*=', r'secret\s*=', r'api_key\s*=',
                r'token\s*=', r'credential', r'authorization'
            ],
            'unsafe_automation': [
                r'while\s+True:', r'for\s+.*\s+in\s+.*:.*\n.*\n.*\n.*\n.*',  # حلقات عميقة
                r'\.delete', r'\.drop', r'\.truncate', r'\.destroy'
            ]
        }
        
        self.allowed_domains = ['localhost', '127.0.0.1', '*.test', '*.example.com']
        
        self.violation_history: List[Dict] = []
    
    async def review_action(self, action_type: str, action_content: str, 
                            context: Dict) -> SafetyVerdict:
        """
        مراجعة إجراء قبل التنفيذ
        """
        violations = []
        risk_level = RiskLevel.SAFE
        
        # 1. فحص التضمين البرمجي الخطر
        if action_type in ['code_generation', 'code_modification', 'refactoring']:
            code_violations = self._check_code_injection(action_content)
            if code_violations:
                violations.extend(code_violations)
                risk_level = RiskLevel.CRITICAL
        
        # 2. فحص العمليات التدميرية
        destructive_violations = self._check_destructive_ops(action_content)
        if destructive_violations:
            violations.extend(destructive_violations)
            risk_level = max(risk_level, RiskLevel.HIGH, key=lambda x: list(RiskLevel).index(x))
        
        # 3. فحص تسرب البيانات الحساسة
        credential_violations = self._check_credential_leakage(action_content, context)
        if credential_violations:
            violations.extend(credential_violations)
            risk_level = RiskLevel.CRITICAL
        
        # 4. فحص الأتمتة غير الآمنة
        automation_violations = self._check_unsafe_automation(action_content)
        if automation_violations:
            violations.extend(automation_violations)
            risk_level = max(risk_level, RiskLevel.MEDIUM, key=lambda x: list(RiskLevel).index(x))
        
        # 5. فحص السياق (من هو المستخدم، ما هي الصلاحيات)
        context_violations = self._check_context(context)
        if context_violations:
            violations.extend(context_violations)
        
        # 6. تسجيل المخالفة
        if violations:
            self.violation_history.append({
                'timestamp': __import__('time').time(),
                'action_type': action_type,
                'violations': violations,
                'risk_level': risk_level.value
            })
            
            logger.warning(f"Safety violation: {violations[:2]} (risk={risk_level.value})")
        
        # 7. اتخاذ القرار
        if risk_level == RiskLevel.CRITICAL:
            allowed = False
            reason = "CRITICAL: Action blocked due to security risk"
        elif risk_level == RiskLevel.HIGH and not self._is_emergency_override(context):
            allowed = False
            reason = "HIGH: Action requires manual review"
        else:
            allowed = True
            reason = "Action approved" if not violations else "Action approved with warnings"
        
        suggestions = self._generate_suggestions(violations)
        
        return SafetyVerdict(
            allowed=allowed,
            risk_level=risk_level,
            reason=reason,
            suggestions=suggestions
        )
    
    def _check_code_injection(self, code: str) -> List[str]:
        """فحص تضمين برمجي خطر"""
        violations = []
        
        for pattern_name, patterns in self.blocked_patterns.items():

            for node in ast.walk(tree):

    def _check_destructive_ops(self, content: str) -> List[str]:
        """فحص العمليات التدميرية"""
        violations = []

        suspicious_values = re.findall(r'[\'"]([A-Za-z0-9]{32,})[\'"]', content)
        if suspicious_values:

    def _check_unsafe_automation(self, content: str) -> List[str]:

"""
Uncertainty Engine V8 - epistemic uncertainty propagation
"""

import numpy as np
from typing import Dict, List, Optional, Any, Tuple
from dataclasses import dataclass
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


@dataclass
class UncertainPrediction:
    """توقع مع عدم يقين"""
    value: Any
    mean_confidence: float
    std_uncertainty: float
    epistemic_uncertainty: float  # عدم يقين معرفي (نقص المعرفة)
    aleatoric_uncertainty: float  # عدم يقين عشوائي (ضوضاء في البيانات)
    calibration_score: float      # مدى دقة معايرة الثقة


class UncertaintyEngineV8:
    """
    محرك عدم اليقين - يتتبع ويقدم عدم اليقين في جميع التنبؤات
    """
    
    def __init__(self):
        self.prediction_history: List[UncertainPrediction] = []
        self.calibration_history: List[float] = []
        
    async def quantify(self, predictions: List[Tuple[Any, float]]) -> UncertainPrediction:
        """
        قياس عدم اليقين من تنبؤات متعددة
        """
        if not predictions:
            return UncertainPrediction(
                value=None,
                mean_confidence=0.0,
                std_uncertainty=1.0,
                epistemic_uncertainty=1.0,
                aleatoric_uncertainty=0.0,
                calibration_score=0.5
            )
        
        values = [p[0] for p in predictions]
        confidences = [p[1] for p in predictions]
        
        mean_conf = np.mean(confidences)
        std_conf = np.std(confidences)
        
        # عدم يقين معرفي = تباين بين التنبؤات
        epistemic = std_conf
        
        # عدم يقين عشوائي = متوسط 1 - الثقة
        aleatoric = np.mean([1 - c for c in confidences])
        
        # اختيار القيمة الأكثر شيوعاً (للتنبؤات الفئوية) أو المتوسط (للتنبؤات الرقمية)
        if all(isinstance(v, (int, float)) for v in values):
            final_value = np.mean(values)
        else:
            from collections import Counter
            final_value = Counter(values).most_common(1)[0][0]
        
        # حساب درجة المعايرة
        calibration = await self._compute_calibration()
        
        prediction = UncertainPrediction(
            value=final_value,
            mean_confidence=mean_conf,
            std_uncertainty=std_conf,
            epistemic_uncertainty=epistemic,
            aleatoric_uncertainty=aleatoric,
            calibration_score=calibration
        )
        
        self.prediction_history.append(prediction)
        if len(self.prediction_history) > 1000:
            self.prediction_history = self.prediction_history[-1000:]
        
        return prediction
    
    async def _compute_calibration(self) -> float:
        """
        حساب مدى دقة معايرة الثقة
        كلما اقتربت من 1، كانت الثقة أكثر دقة
        """
        if len(self.prediction_history) < 10:
            return 0.7  # افتراضي
        
        recent = self.prediction_history[-100:]
        
        # تبسيط: مقارنة بين الثقة المتوقعة والنجاح الفعلي
        # في الإنتاج: حساب ECE (Expected Calibration Error)
        
        return 0.85  # تبسيط
    
    async def is_confident(self, prediction: UncertainPrediction, threshold: float = 0.7) -> bool:
        """هل النظام واثق من هذا التنبؤ؟"""
        return (prediction.mean_confidence > threshold and 
                prediction.epistemic_uncertainty < 0.3)
    
    async def should_defer_to_user(self, prediction: UncertainPrediction) -> bool:
        """هل يجب تأجيل القرار للمستخدم؟"""
        return (prediction.epistemic_uncertainty > 0.5 or 
                prediction.mean_confidence < 0.6 or
                prediction.calibration_score < 0.6)
    
    async def get_stats(self) -> Dict:
        """إحصائيات عدم اليقين"""
        if not self.prediction_history:
            return {'total_predictions': 0}
        
        recent = self.prediction_history[-100:]
        
        return {
            'total_predictions': len(self.prediction_history),
            'avg_epistemic_uncertainty': np.mean([p.epistemic_uncertainty for p in recent]),
            'avg_aleatoric_uncertainty': np.mean([p.aleatoric_uncertainty for p in recent]),
            'avg_calibration': np.mean([p.calibration_score for p in recent]),
            'high_uncertainty_rate': sum(1 for p in recent if p.epistemic_uncertainty > 0.5) / len(recent)
        }

        if not trace:
            return "No reasoning trace found"
        
        explanation = await self._generate_explanation(trace)
        self.explanations[step_id] = explanation
        
        return explanation
    
    async def _generate_explanation(self, trace: ReasoningTrace) -> str:
        """

        explanation_parts = [

            f"  - I considered alternatives: {', '.join(trace.alternatives_considered[:3])}",
            f"  - My reasoning: {trace.reasoning[:200]}",
            f"  - Confidence level: {trace.confidence:.2f}"
        ]
        
        if trace.parent_step_id:
            parent_trace = self._find_trace(trace.parent_step_id)
            if parent_trace:

        return "\n".join(explanation_parts)

        """حلقة الدمج الخلفية"""
        while self.running:
            # تكيف النموذج الظلي
            await self.adapt_shadow()
            
            # محاولة الدمج
            await self.validate_and_merge()
            
            await asyncio.sleep(300)  # كل 5 دقائق
    
    async def start(self):
        """بدء محرك التكيف"""
        self.running = True
        self.merge_task = asyncio.create_task(self._merge_loop())
        logger.info("Shadow adaptation engine started")
    
    async def get_stats(self) -> Dict:
        """إحصائيات التكيف"""
        return {
            **self.stats,
            'buffer_size': len(self.experience_buffer),
            'checkpoints': len(self.checkpoints),
            'last_checkpoint': self.checkpoints[-1].timestamp if self.checkpoints else None
        }
    
    async def shutdown(self):
        """إيقاف المحرك"""
        self.running = False
        if self.merge_task:

import asyncio

class CognitiveEngineeringTwinV8:
    """
    التوأم المعرفي الهندسي V8 - النظام الديناميكي الكامل
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        
        # المنسق المعرفي المركزي
        self.orchestrator = CognitiveOrchestratorV8()
        
        # طبقات الذكاء المتخصصة
        self.audio = AudioIntelligence()
        self.visual = VisualIntelligence()
        self.code = CodeIntelligence()
        self.behavior = BehavioralIntelligence()
        
        # المحرك الجديد
        self.fusion_engine = CognitiveFusionEngineV8()
        self.memory = HierarchicalMemoryV8(user_id)
        self.safety_governor = SafetyGovernorV8()
        self.uncertainty_engine = UncertaintyEngineV8()
        self.meta_cognition = MetaCognitionLayerV8()
        
        # ملاحظ: shadow_adaptation يتطلب نموذج PyTorch، تم تعطيله في هذا الإصدار المبسط
        
        self.stats = {
            'total_perceptions': 0,
            'avg_fusion_confidence': 0.0,
            'avg_latency_ms': 0.0
        }
        
        self.is_running = False
    
    async def initialize(self):
        """تهيئة النظام"""
        await self.orchestrator.start()
        await self.memory.start()
        self.is_running = True
        logger.info(f"Cognitive Twin V8 initialized for user {self.user_id}")
    
    async def perceive_and_fuse(self, 
                                 audio_data: Optional[np.ndarray] = None,
                                 screen_data: Optional[np.ndarray] = None,
                                 code_data: Optional[str] = None,
                                 mouse_x: Optional[float] = None,
                                 mouse_y: Optional[float] = None,
                                 key_presses: Optional[List[str]] = None) -> Dict:
        """
        إدراك متعدد الحواس مع دمج معرفي - نقطة الدخول الرئيسية
        """
        start_time = time.time()
        
        # 1. التوجيه الديناميكي (تحديد النماذج المطلوبة)
        context = {
            'has_audio': audio_data is not None,
            'has_screen': screen_data is not None,
            'has_code': code_data is not None,
            'has_behavior': mouse_x is not None or key_presses is not None
        }
        active_modalities = await self.orchestrator.router.route(context)
        
        # 2. المعالجة المتوازية عبر asyncio.gather
        results = {}
        tasks = []
        
        if 'audio' in active_modalities and audio_data is not None:
            task_id = await self.orchestrator.schedule_task(
                Modality.AUDIO, 
                lambda ctx: self.audio.process_audio(ctx['audio']),
                Priority.HIGH,
                {'audio': audio_data}
            )
            if task_id:
                tasks.append(('audio', task_id))
        
        if 'vision' in active_modalities and screen_data is not None:
            task_id = await self.orchestrator.schedule_task(
                Modality.VISION,
                lambda ctx: self.visual.process_screen(ctx['screen']),
                Priority.HIGH,
                {'screen': screen_data}
            )
            if task_id:
                tasks.append(('vision', task_id))
        
        if 'code' in active_modalities and code_data is not None:
            task_id = await self.orchestrator.schedule_task(
                Modality.CODE,
                lambda ctx: self.code.process_code(ctx['code']),
                Priority.NORMAL,
                {'code': code_data}
            )
            if task_id:
                tasks.append(('code', task_id))
        
        if 'behavior' in active_modalities and (mouse_x is not None or key_presses is not None):
            task_id = await self.orchestrator.schedule_task(
                Modality.BEHAVIOR,
                lambda ctx: self.behavior.process_behavior(
                    ctx.get('mouse_x', 0), ctx.get('mouse_y', 0), ctx.get('key_presses', [])
                ),
                Priority.NORMAL,
                {'mouse_x': mouse_x, 'mouse_y': mouse_y, 'key_presses': key_presses or []}
            )
            if task_id:
                tasks.append(('behavior', task_id))
        
        # انتظار جميع المهام بالتوازي
        for modality_name, task_id in tasks:
            result = await self.orchestrator.get_task_result(task_id)
            if result:
                results[modality_name] = result
        
        # 3. الدمج المعرفي (الأهم)
        fused = await self.fusion_engine.fuse(
            audio_result=results.get('audio'),
            vision_result=results.get('vision'),
            code_result=results.get('code'),
            behavior_result=results.get('behavior')
        )
        
        # 4. تخزين في الذاكرة
        await self.memory.store_episodic({
            'timestamp': time.time(),
            'fused_state': fused.inferred_mental_state,
            'confidence': fused.confidence,
            'modalities': list(results.keys())
        }, importance=fused.confidence)
        
        # 5. قياس عدم اليقين
        uncertainty = await self.uncertainty_engine.quantify([
            (fused.inferred_mental_state, fused.confidence),
            (fused.interpretation, fused.confidence)
        ])
        
        # 6. تحديث الإحصائيات
        latency_ms = (time.time() - start_time) * 1000
        self.stats['total_perceptions'] += 1
        self.stats['avg_fusion_confidence'] = (
            self.stats['avg_fusion_confidence'] * 0.95 + fused.confidence * 0.05
        )
        self.stats['avg_latency_ms'] = (
            self.stats['avg_latency_ms'] * 0.95 + latency_ms * 0.05
        )
        
        return {
            'fused_perception': {
                'mental_state': fused.inferred_mental_state,
                'interpretation': fused.interpretation,
                'confidence': fused.confidence,
                'urgency': fused.urgency_score
            },
            'cross_modal_correlations': fused.cross_modal_correlations,
            'uncertainty': {
                'epistemic': uncertainty.epistemic_uncertainty,
                'aleatoric': uncertainty.aleatoric_uncertainty,
                'is_confident': await self.uncertainty_engine.is_confident(uncertainty)
            },
            'latency_ms': latency_ms
        }
    
    async def act(self, action_type: str, action_content: str, context: Dict = None) -> Dict:
        """
        تنفيذ إجراء مع فحص السلامة
        """
        # 1. مراجعة السلامة
        verdict = await self.safety_governor.review_action(
            action_type, action_content, context or {}
        )
        
        if not verdict.allowed:
            return {
                'executed': False,
                'blocked': True,
                'reason': verdict.reason,
                'suggestions': verdict.suggestions
            }
        
        # 2. تنفيذ الإجراء
        # في الإنتاج: التنفيذ الفعلي
        return {
            'executed': True,
            'blocked': False,
            'risk_level': verdict.risk_level.value,
            'warnings': verdict.suggestions if verdict.risk_level != RiskLevel.SAFE else []
        }
    
    async def explain_reasoning(self, step_id: str) -> str:
        """شرح مسار التفكير"""
        return await self.meta_cognition.explain(step_id)
    
    async def get_system_status(self) -> Dict:
        """الحالة الكاملة للنظام"""
        return {
            'user_id': self.user_id,
            'is_running': self.is_running,
            'stats': self.stats,
            'orchestrator': await self.orchestrator.get_system_health(),
            'fusion': await self.fusion_engine.get_stats(),
            'memory': await self.memory.get_stats(),
            'safety': await self.safety_governor.get_stats(),
            'uncertainty': await self.uncertainty_engine.get_stats(),
            'meta_cognition': await self.meta_cognition.get_stats()
        }
    
    async def shutdown(self):
        """إيقاف النظام"""
        self.is_running = False
        await self.orchestrator.shutdown()
        await self.memory.shutdown()
        logger.info(f"Cognitive Twin V8 shutdown for user {self.user_id}")

"""
Cognitive Orchestrator V8 - المنسق المعرفي المركزي
مسؤول عن: scheduling, prioritization, GPU allocation, cognitive budget, routing
"""

import asyncio
import time
import psutil
import GPUtil

from enum import Enum
from dataclasses import dataclass, field
from collections import defaultdict
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


class Modality(Enum):
    AUDIO = "audio"
    VISION = "vision"
    CODE = "code"
    BEHAVIOR = "behavior"
    PREDICTIVE = "predictive"


class Priority(Enum):
    CRITICAL = 0
    HIGH = 1
    NORMAL = 2
    LOW = 3
    BACKGROUND = 4


@dataclass
class CognitiveBudget:
    """ميزانية المعالجة المعرفية"""
    total_gpu_memory_mb: float = 24000
    available_gpu_memory_mb: float = 24000
    total_cpu_percent: float = 100.0
    available_cpu_percent: float = 100.0
    allocated_resources: Dict[str, float] = field(default_factory=dict)
    
    def can_allocate(self, required_mb: float, required_cpu: float) -> bool:
        return (self.available_gpu_memory_mb >= required_mb and 
                self.available_cpu_percent >= required_cpu)
    
    def allocate(self, modality: str, required_mb: float, required_cpu: float) -> bool:
        if self.can_allocate(required_mb, required_cpu):
            self.available_gpu_memory_mb -= required_mb
            self.available_cpu_percent -= required_cpu
            self.allocated_resources[modality] = required_mb
            return True
        return False
    
    def release(self, modality: str):
        if modality in self.allocated_resources:
            self.available_gpu_memory_mb += self.allocated_resources[modality]
            del self.allocated_resources[modality]


class DynamicCognitiveRouter:
    """
    التوجيه المعرفي الديناميكي - يقرر أي النماذج يتم تفعيلها
    """
    
    def __init__(self):
        self.routing_rules = self._load_routing_rules()
        self.active_stack: List[str] = []
        
    def _load_routing_rules(self) -> Dict:
        """قواعد التوجيه الديناميكية"""
        return {
            'coding': {
                'trigger': ['code', 'implementation', 'function', 'class', 'algorithm'],
                'activate': ['code', 'behavior', 'predictive'],
                'deactivate': ['audio', 'vision']
            },
            'debugging': {
                'trigger': ['error', 'bug', 'debug', 'exception', 'traceback', 'backspace'],
                'activate': ['code', 'behavior', 'predictive', 'vision'],
                'deactivate': ['audio']
            },
            'meeting': {
                'trigger': ['meeting', 'discussion', 'presentation', 'slides'],
                'activate': ['audio', 'memory', 'summarization'],
                'deactivate': ['code', 'vision']
            },
            'research': {
                'trigger': ['paper', 'arxiv', 'research', 'benchmark', 'state-of-the-art'],
                'activate': ['code', 'predictive', 'memory'],
                'deactivate': ['audio', 'behavior']
            },
            'design': {
                'trigger': ['architecture', 'design', 'diagram', 'flowchart', 'UML'],
                'activate': ['vision', 'code', 'predictive'],
                'deactivate': ['audio', 'behavior']
            }
        }
    
    async def route(self, context: Dict) -> List[str]:
        """
        تحديد النماذج التي يجب تفعيلها بناءً على السياق
        """
        # تحليل السياق
        context_text = str(context).lower()
        
        best_match = None
        best_score = 0.0
        
        for mode, rule in self.routing_rules.items():
            score = sum(1 for trigger in rule['trigger'] if trigger in context_text)
            if score > best_score:
                best_score = score
                best_match = mode
        
        if best_match and best_score > 0:
            self.active_stack = self.routing_rules[best_match]['activate']
            logger.info(f"Dynamic routing: {best_match} mode activated -> {self.active_stack}")
        else:
            # الوضع الافتراضي
            self.active_stack = ['code', 'behavior']
        
        return self.active_stack


class CognitiveOrchestratorV8:
    """
    Cognitive Orchestrator V8 - المنسق المعرفي المركزي
    """
    
    def __init__(self):
        self.budget = CognitiveBudget()
        self.router = DynamicCognitiveRouter()
        
        # جداول المهام
        self.task_queue: asyncio.Queue = asyncio.Queue()
        self.running_tasks: Dict[str, asyncio.Task] = {}
        self.task_results: Dict[str, Any] = {}
        
        # أولويات المهام
        self.priorities: Dict[str, Priority] = {}
        
        # إحصائيات
        self.stats = {
            'tasks_scheduled': 0,
            'tasks_completed': 0,
            'tasks_failed': 0,
            'avg_wait_time_ms': 0.0,
            'gpu_utilization_history': []
        }
        
        # حلقة المعالجة
        self.processing_task = None
        self.running = False
        
        # تكاليف النماذج التقريبية (MB VRAM, % CPU)
        self.model_costs = {
            Modality.AUDIO: (2048, 15),      # WhisperX
            Modality.VISION: (4096, 20),     # ScreenAI
            Modality.CODE: (3072, 25),       # Tree-sitter + Joern
            Modality.BEHAVIOR: (1024, 10),   # Mamba
            Modality.PREDICTIVE: (2048, 15)  # Predictive models
        }
    
    async def schedule_task(self, modality: Modality, task_func, 
                            priority: Priority = Priority.NORMAL,
                            context: Dict = None) -> str:
        """
        جدولة مهمة معرفية
        """
        task_id = f"{modality.value}_{int(time.time() * 1000)}_{self.stats['tasks_scheduled']}"
        
        # التحقق من الميزانية
        cost_mb, cost_cpu = self.model_costs.get(modality, (1024, 10))
        if not self.budget.can_allocate(cost_mb, cost_cpu):
            logger.warning(f"Insufficient resources for {modality.value}")
            return None
        
        # إضافة إلى قائمة الانتظار
        await self.task_queue.put({
            'id': task_id,
            'modality': modality,
            'task_func': task_func,
            'priority': priority,
            'context': context or {},
            'enqueued_at': time.time()
        })
        
        self.priorities[task_id] = priority
        self.stats['tasks_scheduled'] += 1
        
        return task_id
    
    async def _processing_loop(self):
        """حلقة المعالجة الرئيسية - تستهلك المهام من قائمة الانتظار"""
        while self.running:
            try:
                # الحصول على المهمة التالية (مع أولوية)
                task_item = await self._get_next_task()
                if not task_item:
                    await asyncio.sleep(0.01)
                    continue
                
                task_id = task_item['id']
                modality = task_item['modality']
                cost_mb, cost_cpu = self.model_costs.get(modality, (1024, 10))
                
                # تخصيص الموارد
                if self.budget.allocate(modality.value, cost_mb, cost_cpu):
                    wait_time = (time.time() - task_item['enqueued_at']) * 1000
                    self.stats['avg_wait_time_ms'] = (
                        self.stats['avg_wait_time_ms'] * 0.95 + wait_time * 0.05
                    )
                    
                    # تنفيذ المهمة
                    try:
                        result = await task_item['task_func'](task_item['context'])
                        self.task_results[task_id] = result
                        self.stats['tasks_completed'] += 1
                    except Exception as e:
                        self.task_results[task_id] = {'error': str(e)}
                        self.stats['tasks_failed'] += 1
                        logger.error(f"Task {task_id} failed: {e}")
                    finally:
                        # تحرير الموارد
                        self.budget.release(modality.value)
                else:
                    # إعادة المهمة إلى قائمة الانتظار
                    await self.task_queue.put(task_item)
                    await asyncio.sleep(0.05)
                    
            except Exception as e:
                logger.error(f"Processing loop error: {e}")
                await asyncio.sleep(0.1)
    
    async def _get_next_task(self) -> Optional[Dict]:
        """الحصول على المهمة التالية مع مراعاة الأولوية"""
        # تجميع المهام حسب الأولوية
        priority_buckets = defaultdict(list)
        temp_queue = []
        
        # استخراج جميع المهام من قائمة الانتظار
        while not self.task_queue.empty():
            try:
                task = self.task_queue.get_nowait()
                priority_buckets[self.priorities.get(task['id'], Priority.NORMAL)].append(task)
                temp_queue.append(task)
            except asyncio.QueueEmpty:
                break
        
        # إعادة المهام إلى قائمة الانتظار (إذا لم نأخذ مهمة)
        if not priority_buckets:
            return None
        
        # اختيار المهمة ذات الأولوية القصوى
        for priority in [Priority.CRITICAL, Priority.HIGH, Priority.NORMAL, Priority.LOW, Priority.BACKGROUND]:
            if priority in priority_buckets and priority_buckets[priority]:
                selected = priority_buckets[priority][0]
                # إعادة باقي المهام
                remaining = [t for t in temp_queue if t['id'] != selected['id']]
                for task in remaining:
                    await self.task_queue.put(task)
                return selected
        
        return None
    
    async def get_task_result(self, task_id: str, timeout_ms: int = 5000) -> Optional[Any]:
        """انتظار نتيجة مهمة"""
        start_time = time.time()
        while task_id not in self.task_results:
            if (time.time() - start_time) * 1000 > timeout_ms:
                return None
            await asyncio.sleep(0.01)
        return self.task_results.pop(task_id)
    
    async def start(self):
        """بدء المنسق"""
        self.running = True
        self.processing_task = asyncio.create_task(self._processing_loop())
        logger.info("Cognitive Orchestrator started")
    
    async def get_system_health(self) -> Dict:
        """الحصول على صحة النظام"""
        try:
            gpus = GPUtil.getGPUs()
            gpu_util = gpus[0].load * 100 if gpus else 0
            gpu_memory = gpus[0].memoryUsed if gpus else 0
        except:
            gpu_util = 0
            gpu_memory = 0
        
        self.stats['gpu_utilization_history'].append({
            'timestamp': time.time(),
            'utilization': gpu_util
        })
        if len(self.stats['gpu_utilization_history']) > 100:
            self.stats['gpu_utilization_history'] = self.stats['gpu_utilization_history'][-100:]
        
        return {
            'budget': {
                'available_gpu_mb': self.budget.available_gpu_memory_mb,
                'available_cpu_percent': self.budget.available_cpu_percent,
                'allocated': self.budget.allocated_resources
            },
            'queue_size': self.task_queue.qsize(),
            'running_tasks': len(self.running_tasks),
            'stats': self.stats,
            'gpu_utilization': gpu_util,
            'gpu_memory_mb': gpu_memory
        }
    
    async def shutdown(self):
        """إيقاف المنسق"""
        self.running = False
        if self.processing_task:
            self.processing_task.cancel()
        logger.info("Cognitive Orchestrator shutdown")

"""
Cognitive Fusion Engine V8 - الدمج المعرفي متعدد الحواس
Cross-modal attention, temporal alignment, conflict resolution
"""

import asyncio
import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np
from typing import Dict, List, Optional, Any, Tuple
from dataclasses import dataclass
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


@dataclass
class FusedPerception:
    """الإدراك المدمج بعد الدمج المعرفي"""
    cognitive_state: np.ndarray
    interpretation: str
    confidence: float
    cross_modal_correlations: Dict[str, float]
    detected_anomalies: List[str]
    inferred_mental_state: str  # focused, confused, overloaded, distracted, debugging
    urgency_score: float
    timestamp: float


class CrossModalAttention(nn.Module):
    """
    الانتباه متعدد الحواس - يربط بين الإدراكات المختلفة
    """
    
    def __init__(self, d_model: int = 512, n_heads: int = 8):
        super().__init__()
        self.d_model = d_model
        self.n_heads = n_heads
        
        self.audio_proj = nn.Linear(256, d_model)
        self.vision_proj = nn.Linear(256, d_model)
        self.code_proj = nn.Linear(256, d_model)
        self.behavior_proj = nn.Linear(256, d_model)
        
        self.cross_attention = nn.MultiheadAttention(d_model, n_heads, batch_first=True)
        
    def forward(self, audio_feat, vision_feat, code_feat, behavior_feat):
        # إسقاط جميع الميزات إلى نفس البعد
        audio_emb = self.audio_proj(audio_feat).unsqueeze(1)
        vision_emb = self.vision_proj(vision_feat).unsqueeze(1)
        code_emb = self.code_proj(code_feat).unsqueeze(1)
        behavior_emb = self.behavior_proj(behavior_feat).unsqueeze(1)
        
        # دمج جميع الوسائط
        combined = torch.cat([audio_emb, vision_emb, code_emb, behavior_emb], dim=1)
        
        # الانتباه المتقاطع
        attended, attention_weights = self.cross_attention(combined, combined, combined)
        
        return attended, attention_weights


class CognitiveFusionEngineV8:
    """
    Cognitive Fusion Engine V8 - يدمج جميع الحواس في فهم واحد موحد
    """
    
    def __init__(self):
        self.cross_attention = CrossModalAttention()
        self.temporal_aligner = TemporalAligner()
        self.conflict_resolver = ConflictResolver()
        
        # محول الحالة العقلية
        self.mental_state_classifier = nn.Sequential(
            nn.Linear(512, 256),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(256, 6)  # 6 حالات عقلية
        )
        
        self.mental_states = [
            'focused', 'confused', 'overloaded', 
            'distracted', 'debugging', 'designing'
        ]
        
        self.stats = {
            'fusions': 0,
            'avg_confidence': 0.0,
            'state_distribution': {state: 0 for state in mental_states}
        }
    
    async def fuse(self, audio_result: Optional[Dict] = None,
                         vision_result: Optional[Dict] = None,
                         code_result: Optional[Dict] = None,
                         behavior_result: Optional[Dict] = None) -> FusedPerception:
        """
        دمج جميع الإدراكات في فهم واحد موحد
        """
        # 1. استخراج الميزات من كل وسيط
        audio_feat = self._extract_features(audio_result, default_dim=256) if audio_result else torch.zeros(1, 256)
        vision_feat = self._extract_features(vision_result, default_dim=256) if vision_result else torch.zeros(1, 256)
        code_feat = self._extract_features(code_result, default_dim=256) if code_result else torch.zeros(1, 256)
        behavior_feat = self._extract_features(behavior_result, default_dim=256) if behavior_result else torch.zeros(1, 256)
        
        # 2. المحاذاة الزمنية
        aligned = await self.temporal_aligner.align(
            audio_feat, vision_feat, code_feat, behavior_feat
        )
        
        # 3. الانتباه متعدد الحواس
        with torch.no_grad():
            fused, attention_weights = self.cross_attention(
                aligned['audio'], aligned['vision'], 
                aligned['code'], aligned['behavior']
            )
        
        # 4. تصنيف الحالة العقلية
        mental_state_logits = self.mental_state_classifier(fused[:, 0, :])
        mental_state_idx = torch.argmax(mental_state_logits, dim=1).item()
        mental_state = self.mental_states[mental_state_idx]
        confidence = torch.softmax(mental_state_logits, dim=1)[0, mental_state_idx].item()
        
        # 5. كشف الارتباطات بين الحواس
        correlations = self._compute_cross_correlations(attention_weights)
        
        # 6. كشف الشذوذ (تناقضات بين الحواس)
        anomalies = self._detect_anomalies(audio_result, vision_result, code_result, behavior_result)
        
        # 7. توليد تفسير نصي
        interpretation = self._generate_interpretation(
            mental_state, correlations, anomalies, 
            audio_result, vision_result, code_result, behavior_result
        )
        
        # 8. حساب درجة الاستعجال
        urgency = self._compute_urgency(mental_state, behavior_result, code_result)
        
        # 9. تحديث الإحصائيات
        self.stats['fusions'] += 1
        self.stats['avg_confidence'] = self.stats['avg_confidence'] * 0.95 + confidence * 0.05
        self.stats['state_distribution'][mental_state] += 1
        
        return FusedPerception(
            cognitive_state=fused[:, 0, :].numpy(),
            interpretation=interpretation,
            confidence=confidence,
            cross_modal_correlations=correlations,
            detected_anomalies=anomalies,
            inferred_mental_state=mental_state,
            urgency_score=urgency,
            timestamp=time.time()
        )
    
    def _extract_features(self, result: Dict, default_dim: int = 256) -> torch.Tensor:
        """استخراج الميزات من نتيجة وسيط"""
        if not result:
            return torch.zeros(1, default_dim)
        
        # محاولة استخراج متجه الميزات
        if 'cognitive_vector' in result:
            return torch.tensor(result['cognitive_vector']).float().unsqueeze(0)
        elif 'embedding' in result:
            return torch.tensor(result['embedding']).float().unsqueeze(0)
        else:
            # ميزات افتراضية
            return torch.randn(1, default_dim) * 0.1
    
    def _compute_cross_correlations(self, attention_weights: torch.Tensor) -> Dict[str, float]:
        """حساب الارتباطات بين الحواس المختلفة"""
        if attention_weights is None or attention_weights.numel() == 0:
            return {}
        
        weights = attention_weights[0].cpu().numpy()
        modalities = ['audio', 'vision', 'code', 'behavior']
        
        correlations = {}
        for i, m1 in enumerate(modalities):
            for j, m2 in enumerate(modalities):
                if i < j:
                    corr = float(weights[i, j]) if i < len(weights) and j < len(weights[i]) else 0.0

        return correlations

class MemoryConsolidationStrategy:
    """
    استراتيجية دمج الذاكرة - تمنع الـ catastrophic forgetting
    """
    
    def __init__(self):
        self.consolidation_interval = 3600  # كل ساعة
        self.last_consolidation = time.time()
        self.pending_consolidations: List[MemoryEntry] = []
        
    async def should_consolidate(self) -> bool:
        """هل حان وقت الدمج؟"""
        return (time.time() - self.last_consolidation) > self.consolidation_interval
    
    async def consolidate(self, shadow_memory: List[MemoryEntry], 
                           main_memory: List[MemoryEntry]) -> List[MemoryEntry]:
        """
        دمج الذاكرة الظلية مع الذاكرة الرئيسية
        - يتم الدمج فقط بعد التحقق من الاستقرار
        - يمنع الـ catastrophic forgetting
        """
        if not shadow_memory:
            return main_memory
        
        # 1. التحقق من استقرار التغييرات
        stable_entries = []
        for entry in shadow_memory:
            if await self._is_stable(entry):
                stable_entries.append(entry)
        
        # 2. دمج الإدخالات المستقرة فقط
        for entry in stable_entries:
            existing = self._find_entry(main_memory, entry.id)
            if existing:
                # دمج مع وجود سابق
                existing.importance = max(existing.importance, entry.importance)
                existing.access_count += entry.access_count
                existing.consolidation_count += 1
            else:
                # إضافة جديدة
                main_memory.append(entry)
        
        # 3. تنظيف الذاكرة الظلية
        shadow_memory.clear()
        self.last_consolidation = time.time()
        
        logger.info(f"Memory consolidation: merged {len(stable_entries)} entries")
        
        return main_memory
    
    async def _is_stable(self, entry: MemoryEntry) -> bool:
        """التحقق من استقرار إدخال الذاكرة"""
        # الإدخال مستقر إذا:
        # 1. تم الوصول إليه عدة مرات
        # 2. عمره كافٍ (ليس جديداً جداً)
        # 3. ليس له تناقضات مع ذكريات أخرى
        
        if entry.access_count < 3:
            return False
        
        if time.time() - entry.timestamp < 300:  # أقل من 5 دقائق
            return False
        
        return True
    
    def _find_entry(self, memory: List[MemoryEntry], entry_id: str) -> Optional[MemoryEntry]:
        """البحث عن إدخال في الذاكرة"""
        for entry in memory:
            if entry.id == entry_id:
                return entry
        return None

class RiskLevel(Enum):
    SAFE = "safe"
    LOW = "low"
    MEDIUM = "medium"

@dataclass

    """قرار السلامة"""
    allowed: bool
    risk_level: RiskLevel
    reason: str

class SafetyGovernorV8:
    """
    Safety Governor - يفحص جميع الإجراءات قبل التنفيذ
    """
    
    def __init__(self):
        self.blocked_patterns = {
            'code_injection': [
                r'eval\s*\(', r'exec\s*\(', r'__import__\s*\(',
                r'compile\s*\(', r'globals\(\)', r'locals\(\)'
            ],
            'destructive_ops': [

                r'os\.rmdir', r'pathlib.*\.unlink', r'os\.unlink'
            ],
            'credential_leakage': [
                r'password\s*=', r'secret\s*=', r'api_key\s*=',
                r'token\s*=', r'credential', r'authorization'
            ],
            'unsafe_automation': [
                r'while\s+True:', r'for\s+.*\s+in\s+.*:.*\n.*\n.*\n.*\n.*',  # حلقات عميقة
                r'\.delete', r'\.drop', r'\.truncate', r'\.destroy'
            ]
        }
        
        self.allowed_domains = ['localhost', '127.0.0.1', '*.test', '*.example.com']
        
        self.violation_history: List[Dict] = []
    
    async def review_action(self, action_type: str, action_content: str, 
                            context: Dict) -> SafetyVerdict:
        """
        مراجعة إجراء قبل التنفيذ
        """
        violations = []
        risk_level = RiskLevel.SAFE
        
        # 1. فحص التضمين البرمجي الخطر
        if action_type in ['code_generation', 'code_modification', 'refactoring']:
            code_violations = self._check_code_injection(action_content)
            if code_violations:
                violations.extend(code_violations)
                risk_level = RiskLevel.CRITICAL
        
        # 2. فحص العمليات التدميرية
        destructive_violations = self._check_destructive_ops(action_content)
        if destructive_violations:
            violations.extend(destructive_violations)
            risk_level = max(risk_level, RiskLevel.HIGH, key=lambda x: list(RiskLevel).index(x))
        
        # 3. فحص تسرب البيانات الحساسة
        credential_violations = self._check_credential_leakage(action_content, context)
        if credential_violations:
            violations.extend(credential_violations)
            risk_level = RiskLevel.CRITICAL
        
        # 4. فحص الأتمتة غير الآمنة
        automation_violations = self._check_unsafe_automation(action_content)
        if automation_violations:
            violations.extend(automation_violations)
            risk_level = max(risk_level, RiskLevel.MEDIUM, key=lambda x: list(RiskLevel).index(x))
        
        # 5. فحص السياق (من هو المستخدم، ما هي الصلاحيات)
        context_violations = self._check_context(context)
        if context_violations:
            violations.extend(context_violations)
        
        # 6. تسجيل المخالفة
        if violations:
            self.violation_history.append({
                'timestamp': __import__('time').time(),
                'action_type': action_type,
                'violations': violations,
                'risk_level': risk_level.value
            })
            
            logger.warning(f"Safety violation: {violations[:2]} (risk={risk_level.value})")
        
        # 7. اتخاذ القرار
        if risk_level == RiskLevel.CRITICAL:
            allowed = False
            reason = "CRITICAL: Action blocked due to security risk"
        elif risk_level == RiskLevel.HIGH and not self._is_emergency_override(context):
            allowed = False
            reason = "HIGH: Action requires manual review"
        else:
            allowed = True
            reason = "Action approved" if not violations else "Action approved with warnings"
        
        suggestions = self._generate_suggestions(violations)
        
        return SafetyVerdict(
            allowed=allowed,
            risk_level=risk_level,
            reason=reason,
            suggestions=suggestions
        )
    
    def _check_code_injection(self, code: str) -> List[str]:
        """فحص تضمين برمجي خطر"""
        violations = []
        
        for pattern_name, patterns in self.blocked_patterns.items():
            if pattern_name == 'code_injection':
                for pattern in patterns:
                    if re.search(pattern, code, re.IGNORECASE):
                        violations.append(f"Code injection risk: {pattern}")
        
        # فحص AST للكود
        try:
            tree = ast.parse(code)
            for node in ast.walk(tree):
                if isinstance(node, ast.Call):
                    if isinstance(node.func, ast.Name):
                        if node.func.id in ['eval', 'exec', 'compile', '__import__']:
                            violations.append(f"Dangerous function call: {node.func.id}")
        except:
            pass  # فشل تحليل AST
        
        return violations
    
    def _check_destructive_ops(self, content: str) -> List[str]:
        """فحص العمليات التدميرية"""
        violations = []
        
        for pattern in self.blocked_patterns['destructive_ops']:
            if re.search(pattern, content, re.IGNORECASE):
                violations.append(f"Destructive operation: {pattern}")
        
        return violations
    
    def _check_credential_leakage(self, content: str, context: Dict) -> List[str]:
        """فحص تسرب البيانات الحساسة"""
        violations = []
        
        # فحص الأنماط المباشرة
        for pattern in self.blocked_patterns['credential_leakage']:
            if re.search(pattern, content, re.IGNORECASE):
                violations.append(f"Potential credential leakage: {pattern}")
        
        # فحص القيم المشبوهة (مفاتيح API، كلمات مرور)
        suspicious_values = re.findall(r'[\'"]([A-Za-z0-9]{32,})[\'"]', content)
        if suspicious_values:
            violations.append(f"Suspicious long string (potential key/token): {suspicious_values[0][:10]}...")
        
        return violations
    
    def _check_unsafe_automation(self, content: str) -> List[str]:
        """فحص الأتمتة غير الآمنة"""
        violations = []
        
        # فحص الحلقات اللانهائية المحتملة
        for pattern in self.blocked_patterns['unsafe_automation']:
            if re.search(pattern, content, re.IGNORECASE | re.DOTALL):
                violations.append(f"Unsafe automation pattern: {pattern[:30]}...")
        
        return violations
    
    def _check_context(self, context: Dict) -> List[str]:
        """فحص السياق والصلاحيات"""
        violations = []
        
        # التحقق من صلاحيات المستخدم
        if context.get('privilege_level', 'user') == 'admin':
            # تنبيه للمستخدمين ذوي الصلاحيات العالية
            violations.append("Admin privilege mode - extra caution required")
        
        # التحقق من بيئة الإنتاج
        if context.get('environment') == 'production':
            violations.append("Production environment - all actions are logged")
        
        return violations
    
    def _is_emergency_override(self, context: Dict) -> bool:
        """التحقق من حالة الطوارئ التي تسمح بتجاوز السلامة"""
        return context.get('emergency_override', False) and context.get('authorized_user', False)
    
    def _generate_suggestions(self, violations: List[str]) -> List[str]:
        """توليد اقتراحات بديلة آمنة"""
        suggestions = []
        
        for violation in violations:
            if 'eval' in violation or 'exec' in violation:
                suggestions.append("Use safe alternative: ast.literal_eval() or json.loads()")
            elif 'os.system' in violation:
                suggestions.append("Use subprocess.run() with proper argument sanitization")
            elif 'password' in violation.lower():
                suggestions.append("Use environment variables or secret manager for credentials")
            elif 'while True' in violation:
                suggestions.append("Add a break condition or timeout to prevent infinite loops")
        
        return list(set(suggestions))  # إزالة التكرارات
    
    async def get_stats(self) -> Dict:
        """إحصائيات حاكم السلامة"""
        total = len(self.violation_history)
        if total == 0:
            return {'total_violations': 0, 'violation_rate': 0.0}
        
        critical_count = sum(1 for v in self.violation_history if v['risk_level'] == 'critical')
        
        return {
            'total_violations': total,
            'critical_violations': critical_count,
            'violation_rate': critical_count / total,
            'recent_violations': self.violation_history[-10:]
        }

@dataclass
class UncertainPrediction:
    """توقع مع عدم يقين"""
    value: Any
    mean_confidence: float
    std_uncertainty: float
    epistemic_uncertainty: float  # عدم يقين معرفي (نقص المعرفة)
    aleatoric_uncertainty: float  # عدم يقين عشوائي (ضوضاء في البيانات)
    calibration_score: float      # مدى دقة معايرة الثقة


class UncertaintyEngineV8:
    """
    محرك عدم اليقين - يتتبع ويقدم عدم اليقين في جميع التنبؤات
    """
    
    def __init__(self):
        self.prediction_history: List[UncertainPrediction] = []
        self.calibration_history: List[float] = []
        
    async def quantify(self, predictions: List[Tuple[Any, float]]) -> UncertainPrediction:
        """
        قياس عدم اليقين من تنبؤات متعددة
        """
        if not predictions:
            return UncertainPrediction(
                value=None,
                mean_confidence=0.0,
                std_uncertainty=1.0,
                epistemic_uncertainty=1.0,
                aleatoric_uncertainty=0.0,
                calibration_score=0.5
            )
        
        values = [p[0] for p in predictions]
        confidences = [p[1] for p in predictions]
        
        mean_conf = np.mean(confidences)
        std_conf = np.std(confidences)
        
        # عدم يقين معرفي = تباين بين التنبؤات
        epistemic = std_conf
        
        # عدم يقين عشوائي = متوسط 1 - الثقة
        aleatoric = np.mean([1 - c for c in confidences])
        
        # اختيار القيمة الأكثر شيوعاً (للتنبؤات الفئوية) أو المتوسط (للتنبؤات الرقمية)
        if all(isinstance(v, (int, float)) for v in values):
            final_value = np.mean(values)
        else:
            from collections import Counter
            final_value = Counter(values).most_common(1)[0][0]
        
        # حساب درجة المعايرة
        calibration = await self._compute_calibration()
        
        prediction = UncertainPrediction(
            value=final_value,
            mean_confidence=mean_conf,
            std_uncertainty=std_conf,
            epistemic_uncertainty=epistemic,
            aleatoric_uncertainty=aleatoric,
            calibration_score=calibration
        )
        
        self.prediction_history.append(prediction)
        if len(self.prediction_history) > 1000:
            self.prediction_history = self.prediction_history[-1000:]
        
        return prediction
    
    async def _compute_calibration(self) -> float:
        """
        حساب مدى دقة معايرة الثقة
        كلما اقتربت من 1، كانت الثقة أكثر دقة
        """
        if len(self.prediction_history) < 10:
            return 0.7  # افتراضي
        
        recent = self.prediction_history[-100:]
        
        # تبسيط: مقارنة بين الثقة المتوقعة والنجاح الفعلي
        # في الإنتاج: حساب ECE (Expected Calibration Error)
        
        return 0.85  # تبسيط
    
    async def is_confident(self, prediction: UncertainPrediction, threshold: float = 0.7) -> bool:
        """هل النظام واثق من هذا التنبؤ؟"""
        return (prediction.mean_confidence > threshold and 
                prediction.epistemic_uncertainty < 0.3)
    
    async def should_defer_to_user(self, prediction: UncertainPrediction) -> bool:
        """هل يجب تأجيل القرار للمستخدم؟"""
        return (prediction.epistemic_uncertainty > 0.5 or 
                prediction.mean_confidence < 0.6 or
                prediction.calibration_score < 0.6)
    
    async def get_stats(self) -> Dict:
        """إحصائيات عدم اليقين"""
        if not self.prediction_history:
            return {'total_predictions': 0}

            'avg_epistemic_uncertainty': np.mean([p.epistemic_uncertainty for p in recent]),
            'avg_aleatoric_uncertainty': np.mean([p.aleatoric_uncertainty for p in recent]),
            'avg_calibration': np.mean([p.calibration_score for p in recent]),
            'high_uncertainty_rate': sum(1 for p in recent if p.epistemic_uncertainty > 0.5) / len(recent)
        }

"""

@dataclass
class ReasoningTrace:
    """تتبع مسار التفكير"""
    step_id: str
    step_type: str
    input: Any
    output: Any
    confidence: float
    reasoning: str
    alternatives_considered: List[str]
    timestamp: float
    parent_step_id: Optional[str] = None


class MetaCognitionLayerV8:
    """
    طبقة ما وراء المعرفة - تشرح وتصحح مسارات التفكير
    """
    
    def __init__(self):
        self.reasoning_traces: List[ReasoningTrace] = []
        self.explanations: Dict[str, str] = {}
        
    async def trace_reasoning(self, step_id: str, step_type: str,
                               input_data: Any, output: Any,
                               confidence: float, reasoning: str,
                               alternatives: List[str],
                               parent_step: Optional[str] = None) -> str:
        """
        تتبع مسار التفكير
        """
        trace = ReasoningTrace(
            step_id=step_id,
            step_type=step_type,
            input=input_data,
            output=output,
            confidence=confidence,
            reasoning=reasoning,
            alternatives_considered=alternatives,
            timestamp=time.time(),
            parent_step_id=parent_step
        )
        
        self.reasoning_traces.append(trace)
        
        # توليد تفسير
        explanation = await self._generate_explanation(trace)
        self.explanations[step_id] = explanation
        
        return step_id
    
    async def explain(self, step_id: str) -> str:
        """
        شرح سبب اتخاذ مسار تفكير معين
        """
        if step_id in self.explanations:
            return self.explanations[step_id]
        
        trace = self._find_trace(step_id)
        if not trace:
            return "No reasoning trace found"
        
        explanation = await self._generate_explanation(trace)
        self.explanations[step_id] = explanation
        
        return explanation
    
    async def _generate_explanation(self, trace: ReasoningTrace) -> str:
        """
        توليد تفسير لمسار التفكير
        """
        explanation_parts = [
            f"I chose to {trace.step_type} because:",
            f"  - Input analysis showed: {str(trace.input)[:100]}",
            f"  - I considered alternatives: {', '.join(trace.alternatives_considered[:3])}",
            f"  - My reasoning: {trace.reasoning[:200]}",
            f"  - Confidence level: {trace.confidence:.2f}"
        ]
        
        if trace.parent_step_id:
            parent_trace = self._find_trace(trace.parent_step_id)
            if parent_trace:
                explanation_parts.append(f"  - This decision was influenced by previous step: {parent_trace.step_type}")
        
        return "\n".join(explanation_parts)
    
    async def debug_cognition(self, step_id: str) -> Dict:
        """
        تصحيح أخطاء التفكير - تحليل لماذا فكر بشكل خاطئ
        """
        trace = self._find_trace(step_id)
        if not trace:
            return {'error': 'Trace not found'}
        
        debug_info = {
            'step_id': step_id,
            'step_type': trace.step_type,
            'confidence': trace.confidence,
            'is_confident': trace.confidence > 0.7,
            'alternative_paths': trace.alternatives_considered,
            'potential_issues': []
        }
        
        # تحليل المشاكل المحتملة
        if trace.confidence < 0.5:
            debug_info['potential_issues'].append("Low confidence - insufficient evidence")
        
        if len(trace.alternatives_considered) == 0:
            debug_info['potential_issues'].append("No alternatives considered - possible tunnel vision")
        
        if 'error' in str(trace.output).lower():
            debug_info['potential_issues'].append("Output contains error - reasoning may be flawed")
        
        return debug_info
    
    async def get_reasoning_graph(self, root_step_id: str) -> Dict:
        """
        الحصول على الرسم البياني الكامل لمسار التفكير
        """
        graph = {
            'nodes': [],
            'edges': []
        }
        
        def traverse(step_id: str):
            trace = self._find_trace(step_id)
            if not trace:
                return
            
            graph['nodes'].append({
                'id': trace.step_id,
                'type': trace.step_type,
                'confidence': trace.confidence,
                'timestamp': trace.timestamp
            })
            
            if trace.parent_step_id:
                graph['edges'].append({
                    'from': trace.parent_step_id,
                    'to': trace.step_id
                })
        
        traverse(root_step_id)
        
        return graph
    
    def _find_trace(self, step_id: str) -> Optional[ReasoningTrace]:
        """البحث عن تتبع التفكير"""
        for trace in self.reasoning_traces:
            if trace.step_id == step_id:
                return trace
        return None
    
    async def get_stats(self) -> Dict:
        """إحصائيات ما وراء المعرفة"""
        return {
            'total_traces': len(self.reasoning_traces),
            'total_explanations': len(self.explanations),
            'recent_traces': [
                {'id': t.step_id, 'type': t.step_type, 'confidence': t.confidence}
                for t in self.reasoning_traces[-10:]
            ]
        }

                'epistemic': uncertainty.epistemic_uncertainty,
                'aleatoric': uncertainty.aleatoric_uncertainty,
                'is_confident': await self.uncertainty_engine.is_confident(uncertainty)
            },
            'latency_ms': latency_ms
        }
    
    async def act(self, action_type: str, action_content: str, context: Dict = None) -> Dict:
        """
        تنفيذ إجراء مع فحص السلامة
        """
        # 1. مراجعة السلامة
        verdict = await self.safety_governor.review_action(
            action_type, action_content, context or {}
        )
        
        if not verdict.allowed:
            return {
                'executed': False,
                'blocked': True,
                'reason': verdict.reason,
                'suggestions': verdict.suggestions
            }
        
        # 2. تنفيذ الإجراء
        # في الإنتاج: التنفيذ الفعلي
        return {
            'executed': True,
            'blocked': False,
            'risk_level': verdict.risk_level.value,
            'warnings': verdict.suggestions if verdict.risk_level != RiskLevel.SAFE else []
        }
    
    async def explain_reasoning(self, step_id: str) -> str:
        """شرح مسار التفكير"""
        return await self.meta_cognition.explain(step_id)
    
    async def get_system_status(self) -> Dict:
        """الحالة الكاملة للنظام"""
        return {
            'user_id': self.user_id,
            'is_running': self.is_running,
            'stats': self.stats,
            'orchestrator': await self.orchestrator.get_system_health(),
            'fusion': await self.fusion_engine.get_stats(),
            'memory': await self.memory.get_stats(),
            'safety': await self.safety_governor.get_stats(),
            'uncertainty': await self.uncertainty_engine.get_stats(),
            'meta_cognition': await self.meta_cognition.get_stats()
        }
    
    async def shutdown(self):
        """إيقاف النظام"""
        self.is_running = False
        await self.orchestrator.shutdown()
        await self.memory.shutdown()
        logger.info(f"Cognitive Twin V8 shutdown for user {self.user_id}")
