# backend/interface/focus_manager.py
import time
from collections import deque
import numpy as np

class FocusManager:
    def __init__(self):
        self.focus_history = deque(maxlen=1000)
        self.distractions = []
        self.current_focus_score = 0.8
        self.last_interaction = time.time()
        self.interaction_times = []
        
        # عتبات التشتت
        self.distraction_thresholds = {
            'eye_movement': 0.7,      # حركة عين عالية
            'tab_switches': 5,         # تبديل تبويبات كثير
            'pause_time': 30,          # توقف طويل
            'scroll_speed': 0.8,       # تمرير سريع
            'typing_errors': 0.3       # أخطاء كتابة كثيرة
        }
        
    async def update_focus(self, metrics: Dict) -> float:
        """تحديث درجة التركيز بناءً على المقاييس"""
        # حساب درجة التركيز
        focus_score = 1.0
        
        if metrics.get('eye_movement', 0) > self.distraction_thresholds['eye_movement']:
            focus_score -= 0.2
        
        if metrics.get('tab_switches', 0) > self.distraction_thresholds['tab_switches']:
            focus_score -= 0.15
        
        pause_time = time.time() - self.last_interaction
        if pause_time > self.distraction_thresholds['pause_time']:
            focus_score -= min(0.5, pause_time / 60)
        
        if metrics.get('typing_errors', 0) > self.distraction_thresholds['typing_errors']:
            focus_score -= 0.1
        
        self.current_focus_score = max(0.0, min(1.0, focus_score))
        self.focus_history.append({
            'timestamp': time.time(),
            'score': self.current_focus_score,
            'metrics': metrics
        })
        
        self.last_interaction = time.time()
        self.interaction_times.append(self.last_interaction)
        
        # التحقق من الحاجة للتدخل
        if self.current_focus_score < 0.5:
            await self._intervene()
        
        return self.current_focus_score
    
    async def _intervene(self) -> None:
        """التدخل لاستعادة التركيز"""
        interventions = [
            {'type': 'suggestion', 'message': 'It seems you are distracted. Would you like to take a short break?'},
            {'type': 'summary', 'message': 'Here is a summary of what you have done so far...'},
            {'type': 'focus_mode', 'action': 'minimize_distractions'},
            {'type': 'question', 'message': 'What is the main problem you are trying to solve?'}
        ]
        
        # اختيار التدخل المناسب
        import random
        intervention = random.choice(interventions)
        
        await self._apply_intervention(intervention)
    
    async def _apply_intervention(self, intervention: Dict) -> None:
        """تطبيق التدخل"""
        # تسجيل التدخل
        self.distractions.append({
            'timestamp': time.time(),
            'intervention': intervention,
            'focus_score': self.current_focus_score
        })
        
        # إرسال التدخل إلى الواجهة
        # يتم التعامل مع هذا في طبقة الواجهة
        pass
    
    def get_focus_report(self) -> Dict:
        """الحصول على تقرير التركيز"""
        if len(self.focus_history) == 0:
            return {'average_focus': 0.8, 'distractions': 0, 'recommendations': []}
        
        scores = [f['score'] for f in self.focus_history]
        
        return {
            'average_focus': np.mean(scores),
            'min_focus': np.min(scores),
            'max_focus': np.max(scores),
            'distractions_count': len(self.distractions),
            'total_focus_time': sum(self.interaction_times[i+1] - self.interaction_times[i] 
                                    for i in range(len(self.interaction_times)-1)),
            'recommendations': self._generate_recommendations()
        }
    
    def _generate_recommendations(self) -> List[str]:
        """توليد توصيات لتحسين التركيز"""
        recommendations = []
        
        if self.current_focus_score < 0.5:
            recommendations.append("Take a 5-minute break to reset your focus")
        
        if len(self.distractions) > 10:
            recommendations.append("Consider closing unnecessary tabs and applications")
        
        if self.current_focus_score > 0.9:
            recommendations.append("Great focus! Keep up the momentum")
        
        return recommendations

# backend/interface/topic_adaptation.py
import asyncio
from typing import Dict, List

class TopicAdaptationEngine:
    def __init__(self):
        self.current_topic = None
        self.topic_history = []
        self.topic_tools = {
            'lowlevel': {
                'tools': ['memory_viewer', 'performance_profiler', 'assembly_viewer'],
                'panels': ['registers', 'stack', 'heap', 'disassembly'],
                'shortcuts': ['Ctrl+M', 'Ctrl+P', 'Ctrl+A'],
                'snippets': ['dma_transfer', 'interrupt_handler', 'lock_free_queue']
            },
            'networking': {
                'tools': ['packet_analyzer', 'protocol_diagram', 'socket_monitor'],
                'panels': ['packets', 'protocols', 'connections', 'latency'],
                'shortcuts': ['Ctrl+P', 'Ctrl+D', 'Ctrl+C'],
                'snippets': ['socket_server', 'http_client', 'tcp_listener']
            },
            'ml': {
                'tools': ['tensor_visualizer', 'training_monitor', 'model_graph'],
                'panels': ['layers', 'parameters', 'metrics', 'embeddings'],
                'shortcuts': ['Ctrl+T', 'Ctrl+M', 'Ctrl+G'],
                'snippets': ['neural_network', 'training_loop', 'data_loader']
            },
            'web': {
                'tools': ['browser_preview', 'console', 'network_inspector'],
                'panels': ['elements', 'styles', 'scripts', 'storage'],
                'shortcuts': ['Ctrl+Shift+I', 'Ctrl+Shift+C', 'Ctrl+Shift+J'],
                'snippets': ['fetch_api', 'react_component', 'express_server']
            }
        }
        
    async def detect_topic(self, code: str, speech: str, screen: np.ndarray) -> str:
        """كشف الموضوع الحالي من المدخلات"""
        # تحليل الكود
        code_topic = await self._analyze_code_topic(code)
        
        # تحليل الكلام
        speech_topic = await self._analyze_speech_topic(speech) if speech else None
        
        # تحليل الشاشة
        screen_topic = await self._analyze_screen_topic(screen) if screen is not None else None
        
        # دمج النتائج
        topics = [t for t in [code_topic, speech_topic, screen_topic] if t]
        if not topics:
            return 'general'
        
        # اختيار الموضوع الأكثر تكراراً
        from collections import Counter
        topic_counts = Counter(topics)
        detected_topic = topic_counts.most_common(1)[0][0]
        
        # تحديث التاريخ
        if detected_topic != self.current_topic:
            self.topic_history.append({
                'timestamp': time.time(),
                'from': self.current_topic,
                'to': detected_topic,
                'confidence': topic_counts[detected_topic] / len(topics)
            })
            self.current_topic = detected_topic
            await self._adapt_environment()
        
        return detected_topic
    
    async def _adapt_environment(self) -> None:
        """تكيف البيئة مع الموضوع الجديد"""
        if not self.current_topic:
            return
        
        tools = self.topic_tools.get(self.current_topic, self.topic_tools['lowlevel'])
        
        # تحديث الأدوات
        await self._update_tools(tools['tools'])
        
        # تحديث الألواح
        await self._update_panels(tools['panels'])
        
        # تحديث الاختصارات
        await self._update_shortcuts(tools['shortcuts'])
        
        # تحديث الـ snippets
        await self._update_snippets(tools['snippets'])
        
        # إشعار المستخدم
        await self._notify_topic_change()
    
    async def _analyze_code_topic(self, code: str) -> str:
        """تحليل الكود لتحديد الموضوع"""
        code_lower = code.lower()
        
        topics = {
            'lowlevel': ['dma', 'nic', 'interrupt', 'kernel', 'driver', 'mmap', 'ioctl', 'asm'],
            'networking': ['socket', 'tcp', 'udp', 'http', 'server', 'client', 'packet', 'protocol'],
            'ml': ['tensor', 'neural', 'layer', 'activation', 'gradient', 'loss', 'accuracy', 'model'],
            'web': ['html', 'css', 'javascript', 'react', 'vue', 'angular', 'node', 'express']
        }
        
        scores = {}
        for topic, keywords in topics.items():
            score = sum(1 for kw in keywords if kw in code_lower)
            scores[topic] = score
        
        if max(scores.values()) == 0:
            return 'general'
        
        return max(scores, key=scores.get)
    
    async def _analyze_speech_topic(self, speech: str) -> str:
        """تحليل الكلام لتحديد الموضوع"""
        speech_lower = speech.lower()
        
        topics = {
            'lowlevel': ['memory', 'dma', 'nic', 'kernel', 'driver', 'hardware', 'low level'],
            'networking': ['network', 'packet', 'protocol', 'socket', 'tcp', 'http', 'server'],
            'ml': ['neural', 'network', 'training', 'model', 'data', 'accuracy', 'loss'],
            'web': ['web', 'html', 'css', 'javascript', 'react', 'api', 'frontend', 'backend']
        }
        
        scores = {}
        for topic, keywords in topics.items():
            score = sum(1 for kw in keywords if kw in speech_lower)
            scores[topic] = score
        
        if max(scores.values()) == 0:
            return 'general'
        
        return max(scores, key=scores.get)
