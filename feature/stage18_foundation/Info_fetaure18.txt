# backend/guidance/cognitive_guidance_layer.py
import asyncio
from typing import Dict, List
from datetime import datetime
from collections import deque
import re

class CognitiveGuidanceLayer:
    """
    الإرشاد المعرفي الذكي والتوعية التكيفية
    رفيق معرفي هندسي يعمل بإذن المستخدم لمساعدته على التطور
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        self.user_consent = True  # موافقة المستخدم
        self.activity_monitor = ActivityMonitor()
        self.cognitive_coach = CognitiveCoach()
        self.time_awareness = TimeAwareness()
        self.code_sensitivity = AdaptiveCodeSensitivity()
        self.personal_memory = PersonalEngineeringMemory(user_id)
        self.enhancement_engine = HumanEnhancementEngine()
        
        # سجل النشاط (للشفافية)
        self.activity_log = []
        
    async def process_activity(self, activity: Dict) -> Dict:
        """
        معالجة نشاط المستخدم وتقديم الإرشاد المناسب
        """
        # 1. تسجيل النشاط (بموافقة المستخدم)
        if self.user_consent:
            await self.activity_monitor.record(activity)
            await self._log_activity('activity_recorded', activity)
        
        # 2. تحليل النشاط (بدون تخزين معلومات حساسة)
        analysis = await self.activity_monitor.analyze(activity)
        
        # 3. توليد الإرشاد المعرفي
        guidance = await self.cognitive_coach.generate_guidance(analysis)
        
        # 4. التحقق من الوعي الزمني
        time_alert = await self.time_awareness.check_fatigue(analysis)
        
        # 5. التحقق من حساسية الكود
        code_insights = await self.code_sensitivity.analyze(activity.get('code', ''))
        
        # 6. تحديث الذاكرة الشخصية
        await self.personal_memory.update(activity, analysis, guidance)
        
        # 7. تقديم التوصيات (بدون إزعاج)
        recommendations = await self._generate_recommendations(
            analysis, guidance, time_alert, code_insights
        )
        
        return {
            'guidance': guidance,
            'time_alert': time_alert,
            'code_insights': code_insights,
            'recommendations': recommendations,
            'timestamp': datetime.now().isoformat()
        }
    
    async def _generate_recommendations(self, analysis: Dict, guidance: Dict, 
                                        time_alert: Dict, code_insights: Dict) -> List:
        """
        توليد توصيات ذكية (تركز على التطوير وليس النقد)
        """
        recommendations = []
        
        # توصيات لتحسين التفكير
        if guidance.get('thinking_improvement'):
            recommendations.append({
                'type': 'cognitive',
                'message': guidance['thinking_improvement'],
                'priority': 'medium'
            })
        
        # توصيات للراحة
        if time_alert.get('needs_break'):
            recommendations.append({
                'type': 'wellbeing',
                'message': 'يبدو أنك بحاجة إلى استراحة قصيرة. العقل المرتاح يفكر بشكل أفضل.',
                'priority': 'high'
            })
        
        # توصيات أمنية
        if code_insights.get('security_risks'):
            recommendations.append({
                'type': 'security',
                'message': f'اكتشفت مشكلة أمنية محتملة: {code_insights["security_risks"][0]}',
                'priority': 'high'
            })
        
        return recommendations


class ActivityMonitor:
    """
    مراقب النشاط - يحلل ما يفعله المستخدم (بموافقته)
    """
    
    def __init__(self):
        self.activity_history = deque(maxlen=1000)
        
    async def record(self, activity: Dict):
        """
        تسجيل نشاط المستخدم (بدون معلومات حساسة)
        """
        # إزالة أي معلومات حساسة محتملة
        safe_activity = self._sanitize_activity(activity)
        self.activity_history.append(safe_activity)
    
    async def analyze(self, activity: Dict) -> Dict:
        """
        تحليل النشاط لاستخراج أنماط مفيدة
        """
        # تحليل الأخطاء المتكررة
        error_patterns = self._detect_error_patterns()
        
        # تحليل أنماط التركيز
        focus_patterns = self._detect_focus_patterns()
        
        # تحليل سرعة العمل
        work_speed = self._calculate_work_speed()
        
        return {
            'error_patterns': error_patterns,
            'focus_patterns': focus_patterns,
            'work_speed': work_speed,
            'current_activity_type': activity.get('type', 'coding')
        }
    
    def _sanitize_activity(self, activity: Dict) -> Dict:
        """
        إزالة المعلومات الحساسة من النشاط
        """
        # لا نخزن محتوى الكود الكامل، فقط أنماطه
        if 'code' in activity:
            activity['code_hash'] = hash(activity['code'][:100])
            activity['code_length'] = len(activity['code'])
            del activity['code']
        
        return activity


class CognitiveCoach:
    """
    المدرب المعرفي - يدفع المستخدم للتفكير بدلاً من إعطاء الحل مباشرة
    """
    
    async def generate_guidance(self, analysis: Dict) -> Dict:
        """
        توليد إرشاد معرفي يشجع على التفكير
        """
        error_patterns = analysis.get('error_patterns', [])
        
        if not error_patterns:
            return {'thinking_improvement': None}
        
        # إرشاد يعتمد على السياق
        if 'race_condition' in str(error_patterns).lower():
            return {
                'thinking_improvement': 'هناك احتمال لـ race condition في الكود. '
                                       'راجع ترتيب الوصول إلى الموارد المشتركة.',
                'hint': 'ما الذي قد يحدث إذا وصلت خيطان إلى نفس المورد في نفس الوقت؟'
            }
        
        if 'memory_leak' in str(error_patterns).lower():
            return {
                'thinking_improvement': 'لاحظت وجود تسرب محتمل في الذاكرة. '
                                       'تأكد من تحرير الموارد بعد استخدامها.',
                'hint': 'هل جميع الـ resources التي فتحتها تم إغلاقها؟'
            }
        
        if 'security' in str(error_patterns).lower():
            return {
                'thinking_improvement': 'يوجد مشكلة أمنية محتملة في هذا الجزء.',
                'hint': 'هل يتم التحقق من صحة المدخلات بشكل كافٍ؟'
            }
        
        return {'thinking_improvement': None}


class AdaptiveCodeSensitivity:
    """
    الحساسية التكيفية للكود - تفهم النية وتكتشف المشكلات
    """
    
    async def analyze(self, code: str) -> Dict:
        """
        تحليل الكود واكتشاف المشكلات المحتملة
        """
        if not code:
            return {}
        
        security_risks = []
        performance_issues = []
        
        # فحص ثغرات أمنية شائعة
        if 'eval(' in code or 'exec(' in code:
            security_risks.append('استخدام eval/exec قد يكون خطراً أمنياً')
        
        if 'password' in code.lower() and 'hardcoded' in code.lower():
            security_risks.append('كلمة مرور مكتوبة بشكل ثابت في الكود')
        
        # فحص مشكلات أداء
        if 'for' in code and 'range(len(' in code:
            performance_issues.append('استخدام range(len()) يمكن تحسينه باستخدام enumerate')
        
        if 'while True' in code and 'break' not in code:
            performance_issues.append('حلقة while True قد تسبب تجمد البرنامج')
        
        return {
            'security_risks': security_risks,
            'performance_issues': performance_issues,
            'quality_score': self._calculate_quality_score(code)
        }


class PersonalEngineeringMemory:
    """
    الذاكرة الشخصية الهندسية - تخزن نمط عمل المستخدم لتطوير النسخة الرقمية
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        self.knowledge_graph = Neo4jConnection()
        self.vector_memory = QdrantConnection()
        
    async def update(self, activity: Dict, analysis: Dict, guidance: Dict):
        """
        تحديث الذاكرة الشخصية بالتجارب الجديدة
        """
        # تخزين نمط حل المشكلات
        await self._store_problem_solving_pattern(activity, analysis)
        
        # تخزين الأخطاء المتكررة
        if analysis.get('error_patterns'):
            await self._store_error_patterns(analysis['error_patterns'])
        
        # تخزين نجاحات المستخدم
        if guidance.get('success'):
            await self._store_success_pattern(activity)
        
        # تحديث المتجه الشخصي
        await self._update_personal_vector(activity, analysis)


class HumanEnhancementEngine:

    """

        """
        تحسين القدرات المعرفية للمستخدم
        """
        # تحديد نقاط القوة الحالية
        strengths = user_profile.get('strengths', [])

        growth_areas = user_profile.get('growth_areas', [])
        
        # بناء خطة تطوير مخصصة
        development_plan = {
            'focus_areas': growth_areas[:3],

            'estimated_improvement': '20-30% in 4 weeks',

        }

# backend/part19_v19/cognitive_state_model.py
"""
Cognitive State Model V19 - يمثل الحالة العقلية للمستخدم لحظياً
يقيس: التركيز، الحمل المعرفي، المهارة، الإرهاق، الارتباك
"""

import time
import numpy as np
from typing import Dict, List, Optional
from dataclasses import dataclass, field
from collections import deque
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


@dataclass
class CognitiveState:
    """الحالة العقلية للمستخدم"""
    focus_level: float = 1.0      # 0-1: مستوى التركيز
    cognitive_load: float = 0.0   # 0-1: الحمل المعرفي
    skill_level: float = 0.5      # 0-1: مستوى المهارة في المهمة الحالية
    fatigue: float = 0.0          # 0-1: مستوى الإرهاق
    confusion_score: float = 0.0  # 0-1: درجة الارتباك
    timestamp: float = field(default_factory=time.time)
    
    def to_dict(self) -> Dict:
        return {
            'focus_level': round(self.focus_level, 2),
            'cognitive_load': round(self.cognitive_load, 2),
            'skill_level': round(self.skill_level, 2),
            'fatigue': round(self.fatigue, 2),
            'confusion_score': round(self.confusion_score, 2),
            'timestamp': self.timestamp
        }


class CognitiveStateModel:
    """
    Cognitive State Model V19 - نموذج الحالة العقلية للمستخدم
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        self.current_state = CognitiveState()
        self.state_history: List[CognitiveState] = []
        
        # مقاييس السلوك
        self.typing_speed_history: deque = deque(maxlen=100)
        self.error_rate_history: deque = deque(maxlen=100)
        self.navigation_history: deque = deque(maxlen=100)
        self.pause_history: deque = deque(maxlen=100)
        
        self.stats = {
            'updates': 0,
            'avg_focus': 0.0,
            'avg_load': 0.0
        }
    
    async def update(self, activity: Dict) -> CognitiveState:
        """
        تحديث الحالة العقلية بناءً على نشاط المستخدم
        """
        # استخراج المقاييس من النشاط
        typing_speed = activity.get('typing_speed', 30)  # كلمة في الدقيقة
        error_rate = activity.get('error_rate', 0.1)     # نسبة الأخطاء
        navigation_count = activity.get('navigation_count', 0)
        pause_duration = activity.get('pause_duration', 0)
        task_complexity = activity.get('task_complexity', 0.5)
        
        # تسجيل التاريخ
        self.typing_speed_history.append(typing_speed)
        self.error_rate_history.append(error_rate)
        self.navigation_history.append(navigation_count)
        self.pause_history.append(pause_duration)
        
        # حساب الحالة الجديدة
        focus = self._calculate_focus(typing_speed, error_rate, navigation_count)
        cognitive_load = self._calculate_cognitive_load(task_complexity, error_rate, navigation_count)
        skill_level = self._calculate_skill_level(typing_speed, error_rate)
        fatigue = self._calculate_fatigue(pause_duration, len(self.pause_history))
        confusion = self._calculate_confusion(error_rate, typing_speed, navigation_count)
        
        self.current_state = CognitiveState(
            focus_level=focus,
            cognitive_load=cognitive_load,
            skill_level=skill_level,
            fatigue=fatigue,
            confusion_score=confusion,
            timestamp=time.time()
        )
        
        # تخزين التاريخ
        self.state_history.append(self.current_state)
        if len(self.state_history) > 1000:
            self.state_history = self.state_history[-1000:]
        
        # تحديث الإحصائيات
        self.stats['updates'] += 1
        self.stats['avg_focus'] = self.stats['avg_focus'] * 0.95 + focus * 0.05
        self.stats['avg_load'] = self.stats['avg_load'] * 0.95 + cognitive_load * 0.05
        
        return self.current_state
    
    def _calculate_focus(self, typing_speed: float, error_rate: float, 
                         navigation_count: int) -> float:
        """
        حساب مستوى التركيز
        سرعة كتابة عالية + أخطاء قليلة + تنقل قليل = تركيز عالٍ
        """
        speed_factor = min(1.0, typing_speed / 60)
        error_factor = 1.0 - min(0.5, error_rate)
        nav_factor = 1.0 - min(0.3, navigation_count / 20)
        
        focus = (speed_factor * 0.3 + error_factor * 0.4 + nav_factor * 0.3)
        return max(0.0, min(1.0, focus))
    
    def _calculate_cognitive_load(self, complexity: float, error_rate: float,
                                   navigation_count: int) -> float:
        """
        حساب الحمل المعرفي
        تعقيد عالٍ + أخطاء كثيرة + تنقل كثير = حمل معرفي عالٍ
        """
        load = (complexity * 0.4 + error_rate * 0.3 + min(1.0, navigation_count / 15) * 0.3)
        return max(0.0, min(1.0, load))
    
    def _calculate_skill_level(self, typing_speed: float, error_rate: float) -> float:
        """
        حساب مستوى المهارة بناءً على سرعة الكتابة ودقتها
        """
        if not self.typing_speed_history:
            return 0.5
        
        avg_speed = np.mean(self.typing_speed_history)
        avg_errors = np.mean(self.error_rate_history)
        
        speed_skill = min(1.0, avg_speed / 60)
        accuracy_skill = 1.0 - min(0.5, avg_errors)
        
        return (speed_skill + accuracy_skill) / 2
    
    def _calculate_fatigue(self, pause_duration: float, history_length: int) -> float:
        """
        حساب مستوى الإرهاق بناءً على فترات التوقف
        """
        if history_length < 10:
            return 0.0
        
        avg_pause = np.mean(list(self.pause_history)[-20:])
        fatigue = min(1.0, avg_pause / 120)  # 2 دقيقة = إرهاق عالٍ
        
        return fatigue
    
    def _calculate_confusion(self, error_rate: float, typing_speed: float,
                              navigation_count: int) -> float:
        """
        حساب درجة الارتباك
        أخطاء كثيرة + سرعة بطيئة + تنقل كثير = ارتباك
        """
        error_factor = min(0.5, error_rate) * 2
        speed_factor = max(0, 1.0 - typing_speed / 40)
        nav_factor = min(0.4, navigation_count / 10)
        
        confusion = (error_factor * 0.4 + speed_factor * 0.3 + nav_factor * 0.3)
        return max(0.0, min(1.0, confusion))
    
    async def needs_intervention(self) -> bool:
        """
        هل يحتاج المستخدم للتدخل؟
        """
        return (self.current_state.confusion_score > 0.7 or
                self.current_state.cognitive_load > 0.8 or
                self.current_state.fatigue > 0.7)
    
    async def get_intervention_type(self) -> str:
        """
        تحديد نوع التدخل المناسب
        """
        if self.current_state.confusion_score > 0.7:
            return "hint"
        elif self.current_state.cognitive_load > 0.8:
            return "simplify"
        elif self.current_state.fatigue > 0.7:
            return "break"
        elif self.current_state.focus_level < 0.4:
            return "focus_alert"
        else:
            return "none"
    
    async def get_state(self) -> Dict:
        """الحصول على الحالة الحالية"""
        return self.current_state.to_dict()
    
    async def get_stats(self) -> Dict:
        """إحصائيات النموذج"""
        return {
            **self.stats,
            'history_length': len(self.state_history),
            'current_state': self.current_state.to_dict()
        }

# backend/part19_v19/event_capture_layer.py
"""
Event Capture Layer V19 - يجمع الأحداث من IDE/Web/API
يدعم: VSCode Extension, Browser Bridge, API Hooks
"""

import asyncio
import time
import uuid
from typing import Dict, List, Optional, Any
from dataclasses import dataclass, field
from enum import Enum
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


class EventType(Enum):
    CODE_CHANGE = "code_change"
    CURSOR_MOVE = "cursor_move"
    FILE_OPEN = "file_open"
    FILE_SAVE = "file_save"
    COMPILE = "compile"
    ERROR = "error"
    DEBUG = "debug"
    TEST_RUN = "test_run"
    SEARCH = "search"
    NAVIGATION = "navigation"


@dataclass
class Event:
    """حدث من المستخدم"""
    id: str
    type: EventType
    data: Dict
    timestamp: float
    user_id: str
    session_id: str
    source: str  # 'vscode', 'web', 'api'


class EventCaptureLayer:
    """
    Event Capture Layer V19 - طبقة التقاط الأحداث
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        self.session_id = str(uuid.uuid4())
        self.events: List[Event] = []
        self.max_events = 10000
        
        # قائمة انتظار الأحداث
        self.event_queue: asyncio.Queue = asyncio.Queue()
        self.is_running = False
        self.processor_task = None
        
        self.stats = {
            'events_captured': 0,
            'events_processed': 0,
            'by_type': {t.value: 0 for t in EventType}
        }
    
    async def capture_from_vscode(self, payload: Dict) -> Optional[Event]:
        """
        التقاط حدث من VSCode Extension
        """
        return await self._capture_event(EventType(payload.get('type', 'code_change')), 
                                          payload.get('data', {}), 
                                          'vscode')
    
    async def capture_from_web(self, payload: Dict) -> Optional[Event]:
        """
        التقاط حدث من Web Interface
        """
        return await self._capture_event(EventType(payload.get('type', 'code_change')),
                                          payload.get('data', {}),
                                          'web')
    
    async def capture_from_api(self, payload: Dict) -> Optional[Event]:
        """
        التقاط حدث من API
        """
        return await self._capture_event(EventType(payload.get('type', 'code_change')),
                                          payload.get('data', {}),
                                          'api')
    
    async def _capture_event(self, event_type: EventType, data: Dict, 
                              source: str) -> Event:
        """
        التقاط حدث داخلياً
        """
        event = Event(
            id=f"evt_{int(time.time()*1000)}_{self.stats['events_captured']}",
            type=event_type,
            data=self._sanitize_data(data),
            timestamp=time.time(),
            user_id=self.user_id,
            session_id=self.session_id,
            source=source
        )
        
        self.events.append(event)
        if len(self.events) > self.max_events:
            self.events = self.events[-self.max_events:]
        
        await self.event_queue.put(event)
        
        self.stats['events_captured'] += 1
        self.stats['by_type'][event_type.value] += 1
        
        return event
    
    def _sanitize_data(self, data: Dict) -> Dict:
        """
        تنظيف البيانات من المعلومات الحساسة
        """
        # لا نخزن محتوى الكود الكامل
        if 'code' in data and len(data['code']) > 500:
            data['code_hash'] = hash(data['code'][:500])
            data['code_length'] = len(data['code'])
            del data['code']
        
        # لا نخزن مسارات كاملة
        if 'file_path' in data:
            data['file_name'] = data['file_path'].split('/')[-1]
            del data['file_path']
        
        return data
    
    async def get_events(self, event_type: Optional[EventType] = None,
                         since: Optional[float] = None,
                         limit: int = 100) -> List[Event]:
        """
        الحصول على الأحداث المخزنة
        """
        result = self.events
        
        if event_type:
            result = [e for e in result if e.type == event_type]
        
        if since:
            result = [e for e in result if e.timestamp >= since]
        
        return result[-limit:]
    
    async def get_recent_activity(self, seconds: int = 60) -> Dict:
        """
        الحصول على النشاط الأخير للمستخدم
        """
        since = time.time() - seconds
        recent = await self.get_events(since=since)
        
        if not recent:
            return {'has_activity': False, 'events_count': 0}
        
        # تحليل الأنشطة
        code_changes = [e for e in recent if e.type == EventType.CODE_CHANGE]
        errors = [e for e in recent if e.type == EventType.ERROR]
        navigations = [e for e in recent if e.type == EventType.NAVIGATION]
        
        return {
            'has_activity': True,
            'events_count': len(recent),
            'code_changes': len(code_changes),
            'errors': len(errors),
            'navigations': len(navigations),
            'error_rate': len(errors) / max(1, len(recent)),
            'activity_intensity': min(1.0, len(recent) / 100)
        }
    
    async def start(self):
        """بدء معالجة الأحداث"""
        self.is_running = True
        self.processor_task = asyncio.create_task(self._process_loop())
        logger.info("EventCaptureLayer started")
    
    async def _process_loop(self):
        """حلقة معالجة الأحداث"""
        while self.is_running:
            try:
                event = await self.event_queue.get()
                # معالجة الحدث (في الإنتاج: إرسال إلى Kafka)
                self.stats['events_processed'] += 1
            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.error(f"Event processing error: {e}")
    
    async def stop(self):
        """إيقاف معالجة الأحداث"""
        self.is_running = False
        if self.processor_task:
            self.processor_task.cancel()
    
    async def get_stats(self) -> Dict:
        """إحصائيات طبقة التقاط الأحداث"""
        return {
            **self.stats,
            'queue_size': self.event_queue.qsize(),
            'session_id': self.session_id
        }

# backend/part19_v19/context_engine.py
"""
Context Engine V19 - يبني فهم كامل لما يفعله المستخدم الآن
يدمج: Project Graph, Task State, Code State, Intent Inference
"""

import time
from typing import Dict, List, Optional, Any
from dataclasses import dataclass, field
import logging

from .cognitive_state_model import CognitiveState

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


@dataclass
class Context:
    """السياق الكامل للمستخدم"""
    current_project: str
    active_file: str
    task_type: str  # 'coding', 'debugging', 'refactoring', 'learning', 'reviewing'
    intent: str     # 'implement', 'fix', 'optimize', 'understand', 'test'
    complexity: float
    estimated_time_remaining: float
    historical_patterns: List[str]
    related_files: List[str]
    timestamp: float


class ContextEngine:
    """
    Context Engine V19 - محرك فهم السياق
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        self.project_graph: Dict = {}  # في الإنتاج: Neo4j
        self.task_history: List[Dict] = []
        
        self.stats = {
            'contexts_built': 0,
            'avg_build_time_ms': 0.0
        }
    
    async def build_context(self, event: Dict, cognitive_state: CognitiveState,
                             memory: Dict) -> Context:
        """
        بناء السياق من الحدث والحالة المعرفية والذاكرة
        """
        start_time = time.time()
        
        # 1. تحديد المشروع الحالي
        current_project = await self._get_current_project(event, memory)
        
        # 2. تحديد الملف النشط
        active_file = event.get('data', {}).get('file_name', 'unknown')
        
        # 3. تحديد نوع المهمة
        task_type = await self._detect_task_type(event, cognitive_state)
        
        # 4. استدلال النية
        intent = await self._infer_intent(event, cognitive_state, task_type)
        
        # 5. تقدير التعقيد
        complexity = await self._estimate_complexity(event, cognitive_state)
        
        # 6. تقدير الوقت المتبقي
        estimated_time = await self._estimate_time_remaining(task_type, complexity)
        
        # 7. استخراج الأنماط التاريخية
        patterns = await self._get_historical_patterns(task_type)
        
        # 8. تحديد الملفات المرتبطة
        related_files = await self._get_related_files(active_file)
        
        context = Context(
            current_project=current_project,
            active_file=active_file,
            task_type=task_type,
            intent=intent,
            complexity=complexity,
            estimated_time_remaining=estimated_time,
            historical_patterns=patterns,
            related_files=related_files,
            timestamp=time.time()
        )
        
        # تسجيل السياق
        self.task_history.append({
            'timestamp': time.time(),
            'task_type': task_type,
            'intent': intent,
            'complexity': complexity
        })
        
        if len(self.task_history) > 1000:
            self.task_history = self.task_history[-1000:]
        
        build_time_ms = (time.time() - start_time) * 1000
        self.stats['contexts_built'] += 1
        self.stats['avg_build_time_ms'] = (
            self.stats['avg_build_time_ms'] * 0.95 + build_time_ms * 0.05
        )
        
        return context
    
    async def _get_current_project(self, event: Dict, memory: Dict) -> str:
        """تحديد المشروع الحالي"""
        project = memory.get('current_project', 'unknown')
        return project
    
    async def _detect_task_type(self, event: Dict, state: CognitiveState) -> str:
        """تحديد نوع المهمة"""
        event_data = event.get('data', {})
        
        if event_data.get('error'):
            return 'debugging'
        elif 'test' in event_data.get('file_name', ''):
            return 'testing'
        elif state.confusion_score > 0.5:
            return 'learning'
        elif 'refactor' in str(event_data).lower():
            return 'refactoring'
        else:
            return 'coding'
    
    async def _infer_intent(self, event: Dict, state: CognitiveState,
                             task_type: str) -> str:
        """استدلال نية المستخدم"""
        if task_type == 'debugging':
            return 'fix'
        elif task_type == 'refactoring':
            return 'optimize'
        elif task_type == 'learning':
            return 'understand'
        elif task_type == 'testing':
            return 'test'
        else:
            return 'implement'
    
    async def _estimate_complexity(self, event: Dict, state: CognitiveState) -> float:
        """تقدير تعقيد المهمة"""
        # استخدام الحمل المعرفي كبديل للتعقيد
        return state.cognitive_load
    
    async def _estimate_time_remaining(self, task_type: str, complexity: float) -> float:
        """تقدير الوقت المتبقي بالدقائق"""
        base_times = {
            'coding': 30,
            'debugging': 15,
            'refactoring': 45,
            'learning': 60,
            'testing': 20
        }
        base = base_times.get(task_type, 30)
        return base * (0.5 + complexity)
    
    async def _get_historical_patterns(self, task_type: str) -> List[str]:

class GuidanceDecisionEngine:
    """
    Guidance Decision Engine V19 - يقرر متى وكيف يتدخل النظام
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        self.decision_history: list = []
        self.stats = {
            'decisions_made': 0,
            'interventions_sent': 0,
            'avg_decision_time_ms': 0.0
        }
    
    async def decide(self, cognitive_state: CognitiveState, context: Context) -> Intervention:
        """
        اتخاذ قرار التدخل بناءً على الحالة المعرفية والسياق

            )

        intervention_type = await self._determine_intervention_type(cognitive_state, context)
        
        # 3. توليد محتوى التدخل
        intervention = await self._generate_intervention(intervention_type, cognitive_state, context)
        
        # تسجيل القرار
        self.decision_history.append({
            'timestamp': time.time(),
            'cognitive_state': cognitive_state.to_dict(),
            'context': {
                'task_type': context.task_type,
                'intent': context.intent,
                'complexity': context.complexity
            },
            'intervention': intervention.type
        })
        
        if len(self.decision_history) > 1000:
            self.decision_history = self.decision_history[-1000:]
        
        if intervention.type != "none":
            self.stats['interventions_sent'] += 1
        
        self.stats['decisions_made'] += 1
        decision_time_ms = (time.time() - start_time) * 1000
        self.stats['avg_decision_time_ms'] = (

"""
Privacy Firewall V19 - حماية خصوصية المستخدم
Consent Manager, Data Filtering, Hash Sensitive Info, User Control

import hashlib
import time
from typing import Dict, Any, Optional
from dataclasses import dataclass
from enum import Enum
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


class ConsentLevel(Enum):
    NONE = "none"
    BASIC = "basic"
    FULL = "full"


@dataclass
class PrivacySettings:
    """إعدادات الخصوصية للمستخدم"""
    consent_level: ConsentLevel = ConsentLevel.BASIC
    allow_code_storage: bool = False
    allow_pattern_analysis: bool = True
    allow_personalization: bool = True
    data_retention_days: int = 30


class PrivacyFirewall:
    """
    Privacy Firewall V19 - جدار حماية الخصوصية
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        self.settings = PrivacySettings()
        self.consent_given_at: Optional[float] = None
        self.consent_revoked_at: Optional[float] = None
        
        self.stats = {
            'requests_filtered': 0,
            'data_anonymized': 0,
            'consent_checks': 0
        }
    
    async def request_consent(self) -> bool:
        """
        طلب موافقة المستخدم
        """
        # في الإنتاج: عرض واجهة طلب موافقة
        logger.info(f"Requesting consent from user {self.user_id}")
        
        # تبسيطاً: قبول تلقائي (في الإنتاج: انتظار رد المستخدم)
        self.settings.consent_level = ConsentLevel.BASIC
        self.consent_given_at = time.time()
        
        return True
    
    async def update_settings(self, new_settings: Dict) -> bool:
        """
        تحديث إعدادات الخصوصية
        """
        if 'consent_level' in new_settings:
            self.settings.consent_level = ConsentLevel(new_settings['consent_level'])
        
        if 'allow_code_storage' in new_settings:
            self.settings.allow_code_storage = new_settings['allow_code_storage']
        
        if 'allow_pattern_analysis' in new_settings:
            self.settings.allow_pattern_analysis = new_settings['allow_pattern_analysis']
        
        if 'allow_personalization' in new_settings:
            self.settings.allow_personalization = new_settings['allow_personalization']
        
        logger.info(f"Privacy settings updated for user {self.user_id}")
        return True
    
    async def filter_data(self, data: Dict) -> Dict:
        """
        تصفية البيانات حسب إعدادات الخصوصية
        """
        self.stats['consent_checks'] += 1
        
        if self.settings.consent_level == ConsentLevel.NONE:
            return {}
        
        filtered = {}
        
        for key, value in data.items():
            # تصفية حسب النوع
            if key == 'code' and not self.settings.allow_code_storage:
                # تخزين التجزئة فقط
                filtered['code_hash'] = hashlib.sha256(str(value).encode()).hexdigest()[:16]
                filtered['code_length'] = len(str(value))
                self.stats['data_anonymized'] += 1
            elif key in ['password', 'token', 'secret', 'api_key']:
                # لا نخزن أبداً
                continue
            else:
                filtered[key] = value
        
        self.stats['requests_filtered'] += 1
        
        return filtered
    
    async def revoke_consent(self) -> bool:
        """
        سحب موافقة المستخدم
        """
        self.settings.consent_level = ConsentLevel.NONE
        self.consent_revoked_at = time.time()
        logger.info(f"Consent revoked for user {self.user_id}")
        return True
    
    async def get_settings(self) -> Dict:
        """الحصول على إعدادات الخصوصية الحالية"""
        return {
            'consent_level': self.settings.consent_level.value,
            'allow_code_storage': self.settings.allow_code_storage,
            'allow_pattern_analysis': self.settings.allow_pattern_analysis,
            'allow_personalization': self.settings.allow_personalization,
            'data_retention_days': self.settings.data_retention_days,
            'consent_given_at': self.consent_given_at,
            'consent_revoked_at': self.consent_revoked_at
        }
    
    async def get_stats(self) -> Dict:
        """إحصائيات جدار الحماية"""
        return {
            **self.stats,
            'current_consent': self.settings.consent_level.value
        }

        }

        self.action_history: deque = deque(maxlen=1000)

        self.error_history: deque = deque(maxlen=100)

        # 2. توقع الإجراء التالي
        next_action = max(action_probs, key=action_probs.get)
        confidence = action_probs[next_action]
        
        # 3. توقع الخطأ المحتمل
        expected_error = await self._predict_error(current_state, context)
        
        # 4. توليد إرشاد استباقي
        guidance = await self._generate_proactive_guidance(next_action, expected_error)
        
        prediction = Prediction(
            next_action=next_action,
            confidence=confidence,
            expected_error=expected_error,
            suggested_guidance=guidance,
            time_horizon_seconds=30
        )
        
        # تحديث الإحصائيات
        self.stats['predictions_made'] += 1
        prediction_time_ms = (time.time() - start_time) * 1000
        self.stats['avg_prediction_time_ms'] = (
            self.stats['avg_prediction_time_ms'] * 0.95 + prediction_time_ms * 0.05
        )
        
        return prediction

        """

        # إجراءات حسب الحالة
        if state.confusion_score > 0.6:
            return {'pause': 0.4, 'search': 0.3, 'debug': 0.2, 'continue_typing': 0.1}
        elif state.focus_level < 0.4:
            return {'pause': 0.5, 'search': 0.2, 'continue_typing': 0.2, 'debug': 0.1}
        elif state.cognitive_load > 0.7:
            return {'pause': 0.3, 'refactor': 0.25, 'continue_typing': 0.25, 'debug': 0.2}
        else:
            return {'continue_typing': 0.6, 'test': 0.15, 'refactor': 0.15, 'debug': 0.1}
    
    async def _predict_error(self, state: CognitiveState, context: Dict) -> Optional[str]:
        """

        elif state.cognitive_load > 0.8:

        elif state.focus_level < 0.5:

        elif next_action == 'search':

# مثال على تكامل مع VSCode Extension
# extension.ts (TypeScript)

import * as vscode from 'vscode';

export function activate(context: vscode.ExtensionContext) {
    // مراقبة تغييرات الكود
    vscode.workspace.onDidChangeTextDocument(async (event) => {
        const eventData = {
            type: 'code_change',
            data: {
                file_name: event.document.fileName,
                changes: event.contentChanges.length,
                language: event.document.languageId
            }
        };
        
        // إرسال إلى Cognitive Guidance System
        await fetch('http://localhost:8000/event', {
            method: 'POST',
            body: JSON.stringify(eventData),
            headers: {'Content-Type': 'application/json'}
        });
    });
}
