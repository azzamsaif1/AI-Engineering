# backend/evaluation/cognitive_engineering_evolution_index.py
import numpy as np
from typing import Dict, List
from datetime import datetime, timedelta

class CognitiveEngineeringEvolutionIndex:
    """
    مؤشر التطور المعرفي الهندسي
    نظام عادل وصحي نفسياً لقياس التطور البشري
    لا يقيس "من هو آلة أكثر"، بل كيف يتطور الإنسان
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        
        # الأبعاد المقاسة (عادلة وشاملة)
        self.dimensions = {
            'analysis_speed': AnalysisSpeedMetric(),
            'thinking_quality': ThinkingQualityMetric(),
            'precision': PrecisionMetric(),
            'speed': SpeedMetric(),
            'continuous_evolution': ContinuousEvolutionMetricV2(),
            'human_ai_sync': HumanAISyncMetric()
        }
        
        # أوزان متوازنة
        self.weights = {
            'analysis_speed': 0.15,
            'thinking_quality': 0.25,
            'precision': 0.20,
            'speed': 0.10,
            'continuous_evolution': 0.20,
            'human_ai_sync': 0.10
        }
        
        # تاريخ القياسات (لرحلة التطور)
        self.history = []
        
    async def evaluate(self, user_data: Dict) -> Dict:
        """
        تقييم عادل يركز على التطور وليس المقارنة
        """
        scores = {}
        
        for name, metric in self.dimensions.items():
            scores[name] = await metric.calculate(user_data)
        
        # المؤشر الكلي
        total_score = sum(scores[name] * self.weights[name] for name in scores)
        
        # تحديد مستوى التطور (ليس تصنيفاً، بل مرحلة في الرحلة)
        level = self._get_development_level(total_score)
        
        # إنشاء تقرير تشجيعي
        report = {
            'user_id': self.user_id,
            'timestamp': datetime.now().isoformat(),
            'development_level': level,
            'dimensions': scores,
            'next_milestones': self._get_next_milestones(scores),
            'recommended_challenges': await self._generate_recommended_challenges(scores),
            'progress_over_time': self._calculate_progress_timeline(),
            'human_ai_collaboration_score': scores.get('human_ai_sync', 0.5)
        }
        
        self.history.append(report)
        
        return report
    
    def _get_development_level(self, score: float) -> Dict:
        """
        تحديد مستوى التطور (بطريقة تشجيعية وليست تصنيفية)
        """
        if score < 0.3:
            return {
                'name': 'Explorer',
                'description': 'بداية رحلة التطور الهندسي، كل خطوة تبنيك',
                'next_focus': 'بناء الأساسيات الهندسية',
                'encouragement': 'أنت في بداية رحلة رائعة، كل خبير كان مبتدئاً'
            }
        elif score < 0.5:
            return {
                'name': 'Builder',
                'description': 'تبني أساساً قوياً للمهارات الهندسية',
                'next_focus': 'تعميق فهم الأنظمة المعقدة',
                'encouragement': 'تقدم ممتاز! استمر في بناء قدراتك'
            }
        elif score < 0.7:
            return {
                'name': 'Architect',
                'description': 'تفهم الأنظمة وتبني حلولاً متقنة',
                'next_focus': 'توسيع نطاق التأثير والمشاريع',
                'encouragement': 'مستوى عالٍ من الهندسة، أنت تبني أنظمة مؤثرة'
            }
        elif score < 0.9:
            return {
                'name': 'Master',
                'description': 'خبرة عميقة وقدرة على حل المشكلات المعقدة',
                'next_focus': 'الابتكار وقيادة الفرق الهندسية',
                'encouragement': 'إتقان هندسي رفيع، أنت تلهم الآخرين'
            }
        else:
            return {
                'name': 'Visionary',
                'description': 'رؤية هندسية استثنائية وقدرة على تشكيل المستقبل',
                'next_focus': 'توسيع حدود الهندسة والإبداع',
                'encouragement': 'أنت في قمة الهندسة، استمر في دفع الحدود'
            }
    
    def _get_next_milestones(self, scores: Dict) -> List:
        """
        تحديد المعالم التالية للتطور (بطريقة تشجيعية)
        """
        milestones = []
        
        # تحديد أضعف بُعد (كفرصة للتطور وليس كنقد)
        weakest = min(scores, key=scores.get)
        weakest_score = scores[weakest]
        
        milestone_map = {
            'analysis_speed': {
                'title': 'تسريع التحليل الهندسي',
                'description': 'حل تحديات تحليل الكود ضمن وقت محدد',
                'suggested_practice': 'تدرب على قراءة وتحليل codebases مختلفة'
            },
            'thinking_quality': {
                'title': 'تعمق في التفكير المنهجي',
                'description': 'بناء أنظمة معقدة باستخدام مبادئ العمارة البرمجية',
                'suggested_practice': 'ادرس أنماط التصميم والعمارة البرمجية'
            },
            'precision': {
                'title': 'زيادة الدقة والموثوقية',
                'description': 'تطوير حلول تغطي جميع الحالات الحدودية',
                'suggested_practice': 'مارس الـ test-driven development'
            },
            'speed': {
                'title': 'تسريع وتيرة التطوير',
                'description': 'حل تحديات برمجية ضمن وقت محدد',
                'suggested_practice': 'شارك في تحديات برمجية قصيرة'
            },
            'continuous_evolution': {
                'title': 'تسريع منحنى التعلم',
                'description': 'تعلم تقنيات جديدة وتطبيقها في مشاريع حقيقية',
                'suggested_practice': 'خصص وقتاً يومياً لتعلم شيء جديد'
            },
            'human_ai_sync': {
                'title': 'تعزيز التعاون مع النسخة الرقمية',
                'description': 'استخدم النسخة الرقمية لتوسيع قدراتك',
                'suggested_practice': 'تعاون مع نسختك الرقمية في مشاريع معقدة'
            }
        }
        
        milestones.append({
            'focus_area': weakest,
            'current_score': weakest_score,
            'target_score': min(1.0, weakest_score + 0.2),
            'milestone': milestone_map.get(weakest, {}),
            'encouragement': f'تركيزك على {weakest.replace("_", " ")} سيسرع تطورك بشكل كبير'
        })
        
        return milestones

class HumanAISyncMetric:
    """قياس التكامل الصحي بين الإنسان والنسخة الرقمية"""
    
    async def calculate(self, user_data: Dict) -> float:
        """
        قياس كيف يتعاون المستخدم مع نسخته الرقمية
        بشكل صحي وإيجابي
        """
        # مدى استخدام المستخدم لنسخته الرقمية
        twin_usage = user_data.get('twin_usage_frequency', 0.5)
        
        # تحسين الأداء بفضل النسخة الرقمية
        performance_improvement = user_data.get('performance_improvement', 0.1)
        
        # مدى تطور النسخة الرقمية بفضل المستخدم
        twin_evolution = user_data.get('twin_evolution', 0.1)
        
        # الرضا عن التعاون
        collaboration_satisfaction = user_data.get('collaboration_satisfaction', 0.7)
        
        # التطور المشترك
        co_evolution = (performance_improvement + twin_evolution) / 2
        
        score = 0.3 * twin_usage + 0.3 * co_evolution + 0.4 * collaboration_satisfaction
        
        return min(1.0, score)

# backend/generation/universal_software_generator.py
from langgraph import StateGraph
from typing import Dict, List
import asyncio

class UniversalSoftwareGenerator:
    """
    المولد العالمي المستقل للبرمجيات
    يحول أي وصف نصي إلى مشروع كامل
    """
    
    def __init__(self):
        # مراحل التوليد
        self.stages = {
            'understanding': IdeaUnderstandingStage(),
            'design': ArchitectureDesignStage(),
            'technology_selection': TechnologySelectionStage(),
            'generation': CodeGenerationStage(),
            'library_creation': LibraryCreationStage(),
            'language_development': LanguageDevelopmentStage()
        }
        
        # تكامل الأدوات
        self.tool_integration = MassiveToolIntegrationV2()
        
        # تنسيق المراحل
        self.orchestrator = StateGraph(GenerationState)
        self._setup_orchestration()
        
    async def generate_from_description(self, description: str) -> Dict:
        """
        تحويل وصف نصي إلى مشروع كامل
        """
        initial_state = {
            'description': description,
            'requirements': {},
            'design': {},
            'technologies': {},
            'code': {},
            'libraries': [],
            'language_spec': None,
            'final_project': None
        }
        
        result = await self.orchestrator.ainvoke(initial_state)
        
        return result

class IdeaUnderstandingStage:
    """
    مرحلة فهم الفكرة - تحويل النص الطبيعي إلى متطلبات هندسية
    """
    
    async def execute(self, state: Dict) -> Dict:
        description = state['description']
        
        # استخدام DeepSeek R1 لفهم الفكرة
        prompt = f"""
        Analyze this project idea and extract:
        1. Core functionality
        2. Technical requirements
        3. User types
        4. Constraints
        5. Scale expectations
        6. Security requirements
        
        Idea: {description}
        
        Return structured requirements.
        """
        
        # استدعاء النموذج
        requirements = await self._call_model(prompt)
        
        state['requirements'] = requirements
        return state

class LibraryCreationStage:
    """
    مرحلة إنشاء المكتبات - بناء مكتبات وأطر عمل جديدة
    """
    
    async def execute(self, state: Dict) -> Dict:
        """
        إنشاء مكتبات جديدة بناءً على متطلبات المشروع
        """
        requirements = state.get('requirements', {})
        
        # تحليل الحاجة لمكتبة جديدة
        need_new_library = self._analyze_library_need(requirements)
        
        if need_new_library:
            # تصميم المكتبة
            library_spec = await self._design_library(requirements)
            
            # تنفيذ المكتبة
            library_code = await self._implement_library(library_spec)
            
            # توثيق المكتبة
            library_docs = await self._document_library(library_code)
            
            # نشر المكتبة (محلياً)
            library_package = await self._package_library(library_code, library_docs)
            
            state['libraries'].append({
                'spec': library_spec,
                'code': library_code,
                'docs': library_docs,
                'package': library_package
            })
        
        return state

class LanguageDevelopmentStage:
    """
    مرحلة تطوير اللغة - بناء لغات برمجة وأدوات جديدة
    """
    
    async def execute(self, state: Dict) -> Dict:
        """
        تطوير لغة برمجة جديدة أو DSL للمشروع
        """
        requirements = state.get('requirements', {})
        
        # تحليل الحاجة للغة جديدة
        need_new_language = self._analyze_language_need(requirements)
        
        if need_new_language:
            # تصميم اللغة
            language_spec = await self._design_language(requirements)
            
            # بناء الـ Compiler/Interpreter
            compiler = await self._build_compiler(language_spec)
            
            # بناء الـ Transpiler (إذا لزم الأمر)
            transpiler = await self._build_transpiler(language_spec)
            
            # بناء الأدوات (LSP, Formatter, Linter)
            tools = await self._build_language_tools(language_spec)
            
            state['language_spec'] = {
                'spec': language_spec,
                'compiler': compiler,
                'transpiler': transpiler,
                'tools': tools
            }
        
        return state

"""
Cognitive Engineering Evolution Index V15 - مؤشر تطور عادل وصحي نفسياً
مبسط: 4 مقاييس فقط، يركز على التطور وليس المقارنة
"""

import time
import numpy as np
from typing import Dict, List, Optional
from dataclasses import dataclass
from collections import deque
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


@dataclass
class EvolutionMetrics:
    """مقاييس التطور (مبسطة وعادلة)"""
    problem_solving: float  # قدرة حل المشكلات
    code_understanding: float  # فهم الكود
    learning_speed: float  # سرعة التعلم
    consistency: float  # الاتساق والموثوقية
    
    @property
    def total_score(self) -> float:
        """الدرجة الكلية (متوسط بسيط)"""
        return (self.problem_solving + self.code_understanding + 
                self.learning_speed + self.consistency) / 4


class CognitiveEvolutionIndex:
    """
    مؤشر التطور المعرفي الهندسي
    - عادل: يقيس التطور الشخصي وليس المقارنة مع الآخرين
    - صحي نفسياً: تشجيعي وليس تصنيفي
    - بسيط: 4 مقاييس فقط
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        
        # تاريخ التطور (لرحلة المستخدم)
        self.history: List[EvolutionMetrics] = []
        self.recent_scores: deque = deque(maxlen=20)
        
        # مستوى البدء الافتراضي (لا يوجد "صفر" محبط)
        self.starting_level = 0.3
        
    async def evaluate(self, user_data: Dict) -> Dict:
        """
        تقييم عادل يركز على التطور وليس المقارنة
        """
        # حساب المقاييس الأربعة
        problem_solving = await self._calculate_problem_solving(user_data)
        code_understanding = await self._calculate_code_understanding(user_data)
        learning_speed = await self._calculate_learning_speed(user_data)
        consistency = await self._calculate_consistency(user_data)
        
        metrics = EvolutionMetrics(
            problem_solving=problem_solving,
            code_understanding=code_understanding,
            learning_speed=learning_speed,
            consistency=consistency
        )
        
        self.history.append(metrics)
        self.recent_scores.append(metrics.total_score)
        
        # تحديد مرحلة التطور (تشجيعية)
        level_info = self._get_evolution_level(metrics.total_score)
        
        # حساب التقدم الشخصي (الأهم - يقارن المستخدم بنفسه السابقة)
        personal_progress = self._calculate_personal_progress()
        
        return {
            'user_id': self.user_id,
            'timestamp': time.time(),
            'current_level': level_info,
            'metrics': {
                'problem_solving': round(problem_solving, 2),
                'code_understanding': round(code_understanding, 2),
                'learning_speed': round(learning_speed, 2),
                'consistency': round(consistency, 2)
            },
            'total_score': round(metrics.total_score, 2),
            'personal_progress': personal_progress,
            'next_milestone': self._get_next_milestone(metrics),
            'encouragement': self._generate_encouragement(metrics, personal_progress),
            'recommended_focus': self._get_recommended_focus(metrics)
        }
    
    async def _calculate_problem_solving(self, data: Dict) -> float:
        """قدرة حل المشكلات"""
        # من تحليل حلول المستخدم
        success_rate = data.get('problem_success_rate', 0.7)
        complexity_handled = data.get('max_complexity_handled', 0.5)
        return (success_rate + complexity_handled) / 2
    
    async def _calculate_code_understanding(self, data: Dict) -> float:
        """فهم الكود"""
        # من تحليل قراءة الكود
        analysis_accuracy = data.get('code_analysis_accuracy', 0.6)
        pattern_recognition = data.get('pattern_recognition', 0.5)
        return (analysis_accuracy + pattern_recognition) / 2
    
    async def _calculate_learning_speed(self, data: Dict) -> float:
        """سرعة التعلم"""
        # من تحليل تعلم تقنيات جديدة
        time_to_learn = data.get('time_to_learn_new_tech_days', 14)
        retention = data.get('knowledge_retention', 0.7)
        
        # كلما قل الوقت، زادت السرعة
        speed_score = max(0.1, 1.0 - (time_to_learn / 30))
        return (speed_score + retention) / 2
    
    async def _calculate_consistency(self, data: Dict) -> float:
        """الاتساق والموثوقية"""
        reliability = data.get('reliability_score', 0.7)
        low_variance = data.get('performance_variance', 0.2)
        
        consistency_score = 1.0 - min(0.5, low_variance)
        return (reliability + consistency_score) / 2
    
    def _get_evolution_level(self, score: float) -> Dict:
        """
        مراحل التطور (تشجيعية وليست تصنيفية)
        """
        if score < 0.3:
            return {
                'name': 'Explorer',
                'description': 'بداية رحلة التطور الهندسي',
                'badge': '🌱',
                'message': 'كل خطوة تبنيك، استمر في الاستكشاف'
            }
        elif score < 0.5:
            return {
                'name': 'Builder',
                'description': 'تبني أساساً قوياً',
                'badge': '🏗️',
                'message': 'تقدم ممتاز! أنت تبني قدراتك'
            }
        elif score < 0.7:
            return {
                'name': 'Architect',
                'description': 'تفهم الأنظمة وتبني حلولاً متقنة',
                'badge': '🏛️',
                'message': 'مستوى عالٍ من الهندسة، استمر'
            }
        elif score < 0.9:
            return {
                'name': 'Master',
                'description': 'خبرة عميقة وقدرة على حل المشكلات المعقدة',
                'badge': '🎓',
                'message': 'إتقان هندسي رفيع، أنت تلهم الآخرين'
            }
        else:
            return {
                'name': 'Visionary',
                'description': 'رؤية هندسية استثنائية',
                'badge': '🔮',
                'message': 'أنت في قمة الهندسة، استمر في دفع الحدود'
            }
    
    def _calculate_personal_progress(self) -> Dict:
        """
        حساب التقدم الشخصي (يقارن المستخدم بنفسه السابقة)
        """
        if len(self.history) < 2:
            return {'improvement': 0.0, 'trend': 'just_started'}
        
        first_score = self.history[0].total_score
        current_score = self.history[-1].total_score
        improvement = current_score - first_score
        
        if improvement > 0.1:
            trend = 'accelerating'
        elif improvement > 0.03:
            trend = 'improving'
        elif improvement > -0.03:
            trend = 'steady'
        else:
            trend = 'needs_support'
        
        return {
            'improvement': round(improvement, 2),
            'from_score': round(first_score, 2),
            'to_score': round(current_score, 2),
            'trend': trend
        }
    
    def _get_next_milestone(self, metrics: EvolutionMetrics) -> Dict:
        """تحديد المعلم التالي للتطور"""
        # أضعف بُعد (كفرصة وليس كنقد)
        weakest = min([
            ('problem_solving', metrics.problem_solving),
            ('code_understanding', metrics.code_understanding),
            ('learning_speed', metrics.learning_speed),
            ('consistency', metrics.consistency)
        ], key=lambda x: x[1])
        
        targets = {
            'problem_solving': {
                'title': 'تحديات حل المشكلات',
                'suggestion': 'جرب مشاريع أكثر تعقيداً',
                'resource': 'مشاريع مفتوحة المصدر'
            },
            'code_understanding': {
                'title': 'قراءة وتحليل الكود',
                'suggestion': 'اقرأ codebase مشاريع كبيرة',
                'resource': 'GitHub Explore'
            },
            'learning_speed': {
                'title': 'تسريع التعلم',
                'suggestion': 'تعلم تقنية جديدة كل أسبوع',
                'resource': 'دورات قصيرة'
            },
            'consistency': {
                'title': 'الاستمرارية والموثوقية',
                'suggestion': 'التزم بجدول يومي',
                'resource': 'مشاريع شخصية منتظمة'
            }
        }
        
        return {
            'focus': weakest[0],
            'current_score': weakest[1],
            'target_score': min(1.0, weakest[1] + 0.15),
            **targets.get(weakest[0], {})
        }
    
    def _generate_encouragement(self, metrics: EvolutionMetrics, progress: Dict) -> str:
        """توليد رسالة تشجيعية"""
        if progress['improvement'] > 0.1:
            return f"🎉 تقدم رائع! تحسنت بنسبة {progress['improvement']*100:.0f}%"
        elif progress['improvement'] > 0.03:
            return f"📈 استمر في هذا المنحنى! أنت تتطور بشكل ثابت"
        elif len(self.history) < 3:
            return f"🌟 بداية قوية! استمر في بناء قدراتك"
        else:
            return f"💪 التطور يحتاج وقتاً و صبراً، أنت في رحلتك الخاصة"
    
    def _get_recommended_focus(self, metrics: EvolutionMetrics) -> str:
        """تحديد مجال التركيز الموصى به"""
        weakest = min([
            metrics.problem_solving,
            metrics.code_understanding,
            metrics.learning_speed,
            metrics.consistency
        ])
        
        if weakest == metrics.problem_solving:
            return "حل تحديات برمجية يومياً"
        elif weakest == metrics.code_understanding:
            return "اقرأ و حلل كود مشاريع مفتوحة المصدر"
        elif weakest == metrics.learning_speed:
            return "خصص وقتاً يومياً لتعلم شيء جديد"
        else:
            return "ابنِ مشروعاً جانبياً والتزم بتطويره"
    
    async def get_report(self) -> Dict:
        """تقرير التطور الكامل"""
        if not self.history:
            return {'message': 'لا توجد بيانات كافية بعد'}
        
        latest = self.history[-1]
        
        return {
            'user_id': self.user_id,
            'current_score': latest.total_score,
            'journey_length': len(self.history),
            'level': self._get_evolution_level(latest.total_score),
            'progress': self._calculate_personal_progress(),
            'recommendation': self._get_recommended_focus(latest)
        }

class SandboxExecutor:
    """
    Sandbox Executor - تنفيذ آمن في Docker
    """
    
    def __init__(self):
        self.docker_available = self._check_docker()
        
    def _check_docker(self) -> bool:
        """التحقق من توفر Docker"""
        try:
            import docker
            client = docker.from_env()
            client.ping()
            return True
        except:
            return False
    
    async def execute(self, code: str, timeout_sec: int = 30) -> Dict:
        """
        تنفيذ كود في بيئة معزولة
        """
        if self.docker_available:
            return await self._execute_docker(code, timeout_sec)
        else:
            return await self._execute_subprocess(code, timeout_sec)
    
    async def _execute_docker(self, code: str, timeout_sec: int) -> Dict:
        """تنفيذ في Docker"""
        try:
            import docker
            import tempfile
            import os
            
            client = docker.from_env()
            
            # إنشاء ملف مؤقت
            with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
                f.write(code)
                temp_file = f.name
            
            # تشغيل الحاوية
            container = client.containers.run(
                image="python:3.11-slim",
                command=f"python /code/{os.path.basename(temp_file)}",
                volumes={os.path.dirname(temp_file): {'bind': '/code', 'mode': 'ro'}},
                working_dir='/code',
                mem_limit="512m",
                network_disabled=True,
                read_only=True,
                detach=True,
                remove=False
            )
            
            try:
                result = container.wait(timeout=timeout_sec)
                logs = container.logs().decode()
                
                return {
                    'success': result['StatusCode'] == 0,
                    'output': logs,
                    'error': "" if result['StatusCode'] == 0 else logs,
                    'exit_code': result['StatusCode']
                }
            finally:
                container.remove(force=True)
                os.unlink(temp_file)
                
        except Exception as e:
            return {'success': False, 'output': '', 'error': str(e), 'exit_code': -1}
    
    async def _execute_subprocess(self, code: str, timeout_sec: int) -> Dict:
        """تنفيذ في subprocess (fallback)"""
        import subprocess
        import tempfile
        
        with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
            f.write(code)
            temp_file = f.name
        
        try:
            result = subprocess.run(
                ["python", temp_file],
                capture_output=True,
                timeout=timeout_sec
            )
            
            return {
                'success': result.returncode == 0,
                'output': result.stdout.decode(),
                'error': result.stderr.decode(),
                'exit_code': result.returncode
            }
        except subprocess.TimeoutExpired:
            return {'success': False, 'output': '', 'error': 'Timeout', 'exit_code': -1}
        finally:
            import os
            if os.path.exists(temp_file):
                os.unlink(temp_file)


class AutonomousLoopEngine:
    """
    Autonomous Loop Engine V15 - المحرك التنفيذي الذاتي
    الحلقة المغلقة: Plan → Generate → Execute → Validate → Fix → Repeat
    """
    
    def __init__(self, max_iterations: int = 3):
        self.llm = LLMClient()
        self.sandbox = SandboxExecutor()
        self.max_iterations = max_iterations
        
        self.stats = {
            'total_executions': 0,
            'successful_executions': 0,
            'failed_executions': 0,
            'avg_iterations': 0.0,
            'avg_time_ms': 0.0
        }
    
    async def run(self, task: str) -> ExecutionResult:
        """
        تنفيذ مهمة كاملة مع حلقة تصحيح ذاتي
        """
        start_time = time.time()
        iterations = 0
        current_code = ""
        last_error = ""
        
        for i in range(self.max_iterations):
            iterations = i + 1
            logger.info(f"Iteration {iterations}/{self.max_iterations} for task: {task[:50]}...")
            
            # 1. PLAN - تخطيط
            if i == 0:
                plan = await self._plan(task)
                logger.info(f"Plan generated: {plan[:100]}...")
            else:
                plan = await self._replan(task, current_code, last_error)
            
            # 2. GENERATE - توليد الكود
            code = await self._generate(plan, task)
            current_code = code
            
            # 3. EXECUTE - تنفيذ في Sandbox
            result = await self.sandbox.execute(code)
            
            # 4. VALIDATE - تحقق
            if result['success']:
                logger.info(f"Execution successful!")
                
                execution_result = ExecutionResult(
                    success=True,
                    output=result['output'],
                    error="",
                    execution_time_ms=(time.time() - start_time) * 1000,
                    iterations=iterations,
                    code=code
                )
                
                self._update_stats(execution_result)
                return execution_result
            else:
                last_error = result['error']
                logger.warning(f"Execution failed: {last_error[:100]}... Retrying...")
        
        # فشل بعد كل المحاولات
        execution_result = ExecutionResult(
            success=False,
            output="",
            error=last_error,
            execution_time_ms=(time.time() - start_time) * 1000,
            iterations=iterations,
            code=current_code
        )
        
        self._update_stats(execution_result)
        return execution_result
    
    async def _plan(self, task: str) -> str:
        """تخطيط المهمة"""
        prompt = f"""
        Create a simple plan for this programming task:
        Task: {task}
        
        Output format (plain text, 3-5 steps):
        1. First step
        2. Second step
        ...
        """
        return await self.llm.generate(prompt)
    
    async def _replan(self, task: str, code: str, error: str) -> str:
        """إعادة تخطيط بعد فشل"""
        prompt = f"""
        The previous attempt failed. Create a revised plan.
        
        Original task: {task}
        Previous code: {code[:500]}
        Error: {error}
        
        New plan (3-5 steps):
        """
        return await self.llm.generate(prompt)
    
    async def _generate(self, plan: str, task: str) -> str:
        """توليد الكود من الخطة"""
        prompt = f"""
        Generate Python code for this task.
        
        Task: {task}
        Plan: {plan}
        
        Requirements:
        - Complete, runnable Python code
        - Include main() function
        - Handle errors gracefully
        
        Code:
        """
        return await self.llm.generate(prompt, max_tokens=1500)
    
    def _update_stats(self, result: ExecutionResult):
        """تحديث الإحصائيات"""
        self.stats['total_executions'] += 1
        if result.success:
            self.stats['successful_executions'] += 1
        else:
            self.stats['failed_executions'] += 1
        
        self.stats['avg_iterations'] = (
            self.stats['avg_iterations'] * 0.9 + result.iterations * 0.1
        )
        self.stats['avg_time_ms'] = (
            self.stats['avg_time_ms'] * 0.9 + result.execution_time_ms * 0.1
        )

@dataclass
class GeneratedProject:
    """مشروع مولّد"""
    id: str
    idea: str
    requirements: Dict
    architecture: Dict
    code: Dict[str, str]  # filename -> content

class UniversalGenerator:
    """
    Universal Software Generator V15
    يحول أي فكرة إلى مشروع كامل (4 مراحل فقط)
    """
    
    def __init__(self, llm_client=None):
        self.llm = llm_client or self._create_llm_client()
        self.generation_history: List[GeneratedProject] = []
        
        self.stats = {
            'projects_generated': 0,
            'avg_generation_time_ms': 0.0,
            'success_rate': 1.0
        }
    
    def _create_llm_client(self):
        """إنشاء عميل LLM"""
        class SimpleLLM:
            async def generate(self, prompt: str) -> str:
                await asyncio.sleep(0.5)
                return '{"status": "ok"}'
        return SimpleLLM()
    
    async def generate(self, idea: str) -> GeneratedProject:
        """
        توليد مشروع كامل من فكرة
        """
        start_time = time.time()
        project_id = f"proj_{int(time.time()*1000)}"
        
        logger.info(f"Generating project from idea: {idea[:100]}...")
        
        # ========== STAGE 1: UNDERSTAND ==========
        requirements = await self._understand(idea)
        logger.info(f"Requirements extracted: {len(requirements)} items")
        
        # ========== STAGE 2: DESIGN ==========
        architecture = await self._design(requirements)
        logger.info(f"Architecture designed")
        
        # ========== STAGE 3: GENERATE ==========
        code = await self._generate_code(architecture, requirements)
        logger.info(f"Code generated: {len(code)} files")
        
        # ========== STAGE 4: PACKAGE ==========
        package_info = await self._package(code, requirements)
        logger.info(f"Project packaged")
        
        project = GeneratedProject(
            id=project_id,
            idea=idea,
            requirements=requirements,
            architecture=architecture,
            code=code,
            package_info=package_info,
            generated_at=time.time()
        )
        
        self.generation_history.append(project)
        self.stats['projects_generated'] += 1
        
        generation_time_ms = (time.time() - start_time) * 1000
        self.stats['avg_generation_time_ms'] = (
            self.stats['avg_generation_time_ms'] * 0.95 + generation_time_ms * 0.05
        )
        
        return project

        """

"""
Orchestrator V15 - المنسق المركزي
يدمج الأنظمة الثلاثة في حلقة واحدة
"""

import asyncio
import time
from typing import Dict, Optional
import logging

from .cognitive_engine import CognitiveEvolutionIndex
from .autonomous_loop_engine import AutonomousLoopEngine
from .universal_generator import UniversalGenerator

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


class TechLinguaOrchestrator:
    """
    TechLingua Orchestrator V15 - المنسق المركزي
    """
    
    def __init__(self):
        self.cognitive_engine = None  # يتم إنشاؤه لكل مستخدم
        self.execution_engine = AutonomousLoopEngine(max_iterations=3)
        self.generator = UniversalGenerator()
        
        self.user_sessions: Dict[str, Dict] = {}
        self.stats = {
            'total_tasks': 0,
            'successful_tasks': 0,
            'failed_tasks': 0,
            'start_time': time.time()
        }
    
    async def process(self, user_id: str, task: str, task_type: str = "execute") -> Dict:
        """
        معالجة طلب مستخدم
        """
        self.stats['total_tasks'] += 1
        
        # إنشاء جلسة للمستخدم الجديد
        if user_id not in self.user_sessions:
            self.user_sessions[user_id] = {
                'cognitive': CognitiveEvolutionIndex(user_id),
                'history': []
            }
        
        session = self.user_sessions[user_id]
        
        try:
            if task_type == "execute":
                result = await self.execution_engine.run(task)
            elif task_type == "generate":
                project = await self.generator.generate(task)
                result = {'success': True, 'project_id': project.id, 'files': len(project.code)}
            elif task_type == "evaluate":
                # محاكاة بيانات تقييم
                user_data = {'problem_success_rate': 0.8, 'code_analysis_accuracy': 0.7}
                evaluation = await session['cognitive'].evaluate(user_data)
                result = evaluation
            else:
                result = {'success': False, 'error': f'Unknown task type: {task_type}'}
            
            # تحديث الإحصائيات
            if result.get('success', False):
                self.stats['successful_tasks'] += 1
            else:
                self.stats['failed_tasks'] += 1
            
            # تسجيل في تاريخ المستخدم
            session['history'].append({
                'timestamp': time.time(),
                'task': task[:100],
                'task_type': task_type,
                'result': result.get('success', False)
            })
            
            return {
                'success': True,
                'user_id': user_id,
                'task_type': task_type,
                'result': result,
                'stats': self.stats
            }
            
        except Exception as e:
            self.stats['failed_tasks'] += 1
            logger.error(f"Processing error: {e}")
            
            return {
                'success': False,
                'user_id': user_id,
                'error': str(e),
                'stats': self.stats
            }
    
    async def get_user_report(self, user_id: str) -> Dict:
        """الحصول على تقرير تطور المستخدم"""
        if user_id not in self.user_sessions:
            return {'error': 'User not found'}
        
        session = self.user_sessions[user_id]
        report = await session['cognitive'].get_report()
        
        return {
            'user_id': user_id,
            'cognitive_report': report,
            'tasks_completed': len(session['history']),
            'success_rate': sum(1 for h in session['history'] if h['result']) / max(1, len(session['history']))
        }
    
    async def get_stats(self) -> Dict:
        """إحصائيات النظام"""
        uptime_seconds = time.time() - self.stats['start_time']
        
        return {
            **self.stats,
            'uptime_hours': round(uptime_seconds / 3600, 1),
            'active_users': len(self.user_sessions),
            'execution_engine': await self.execution_engine.get_stats(),
            'generator': await self.generator.get_stats()
        }
