# backend/part5_v5/control_plane.py
"""
Control Plane - العقل المركزي للنظام
مسؤول عن: من يعمل؟ لماذا؟ بأي أولوية؟ متى نتوقف؟
"""

import asyncio
import time
from typing import Dict, List, Optional, Any, Tuple
from enum import Enum
from dataclasses import dataclass, field
from collections import deque
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


class ComponentState(Enum):
    """حالات المكونات في النظام"""
    IDLE = "idle"
    ACTIVE = "active"
    PAUSED = "paused"
    DEGRADED = "degraded"
    FAILED = "failed"


class Priority(Enum):
    """مستويات الأولوية للعمليات"""
    CRITICAL = 0      # لا يمكن تأجيلها
    HIGH = 1          # مهمة جداً
    NORMAL = 2        # عادية
    LOW = 3           # يمكن تأجيلها
    BACKGROUND = 4    # تعمل فقط عند وجود فائض


@dataclass
class Component:
    """تمثيل مكون في النظام"""
    name: str
    priority: Priority
    state: ComponentState = ComponentState.IDLE
    last_active: float = field(default_factory=time.time)
    compute_budget_ms: float = 200.0
    current_latency_ms: float = 0.0
    error_count: int = 0
    consecutive_failures: int = 0


@dataclass
class OrchestrationDecision:
    """قرار التنسيق من Control Plane"""
    component_to_activate: str
    components_to_pause: List[str]
    reason: str
    priority: Priority
    timestamp: float = field(default_factory=time.time)


class ControlPlane:
    """
    Control Plane - العقل المركزي للنظام
    هذا هو المكون الأهم في الجزء الخامس
    """
    
    def __init__(self):
        # تسجيل جميع مكونات النظام
        self.components: Dict[str, Component] = {}
        self._register_all_components()
        
        # قائمة انتظار القرارات
        self.decision_queue: deque = deque(maxlen=100)
        
        # حالة النظام العالمية
        self.global_state: Dict[str, Any] = {
            'mode': 'normal',  # normal, degraded, emergency
            'total_latency_budget_ms': 200.0,
            'current_latency_ms': 0.0,
            'active_components': 0,
            'last_orchestration': time.time()
        }
        
        # إحصائيات
        self.stats = {
            'orchestrations': 0,
            'conflicts_resolved': 0,
            'failures_handled': 0,
            'recoveries': 0
        }
        
        # مكونات النظام
        self.ingestion_layer = None
        self.modality_router = None
        self.memory_system = None
        self.agent_governor = None
        self.reasoning_verifier = None
        self.evolution_gate = None
        self.competition_engine = None
        
    def _register_all_components(self):
        """تسجيل جميع مكونات النظام في Control Plane"""
        self.components = {
            # طبقة الاستقبال
            'kafka_ingestion': Component('kafka_ingestion', Priority.CRITICAL),
            'redis_cache': Component('redis_cache', Priority.HIGH),
            'fluss_archive': Component('fluss_archive', Priority.LOW),
            
            # طبقة التوجيه
            'modality_router': Component('modality_router', Priority.CRITICAL),
            
            # معالجات الوسائط
            'whisper_asr': Component('whisper_asr', Priority.HIGH, compute_budget_ms=150),
            'qwen_vision': Component('qwen_vision', Priority.HIGH, compute_budget_ms=200),
            'deepseek_code': Component('deepseek_code', Priority.NORMAL, compute_budget_ms=300),
            'tree_sitter': Component('tree_sitter', Priority.HIGH, compute_budget_ms=50),
            'behavior_lstm': Component('behavior_lstm', Priority.NORMAL, compute_budget_ms=50),
            
            # الذاكرة
            'qdrant_memory': Component('qdrant_memory', Priority.NORMAL),
            'neo4j_graph': Component('neo4j_graph', Priority.LOW),
            'episodic_buffer': Component('episodic_buffer', Priority.NORMAL),
            'memory_arbitration': Component('memory_arbitration', Priority.NORMAL),
            
            # الوكلاء
            'planner_agent': Component('planner_agent', Priority.HIGH),
            'executor_agents': Component('executor_agents', Priority.NORMAL),
            'critic_agent': Component('critic_agent', Priority.HIGH),
            'agent_governor': Component('agent_governor', Priority.CRITICAL),
            
            # التفكير والتحقق
            'reasoning_engine': Component('reasoning_engine', Priority.NORMAL),
            'verifier': Component('verifier', Priority.HIGH),
            
            # التطور والمنافسة
            'evolution_gate': Component('evolution_gate', Priority.LOW),
            'competition_engine': Component('competition_engine', Priority.LOW)
        }
    
    async def orchestrate(self, trigger: str, context: Dict) -> OrchestrationDecision:
        """
        القرار المركزي: من يعمل الآن؟
        """
        decision = None
        
        # 1. التحقق من حالة النظام
        system_health = await self._assess_system_health()
        
        # 2. تحديد المكون النشط بناءً على الزناد
        if trigger == 'user_input':
            decision = await self._orchestrate_for_user_input(context)
        elif trigger == 'system_tick':
            decision = await self._orchestrate_for_system_tick(system_health)
        elif trigger == 'error':
            decision = await self._orchestrate_for_error(context)
        elif trigger == 'evolution_trigger':
            decision = await self._orchestrate_for_evolution(system_health)
        else:
            decision = await self._orchestrate_default()
        
        # 3. تنفيذ القرار
        if decision:
            await self._apply_decision(decision)
            self.decision_queue.append(decision)
            self.stats['orchestrations'] += 1
        
        return decision
    
    async def _orchestrate_for_user_input(self, context: Dict) -> OrchestrationDecision:
        """
        تنسيق استجابة النظام لمدخل المستخدم
        """
        input_type = context.get('input_type', 'unknown')
        
        components_to_pause = []
        component_to_activate = None
        
        if input_type == 'audio':
            # أولوية عالية لـ ASR
            component_to_activate = 'whisper_asr'
            components_to_pause = ['qwen_vision', 'deepseek_code', 'evolution_gate']
            
        elif input_type == 'screen':
            component_to_activate = 'qwen_vision'
            components_to_pause = ['whisper_asr', 'evolution_gate']
            
        elif input_type == 'code':
            component_to_activate = 'deepseek_code'
            components_to_pause = ['qwen_vision', 'evolution_gate']
            
        elif input_type == 'behavior':
            component_to_activate = 'behavior_lstm'
            components_to_pause = ['evolution_gate']
            
        else:
            component_to_activate = 'modality_router'
        
        return OrchestrationDecision(
            component_to_activate=component_to_activate,
            components_to_pause=components_to_pause,
            reason=f"User input type: {input_type}",
            priority=Priority.CRITICAL
        )
    
    async def _orchestrate_for_system_tick(self, system_health: Dict) -> OrchestrationDecision:
        """
        تنسيق دوري للنظام
        """
        # إذا كان النظام في حالة جيدة، نفعل الخلفية
        if system_health['status'] == 'healthy':
            return OrchestrationDecision(
                component_to_activate='memory_arbitration',
                components_to_pause=[],
                reason="Background memory maintenance",
                priority=Priority.LOW
            )
        else:
            # نظام متدهور - ندخل وضع الطوارئ
            return OrchestrationDecision(
                component_to_activate='agent_governor',
                components_to_pause=['evolution_gate', 'fluss_archive'],
                reason="System degraded - entering emergency mode",
                priority=Priority.CRITICAL
            )
    
    async def _orchestrate_for_error(self, context: Dict) -> OrchestrationDecision:
        """
        تنسيق استجابة للأخطاء
        """
        failed_component = context.get('component', 'unknown')
        
        # تعطيل المكون الفاشل
        if failed_component in self.components:
            self.components[failed_component].state = ComponentState.FAILED
            self.components[failed_component].consecutive_failures += 1
            self.stats['failures_handled'] += 1
        
        # تفعيل نظام الطوارئ
        return OrchestrationDecision(
            component_to_activate='agent_governor',
            components_to_pause=list(self.components.keys()),
            reason=f"Error in {failed_component} - activating emergency mode",
            priority=Priority.CRITICAL
        )
    
    async def _orchestrate_for_evolution(self, system_health: Dict) -> OrchestrationDecision:
        """
        تنسيق للتطور الذاتي - مع حواجز أمان
        """
        # لا نطور إذا كان النظام غير مستقر
        if system_health['status'] != 'healthy':
            return OrchestrationDecision(
                component_to_activate='',
                components_to_pause=['evolution_gate'],
                reason="System unstable - evolution blocked",
                priority=Priority.HIGH
            )
        
        return OrchestrationDecision(
            component_to_activate='evolution_gate',
            components_to_pause=[],
            reason="Scheduled evolution",
            priority=Priority.LOW
        )
    
    async def _orchestrate_default(self) -> OrchestrationDecision:
        """التنسيق الافتراضي"""
        return OrchestrationDecision(
            component_to_activate='modality_router',
            components_to_pause=[],
            reason="Default orchestration",
            priority=Priority.NORMAL
        )
    
    async def _apply_decision(self, decision: OrchestrationDecision):
        """تطبيق قرار التنسيق"""
        # تفعيل المكون المختار
        if decision.component_to_activate and decision.component_to_activate in self.components:
            self.components[decision.component_to_activate].state = ComponentState.ACTIVE
            self.components[decision.component_to_activate].last_active = time.time()
        
        # إيقاف المكونات المطلوبة
        for comp_name in decision.components_to_pause:
            if comp_name in self.components:
                self.components[comp_name].state = ComponentState.PAUSED
        
        # تحديث الحالة العالمية
        self.global_state['active_components'] = sum(
            1 for c in self.components.values() if c.state == ComponentState.ACTIVE
        )
        self.global_state['last_orchestration'] = time.time()
        
        logger.info(f"Orchestration: Active={decision.component_to_activate}, "
                   f"Paused={decision.components_to_pause}, Reason={decision.reason}")
    
    async def _assess_system_health(self) -> Dict:
        """تقييم صحة النظام"""
        # حساب متوسط زمن الاستجابة
        active_comps = [c for c in self.components.values() if c.state == ComponentState.ACTIVE]
        if active_comps:
            avg_latency = sum(c.current_latency_ms for c in active_comps) / len(active_comps)
            self.global_state['current_latency_ms'] = avg_latency
        
        # تحديد حالة النظام
        if self.global_state['current_latency_ms'] > self.global_state['total_latency_budget_ms'] * 1.5:
            status = 'degraded'
        elif self.global_state['current_latency_ms'] > self.global_state['total_latency_budget_ms'] * 2:
            status = 'critical'
        else:
            status = 'healthy'
        
        return {
            'status': status,
            'latency_ms': self.global_state['current_latency_ms'],
            'budget_ms': self.global_state['total_latency_budget_ms'],
            'active_components': self.global_state['active_components']
        }
    
    def report_component_latency(self, component_name: str, latency_ms: float):
        """الإبلاغ عن زمن استجابة مكون"""
        if component_name in self.components:
            self.components[component_name].current_latency_ms = latency_ms
    
    def report_component_error(self, component_name: str, error: Exception):
        """الإبلاغ عن خطأ في مكون"""
        if component_name in self.components:
            self.components[component_name].error_count += 1
            self.components[component_name].consecutive_failures += 1
            
            # إذا تجاوز الحد، ندخل وضع الطوارئ
            if self.components[component_name].consecutive_failures > 3:
                asyncio.create_task(self.orchestrate('error', {'component': component_name}))
    
    def get_status(self) -> Dict:
        """الحصول على حالة Control Plane"""
        return {
            'global_state': self.global_state,
            'stats': self.stats,
            'components': {
                name: {
                    'state': comp.state.value,
                    'priority': comp.priority.value,
                    'latency_ms': comp.current_latency_ms,
                    'errors': comp.error_count
                }
                for name, comp in self.components.items()
            },
            'recent_decisions': [
                {
                    'activate': d.component_to_activate,
                    'pause': d.components_to_pause,
                    'reason': d.reason
                }
                for d in list(self.decision_queue)[-5:]
            ]
        }

"""
Modality Router - يقرر أي نموذج يعمل لكل مدخل
يمنع تشغيل جميع النماذج معاً (Model Stack Paralysis)
"""

import time
from typing import Dict, List, Optional, Any, Tuple
from enum import Enum
from dataclasses import dataclass
import asyncio


class Modality(Enum):
    """أنواع المدخلات الممكنة"""
    AUDIO = "audio"
    VIDEO = "video"
    SCREEN = "screen"
    CODE = "code"
    BEHAVIOR = "behavior"
    TEXT = "text"


@dataclass
class ModelRoute:
    """توجيه لنموذج معين"""
    modality: Modality
    model_name: str
    priority: int  # 1 = highest
    estimated_latency_ms: float
    compute_cost: float  # relative
    fallback_model: Optional[str] = None


class ModalityRouter:
    """
    Modality Router - الطبقة الحاسمة التي تمنع تشغيل جميع النماذج معاً
    يختار نموذجاً واحداً فقط لكل مدخل حسب الأولوية والميزانية
    """
    
    def __init__(self, total_budget_ms: float = 200.0):
        self.total_budget_ms = total_budget_ms
        
        # تسجيل النماذج المتاحة
        self.model_registry: Dict[Modality, List[ModelRoute]] = {
            Modality.AUDIO: [
                ModelRoute(Modality.AUDIO, "whisper_streaming", 1, 150, 0.8, "whisper_tiny"),
                ModelRoute(Modality.AUDIO, "whisper_tiny", 3, 50, 0.3),
                ModelRoute(Modality.AUDIO, "vad_only", 4, 20, 0.1)
            ],
            Modality.SCREEN: [
                ModelRoute(Modality.SCREEN, "qwen_vl", 1, 200, 0.9, "llava"),
                ModelRoute(Modality.SCREEN, "llava", 2, 150, 0.7),
                ModelRoute(Modality.SCREEN, "ocr_only", 4, 30, 0.2)
            ],
            Modality.CODE: [
                ModelRoute(Modality.CODE, "deepseek_coder", 1, 300, 0.9, "qwen_coder"),
                ModelRoute(Modality.CODE, "qwen_coder", 2, 200, 0.7),
                ModelRoute(Modality.CODE, "tree_sitter", 3, 50, 0.2),
                ModelRoute(Modality.CODE, "regex_fallback", 4, 10, 0.05)
            ],
            Modality.BEHAVIOR: [
                ModelRoute(Modality.BEHAVIOR, "lstm_behavior", 1, 50, 0.3),
                ModelRoute(Modality.BEHAVIOR, "mlp_baseline", 3, 20, 0.1)
            ],
            Modality.VIDEO: [
                ModelRoute(Modality.VIDEO, "mediapipe", 1, 30, 0.2),
                ModelRoute(Modality.VIDEO, "opencv_motion", 3, 15, 0.1)
            ]
        }
        
        # حالة التوجيه الحالية
        self.current_routes: Dict[Modality, ModelRoute] = {}
        self.route_history: List[Dict] = []
        
        # إحصائيات الأداء
        self.model_performance: Dict[str, Dict] = {}
        for routes in self.model_registry.values():
            for route in routes:
                self.model_performance[route.model_name] = {
                    'avg_latency_ms': route.estimated_latency_ms,
                    'success_rate': 1.0,
                    'times_used': 0
                }
        
        # ميزانية الحساب المتبقية
        self.remaining_budget_ms = total_budget_ms
    
    async def route(self, input_data: Dict) -> Tuple[str, Any]:
        """
        توجيه المدخل إلى النموذج المناسب
        المخرج: (model_name, result)
        """
        # تحديد نوع المدخل
        modality = self._detect_modality(input_data)
        
        # اختيار النموذج المناسب
        selected_route = self._select_model(modality)
        
        if not selected_route:
            # فشل في التوجيه - استخدام الـ fallback
            return await self._fallback_route(input_data, modality)
        
        # تسجيل الاستخدام
        self.model_performance[selected_route.model_name]['times_used'] += 1
        self.current_routes[modality] = selected_route
        
        # تقليل الميزانية
        self.remaining_budget_ms -= selected_route.estimated_latency_ms
        
        # تسجيل التاريخ
        self.route_history.append({
            'timestamp': time.time(),
            'modality': modality.value,
            'model': selected_route.model_name,
            'budget_remaining': self.remaining_budget_ms
        })
        
        # إرجاع النموذج المختار (سيتم استدعاؤه من المتصل)
        return selected_route.model_name, input_data
    
    def _detect_modality(self, input_data: Dict) -> Modality:
        """كشف نوع المدخل"""
        if 'audio' in input_data or 'wav' in input_data:
            return Modality.AUDIO
        elif 'screen' in input_data or 'screenshot' in input_data:
            return Modality.SCREEN
        elif 'code' in input_data or 'source_code' in input_data:
            return Modality.CODE
        elif 'video' in input_data or 'frame' in input_data:
            return Modality.VIDEO
        elif 'behavior' in input_data or 'mouse' in input_data or 'keyboard' in input_data:
            return Modality.BEHAVIOR
        else:
            return Modality.TEXT
    
    def _select_model(self, modality: Modality) -> Optional[ModelRoute]:
        """
        اختيار أفضل نموذج حسب:
        1. الأولوية
        2. الميزانية المتبقية
        3. الأداء التاريخي
        """
        available_models = self.model_registry.get(modality, [])
        
        if not available_models:
            return None
        
        # ترتيب حسب الأولوية
        sorted_models = sorted(available_models, key=lambda x: x.priority)
        
        # اختيار أول نموذج يناسب الميزانية
        for model in sorted_models:
            if model.estimated_latency_ms <= self.remaining_budget_ms:
                # تحقق من الأداء التاريخي
                perf = self.model_performance.get(model.model_name, {})
                if perf.get('success_rate', 1.0) > 0.7:  # نجاح >70%
                    return model
        
        # إذا لم يناسب أي نموذج الميزانية، نأخذ الأسرع
        fastest = min(available_models, key=lambda x: x.estimated_latency_ms)
        return fastest
    
    async def _fallback_route(self, input_data: Dict, modality: Modality) -> Tuple[str, Any]:
        """توجيه احتياطي عند فشل التوجيه العادي"""
        # استخدام أخف نموذج ممكن
        fallback_model = "text_embedder"  # أبسط نموذج
        
        self.route_history.append({
            'timestamp': time.time(),
            'modality': modality.value,
            'model': fallback_model,
            'is_fallback': True
        })
        
        return fallback_model, input_data
    
    def update_model_performance(self, model_name: str, latency_ms: float, success: bool):
        """تحديث أداء النموذج بعد الاستخدام"""
        if model_name in self.model_performance:
            perf = self.model_performance[model_name]
            perf['avg_latency_ms'] = perf['avg_latency_ms'] * 0.9 + latency_ms * 0.1
            perf['success_rate'] = perf['success_rate'] * 0.95 + (1.0 if success else 0.0) * 0.05
            perf['times_used'] += 1
    
    def reset_budget(self):
        """إعادة تعيين الميزانية لكل دورة معالجة"""
        self.remaining_budget_ms = self.total_budget_ms
    
    def get_status(self) -> Dict:
        """الحالة الحالية للموجه"""
        return {
            'remaining_budget_ms': self.remaining_budget_ms,
            'total_budget_ms': self.total_budget_ms,
            'current_routes': {
                m.value: r.model_name for m, r in self.current_routes.items()
            },
            'model_performance': self.model_performance,
            'recent_routes': self.route_history[-10:]
        }

"""
Memory Arbitration Engine - يقرر ماذا يبقى وماذا يُنسى في الذاكرة
score = relevance * recency * importance - decay_rate * age
"""

import time
import math
from typing import Dict, List, Any, Optional, Tuple
from dataclasses import dataclass, field
from enum import Enum
import numpy as np


class MemoryType(Enum):
    """أنواع الذاكرة"""
    VECTOR = "vector"
    GRAPH = "graph"
    EPISODIC = "episodic"


@dataclass
class MemoryItem:
    """عنصر في الذاكرة"""
    id: str
    content: Any
    memory_type: MemoryType
    relevance_score: float  # 0-1
    importance: float  # 0-1
    created_at: float
    last_accessed: float
    access_count: int = 0
    decay_rate: float = 0.1  # افتراضي
    
    def compute_current_score(self, current_time: float) -> float:
        """حساب الدرجة الحالية للعنصر"""
        age = current_time - self.created_at
        
        # معامل الحداثة (recency)
        recency = math.exp(-self.decay_rate * age)
        
        # معامل التكرار (frequency)
        frequency = min(1.0, self.access_count / 100)
        
        # الصيغة النهائية
        score = (
            self.relevance_score * 0.4 +
            recency * 0.3 +
            self.importance * 0.2 +
            frequency * 0.1
        ) - (self.decay_rate * age / 86400)  # خصم الاضمحلال اليومي
        
        return max(0.0, min(1.0, score))


class MemoryArbitrationEngine:
    """
    محرك تحكيم الذاكرة - يقرر ماذا يبقى وماذا يُنسى
    هذا يحل مشكلة "الذاكرة بلا سياسة"
    """
    
    def __init__(self, max_items_per_type: int = 10000):
        self.max_items_per_type = max_items_per_type
        
        # تخزين عناصر الذاكرة
        self.memory_items: Dict[MemoryType, List[MemoryItem]] = {
            MemoryType.VECTOR: [],
            MemoryType.GRAPH: [],
            MemoryType.EPISODIC: []
        }
        
        # سياسات الاضمحلال لكل نوع
        self.decay_policies: Dict[MemoryType, float] = {
            MemoryType.VECTOR: 0.05,     # بطيء
            MemoryType.GRAPH: 0.03,      # أبطأ
            MemoryType.EPISODIC: 0.15    # سريع (التجارب القصيرة)
        }
        
        # إحصائيات
        self.stats = {
            'items_added': 0,
            'items_evicted': 0,
            'items_merged': 0,
            'last_arbitration': 0.0
        }
        
        # عتبات الإزالة
        self.eviction_thresholds = {
            MemoryType.VECTOR: 0.2,
            MemoryType.GRAPH: 0.15,
            MemoryType.EPISODIC: 0.1
        }
    
    async def store(self, content: Any, memory_type: MemoryType, 
                   relevance: float, importance: float) -> str:
        """
        تخزين عنصر جديد في الذاكرة
        """
        item_id = f"{memory_type.value}_{int(time.time() * 1000)}_{self.stats['items_added']}"
        
        current_time = time.time()
        memory_item = MemoryItem(
            id=item_id,
            content=content,
            memory_type=memory_type,
            relevance_score=relevance,
            importance=importance,
            created_at=current_time,
            last_accessed=current_time,
            decay_rate=self.decay_policies[memory_type]
        )
        
        self.memory_items[memory_type].append(memory_item)
        self.stats['items_added'] += 1
        
        # التحقق من الحاجة للتحكيم
        if len(self.memory_items[memory_type]) > self.max_items_per_type * 0.9:
            await self.arbitrate()
        
        return item_id
    
    async def retrieve(self, memory_type: MemoryType, query: str, top_k: int = 5) -> List[MemoryItem]:
        """
        استرجاع العناصر الأكثر صلة
        """
        current_time = time.time()
        
        # حساب الدرجات لجميع العناصر
        for item in self.memory_items[memory_type]:
            item.last_accessed = current_time
            item.access_count += 1
        
        # ترتيب حسب الدرجة
        scored_items = [
            (item, item.compute_current_score(current_time))
            for item in self.memory_items[memory_type]
        ]
        scored_items.sort(key=lambda x: x[1], reverse=True)
        
        return [item for item, _ in scored_items[:top_k]]
    
    async def arbitrate(self):
        """
        عملية التحكيم - يقرر ماذا يبقى وماذا يُنسى
        """
        current_time = time.time()
        self.stats['last_arbitration'] = current_time
        
        for memory_type, items in self.memory_items.items():
            if len(items) <= self.max_items_per_type:
                continue
            
            # حساب الدرجات
            scored_items = []
            for item in items:
                score = item.compute_current_score(current_time)
                scored_items.append((item, score))
            
            # ترتيب تنازلي
            scored_items.sort(key=lambda x: x[1], reverse=True)
            
            # الاحتفاظ بأفضل العناصر فقط
            keep_count = int(self.max_items_per_type * 0.8)
            keep_items = scored_items[:keep_count]
            evict_items = scored_items[keep_count:]
            
            # حذف العناصر منخفضة الدرجة
            self.memory_items[memory_type] = [item for item, _ in keep_items]
            self.stats['items_evicted'] += len(evict_items)
            
            # محاولة دمج العناصر المتشابهة
            await self._try_merge_similar(memory_type, keep_items)
    
    async def _try_merge_similar(self, memory_type: MemoryType, items: List[Tuple[MemoryItem, float]]):
        """
        محاولة دمج العناصر المتشابهة لتوفير المساحة
        """
        # تبسيطاً: دمج العناصر ذات المحتوى المتطابق
        content_map = {}
        
        for item, score in items:
            content_hash = str(item.content)[:100]  # تبسيط
            
            if content_hash in content_map:
                # دمج مع العنصر الموجود
                existing = content_map[content_hash]
                existing.importance = max(existing.importance, item.importance)
                existing.access_count += item.access_count
                self.stats['items_merged'] += 1
            else:
                content_map[content_hash] = item
    
    async def forget(self, memory_type: MemoryType, item_id: str) -> bool:
        """
        نسيان عنصر محدد
        """
        items = self.memory_items[memory_type]
        for i, item in enumerate(items):
            if item.id == item_id:
                self.memory_items[memory_type].pop(i)
                self.stats['items_evicted'] += 1
                return True
        return False
    
    async def get_memory_summary(self, memory_type: Optional[MemoryType] = None) -> Dict:
        """
        الحصول على ملخص الذاكرة
        """
        if memory_type:
            items = self.memory_items[memory_type]
            current_time = time.time()
            scores = [item.compute_current_score(current_time) for item in items]
            
            return {
                'type': memory_type.value,
                'count': len(items),
                'max_capacity': self.max_items_per_type,
                'avg_score': np.mean(scores) if scores else 0,
                'min_score': np.min(scores) if scores else 0,
                'max_score': np.max(scores) if scores else 0
            }
        
        return {
            'stats': self.stats,
            'per_type': {
                t.value: {
                    'count': len(items),
                    'max_capacity': self.max_items_per_type,
                    'utilization': len(items) / self.max_items_per_type
                }
                for t, items in self.memory_items.items()
            }
        }

# backend/part5_v5/agent_governor.py
"""
Agent Governor - التحكم المركزي بالوكلاء المتعددين
يمنع التضارب والحلقات اللانهائية
"""

import asyncio
import time
from typing import Dict, List, Any, Optional, Tuple
from enum import Enum
from dataclasses import dataclass, field
from collections import deque


class AgentRole(Enum):
    """أدوار الوكلاء في النظام"""
    PLANNER = "planner"
    EXECUTOR = "executor"
    CRITIC = "critic"
    SPECIALIST = "specialist"


class ConflictType(Enum):
    """أنواع التعارضات بين الوكلاء"""
    CONTRADICTORY_OUTPUT = "contradictory_output"
    RESOURCE_CONTENTION = "resource_contention"
    INFINITE_LOOP = "infinite_loop"
    PRIORITY_INVERSION = "priority_inversion"


@dataclass
class AgentOutput:
    """خرج وكيل"""
    agent_name: str
    role: AgentRole
    content: Any
    confidence: float
    timestamp: float = field(default_factory=time.time)
    priority: int = 2  # 1-5, 1 = highest


@dataclass
class Resolution:
    """قرار حل التعارض"""
    conflict_type: ConflictType
    selected_output: AgentOutput
    rejected_outputs: List[AgentOutput]
    reason: str
    timestamp: float = field(default_factory=time.time)


class AgentGovernor:
    """
    Agent Governor - العقل المنظم للوكلاء المتعددين
    يقرر من يتحدث ومتى وكيف يتم حل التعارضات
    """
    
    def __init__(self):
        # تسجيل الوكلاء
        self.agents: Dict[str, Dict] = {}
        
        # سجل المخرجات
        self.output_history: deque = deque(maxlen=100)
        self.resolution_history: deque = deque(maxlen=50)
        
        # حالة التنفيذ الحالية
        self.current_session: Dict[str, Any] = {
            'active_planner': None,
            'executing_agents': [],
            'pending_outputs': [],
            'start_time': time.time()
        }
        
        # قواعد حل التعارضات
        self.resolution_rules = {
            ConflictType.CONTRADICTORY_OUTPUT: self._resolve_contradiction,
            ConflictType.RESOURCE_CONTENTION: self._resolve_resource_contention,
            ConflictType.INFINITE_LOOP: self._resolve_infinite_loop,
            ConflictType.PRIORITY_INVERSION: self._resolve_priority_inversion
        }
        
        # إحصائيات
        self.stats = {
            'total_outputs': 0,
            'conflicts_detected': 0,
            'conflicts_resolved': 0,
            'loops_prevented': 0
        }
        
        # كشف الحلقات اللانهائية
        self.action_trace: deque = deque(maxlen=20)
    
    def register_agent(self, name: str, role: AgentRole, priority: int = 2):
        """تسجيل وكيل في النظام"""
        self.agents[name] = {
            'role': role,
            'priority': priority,
            'last_output': None,
            'is_active': False,
            'total_activations': 0
        }
    
    async def submit_output(self, agent_name: str, content: Any, confidence: float) -> AgentOutput:
        """
        تقديم خرج من وكيل
        """
        if agent_name not in self.agents:
            raise ValueError(f"Unknown agent: {agent_name}")
        
        agent_info = self.agents[agent_name]
        output = AgentOutput(
            agent_name=agent_name,
            role=agent_info['role'],
            content=content,
            confidence=confidence,
            priority=agent_info['priority']
        )
        
        # تسجيل المخرج
        self.output_history.append(output)
        agent_info['last_output'] = output
        agent_info['total_activations'] += 1
        self.stats['total_outputs'] += 1
        
        # إضافة إلى المخرجات المعلقة
        self.current_session['pending_outputs'].append(output)
        
        # كشف وتنظيف التعارضات
        await self._detect_and_resolve_conflicts()
        
        return output
    
    async def _detect_and_resolve_conflicts(self):
        """
        كشف التعارضات بين المخرجات وحلها
        """
        pending = self.current_session['pending_outputs']
        if len(pending) < 2:
            return
        
        # كشف التعارضات
        conflicts = await self._detect_conflicts(pending)
        
        for conflict in conflicts:
            self.stats['conflicts_detected'] += 1
            
            # حل التعارض
            resolver = self.resolution_rules.get(conflict['type'])
            if resolver:
                resolution = await resolver(conflict)
                self.resolution_history.append(resolution)
                self.stats['conflicts_resolved'] += 1
                
                # تطبيق القرار
                await self._apply_resolution(resolution)
    
    async def _detect_conflicts(self, outputs: List[AgentOutput]) -> List[Dict]:
        """
        كشف التعارضات بين المخرجات
        """
        conflicts = []
        
        # 1. التعارضات المنطقية (مخرجات متناقضة)
        for i, out1 in enumerate(outputs):
            for out2 in outputs[i+1:]:
                if self._are_contradictory(out1, out2):
                    conflicts.append({
                        'type': ConflictType.CONTRADICTORY_OUTPUT,
                        'outputs': [out1, out2],
                        'description': f"Contradiction between {out1.agent_name} and {out2.agent_name}"
                    })
        
        # 2. كشف الحلقات اللانهائية
        if await self._detect_infinite_loop():
            conflicts.append({
                'type': ConflictType.INFINITE_LOOP,
                'outputs': list(self.action_trace)[-5:],
                'description': "Potential infinite loop detected"
            })
        
        return conflicts
    
    def _are_contradictory(self, out1: AgentOutput, out2: AgentOutput) -> bool:
        """
        تحديد ما إذا كان مخرجان متناقضان
        """
        # تبسيطاً: إذا كان النوع نصياً، نبحث عن تعارض
        if isinstance(out1.content, str) and isinstance(out2.content, str):
            # كلمات تشير إلى تعارض
            contradiction_words = ['not', 'wrong', 'incorrect', 'better', 'instead']
            
            content1_lower = out1.content.lower()
            content2_lower = out2.content.lower()
            
            for word in contradiction_words:
                if word in content1_lower and word in content2_lower:
                    return True
        
        return False
    
    async def _detect_infinite_loop(self) -> bool:
        """
        كشف الحلقات اللانهائية في تسلسل الإجراءات
        """
        # إضافة الإجراء الحالي للتتبع
        if self.current_session['active_planner']:
            self.action_trace.append(self.current_session['active_planner'])
        
        # كشف التكرار
        if len(self.action_trace) >= 10:
            # البحث عن نمط متكرر
            recent = list(self.action_trace)[-5:]
            if len(set(recent)) <= 2:  # نفس الإجراء يتكرر
                self.stats['loops_prevented'] += 1
                return True
        
        return False
    
    async def _resolve_contradiction(self, conflict: Dict) -> Resolution:
        """
        حل التعارضات المنطقية
        """
        outputs = conflict['outputs']
        
        # اختيار المخرج بأعلى ثقة وأولوية
        best = max(outputs, key=lambda o: (o.priority, o.confidence))
        rejected = [o for o in outputs if o != best]
        
        return Resolution(

"""
Verifier Agent - يتحقق من صحة الاستدلالات والنتائج
Reason → Execute → Verify → Correct
"""

import asyncio
import time
from typing import Dict, List, Any, Optional, Tuple
from dataclasses import dataclass
from enum import Enum


class VerificationStatus(Enum):
    """حالات التحقق"""
    PASSED = "passed"
    FAILED = "failed"
    PARTIAL = "partial"
    UNCERTAIN = "uncertain"


@dataclass
class VerificationResult:
    """نتيجة التحقق"""
    status: VerificationStatus
    confidence: float
    feedback: str
    corrections: Optional[List[str]]
    execution_time_ms: float


class VerifierAgent:
    """
    وكيل التحقق - يضمن صحة الاستدلالات قبل التنفيذ النهائي
    هذا يمنع الأخطاء المتسلسلة ويضمن الموثوقية
    """
    
    def __init__(self, sandbox_executor):
        self.sandbox = sandbox_executor
        self.verification_rules = self._load_rules()
        self.verification_history: List[Dict] = []
        
    async def verify(self, reasoning_output: str, code: str, context: Dict) -> VerificationResult:
        """
        التحقق من صحة مخرج التفكير والكود
        """
        start_time = time.time()
        
        # 1. التحقق النحوي (syntax check)
        syntax_ok, syntax_errors = await self._check_syntax(code)
        
        if not syntax_ok:
            return VerificationResult(
                status=VerificationStatus.FAILED,
                confidence=0.0,
                feedback=f"Syntax errors: {syntax_errors}",
                corrections=self._suggest_syntax_fixes(syntax_errors),
                execution_time_ms=(time.time() - start_time) * 1000
            )
        
        # 2. التحقق الدلالي (semantic check)
        semantic_ok, semantic_issues = await self._check_semantics(reasoning_output, code)
        
        if not semantic_ok:
            return VerificationResult(
                status=VerificationStatus.FAILED if len(semantic_issues) > 2 else VerificationStatus.PARTIAL,
                confidence=0.5,
                feedback=f"Semantic issues: {semantic_issues[:3]}",
                corrections=self._suggest_semantic_fixes(semantic_issues),
                execution_time_ms=(time.time() - start_time) * 1000
            )
        
        # 3. التحقق بالتنفيذ (execution in sandbox)
        execution_result = await self._execute_in_sandbox(code, context.get('test_cases', []))
        
        if not execution_result['passed']:
            return VerificationResult(
                status=VerificationStatus.FAILED,
                confidence=0.3,
                feedback=f"Execution failed: {execution_result.get('error', 'Unknown error')}",
                corrections=self._suggest_execution_fixes(execution_result),
                execution_time_ms=(time.time() - start_time) * 1000
            )
        
        # 4. التحقق من الكفاءة (efficiency check)
        efficiency_ok, efficiency_score = await self._check_efficiency(code, execution_result)
        
        if not efficiency_ok:
            return VerificationResult(
                status=VerificationStatus.PARTIAL,
                confidence=efficiency_score,
                feedback=f"Efficiency could be improved (score: {efficiency_score:.2f})",
                corrections=self._suggest_efficiency_improvements(code),
                execution_time_ms=(time.time() - start_time) * 1000
            )
        
        # كل شيء يعمل بشكل صحيح
        self._log_verification(reasoning_output, code, True)
        
        return VerificationResult(
            status=VerificationStatus.PASSED,
            confidence=1.0,
            feedback="All verification checks passed",
            corrections=[],
            execution_time_ms=(time.time() - start_time) * 1000
        )
    
    async def _check_syntax(self, code: str) -> Tuple[bool, List[str]]:
        """التحقق النحوي للكود"""
        errors = []
        
        # استخدام AST لفحص الصياغة
        try:
            import ast
            ast.parse(code)
        except SyntaxError as e:
            errors.append(str(e))
        except Exception as e:
            errors.append(f"Unknown syntax error: {e}")
        
        return len(errors) == 0, errors
    
    async def _check_semantics(self, reasoning: str, code: str) -> Tuple[bool, List[str]]:
        """التحقق الدلالي - هل المنطق يتوافق مع الكود؟"""
        issues = []
        
        # التحقق من وجود الوظائف المذكورة في التفكير
        import re
        functions_in_reasoning = re.findall(r'\b(\w+)\s*\(', reasoning)
        
        for func in functions_in_reasoning:
            if func not in code:
                issues.append(f"Function '{func}' mentioned in reasoning but not in code")
        
        # التحقق من الشروط المنطقية
        if 'if' in reasoning and 'if' not in code:
            issues.append("Conditional logic mentioned but not implemented")
        
        if 'loop' in reasoning.lower() and 'for' not in code and 'while' not in code:
            issues.append("Loop mentioned but not implemented")
        
        return len(issues) <= 2, issues  # مسموح ببعض الانحرافات
    
    async def _execute_in_sandbox(self, code: str, test_cases: List) -> Dict:
        """تنفيذ الكود في بيئة معزولة"""
        # في الإنتاج: استخدام Docker أو sandbox حقيقي
        # تبسيطاً: محاكاة
        return {
            'passed': True,
            'outputs': ['OK'] * len(test_cases) if test_cases else ['OK'],
            'execution_time_ms': 50
        }
    
    async def _check_efficiency(self, code: str, execution_result: Dict) -> Tuple[bool, float]:
        """التحقق من كفاءة الكود"""
        # تحليل التعقيد الزمني
        import re
        nested_loops = len(re.findall(r'for.*for', code)) + len(re.findall(r'while.*while', code))
        
        if nested_loops > 2:
            return False, 0.6
        elif nested_loops > 0:
            return True, 0.8
        else:
            return True, 1.0
    
    def _suggest_syntax_fixes(self, errors: List[str]) -> List[str]:
        """اقتراح إصلاحات للأخطاء النحوية"""
        suggestions = []
        for error in errors:
            if "expected ':'" in error:
                suggestions.append("Add colon ':' at end of function/class/loop definition")
            elif "invalid syntax" in error:
                suggestions.append("Check parentheses and brackets")
        return suggestions
    
    def _suggest_semantic_fixes(self, issues: List[str]) -> List[str]:
        """اقتراح إصلاحات للمشاكل الدلالية"""
        suggestions = []
        for issue in issues:
            if "mentioned in reasoning but not in code" in issue:
                suggestions.append("Implement missing functions mentioned in reasoning")
        return suggestions
    
    def _suggest_execution_fixes(self, execution_result: Dict) -> List[str]:
        """اقتراح إصلاحات لمشاكل التنفيذ"""
        return ["Check logic", "Add error handling", "Verify input/output types"]
    
    def _suggest_efficiency_improvements(self, code: str) -> List[str]:
        """اقتراح تحسينات الكفاءة"""
        suggestions = []
        if 'for' in code and 'for' in code[code.find('for')+1:]:
            suggestions.append("Reduce nested loops")
        return suggestions

        """تسجيل التحقق للتحليل لاحقاً"""
        self.verification_history.append({

            'code_length': len(code),
            'passed': passed
        })
        
        # الاحتفاظ بآخر 1000 تحقق فقط
        if len(self.verification_history) > 1000:
            self.verification_history = self.verification_history[-1000:]
    
    def get_stats(self) -> Dict:
        """إحصائيات التحقق"""
        if not self.verification_history:
            return {'pass_rate': 0, 'total': 0}
        
        passed = sum(1 for v in self.verification_history if v['passed'])
        return {
            'total_verifications': len(self.verification_history),
            'pass_count': passed,
            'pass_rate': passed / len(self.verification_history)

            self._check_model_stability

        if proposal.risk_score > self.max_risk:

                proposal_id=proposal.id,

        reward = (
            self.weights['skill_gain'] * skill_gain +
            self.weights['speed_gain'] * speed_gain -
            self.weights['frustration_penalty'] * frustration_level
        )
        
        # تحديد الفائز
        if human_score > twin_score + 5:
            outcome = CompetitionOutcome.HUMAN_WINS
            self.stats['human_wins'] += 1
        elif twin_score > human_score + 5:
            outcome = CompetitionOutcome.TWIN_WINS
            self.stats['twin_wins'] += 1
        else:
            outcome = CompetitionOutcome.TIE
            self.stats['ties'] += 1
        
        # إنشاء سجل الجولة
        round_result = CompetitionRound(
            round_id=f"round_{len(self.rounds) + 1}",
            problem=problem,
            human_solution=human_solution,
            twin_solution=twin_solution,
            human_score=human_score,
            twin_score=twin_score,
            human_time_sec=human_time_sec,
            twin_time_sec=twin_time_sec,
            frustration_level=frustration_level,
            outcome=outcome,
            reward=reward
        )
        
        self.rounds.append(round_result)
        self.stats['total_rounds'] += 1
        
        # تحديث متوسط المكافأة
        all_rewards = [r.reward for r in self.rounds]
        self.stats['avg_reward'] = sum(all_rewards) / len(all_rewards)
        
        # تحديث الاتجاه
        self.stats['trend'] = await self._compute_trend()
        
        return round_result
    
    async def _compute_skill_gain(self, human_score: float, twin_score: float) -> float:
        """
        حساب مكسب المهارة
        كلما كانت المنافسة متقاربة، زاد المكسب
        """
        # الفرق المطلق
        difference = abs(human_score - twin_score) / 100
        
        # المكسب أعلى عندما يكون الفرق صغيراً (منافسة قوية)
        if difference < 0.1:
            return 0.9  # منافسة شديدة
        elif difference < 0.2:
            return 0.7
        elif difference < 0.3:
            return 0.5
        elif difference < 0.4:
            return 0.3
        else:
            return 0.1  # منافسة ضعيفة
    
    async def _compute_speed_gain(self, human_time: float, twin_time: float) -> float:
        """
        حساب مكسب السرعة
        """
        if human_time <= 0 or twin_time <= 0:
            return 0.5
        
        # نسبة السرعة
        if human_time < twin_time:
            # الإنسان أسرع
            speed_ratio = human_time / twin_time
            return min(0.9, 0.5 + speed_ratio)  # مكافأة للسرعة
        else:
            # الآلة أسرع
            speed_ratio = twin_time / human_time
            return max(0.1, 0.5 - speed_ratio)  # خصم للبطء
    
    async def _compute_trend(self) -> str:
        """
        حساب اتجاه الأداء
        """
        if len(self.history_scores['human']) < 5:
            return 'stable'
        
        # حساب المتوسطات الأخيرة
        recent_human = sum(self.history_scores['human'][-5:]) / 5
        previous_human = sum(self.history_scores['human'][-10:-5]) / 5 if len(self.history_scores['human']) >= 10 else recent_human
        
        recent_twin = sum(self.history_scores['twin'][-5:]) / 5
        previous_twin = sum(self.history_scores['twin'][-10:-5]) / 5 if len(self.history_scores['twin']) >= 10 else recent_twin
        
        human_improvement = recent_human - previous_human
        twin_improvement = recent_twin - previous_twin
        
        if human_improvement > 0.05 and twin_improvement < 0.05:
            return 'human_dominating'
        elif twin_improvement > 0.05 and human_improvement < 0.05:
            return 'twin_dominating'
        elif abs(human_improvement) < 0.02 and abs(twin_improvement) < 0.02:
            return 'stable'
        else:
            return 'improving_together'
    
    async def get_adaptive_difficulty(self) -> float:
        """
        ضبط الصعوبة بشكل تكيفي بناءً على نتائج المنافسة
        """
        if len(self.rounds) < 10:
            return 0.5  # صعوبة متوسطة في البداية
        
        # حساب نسبة فوز الإنسان في آخر 10 جولات
        recent_rounds = self.rounds[-10:]
        human_wins = sum(1 for r in recent_rounds if r.outcome == CompetitionOutcome.HUMAN_WINS)
        human_win_rate = human_wins / 10
        
        # ضبط الصعوبة للحفاظ على منافسة متوازنة (هدف 50% فوز لكل جانب)
        if human_win_rate > 0.6:
            # الإنسان يفوز كثيراً - نرفع الصعوبة
            return min(0.9, 0.5 + (human_win_rate - 0.5) * 1.5)
        elif human_win_rate < 0.4:
            # الآلة تفوز كثيراً - نخفض الصعوبة
            return max(0.1, 0.5 - (0.5 - human_win_rate) * 1.5)
        else:
            return 0.5  # متوازن
    
    def get_stats(self) -> Dict:
        """الحصول على إحصائيات المنافسة"""
        return {
            'total_rounds': self.stats['total_rounds'],
            'human_wins': self.stats['human_wins'],
            'twin_wins': self.stats['twin_wins'],
            'ties': self.stats['ties'],
            'human_win_rate': self.stats['human_wins'] / max(1, self.stats['total_rounds']),
            'twin_win_rate': self.stats['twin_wins'] / max(1, self.stats['total_rounds']),
            'avg_reward': self.stats['avg_reward'],
            'trend': self.stats['trend'],
            'adaptive_difficulty': self._get_adaptive_difficulty_sync()
        }
    
    def _get_adaptive_difficulty_sync(self) -> float:
        """نسخة متزامنة لـ get_adaptive_difficulty"""
        import asyncio
        try:
            loop = asyncio.get_event_loop()
            if loop.is_running():
                # إذا كان الحدث قيد التشغيل، نستخدم قيمة افتراضية
                return 0.5
            return loop.run_until_complete(self.get_adaptive_difficulty())
        except RuntimeError:
            return 0.5

"""
Pipeline النهائي للجزء الخامس V5 - الإصدار الهندسي القابل للبناء
يدمج جميع المكونات في نظام واحد متكامل وموثوق

import asyncio

import logging

from .control_plane import ControlPlane, Priority

from .verifier import VerifierAgent

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


@dataclass
class ProcessingResult:
    """نتيجة معالجة مدخل"""
    input_id: str
    success: bool
    latency_ms: float
    output: Any
    route_used: str
    verification_passed: bool


class TechLinguaMentorV5:
    """
    النظام النهائي V5 - TechLingua Mentor
    الإصدار الهندسي القابل للبناء والتنفيذ
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        self.start_time = time.time()
        
        # المكونات الأساسية
        self.control_plane = ControlPlane()
        self.modality_router = ModalityRouter(total_budget_ms=200.0)
        self.memory_arbitration = MemoryArbitrationEngine()
        self.agent_governor = AgentGovernor()
        self.verifier = None  # سيتم تهيئته مع sandbox
        self.evolution_gate = EvolutionGate()
        self.competition_reward = CompetitionRewardSystem()
        
        # إحصائيات
        self.processed_inputs: List[ProcessingResult] = []
        self.stats = {
            'total_inputs': 0,
            'avg_latency_ms': 0.0,
            'success_rate': 1.0,
            'active_since': self.start_time
        }
        
        # حالة النظام
        self.is_running = True
        
    async def initialize(self):
        """تهيئة النظام بالكامل"""
        # تسجيل الوكلاء في الحوكمة
        self.agent_governor.register_agent("planner", AgentRole.PLANNER, priority=1)
        self.agent_governor.register_agent("code_executor", AgentRole.EXECUTOR, priority=2)
        self.agent_governor.register_agent("critic", AgentRole.CRITIC, priority=1)
        self.agent_governor.register_agent("optimizer", AgentRole.SPECIALIST, priority=3)
        
        # تهيئة المدقق
        self.verifier = VerifierAgent(sandbox_executor=None)  # سيتم إضافة sandbox حقيقي
        
        logger.info(f"TechLinguaMentor V5 initialized for user {self.user_id}")
    
    async def process_input(self, input_data: Dict) -> ProcessingResult:
        """
        معالجة مدخل جديد - نقطة الدخول الرئيسية
        """
        if not self.is_running:
            raise RuntimeError("System is not running")
        
        input_id = f"input_{int(time.time() * 1000)}_{self.stats['total_inputs']}"
        start_time = time.time()
        
        try:
            # 1. Control Plane يقرر من يعمل
            orchestration = await self.control_plane.orchestrate(
                'user_input',
                {'input_type': input_data.get('type', 'unknown')}
            )
            
            # 2. Modality Router يختار النموذج المناسب
            model_name, routed_data = await self.modality_router.route(input_data)
            
            # 3. معالجة حسب النوع (محاكاة)
            # في الإنتاج: استدعاء النموذج الفعلي
            processed_output = await self._process_with_model(model_name, routed_data)
            
            # 4. تخزين في الذاكرة
            memory_id = await self.memory_arbitration.store(
                content=processed_output,
                memory_type=MemoryType.EPISODIC,
                relevance=0.8,
                importance=0.6
            )
            
            # 5. تقديم خرج للحوكمة
            governor_output = await self.agent_governor.submit_output(
                agent_name=model_name,
                content=processed_output,
                confidence=0.85
            )
            
            # 6. التحقق من صحة الخرج
            if self.verifier and 'code' in processed_output:
                verification = await self.verifier.verify(
                    reasoning_output=str(processed_output),
                    code=processed_output.get('code', ''),
                    context=input_data
                )
                verification_passed = verification.status.value == 'passed'
            else:
                verification_passed = True
            
            # 7. تحديث الإحصائيات
            latency_ms = (time.time() - start_time) * 1000
            self._update_stats(latency_ms, True)
            
            # 8. إنشاء النتيجة
            result = ProcessingResult(
                input_id=input_id,
                success=True,
                latency_ms=latency_ms,
                output=governor_output.content,
                route_used=model_name,
                verification_passed=verification_passed
            )
            
            self.processed_inputs.append(result)
            
            # إعادة تعيين الميزانية للدورة التالية
            self.modality_router.reset_budget()
            
            return result
            
        except Exception as e:
            logger.error(f"Error processing input {input_id}: {e}")
            latency_ms = (time.time() - start_time) * 1000
            self._update_stats(latency_ms, False)
            
            # إبلاغ Control Plane بالخطأ
            await self.control_plane.report_component_error('processing', e)
            
            return ProcessingResult(
                input_id=input_id,
                success=False,
                latency_ms=latency_ms,
                output=None,
                route_used='error_fallback',
                verification_passed=False
            )

    async def run_competition(self, problem: str, human_solution: str) -> Dict:

        twin_score = await self._evaluate_solution(twin_solution, problem)
        
        # قياس الإحباط من السلوك (محاكاة)

            'twin_time_sec': 60,
            'frustration_level': frustration
        })
        
        # ضبط الصعوبة للمرة القادمة
        next_difficulty = await self.competition_reward.get_adaptive_difficulty()

    async def self_evolve(self) -> Dict:

        return {

        alpha = 0.95
        self.stats['avg_latency_ms'] = (

            (1 - alpha) * latency_ms

        total = self.stats['total_inputs']
        successes = sum(1 for r in self.processed_inputs if r.success)
        successes += (1 if success else 0)
        self.stats['success_rate'] = successes / total

class LatencyPredictor:
    """يتوقع زمن الاستجابة باستخدام EWMA +季节性"""
    
    def __init__(self, alpha: float = 0.3, beta: float = 0.2):
        self.alpha = alpha
        self.beta = beta
        self.base_latency = 100.0
        self.trend = 0.0
        self.seasonal = [0.0] * 10  # 10 فترات موسمية
        self.history = deque(maxlen=1000)
        
    async def predict(self, telemetry: TelemetryData, pending_tasks: int) -> float:
        """توقع زمن الاستجابة خلال الـ 5 ثوانٍ القادمة"""
        # تحديث المؤشرات
        self.base_latency = self.alpha * telemetry.current_latency_ms + (1 - self.alpha) * self.base_latency
        
        # اتجاه (trend)
        if len(self.history) > 10:
            recent_avg = sum(list(self.history)[-10:]) / 10
            older_avg = sum(list(self.history)[-20:-10]) / 10 if len(self.history) > 20 else recent_avg
            self.trend = self.beta * (recent_avg - older_avg) + (1 - self.beta) * self.trend
        
        # عامل الازدحام
        congestion_factor = 1.0 + (telemetry.queue_depth / 50) + (pending_tasks / 100)
        
        # عامل GPU
        gpu_factor = 1.0 + (telemetry.gpu_utilization / 100) * 0.5
        
        # العامل الموسمي
        seasonal_idx = int(time.time()) % 10
        seasonal_factor = 1.0 + self.seasonal[seasonal_idx]
        
        predicted = (self.base_latency + self.trend) * congestion_factor * gpu_factor * seasonal_factor
        
        self.history.append(telemetry.current_latency_ms)
        
        return min(500, max(10, predicted))


class CongestionPredictor:
    """يتوقع الازدحام باستخدام نماذج بسيطة"""
    
    def __init__(self):
        self.congestion_threshold = 0.7
        self.pattern_history = deque(maxlen=100)
        
    async def will_congest(self, telemetry: TelemetryData, 
                           predicted_latency: float,
                           pending_tasks: List[Dict]) -> Tuple[bool, float]:
        """توقع الازدحام وعودة درجة الثقة"""
        signals = []
        
        # 1. قياسات مباشرة
        if telemetry.queue_depth > 30:
            signals.append(0.8)
        elif telemetry.queue_depth > 15:
            signals.append(0.5)
        
        if telemetry.gpu_utilization > 85:
            signals.append(0.9)
        elif telemetry.gpu_utilization > 70:
            signals.append(0.6)
        
        if telemetry.concurrent_requests > 20:
            signals.append(0.7)
        
        # 2. توقع زمن الاستجابة
        if predicted_latency > 400:
            signals.append(0.8)
        elif predicted_latency > 300:
            signals.append(0.5)
        
        # 3. المهام المعلقة
        high_priority_pending = sum(1 for t in pending_tasks if t.get('priority', 2) <= 1)
        if high_priority_pending > 5:
            signals.append(0.7)
        
        if not signals:
            return False, 0.0
        
        confidence = sum(signals) / len(signals)
        will_congest = confidence > self.congestion_threshold
        
        # تسجيل للنمط
        self.pattern_history.append({
            'will_congest': will_congest,
            'confidence': confidence,
            'timestamp': time.time()
        })
        
        return will_congest, confidence

class DistributedStateManager:
    """إدارة الحالة الموزعة مع تكرار الأقران"""
    
    def __init__(self, node_id: str, peers: List[str], heartbeat_interval: float = 2.0):
        self.node_id = node_id
        self.peers = peers
        self.heartbeat_interval = heartbeat_interval
        
        # الحالة المحلية
        self.local_state: Dict[str, Any] = {}
        self.local_version: Dict[str, int] = {}
        
        # حالة الأقران
        self.peer_states: Dict[str, Dict] = {}
        self.peer_last_seen: Dict[str, float] = {}
        self.peer_versions: Dict[str, Dict[str, int]] = {}
        
        # انتخابات القائد
        self.is_leader = len(peers) == 0 or node_id == min([node_id] + peers)
        self.leader_id = self.node_id if self.is_leader else None
        self.term = 1
        
        # حلقة الاتصال
        self.running = False
        
    async def start(self):
        """بدء خدمة الحالة الموزعة"""
        self.running = True
        asyncio.create_task(self._heartbeat_loop())
        asyncio.create_task(self._leader_election_loop())
        logger.info(f"DistributedStateManager started on {self.node_id} (leader={self.is_leader})")
    
    async def get_state(self, key: str, timeout_ms: float = 100.0) -> Any:
        """الحصول على حالة مع محاولة من الأقران"""
        # أولاً من المحلي
        if key in self.local_state:
            return self.local_state[key]
        
        # ثم من الأقران (بالتوازي)
        if self.peers:
            tasks = [self._fetch_from_peer(peer, key) for peer in self.peers[:3]]
            results = await asyncio.gather(*tasks, return_exceptions=True)
            
            for result in results:
                if result and not isinstance(result, Exception):
                    # تحديث المحلي
                    await self.set_state(key, result, sync=False)
                    return result
        
        return None
    
    async def set_state(self, key: str, value: Any, sync: bool = True, version: int = None):
        """تعيين حالة ومزامنة مع الأقران"""
        # تحديث الإصدار
        new_version = (self.local_version.get(key, 0) + 1) if version is None else version
        self.local_state[key] = value
        self.local_version[key] = new_version
        
        if sync and self.peers:
            # مزامنة غير متزامنة مع الأقران
            asyncio.create_task(self._broadcast_to_peers(key, value, new_version))
    
    async def _fetch_from_peer(self, peer: str, key: str) -> Any:
        """جلب حالة من نظير"""
        # في الإنتاج: gRPC أو HTTP
        # تبسيطاً: محاكاة
        await asyncio.sleep(0.01)  # محاكاة زمن الشبكة
        return None
    
    async def _broadcast_to_peers(self, key: str, value: Any, version: int):
        """بث الحالة إلى الأقران"""
        for peer in self.peers:
            asyncio.create_task(self._send_to_peer(peer, key, value, version))
    
    async def _send_to_peer(self, peer: str, key: str, value: Any, version: int):
        """إرسال إلى نظير محدد"""
        # في الإنتاج: gRPC
        pass
    
    async def _heartbeat_loop(self):
        """حلقة نبضات القلب"""
        while self.running:
            for peer in self.peers:
                asyncio.create_task(self._send_heartbeat(peer))
            await asyncio.sleep(self.heartbeat_interval)
    
    async def _send_heartbeat(self, peer: str):
        """إرسال نبضة قلب"""
        # في الإنتاج: gRPC
        pass
    
    async def _leader_election_loop(self):
        """حلقة انتخابات القائد"""
        while self.running:
            # التحقق من صحة القائد الحالي
            if self.is_leader:
                # القائد يرسل إشارات
                pass
            else:
                # التحقق من أن القائد لا يزال حياً
                if self.leader_id and self.peer_last_seen.get(self.leader_id, 0) < time.time() - 5:
                    # القائد مات - نبدأ انتخابات جديدة
                    await self._elect_new_leader()
            
            await asyncio.sleep(3)
    
    async def _elect_new_leader(self):
        """انتخاب قائد جديد"""
        self.term += 1
        
        # أبسط خوارزمية: أقوى عقدة
        candidates = [self.node_id] + self.peers
        self.leader_id = min(candidates)  # أقوى = أصغر ID (تبسيط)
        self.is_leader = (self.leader_id == self.node_id)
        
        logger.info(f"New leader elected: {self.leader_id} (term={self.term})")
    
    async def health_check(self) -> Dict:
        """التحقق من صحة العقدة"""
        return {
            'node_id': self.node_id,
            'is_leader': self.is_leader,
            'term': self.term,
            'local_keys': len(self.local_state),
            'peers_connected': len(self.peer_last_seen)
        }

# 3. SELF-OPTIMIZING SCHEDULER (Reinforcement Learning)
# ============================================================================

class SelfOptimizingScheduler:
    """
    Scheduler يتعلم ويحسن نفسه عبر Q-Learning
    """
    
    def __init__(self, state_dim: int = 8, action_space: int = 8):
        self.state_dim = state_dim
        self.action_space = action_space
        
        # Q-Table (تبسيط، في الإنتاج DQN)
        self.q_table = np.zeros((state_dim * 10, action_space))
        self.learning_rate = 0.1
        self.discount = 0.95
        self.exploration_rate = 0.2
        self.exploration_decay = 0.995
        self.min_exploration = 0.05
        
        self.reward_history: List[float] = []
        self.episode_rewards: List[float] = []
        
    def _state_to_index(self, state: np.ndarray) -> int:
        """تحويل الحالة المستمرة إلى مؤشر منفصل"""
        # تجميع كل بُعد إلى 10 فئات
        discretized = 0
        for i, val in enumerate(state[:self.state_dim]):
            bucket = min(9, int(val * 10))
            discretized = discretized * 10 + bucket
        return discretized % (self.state_dim * 10)
    
    async def select_action(self, state: np.ndarray) -> int:
        """اختيار الإجراء الأمثل (ε-greedy)"""
        state_idx = self._state_to_index(state)
        
        if random.random() < self.exploration_rate:
            # استكشاف
            return random.randint(0, self.action_space - 1)
        else:
            # استغلال
            return int(np.argmax(self.q_table[state_idx]))
    
    async def update(self, state: np.ndarray, action: int, 
                     reward: float, next_state: np.ndarray, done: bool = False):
        """تحديث Q-Table"""
        state_idx = self._state_to_index(state)
        next_state_idx = self._state_to_index(next_state)
        
        # Q-Learning update
        best_next = np.max(self.q_table[next_state_idx]) if not done else 0
        td_target = reward + self.discount * best_next
        td_error = td_target - self.q_table[state_idx, action]
        
        self.q_table[state_idx, action] += self.learning_rate * td_error
        
        self.reward_history.append(reward)
        
        if done:
            self.episode_rewards.append(sum(self.reward_history[-100:]))
            # تقليل معدل الاستكشاف
            self.exploration_rate = max(self.min_exploration, 
                                        self.exploration_rate * self.exploration_decay)
    
    def get_best_action_for_state(self, state: np.ndarray) -> int:
        """الحصول على أفضل إجراء لحالة معينة"""
        state_idx = self._state_to_index(state)
        return int(np.argmax(self.q_table[state_idx]))
    
    def get_stats(self) -> Dict:
        """إحصائيات المتعلم"""
        return {
            'exploration_rate': self.exploration_rate,
            'q_table_size': self.q_table.size,
            'avg_recent_reward': np.mean(self.reward_history[-100:]) if self.reward_history else 0,
            'best_episode_reward': max(self.episode_rewards) if self.episode_rewards else 0
        }

            'agent_governor': Component('agent_governor', Priority.CRITICAL),
            'reasoning_engine': Component('reasoning_engine', Priority.NORMAL),

        self.telemetry = new_telemetry
        self.telemetry_history.append(new_telemetry)
        
        # تحديث الحالة الموزعة
        await self.state_manager.set_state('telemetry', new_telemetry.to_dict())
    
    async def orchestrate(self, trigger: str, context: Dict) -> Dict:
        """
        التنسيق التنبؤي الذكي - قلب Control Plane
        """
        start_time = time.time()
        
        # 1. جمع القياسات الحالية
        current_telemetry = await self._collect_telemetry()
        
        # 2. المهام المعلقة
        pending_tasks = context.get('pending_tasks', [])
        
        # 3. توقع زمن الاستجابة
        predicted_latency = await self.latency_predictor.predict(
            current_telemetry, len(pending_tasks)
        )
        
        # 4. توقع الازدحام
        will_congest, congestion_confidence = await self.congestion_predictor.will_congest(
            current_telemetry, predicted_latency, pending_tasks
        )
        self.stats['predictions_made'] += 1
        
        # 5. بناء متجه الحالة لـ RL
        state_vector = self._build_state_vector(current_telemetry, pending_tasks, predicted_latency)
        
        # 6. اختيار الإجراء عبر RL
        action = await self.scheduler.select_action(state_vector)
        
        # 7. اتخاذ القرار النهائي
        decision = await self._make_decision(
            trigger, context, current_telemetry, 
            predicted_latency, will_congest, congestion_confidence, action
        )
        
        # 8. تطبيق القرار
        await self._apply_decision(decision)
        
        # 9. حساب المكافأة وتحسين الـ scheduler
        reward = await self._compute_reward(decision, current_telemetry, predicted_latency)
        next_state = self._build_state_vector(
            await self._collect_telemetry(), [], 0
        )
        await self.scheduler.update(state_vector, action, reward, next_state)
        
        # 10. تحديث الإحصائيات
        self.stats['orchestrations'] += 1
        self.stats['prediction_accuracy'] = (
            self.stats['prediction_accuracy'] * 0.95 +
            (1.0 if (will_congest == (predicted_latency > 300)) else 0.0) * 0.05
        )
        
        decision['latency_ms'] = (time.time() - start_time) * 1000
        decision['predicted_latency_ms'] = predicted_latency
        
        return decision
    
    def _build_state_vector(self, telemetry: TelemetryData, 
                            pending_tasks: List, predicted_latency: float) -> np.ndarray:
        """بناء متجه الحالة للـ RL"""
        vector = [
            telemetry.gpu_utilization / 100.0,
            telemetry.cpu_utilization / 100.0,
            telemetry.memory_usage_mb / 16000.0,
            min(1.0, telemetry.queue_depth / 100.0),
            telemetry.current_latency_ms / 500.0,
            predicted_latency / 500.0,
            min(1.0, len(pending_tasks) / 50.0),
            telemetry.concurrent_requests / 50.0
        ]
        return np.array(vector[:8])

        if telemetry.thermal_throttling or telemetry.gpu_utilization > 95:
            decision['action_taken'] = 'emergency'
            decision['activate'] = None
            decision['pause'] = list(self.components.keys())
            decision['reason'] = 'Emergency: thermal or GPU overload'
            
        # ازدحام متوقع
        elif will_congest and congestion_confidence > 0.7:
            decision['action_taken'] = 'throttle'
            decision['activate'] = None
            decision['pause'] = ['evolution_gate', 'memory_consolidation', 'competition_engine']
            decision['reason'] = f'Predicted congestion (confidence={congestion_confidence:.2f})'
            
        # ميزانية متاحة
        elif predicted_latency < 100 and telemetry.gpu_utilization < 50:
            decision['action_taken'] = 'accelerate'
            decision['activate'] = ['memory_consolidation']
            decision['pause'] = []
            decision['reason'] = 'Low load - accelerating background tasks'
            
        # مدخل مستخدم - أولوية قصوى
        elif trigger == 'user_input':
            input_type = context.get('input_type', 'unknown')
            decision['action_taken'] = 'user_priority'
            decision['activate'] = self._get_component_for_input(input_type)
            decision['pause'] = ['evolution_gate', 'memory_consolidation']
            decision['reason'] = f'User input: {input_type}'

            decision['activate'] = 'memory_arbitration'

        if predicted_well:
            reward += 0.2
        else:

class ComplexityEstimator:
    """يقدر تعقيد المدخل لاختيار النموذج المناسب"""
    
    def __init__(self):
        self.estimators = {
            Modality.CODE: self._estimate_code_complexity,
            Modality.AUDIO: self._estimate_audio_complexity,
            Modality.SCREEN: self._estimate_screen_complexity,
            Modality.BEHAVIOR: self._estimate_behavior_complexity
        }
        
        # ذاكرة التقديرات السابقة
        self.estimation_history: deque = deque(maxlen=1000)
    
    async def estimate(self, modality: Modality, input_data: Dict, 
                       fast_result: Any = None) -> float:
        """تقدير التعقيد (0-1)"""
        estimator = self.estimators.get(modality)
        if estimator:
            complexity = await estimator(input_data, fast_result)
        else:
            complexity = 0.5
        
        # تسجيل
        self.estimation_history.append({
            'modality': modality.value,
            'complexity': complexity,
            'timestamp': time.time()
        })
        
        return min(1.0, max(0.0, complexity))
    
    async def _estimate_code_complexity(self, code_data: Dict, fast_result: Any) -> float:
        """تقدير تعقيد الكود بشكل دقيق"""
        code = code_data.get('code', code_data.get('content', ''))
        
        if not code:
            return 0.2
        
        # 1. المقاييس الأساسية
        lines = len(code.split('\n'))
        
        # 2. تعقيد التحكم
        loops = len(re.findall(r'\b(for|while)\b', code))
        conditionals = len(re.findall(r'\b(if|elif|else|switch|case)\b', code))
        functions = len(re.findall(r'\bdef\s+\w+\s*\(', code))
        classes = len(re.findall(r'\bclass\s+\w+', code))
        
        # 3. تعقيد البيانات
        # عدد المتغيرات الفريدة (تقريبي)
        variables = len(set(re.findall(r'\b[a-z_][a-z0-9_]*(?=\s*=|\s*\[|\s*\.)', code, re.IGNORECASE)))
        
        # 4. عمق التداخل الأقصى
        max_nesting = self._max_nesting_depth(code)
        
        # 5. استخدام المكتبات الخارجية
        imports = len(re.findall(r'\b(import|from)\s+\w+', code))
        
        # 6. إذا كان هناك نتيجة من النموذج السريع، استخدمها لتحسين التقدير
        if fast_result and 'tokens' in fast_result:
            # النموذج السريع أعطانا فكرة عن التعقيد
            fast_complexity = fast_result.get('estimated_complexity', 0.5)
            return 0.6 * fast_complexity + 0.4 * self._compute_metrics_score(
                lines, loops, conditionals, functions, classes, variables, max_nesting, imports
            )
        
        return self._compute_metrics_score(
            lines, loops, conditionals, functions, classes, variables, max_nesting, imports
        )
    
    def _compute_metrics_score(self, lines, loops, conditionals, functions, 
                                classes, variables, max_nesting, imports) -> float:
        """حساب الدرجة من المقاييس"""
        # تسوية كل مقياس
        score = (
            min(1.0, lines / 500) * 0.20 +
            min(1.0, loops / 15) * 0.15 +
            min(1.0, conditionals / 30) * 0.15 +
            min(1.0, functions / 20) * 0.10 +
            min(1.0, classes / 10) * 0.05 +
            min(1.0, variables / 50) * 0.15 +
            min(1.0, max_nesting / 8) * 0.10 +
            min(1.0, imports / 20) * 0.10
        )
        return score
    
    def _max_nesting_depth(self, code: str) -> int:
        """حساب أقصى عمق تداخل"""
        max_depth = 0
        current_depth = 0
        
        # كلمات تزيد العمق
        open_keywords = ['for', 'while', 'if', 'def', 'class', 'with', 'try']
        
        lines = code.split('\n')
        for line in lines:
            stripped = line.strip()
            if not stripped:
                continue
            
            # زيادة العمق
            for keyword in open_keywords:
                if stripped.startswith(keyword) and stripped.endswith(':'):
                    current_depth += 1
                    max_depth = max(max_depth, current_depth)
                    break
            
            # تقليل العمق (تقريبي)
            if stripped == '' or stripped == 'pass' or (stripped and not stripped[0].isspace() and current_depth > 0):
                current_depth = max(0, current_depth - 1)
        
        return max_depth
    
    async def _estimate_audio_complexity(self, audio_data: Dict, fast_result: Any) -> float:
        """تقدير تعقيد الصوت"""
        duration = audio_data.get('duration_sec', 5)
        words_per_minute = audio_data.get('wpm', 150)
        background_noise = audio_data.get('background_noise', 0.3)
        
        complexity = (
            min(1.0, duration / 120) * 0.3 +
            min(1.0, words_per_minute / 250) * 0.3 +
            background_noise * 0.4
        )
        return complexity
    
    async def _estimate_screen_complexity(self, screen_data: Dict, fast_result: Any) -> float:
        """تقدير تعقيم الشاشة"""
        # تبسيطاً
        return 0.5
    
    async def _estimate_behavior_complexity(self, behavior_data: Dict, fast_result: Any) -> float:
        """تقدير تعقيد السلوك"""
        # تبسيطاً
        return 0.4
    
    def get_estimation_accuracy(self) -> float:
        """دقة التقديرات (يمكن قياسها لاحقاً)"""
        if len(self.estimation_history) < 10:
            return 0.7
        return 0.8  # تبسيط

class TelemetryCollector:
    """يجمع القياسات الحية لأداء النماذج"""
    
    def __init__(self):
        self.model_metrics: Dict[str, ModelMetrics] = {}
        self.global_metrics = {
            'total_inferences': 0,
            'total_latency_ms': 0.0,
            'escalation_count': 0,
            'avg_escalation_time_ms': 0.0
        }
        self.recent_escalations: deque = deque(maxlen=100)
    
    def record_model_execution(self, model_name: str, latency_ms: float, 
                                success: bool, gpu_util: float):
        """تسجيل تنفيذ نموذج"""
        if model_name not in self.model_metrics:
            self.model_metrics[model_name] = ModelMetrics(
                latencies=deque(maxlen=100),
                successes=deque(maxlen=100),
                gpu_utils=deque(maxlen=100)
            )
        
        metrics = self.model_metrics[model_name]
        metrics.latencies.append(latency_ms)
        metrics.successes.append(1.0 if success else 0.0)
        metrics.gpu_utils.append(gpu_util)
        metrics.total_calls += 1
        
        # تحديث الإحصائيات
        metrics.avg_latency = sum(metrics.latencies) / len(metrics.latencies)
        metrics.success_rate = sum(metrics.successes) / len(metrics.successes)
        
        # P95
        sorted_latencies = sorted(metrics.latencies)
        p95_idx = int(len(sorted_latencies) * 0.95)
        metrics.p95_latency = sorted_latencies[p95_idx] if sorted_latencies else latency_ms
        
        # تحديث العالمية
        self.global_metrics['total_inferences'] += 1
        self.global_metrics['total_latency_ms'] += latency_ms
    
    def record_escalation(self, from_model: str, to_model: str, 
                          complexity: float, time_ms: float):
        """تسجيل تصعيد"""
        self.global_metrics['escalation_count'] += 1
        self.global_metrics['avg_escalation_time_ms'] = (
            self.global_metrics['avg_escalation_time_ms'] * 0.9 + time_ms * 0.1
        )
        self.recent_escalations.append({
            'from': from_model,
            'to': to_model,
            'complexity': complexity,
            'time_ms': time_ms,
            'timestamp': time.time()
        })
    
    def get_best_model_for_complexity(self, models: List[ModelRoute], 
                                       complexity: float) -> Optional[ModelRoute]:
        """اختيار أفضل نموذج حسب التعقيد والأداء التاريخي"""
        # النماذج المناسبة لنطاق التعقيد
        suitable = [m for m in models if m.complexity_range[0] <= complexity <= m.complexity_range[1]]
        
        if not suitable:
            # خذ أقرب نموذج
            suitable = sorted(models, key=lambda m: abs(m.complexity_range[0] - complexity))
        
        def score(model):
            metrics = self.model_metrics.get(model.model_name)
            if not metrics:
                return 1.0 / (model.estimated_latency_ms + 1)
            
            success_rate = metrics.success_rate
            latency_penalty = metrics.p95_latency / 500
            return success_rate * (1 - latency_penalty * 0.5)
        
        return max(suitable, key=score)
    
    def should_escalate(self, fast_model: str, complexity: float) -> Tuple[bool, float]:
        """تحديد ما إذا كان يجب التصعيد"""
        metrics = self.model_metrics.get(fast_model)
        if not metrics:
            return complexity > 0.4, 0.5
        
        # التصعيد إذا:
        # 1. التعقيد مرتفع
        # 2. النموذج السريع لديه معدل نجاح منخفض للتعقيدات العالية
        if complexity > 0.6:
            return True, 0.8
        elif complexity > 0.4:
            # اعتماداً على الأداء التاريخي
            if metrics.success_rate < 0.7:
                return True, 0.6
        return False, 0.0

Memory Arbitration V5 - النسخة الكاملة مع:
- ANN search حقيقي (Qdrant/FAISS)
- Dynamic decay (emotional weight, novelty, reinforcement)
- Memory consolidation (sleep phase + abstraction)

import time

from typing import Dict, List, Optional, Any, Tuple

class MemoryType(Enum):
    VECTOR = "vector"
    GRAPH = "graph"
    EPISODIC = "episodic"


@dataclass
class MemoryItem:
    """عنصر في الذاكرة مع جميع الخصائص الديناميكية"""
    id: str
    content: Any
    embedding: np.ndarray
    memory_type: MemoryType
    relevance_score: float  # 0-1
    importance: float  # 0-1
    created_at: float
    last_accessed: float
    access_count: int = 0
    base_decay_rate: float = 0.1
    
    # إضافات ديناميكية للاضمحلال الذكي
    emotional_weight: float = 0.0  # 0-1, الأحداث العاطفية تبقى أطول
    novelty_score: float = 0.0  # مدى جدة المعلومة
    reinforcement: float = 0.0  # -1 إلى 1 (سلبي/إيجابي)
    contradictions: int = 0  # عدد التناقضات مع ذكريات أخرى
    utility_score: float = 0.5  # الفائدة المتوقعة مستقبلاً
    
    # للدمج (consolidation)
    abstraction_level: int = 0  # 0=ملموس, 5=مجرد
    merged_from: List[str] = field(default_factory=list)
    
    def compute_dynamic_decay(self, current_time: float) -> float:
        """
        حساب الاضمحلال الديناميكي المتقدم
        يعتمد على خصائص الذاكرة المتعددة
        """
        age = current_time - self.created_at
        days = age / 86400
        
        # 1. اضمحلال أساسي
        base_decay = self.base_decay_rate * days
        
        # 2. تعديل عاطفي (الأحداث العاطفية تبقى)
        emotional_factor = 1.0 - self.emotional_weight * 0.7
        
        # 3. تعديل الجدة (الجديد يضمحل أبطأ)
        novelty_factor = 1.0 + self.novelty_score * 0.5
        
        # 4. تعديل التعزيز
        if self.reinforcement > 0:
            reinforcement_factor = 1.0 - self.reinforcement * 0.6
        else:
            reinforcement_factor = 1.0 + abs(self.reinforcement) * 0.3
        
        # 5. عقوبة التناقضات (الكثير من التناقضات تسرع الاضمحلال)
        contradiction_penalty = 1.0 + (self.contradictions / 10) * 0.3
        
        # 6. الفائدة المتوقعة (المفيد يبقى)
        utility_factor = 1.0 - self.utility_score * 0.4
        
        # 7. التكرار يبطئ الاضمحلال
        frequency_factor = 1.0 - min(0.5, self.access_count / 100) * 0.3
        
        decay = (base_decay * emotional_factor * reinforcement_factor * 
                 contradiction_penalty * utility_factor) / (novelty_factor * frequency_factor)
        
        return max(0.01, min(0.5, decay))
    
    def compute_current_score(self, current_time: float) -> float:
        """حساب الدرجة الحالية للعنصر"""
        decay = self.compute_dynamic_decay(current_time)
        
        # معامل الحداثة (recency)
        recency = math.exp(-decay * (current_time - self.last_accessed) / 3600)
        
        # معامل التكرار
        frequency = min(1.0, self.access_count / 50)
        
        # الصيغة النهائية
        score = (
            self.relevance_score * 0.35 +
            recency * 0.25 +
            self.importance * 0.20 +
            frequency * 0.10 +
            self.utility_score * 0.10
        )
        
        return max(0.0, min(1.0, score))

    def add_item(self, item: MemoryItem):
        """إضافة عنصر إلى الفهرس"""
        self.items[item.id] = item
        self.index_built = False  # يحتاج لإعادة بناء الفهرس
    
    def _rebuild_index(self):
        """إعادة بناء الفهرس (في الإنتاج: FAISS/Qdrant)"""
        if not self.items:
            return
        
        vectors = []
        ids = []
        for item_id, item in self.items.items():
            vectors.append(item.embedding)
            ids.append(item_id)
        
        self.index_vectors = np.array(vectors)
        self.index_ids = np.array(ids)
        self.index_built = True
    
    async def search(self, query_embedding: np.ndarray, top_k: int = 10) -> List[Tuple[str, float]]:
        """البحث عن أقرب الجيران"""
        if not self.items:
            return []
        
        if not self.index_built:
            self._rebuild_index()
        
        # تبسيطاً: بحث خطي مع تحسين NumPy
        if self.index_vectors is not None and len(self.index_vectors) > 0:
            # حساب التشابه جيب التمام
            norms = np.linalg.norm(self.index_vectors, axis=1)
            query_norm = np.linalg.norm(query_embedding)
            
            if query_norm > 0:
                similarities = np.dot(self.index_vectors, query_embedding) / (norms * query_norm)
            else:
                similarities = np.zeros(len(self.index_vectors))
            
            # ترتيب تنازلي
            top_indices = np.argsort(similarities)[-top_k:][::-1]
            
            results = []
            for idx in top_indices:
                if similarities[idx] > 0.3:  # عتبة التشابه
                    results.append((self.index_ids[idx], float(similarities[idx])))
            
            return results
        
        return []
    
    def remove_item(self, item_id: str):
        """حذف عنصر من الفهرس"""
        if item_id in self.items:
            del self.items[item_id]
            self.index_built = False
    
    def get_item(self, item_id: str) -> Optional[MemoryItem]:
        """الحصول على عنصر بالمعرف"""
        return self.items.get(item_id)

class MemoryConsolidationEngine:
    """
    محرك دمج الذاكرة - يحاكي مرحلة النوم
    يقوم بـ:
    - دمج الذكريات المتشابهة
    - تجريد المعلومات
    - إزالة الازدواجية
    """
    
    def __init__(self):
        self.consolidation_history: List[Dict] = []
        self.last_consolidation = time.time()
        self.consolidation_interval = 3600  # كل ساعة
        
    async def consolidate(self, items: List[MemoryItem]) -> List[MemoryItem]:
        """
        دمج الذكريات - العملية الأساسية للـ consolidation
        """
        if len(items) < 10:
            return items
        
        start_time = time.time()
        
        # 1. تجميع الذكريات المتشابهة
        clusters = await self._cluster_similar_items(items)
        
        # 2. دمج كل عنقود
        consolidated = []
        merged_count = 0
        
        for cluster in clusters:
            if len(cluster) == 1:
                consolidated.append(cluster[0])
            else:
                # دمج الذكريات المتشابهة
                merged = await self._merge_cluster(cluster)
                consolidated.append(merged)
                merged_count += len(cluster) - 1
        
        # 3. تحديث التجريد (abstraction) للذكريات القديمة
        for item in consolidated:
            if item.created_at < time.time() - 86400:  # أقدم من يوم
                item.abstraction_level = min(5, item.abstraction_level + 1)
        
        self.consolidation_history.append({
            'timestamp': time.time(),
            'original_count': len(items),
            'consolidated_count': len(consolidated),
            'merged_count': merged_count,
            'duration_ms': (time.time() - start_time) * 1000
        })
        
        return consolidated
    
    async def _cluster_similar_items(self, items: List[MemoryItem]) -> List[List[MemoryItem]]:
        """
        تجميع العناصر المتشابهة باستخدام التشابه الدلالي
        """
        if len(items) < 2:
            return [[item] for item in items]
        
        # تبسيطاً: تجميع حسب التشابه المباشر
        # في الإنتاج: استخدام DBSCAN أو HDBSCAN
        clusters = []
        used = set()
        
        for i, item1 in enumerate(items):
            if i in used:
                continue
            
            cluster = [item1]
            used.add(i)
            
            for j, item2 in enumerate(items[i+1:], i+1):
                if j in used:
                    continue
                
                # حساب التشابه بين التضمينات
                similarity = np.dot(item1.embedding, item2.embedding) / (
                    np.linalg.norm(item1.embedding) * np.linalg.norm(item2.embedding) + 1e-8
                )
                
                if similarity > 0.85:  # تشابه عالٍ جداً
                    cluster.append(item2)
                    used.add(j)
            
            clusters.append(cluster)
        
        return clusters
    
    async def _merge_cluster(self, cluster: List[MemoryItem]) -> MemoryItem:
        """
        دمج عنقود من الذكريات المتشابهة إلى ذاكرة واحدة مجردة
        """
        if not cluster:
            return None
        
        # خذ العنصر الأحدث أو الأكثر أهمية
        base = max(cluster, key=lambda x: (x.importance, x.created_at))
        
        # دمج الخصائص
        for item in cluster:
            if item.id != base.id:
                base.relevance_score = max(base.relevance_score, item.relevance_score)
                base.importance = max(base.importance, item.importance)
                base.access_count += item.access_count
                base.merged_from.append(item.id)
                
                # نقل التعزيزات
                base.reinforcement = (base.reinforcement + item.reinforcement) / 2
                base.utility_score = max(base.utility_score, item.utility_score)
        
        # زيادة مستوى التجريد
        base.abstraction_level = min(5, base.abstraction_level + 1)
        
        return base
    
    def should_consolidate(self) -> bool:
        """هل حان وقت الدمج؟"""
        return (time.time() - self.last_consolidation) > self.consolidation_interval
    
    def get_stats(self) -> Dict:
        """إحصائيات الدمج"""
        if not self.consolidation_history:
            return {'total_consolidations': 0, 'avg_merge_ratio': 0}
        
        recent = self.consolidation_history[-10:]
        avg_merge_ratio = sum(1 - c['consolidated_count']/max(1, c['original_count']) 
                              for c in recent) / len(recent)
        
        return {
            'total_consolidations': len(self.consolidation_history),
            'last_consolidation': self.consolidation_history[-1]['timestamp'] if self.consolidation_history else None,
            'avg_merge_ratio': avg_merge_ratio,
            'consolidation_interval_sec': self.consolidation_interval
        }

    async def retrieve(self, query_embedding: np.ndarray,

        for item_id, similarity in search_results:
            item = self.search_engine.get_item(item_id)
            if item:

                )

                scored_items.append((item, final_score))
            
            scored_items.sort(key=lambda x: x[1], reverse=True)

            existing_ids = {item.id for item in retrieved_items}
            for item, _ in scored_items:
                if item.id not in existing_ids:
                    retrieved_items.append(item)
                    if len(retrieved_items) >= top_k:
                        break
        
        # تحديث الإحصائيات
        self.stats['retrieval_count'] += 1
        latency_ms = (time.time() - start_time) * 1000
        self.stats['avg_retrieval_latency_ms'] = (
            self.stats['avg_retrieval_latency_ms'] * 0.95 + latency_ms * 0.05
        )

    async def arbitrate(self):
        """
        عملية التحكيم - يقرر ماذا يبقى وماذا يُنسى
        """
        current_time = time.time()
        self.stats['last_arbitration'] = current_time
        
        for memory_type, items in self.memory_items.items():
            if len(items) <= self.max_items_per_type:
                continue
            
            # حساب الدرجات
            scored_items = []
            for item in items:
                score = item.compute_current_score(current_time)
                scored_items.append((item, score))
            
            # ترتيب تنازلي
            scored_items.sort(key=lambda x: x[1], reverse=True)
            
            # الاحتفاظ بأفضل العناصر فقط
            keep_count = int(self.max_items_per_type * 0.8)
            keep_items = scored_items[:keep_count]
            evict_items = scored_items[keep_count:]
            
            # حذف العناصر منخفضة الدرجة
            evicted_ids = []
            for item, score in evict_items:
                if score < self.eviction_thresholds.get(memory_type, 0.1):
                    self.search_engine.remove_item(item.id)
                    evicted_ids.append(item.id)
                    self.stats['items_evicted'] += 1
            
            # تحديث القائمة
            self.memory_items[memory_type] = [item for item, _ in keep_items]
            
            # تحديث إحصائيات الدمج
            for item, _ in keep_items:
                if len(item.merged_from) > 0:
                    self.stats['items_merged'] += len(item.merged_from)

        """الحصول على عناصر مشابهة لعنصر معين"""

        similar = []
        for retrieved_id, similarity in results:
            if retrieved_id != item_id:
                retrieved_item = self.search_engine.get_item(retrieved_id)
                if retrieved_item:
                    similar.append(retrieved_item)
        
        return similar[:top_k]
    
    async def reinforce(self, item_id: str, reward: float):
        """تعزيز أو معاقبة ذاكرة معينة"""

            # زيادة الأهمية إذا كان التعزيز إيجابياً
            if reward > 0:
                item.importance = min(1.0, item.importance + reward * 0.05)
                item.utility_score = min(1.0, item.utility_score + reward * 0.1)
    
    async def get_memory_summary(self, memory_type: Optional[MemoryType] = None) -> Dict:

                }
                for t, items in self.memory_items.items()
            }
        }

        """إيقاف المحرك"""
        self.running = False

class SemanticContradictionDetector:
    """
    كشف التناقضات الدلالية بين مخرجات الوكلاء
    """
    
    def __init__(self):
        self.contradiction_patterns = [
            ('yes', 'no'),
            ('true', 'false'),
            ('correct', 'incorrect'),
            ('good', 'bad'),
            ('high', 'low'),
            ('increase', 'decrease'),
            ('better', 'worse'),
            ('optimal', 'suboptimal')
        ]
        
        # نموذج تضمين بسيط (في الإنتاج: sentence-transformers)
        self.embedding_model = None
        
    async def detect_contradiction(self, output1: AgentOutput, 
                                    output2: AgentOutput) -> Tuple[bool, float, str]:
        """
        كشف التناقض بين مخرجين
        """
        content1 = str(output1.content).lower()
        content2 = str(output2.content).lower()
        
        contradictions_found = []
        
        # 1. فحص الأنماط المباشرة
        for pos, neg in self.contradiction_patterns:
            if pos in content1 and neg in content2:
                contradictions_found.append(1.0)
            elif neg in content1 and pos in output2:
                contradictions_found.append(1.0)
        
        # 2. التناقض في درجات الثقة
        if abs(output1.confidence - output2.confidence) > 0.5:
            contradictions_found.append(0.7)
        
        # 3. التناقض في حالة المعتقدات
        for key in set(output1.belief_state.keys()) & set(output2.belief_state.keys()):
            if abs(output1.belief_state.get(key, 0) - output2.belief_state.get(key, 0)) > 0.6:
                contradictions_found.append(0.8)
                break
        
        if not contradictions_found:
            return False, 0.0, ""
        
        confidence = sum(contradictions_found) / len(contradictions_found)
        
        # تحديد نوع التناقض
        if any(c > 0.8 for c in contradictions_found):
            reason = "strong_contradiction"
        else:
            reason = "partial_contradiction"
        
        return True, confidence, reason
    
    async def analyze_belief_conflict(self, output1: AgentOutput, 
                                        output2: AgentOutput) -> Dict:
        """تحليل تناقض المعتقدات"""
        conflict_analysis = {
            'has_conflict': False,
            'conflicting_beliefs': [],
            'shared_beliefs': []
        }
        
        all_keys = set(output1.belief_state.keys()) | set(output2.belief_state.keys())
        
        for key in all_keys:
            val1 = output1.belief_state.get(key, 0.5)
            val2 = output2.belief_state.get(key, 0.5)
            
            if abs(val1 - val2) > 0.5:
                conflict_analysis['has_conflict'] = True
                conflict_analysis['conflicting_beliefs'].append({
                    'belief': key,
                    'value1': val1,
                    'value2': val2,
                    'difference': abs(val1 - val2)
                })
            elif abs(val1 - val2) < 0.2:
                conflict_analysis['shared_beliefs'].append(key)
        
        return conflict_analysis

class BeliefGraph:
    """
    رسم بياني للمعتقدات - يتتبع حالة الإيمان لكل وكيل
    """
    
    def __init__(self):
        self.beliefs: Dict[str, Dict[str, float]] = {}  # agent -> belief_key -> value
        self.belief_consensus: Dict[str, float] = {}  # belief_key -> consensus
        self.belief_history: deque = deque(maxlen=1000)
        
    def update_belief(self, agent_name: str, belief_key: str, value: float, 
                      confidence: float):
        """تحديث معتقد وكيل"""
        if agent_name not in self.beliefs:
            self.beliefs[agent_name] = {}
        
        # تحديث مع مراعاة الثقة
        current = self.beliefs[agent_name].get(belief_key, 0.5)
        self.beliefs[agent_name][belief_key] = current * 0.7 + value * confidence * 0.3
        
        # تحديث الإجماع
        self._update_consensus(belief_key)
        
        # تسجيل التاريخ
        self.belief_history.append({
            'timestamp': time.time(),
            'agent': agent_name,
            'belief': belief_key,
            'value': value,
            'confidence': confidence
        })
    
    def _update_consensus(self, belief_key: str):
        """تحديث الإجماع على معتقد معين"""
        values = []
        for agent_beliefs in self.beliefs.values():
            if belief_key in agent_beliefs:
                values.append(agent_beliefs[belief_key])
        
        if values:
            self.belief_consensus[belief_key] = np.mean(values)
    
    def get_consensus(self, belief_key: str) -> float:
        """الحصول على الإجماع الحالي لمعتقد"""
        return self.belief_consensus.get(belief_key, 0.5)
    
    def get_deviation(self, agent_name: str, belief_key: str) -> float:
        """انحراف وكيل عن الإجماع"""
        consensus = self.get_consensus(belief_key)
        agent_value = self.beliefs.get(agent_name, {}).get(belief_key, consensus)
        return abs(agent_value - consensus)
    
    def get_status(self) -> Dict:
        """حالة الرسم البياني"""
        return {
            'active_beliefs': len(self.belief_consensus),
            'active_agents': len(self.beliefs),
            'consensus_values': {k: round(v, 2) for k, v in list(self.belief_consensus.items())[:10]}
        }

        conflicts.sort(key=lambda x: x.get('severity', 0), reverse=True)
        
        for conflict in conflicts:
            self.stats['conflicts_detected'] += 1
            
            if conflict['type'] == ConflictType.SEMANTIC_MISMATCH:
                self.stats['semantic_conflicts'] += 1
            
            # حل التعارض
            resolver = self.resolution_rules.get(conflict['type'])
            if resolver:
                resolution = await resolver(conflict)
                self.resolution_history.append(resolution)
                self.stats['conflicts_resolved'] += 1
                
                # تطبيق القرار
                await self._apply_resolution(resolution)
    
    async def _detect_all_conflicts(self, outputs: List[AgentOutput]) -> List[Dict]:
        """
        كشف جميع أنواع التعارضات
        """
        conflicts = []
        
        for i, out1 in enumerate(outputs):
            for out2 in outputs[i+1:]:
                # 1. التناقض الدلالي
                is_contradictory, conf, reason = await self.contradiction_detector.detect_contradiction(
                    out1, out2
                )
                if is_contradictory:
                    conflicts.append({
                        'type': ConflictType.SEMANTIC_MISMATCH,
                        'outputs': [out1, out2],
                        'severity': conf,
                        'description': f"Semantic mismatch: {reason}"
                    })
                
                # 2. تناقض المعتقدات
                belief_conflict = await self.contradiction_detector.analyze_belief_conflict(out1, out2)
                if belief_conflict['has_conflict']:
                    conflicts.append({
                        'type': ConflictType.BELIEF_CONFLICT,
                        'outputs': [out1, out2],
                        'severity': 0.7,
                        'description': f"Belief conflict on {len(belief_conflict['conflicting_beliefs'])} beliefs",
                        'analysis': belief_conflict
                    })
        
        # 3. كشف الحلقات اللانهائية
        if await self._detect_infinite_loop():
            conflicts.append({
                'type': ConflictType.INFINITE_LOOP,
                'outputs': list(self.action_trace)[-5:],
                'severity': 1.0,
                'description': "Potential infinite loop detected"
            })
        
        return conflicts
    
    async def _detect_infinite_loop(self) -> bool:
        """كشف الحلقات اللانهائية"""
        if len(self.action_trace) >= 10:
            recent = list(self.action_trace)[-5:]
            if len(set(recent)) <= 2:
                self.stats['loops_prevented'] += 1
                return True
        return False
    
    async def _resolve_contradiction(self, conflict: Dict) -> Resolution:
        """حل التعارضات العادية"""
        outputs = conflict['outputs']
        best = max(outputs, key=lambda o: (o.priority, o.confidence))
        rejected = [o for o in outputs if o != best]
        
        return Resolution(
            conflict_type=ConflictType.CONTRADICTORY_OUTPUT,
            selected_output=best,
            rejected_outputs=rejected,
            reason=f"Selected {best.agent_name} (priority={best.priority}, confidence={best.confidence:.2f})"
        )
    
    async def _resolve_semantic_mismatch(self, conflict: Dict) -> Resolution:
        """حل التناقضات الدلالية - أكثر تعقيداً"""
        outputs = conflict['outputs']
        
        # 1. تحقق من الإجماع في معتقدات الوكلاء الآخرين
        consensus_scores = []
        for out in outputs:
            # متوسط انحراف الوكيل عن الإجماع
            deviations = []
            for key in out.belief_state:
                deviation = self.belief_graph.get_deviation(out.agent_name, key)
                deviations.append(deviation)
            avg_deviation = np.mean(deviations) if deviations else 0.5
            consensus_scores.append((out, 1.0 - avg_deviation))
        
        # 2. اختر الأكثر توافقاً مع الإجماع
        best = max(consensus_scores, key=lambda x: x[1])[0]
        rejected = [o for o in outputs if o != best]
        
        return Resolution(
            conflict_type=ConflictType.SEMANTIC_MISMATCH,
            selected_output=best,
            rejected_outputs=rejected,
            reason=f"Resolved by belief consensus: {best.agent_name}",
            semantic_analysis=conflict.get('analysis', {})
        )
    
    async def _resolve_belief_conflict(self, conflict: Dict) -> Resolution:
        """حل تناقض المعتقدات"""
        outputs = conflict['outputs']
        
        # من لديه تاريخ أكثر نجاحاً؟
        best = max(outputs, key=lambda o: self.agents[o.agent_name].get('success_rate', 0.5))
        rejected = [o for o in outputs if o != best]
        
        # تحديث إجماع المعتقدات
        analysis = conflict.get('analysis', {})
        for belief in analysis.get('conflicting_beliefs', []):
            # خذ قيمة الوكيل الفائز كمرجع
            belief_key = belief.get('belief')
            if belief_key:
                winning_value = best.belief_state.get(belief_key, 0.5)
                self.belief_graph.update_belief('_governor_', belief_key, winning_value, 0.9)
        
        return Resolution(
            conflict_type=ConflictType.BELIEF_CONFLICT,
            selected_output=best,
            rejected_outputs=rejected,
            reason=f"Belief conflict resolved by historical success: {best.agent_name}"
        )
    
    async def _resolve_resource_contention(self, conflict: Dict) -> Resolution:
        """حل التعارضات على الموارد"""
        outputs = conflict['outputs']
        sorted_outputs = sorted(outputs, key=lambda o: (o.priority, o.confidence), reverse=True)
        
        return Resolution(
            conflict_type=ConflictType.RESOURCE_CONTENTION,
            selected_output=sorted_outputs[0],
            rejected_outputs=sorted_outputs[1:],
            reason="Resource contention resolved by priority"
        )
    
    async def _resolve_infinite_loop(self, conflict: Dict) -> Resolution:
        """حل الحلقات اللانهائية"""
        outputs = conflict.get('outputs', [])
        if outputs:
            # خذ وكيلاً مختلفاً عن آخر من كان نشطاً
            last_agent = self.current_session.get('active_planner')
            alternative = next((o for o in outputs if o.agent_name != last_agent), outputs[0])
            
            return Resolution(
                conflict_type=ConflictType.INFINITE_LOOP,
                selected_output=alternative,
                rejected_outputs=[o for o in outputs if o != alternative],
                reason="Infinite loop broken by agent rotation"
            )
        
        return Resolution(
            conflict_type=ConflictType.INFINITE_LOOP,
            selected_output=AgentOutput("governor", AgentRole.CRITIC, "BREAK_LOOP", 1.0),
            rejected_outputs=[],
            reason="Emergency loop break"
        )
    
    async def _resolve_priority_inversion(self, conflict: Dict) -> Resolution:
        """حل انعكاس الأولوية"""
        outputs = conflict['outputs']
        corrected = sorted(outputs, key=lambda o: o.priority)
        
        return Resolution(
            conflict_type=ConflictType.PRIORITY_INVERSION,
            selected_output=corrected[0],
            rejected_outputs=corrected[1:],
            reason="Priority inversion corrected"
        )
    
    async def _apply_resolution(self, resolution: Resolution):
        """تطبيق قرار حل التعارض"""
        self.current_session['pending_outputs'] = [
            o for o in self.current_session['pending_outputs']
            if o not in resolution.rejected_outputs
        ]
        
        # تحديث معدلات نجاح الوكلاء
        for rejected in resolution.rejected_outputs:
            self.agents[rejected.agent_name]['success_rate'] *= 0.9
    
    async def execute_contract(self, agent_name: str, input_data: Any) -> Optional[Any]:
        """
        تنفيذ عقد بين وكلاء
        """
        contract = self.execution_contracts.get(agent_name)
        if not contract:
            return None
        
        self.current_session['current_contract'] = contract
        self.stats['contracts_executed'] += 1
        
        for attempt in range(contract.max_retries):
            try:
                # تنفيذ العقد مع timeout
                result = await asyncio.wait_for(
                    self._call_agent(agent_name, input_data),
                    timeout=contract.timeout_ms / 1000
                )
                return result
                
            except asyncio.TimeoutError:
                if attempt == contract.max_retries - 1 and contract.fallback_agent:
                    return await self.execute_contract(contract.fallback_agent, input_data)
                continue
            except Exception:
                continue
        
        self.stats['contracts_failed'] += 1
        return None
    
    async def _call_agent(self, agent_name: str, input_data: Any) -> Any:
        """استدعاء وكيل"""
        # في الإنتاج: استدعاء فعلي
        await asyncio.sleep(0.01)
        return {"result": f"Executed by {agent_name}"}
    
    def get_active_output(self) -> Optional[AgentOutput]:
        """الحصول على المخرج النشط الحالي"""
        if self.current_session['pending_outputs']:
            return max(self.current_session['pending_outputs'], 
                      key=lambda o: (o.priority, o.confidence))
        return None
    
    def record_action(self, action: str):
        """تسجيل إجراء للتتبع"""
        self.action_trace.append(action)
        self.current_session['active_planner'] = action
    
    def clear_session(self):
        """مسح الجلسة الحالية"""
        self.current_session = {
            'active_planner': None,
            'executing_agents': [],
            'pending_outputs': [],
            'start_time': time.time(),
            'current_contract': None
        }
    
    def get_status(self) -> Dict:
        """الحالة الحالية للحوكمة"""
        return {
            'registered_agents': list(self.agents.keys()),
            'agent_stats': {
                name: {
                    'role': info['role'].value,
                    'total_outputs': info['total_outputs'],
                    'success_rate': info['success_rate']
                }
                for name, info in self.agents.items()
            },
            'stats': self.stats,
            'belief_graph': self.belief_graph.get_status(),
            'current_session': {
                'pending_outputs': len(self.current_session['pending_outputs']),
                'session_duration_sec': time.time() - self.current_session['start_time'],
                'current_contract': self.current_session['current_contract'].agent_name 
                                    if self.current_session['current_contract'] else None
            },
            'recent_resolutions': [
                {
                    'type': r.conflict_type.value,
                    'selected': r.selected_output.agent_name,
                    'reason': r.reason
                }
                for r in list(self.resolution_history)[-5:]
            ]
        }

            # 5. تخزين في الذاكرة مع تضمين

            # 7. تحديث الإحصائيات
            self._update_stats(latency_ms, True)
            
            # 8. إعادة تعيين ميزانية الموجه
            self.modality_router.reset_budget()
            
            result = ProcessingResult(
                input_id=input_id,
                success=True,
